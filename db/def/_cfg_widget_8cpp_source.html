73  void CfgWidget::timerEvent(QTimerEvent* event) {
74 
75  (void)event;
76 
77  cfgParamInfo *theParam;
78  paramWrapper theWrapper, *theWrapper_p;
79  string path;
80  cfgPropertyStruct prop;
81 
82  // go throud the add list
83  addMutex.lock();
84  ignore_change = true;
85  while(addList.size() > 0) {
86  std::map<QString, QVariant> attr;
87  attr["singleStep"] = 0.01;
88  attr["decimals"] = 12;
89  theParam = &(addList[0]);
90 
91  // get the type and add the param to the gui
92  path = string("");
93  //path = string("../");
94  path.append(theParam->group);
95  path.append(string("/"));
96  path.append(theParam->name);
97 
98  prop.propertyIndex = 0;
99  prop.paramId = theParam->id;
100 
101  theWrapper.theParam = *theParam;
102  switch(theParam->type) {
103  case doubleParam:
104  prop.propertyType = doubleProperty;
105  cfg->getProperty(&prop);
106  theWrapper.guiElem = pDialog->addGenericProperty(path,
107  QVariant::Double,
108  prop.dValue, &attr);
109  theWrapper.theProp = prop;
110  break;
111  case intParam:
112  prop.propertyType = intProperty;
113  cfg->getProperty(&prop);
114  theWrapper.guiElem = pDialog->addGenericProperty(path, QVariant::Int, prop.iValue);
115  theWrapper.theProp = prop;
116  break;
117  case boolParam:
118  prop.propertyType = boolProperty;
119  cfg->getProperty(&prop);
120  theWrapper.guiElem = pDialog->addGenericProperty(path, QVariant::Bool,
121  prop.bValue);
122  theWrapper.theProp = prop;
123  break;
124  case stringParam:
125  prop.propertyType = stringProperty;
126  cfg->getProperty(&prop);
127  theWrapper.guiElem = pDialog->addGenericProperty(path, QVariant::String,
128  QString::fromStdString(prop.sValue));
129  theWrapper.theProp = prop;
130  break;
131  default:
132  theWrapper.guiElem = 0;
133  }
134  if(theWrapper.guiElem) paramList.push_back(theWrapper);
135  addList.erase(addList.begin());
136  }
137  ignore_change = false;
138  addMutex.unlock();
139 
140  // check for updates
141  //changeMutex.lock();
142  ignore_change = true;
143  while(changeList.size() > 0) {
144  theWrapper_p = getWrapperById(changeList[0]);
145  if(theWrapper_p && theWrapper_p->guiElem) {
146  cfg->getProperty(&theWrapper_p->theProp);
147  switch(theWrapper_p->theParam.type) {
148  case doubleParam:
149  theWrapper_p->guiElem->setValue(QVariant(theWrapper_p->theProp.dValue));
150  break;
151  case intParam:
152  theWrapper_p->guiElem->setValue(QVariant(theWrapper_p->theProp.iValue));
153  break;
154  case boolParam:
155  theWrapper_p->guiElem->setValue(QVariant(theWrapper_p->theProp.bValue));
156  break;
157  case stringParam:
158  theWrapper_p->guiElem->setValue(QVariant(QString::fromStdString(theWrapper_p->theProp.sValue)));
159  break;
160  case noParam:
161  case dstNrOfParamTypes:
162  break;
163  // don't supply a default case so that the compiler might warn
164  // us if we forget to handle a new enum value.
165  }
166  }
167  changeList.erase(changeList.begin());
168  }
169  ignore_change = false;
170  //changeMutex.unlock();
171 
172  // and go through the remove list ^^
173  removeMutex.lock();
174  ignore_change = true;
175  while (removeList.size() > 0) {
176  theWrapper_p = getWrapperById(removeList[0]);
177  pDialog->removeGenericProperty(theWrapper_p->guiElem);
178  removeList.erase(removeList.begin());
179  }
180  ignore_change = false;
181  removeMutex.unlock();
182  }

194  void CfgWidget::valueChanged(QtProperty *property, const QVariant &value) {
195  if(ignore_change) return;
196  vector<paramWrapper>::iterator iter;
197  double dValue;
198  int iValue;
199  bool bValue;
200  string sValue;
201 
202  for(iter=paramList.begin(); iter!=paramList.end(); ++iter) {
203  if(iter->guiElem == property) {
204  switch(iter->theParam.type) {
205  case doubleParam:
206  dValue = value.toDouble();
207  if(dValue != iter->theProp.dValue) {
208  iter->theProp.dValue = dValue;
209  cfg->setProperty(iter->theProp);
210  }
211  break;
212  case intParam:
213  iValue = value.toInt();
214  if(iValue != iter->theProp.iValue) {
215  iter->theProp.iValue = iValue;
216  cfg->setProperty(iter->theProp);
217  }
218  break;
219  case boolParam:
220  bValue = value.toBool();
221  if(bValue != iter->theProp.bValue) {
222  iter->theProp.bValue = bValue;
223  cfg->setProperty(iter->theProp);
224  }
225  break;
226  case stringParam:
227  sValue = value.toString().toStdString();
228  if(sValue != iter->theProp.sValue) {
229  iter->theProp.sValue = sValue;
230  cfg->setProperty(iter->theProp);
231  }
232 
233  break;
234  default:
235  break;
236  } // switch
237  break;
238  } // if
239  } // for
240  }
