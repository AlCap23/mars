35  bool DataPackageMapping::readPackage(const DataPackage &package) {

39  it != accessors.end(); ++it) {

54  bool DataPackageMapping::writePackage(DataPackage *package) {
55  /* Writing the package is a bit more difficult.
56  * We don't know if the DataItems exist. Even after the first run it might
57  * not exist because it could be an entirly new DataPackage.
58  * For perfomance reasons we abandon the "Look Before You Leap" style in
59  * favour of the "Easier to Ask for Forgiveness Than Permission" style.
60  * Only when setting a DataItem fails do we try to create it.
61  */
62  bool ret = true;
63  for(std::vector<DataItemAccessorBase*>::iterator it = accessors.begin();
64  it != accessors.end(); ++it) {
65  bool tmp = (*it)->setValue(package);
66  if(!tmp) {
67  ret = (ret && (*it)->createValue(package));
68  }
69  }
70  return ret;
71  }
