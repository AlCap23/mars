93  class GetVerticesFunctor {
94  public:
95  // This will store all triangle vertices.
96  std::vector<osg::Vec3> vertices;
97 
98  // When triangle vertices are added to 'vertices', they'll be transformed
99  // by this matrix. This is useful because 'osg::TriangleFunctor' operates
100  // on the model coordinate system, and we want do draw our normals in the
101  // world coordinate system.
102  osg::Matrix transformMatrix;
103 
104  // This will be called once for each triangle in the geometry. As
105  // parameters, it takes the three triangle vertices and a boolean
106  // parameter indicating if the vertices are coming from a "real" vertex
107  // array or from a temporary vertex array created from some other geometry
108  // representation.
109  // The implementation is quite simple: we just store the vertices
110  // (transformed by 'transformMatrix') in a 'std::vector'.
111  void operator() (const osg::Vec3& v1, const osg::Vec3& v2,
112  const osg::Vec3& v3, bool treatVertexDataAsTemporary) {
113  treatVertexDataAsTemporary = treatVertexDataAsTemporary;
114  vertices.push_back (v1 * transformMatrix);
115  vertices.push_back (v2 * transformMatrix);
116  vertices.push_back (v3 * transformMatrix);
117  }
118 
119  vector<osg::Vec3>& getVertices() {
120  return vertices;
121  }
122  };

154  snmesh GuiHelper::convertOsgNodeToSnMesh(osg::Node* node, double scaleX,
155  double scaleY, double scaleZ,
156  double pivotX, double pivotY,
157  double pivotZ) {
158  vector<int> indices;
159  vector<osg::Vec3> OSGvertices;
160  snmesh mesh;
161  //visitor for getting drawables inside node
162  GeodeVisitor* visitor=new GeodeVisitor("PLACEHOLDER");
163  osg::Geode* geode;
164  osg::Node* tmpNode;
165  osg::ref_ptr<osg::Group> osgGroupFromRead = NULL;
166  if ((osgGroupFromRead = node->asGroup()) == 0)
167  fprintf(stderr, "error\n");
168 
169  //get geometries of node
170  int indexcounter = 0;
171  for (size_t m = 0; m < osgGroupFromRead->getNumChildren(); m++) {
172  tmpNode = osgGroupFromRead->getChild(m);
173  tmpNode->accept(*visitor);
174  geode = visitor->getNode();
175  for (size_t j = 0; j<geode->getNumDrawables(); j++) {
176  //initialize functors for getting the drawables
177  osg::TriangleFunctor<GetVerticesFunctor> triangleFunctor_;
178  geode->getDrawable(j)->accept(triangleFunctor_);
179 
180  //Here we get the triangles
181  vector<osg::Vec3>& OSGverticestemp = triangleFunctor_.getVertices();
182  for (unsigned int i = 0; i < OSGverticestemp.size(); i++) {
183  OSGvertices.push_back(OSGverticestemp[i]);
184  indices.push_back(indexcounter);
185  indexcounter++;
186  }
187  }
188  }
189 
190  //store vertices in a mydVector3 structure
191  mars::interfaces::mydVector3 *vertices = 0;
192  if(OSGvertices.size() > 0){
193  vertices = new mars::interfaces::mydVector3[OSGvertices.size()];
194  }
195  // dVector3 *normals = new dVector3[normals_x.size()];
196  int *indexarray = 0;
197  if(indices.size() > 0){
198  indexarray = new int[indices.size()];
199  }
200 
201  //convert osg vertice vector to standard array
202  for (size_t i = 0; i < OSGvertices.size(); i++) {
203  vertices[i][0] = (OSGvertices[i][0] - pivotX) * scaleX;
204  vertices[i][1] = (OSGvertices[i][1] - pivotY) * scaleY;
205  vertices[i][2] = (OSGvertices[i][2] - pivotZ) * scaleZ;
206  }
207 
208  //construct an appropriate index array
209  for (int i = 0; i < indexcounter; i++) {
210  indexarray[i] = indices[i];
211  }
212 
213  mesh.vertices = vertices;
214  mesh.vertexcount = OSGvertices.size();
215  // mesh.normals = normals;
216  mesh.indices = indexarray;
217  mesh.indexcount = indexcounter;
218  delete visitor;
219  return mesh;
220  }

238  void GuiHelper::getPhysicsFromOBJ(mars::interfaces::NodeData* node) {

240  osg::ref_ptr<osg::Group> myCreatedGroup;
241  osg::ref_ptr<osg::Group> myGroupFromRead;

244  bool found = false;
245 
246  completeNode = GuiHelper::readNodeFromFile(node->filename);
247 
248  // check whether it is a osg::Group (.obj file)
249  if((myGroupFromRead = completeNode->asGroup()) != 0){
250  //go through the read node group and combine the parts of the actually
251  //handled node
252  myCreatedGroup = new osg::Group();
253  osg::ref_ptr<osg::StateSet> stateset = myCreatedGroup->getOrCreateStateSet();
254  for (unsigned int i = 0; i < myGroupFromRead->getNumChildren(); i ++) {
255  osg::ref_ptr<osg::Node> myTestingNode = myGroupFromRead->getChild(i);
256  if (myTestingNode == 0) {
257  return;
258  }
259  if (myTestingNode->getName() == node->origName) {
260  myTestingNode->setStateSet(stateset.get());
261  myCreatedGroup->addChild(myTestingNode.get());
262  found = true;
263  } else {
264  if (found) {
265  break;
266  found = false;
267  }
268  }
269  }
270  }
271  // or if it is a osg::Geode (.stl file)
272  else if((myGeodeFromRead = completeNode->asGeode()) != 0) {
273  //if the node was read from a .stl-file it read as geode not as group
274  myCreatedGroup = new osg::Group();
275  osg::ref_ptr<osg::StateSet> stateset = myCreatedGroup->getOrCreateStateSet();
276  completeNode->setStateSet(stateset.get());
277  myCreatedGroup->addChild(completeNode.get());
278  }
279 
280  osg::ComputeBoundsVisitor cbbv;
281  myCreatedGroup.get()->accept(cbbv);
282  osg::BoundingBox bb = cbbv.getBoundingBox();
283  Vector ex;
284  // compute bounding box has to be done in this way
285  (fabs(bb.xMax()) > fabs(bb.xMin())) ? ex.x() = fabs(bb.xMax() - bb.xMin())
286  : ex.x() = fabs(bb.xMin() - bb.xMax());
287  (fabs(bb.yMax()) > fabs(bb.yMin())) ? ex.y() = fabs(bb.yMax() - bb.yMin())
288  : ex.y() = fabs(bb.yMin() - bb.yMax());
289  (fabs(bb.zMax()) > fabs(bb.zMin())) ? ex.z() = fabs(bb.zMax() - bb.zMin())
290  : ex.z() = fabs(bb.zMin() - bb.zMax());
291 
292  //compute scale factor
293  double scaleX = 1, scaleY = 1, scaleZ = 1;
294  if (ex.x() != 0) scaleX = node->visual_size.x() / ex.x();
295  if (ex.y() != 0) scaleY = node->visual_size.y() / ex.y();
296  if (ex.z() != 0) scaleZ = node->visual_size.z() / ex.z();
297 
298  // create transform and group Node for the actual node
299  osg::ref_ptr<osg::PositionAttitudeTransform> transform;
300  osg::ref_ptr<osg::MatrixTransform> tx;
301 
302  transform = new osg::PositionAttitudeTransform();
303  tx = new osg::MatrixTransform;
304  tx->setMatrix(osg::Matrix::scale(scaleX, scaleY, scaleZ));
305  tx->setDataVariance(osg::Node::STATIC);
306  tx->addChild(myCreatedGroup.get());
307 
308  //add the node to a transformation to make him movable
309  transform->addChild(tx.get());
310  transform->setPivotPoint(osg::Vec3(node->pivot.x()*scaleX,
311  node->pivot.y()*scaleY,
312  node->pivot.z()*scaleZ));
313  transform->setPosition(osg::Vec3(node->pos.x() + node->visual_offset_pos.x(),
314  node->pos.y() + node->visual_offset_pos.y(),
315  node->pos.z() + node->visual_offset_pos.z()));
316  //set rotation
317  osg::Quat oquat;
318  Quaternion qrot = node->rot * node->visual_offset_rot;
319  oquat.set(qrot.x(), qrot.y(), qrot.z(), qrot.w());
320  transform->setAttitude(oquat);
321 
322  tempnode.transform = transform.get();
323  tempnode.node = myCreatedGroup.get();
324  tempnode.matrix = tx.get();
325  tempnode.offset = node->visual_offset_pos;
326  tempnode.r_off = node->visual_offset_rot;
327 
328  node->mesh = GuiHelper::convertOsgNodeToSnMesh(tempnode.node.get(),
329  scaleX, scaleY, scaleZ,
330  node->pivot.x(),
331  node->pivot.y(),
332  node->pivot.z());
333  }
