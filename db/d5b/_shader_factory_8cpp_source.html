31  string ShaderFactory::generateShaderSource(ShaderType shader_type) {
32  if (providers.count(shader_type) == 1) { // Provider for given shader type exists
33  IShaderProvider *u = providers[shader_type].get();
34  stringstream code;
35  if (u->getMinVersion() != 0)
36  code << "#version " << u->getMinVersion() << endl;
37  for (set<string>::iterator it = u->getEnabledExtensions().begin();
38  it != u->getEnabledExtensions().end(); ++it)
39  code << "#extension " << *it << " : enable" << endl;
40  for (set<string>::iterator it = u->getDisabledExtensions().begin();
41  it != u->getDisabledExtensions().end(); ++it)
42  code << "#extension " << *it << " : disable" << endl;
43  code << endl;
44 
45  for (set<GLSLUniform>::iterator it = u->getUniforms().begin();
46  it != u->getUniforms().end(); ++it)
47  code << "uniform " << *it << ";" << endl;
48 
49  for (set<GLSLConstant>::iterator it = u->getConstants().begin();
50  it != u->getConstants().end(); ++it)
51  code << "const " << *it << ";" << endl;
52  switch (shader_type) {
53  case SHADER_TYPE_VERTEX:
54  for (set<GLSLVarying>::iterator it = u->getVaryings().begin();
55  it != u->getVaryings().end(); ++it)
56  code << "varying " << *it << ";" << endl;
57  for (set<GLSLAttribute>::iterator it = u->getAttributes().begin();
58  it != u->getAttributes().end(); ++it)
59  code << "attribute " << *it << ";" << endl;
60  break;
61  case SHADER_TYPE_FRAGMENT:
62  for (set<GLSLVarying>::iterator it = u->getVaryings().begin();
63  it != u->getVaryings().end(); ++it)
64  code << "varying " << *it << ";" << endl;
65  break;
66  default:
67  break;
68  }
69  code << endl;
70 
71  std::vector<std::pair<std::string, std::string> > uDeps = u->getDependencies();
72  for (vector<pair<string, string> >::iterator it = uDeps.begin();
73  it != uDeps.end(); ++it)
74  code << it->second;
75 
76  code << endl;
77 
78  code << u->generateDefinitions() << endl;
79 
80  code << providers[shader_type].get()->generateMainSource();
81  return code.str();
82  } else {
83  return ""; // TODO: How to handle missing providers? Exception, empty string?
84  }
85  }

87  osg::Program *ShaderFactory::generateProgram() {
88  osg::Program *program = new osg::Program();
89 
90  if (providers.count(SHADER_TYPE_GEOMETRY) == 1) {
91  osg::Shader *shader = new osg::Shader(osg::Shader::GEOMETRY);
92  program->addShader(shader);
93  shader->setShaderSource(generateShaderSource(SHADER_TYPE_GEOMETRY));
94  }
95  if (providers.count(SHADER_TYPE_VERTEX) == 1) {
96  osg::Shader *shader = new osg::Shader(osg::Shader::VERTEX);
97  program->addShader(shader);
98  shader->setShaderSource(generateShaderSource(SHADER_TYPE_VERTEX));
99  }
100  if (providers.count(SHADER_TYPE_FRAGMENT) == 1) {
101  osg::Shader *shader = new osg::Shader(osg::Shader::FRAGMENT);
102  program->addShader(shader);
103  shader->setShaderSource(generateShaderSource(SHADER_TYPE_FRAGMENT));
104  }
105 
106  return program;
107  }
