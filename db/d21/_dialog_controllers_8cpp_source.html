157  void DialogControllers::on_remove_controller() {
158  filled = false;
159 
160  for (unsigned int i = 0; i < newControllers_p.size(); i++)
161  if (pDialog->currentItem() == newControllers_p[i]) {
162  delete newDialogs[i];
163  newDialogs.erase(newDialogs.begin() + i);
164  newControllers_p.erase(newControllers_p.begin() + i);
165  break;
166  }
167 
168  for (unsigned int i = 0; i < allControllers_p.size(); i++)
169  if (allControllers_p[i] == pDialog->currentItem()) {
170  control->controllers->removeController(allControllers[i].index);
171  allControllers.erase(allControllers.begin() + i);
172  delete allDialogs[i];
173  allDialogs.erase(allDialogs.begin() + i);
174  allControllers_p.erase(allControllers_p.begin() + i);
175  break;
176  }
177 
178  filled = true;
179 
180  if (allControllers_p.size() == 0 && newControllers_p.size() == 0) {
181  removeButton->hide();
182  addButton->hide();
183  } else if (allControllers_p.size() > 0) {
184  pDialog->setCurrentItem(allControllers_p[0]);
185  } else if (newControllers_p.size() > 0) {
186  pDialog->setCurrentItem(newControllers_p[0]);
187  }
188 
189  }

191  void DialogControllers::on_add_controller() {
192  filled = false;
193  for (unsigned int i = 0; i < newControllers_p.size(); i++)
194  if (pDialog->currentItem() == newControllers_p[i]) {
195  newDialogs[i]->accept();
196  allControllers_p.push_back(newControllers_p[i]);
197  allDialogs.push_back(newDialogs[i]);
198  newControllers_p.erase(newControllers_p.begin() + i);
199  newDialogs.erase(newDialogs.begin() + i);
200  topLevelItemChanged(allControllers_p.back());
201  break;
202  }
203 
204  control->controllers->getListController(&allControllers);
205 
206  filled = true;
207  addButton->hide();
208  }
