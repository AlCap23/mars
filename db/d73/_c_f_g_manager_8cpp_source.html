85  bool CFGManager::loadConfigFromStream(istream &in, const char *group) {
86  if(!in.good()) {
87  return false;
88  }
89  try {
90 #ifdef YAML_03_API
91  YAML::Parser parser(in);
92  YAML::Node doc;
93  YAML::Iterator it;
94  string currentGroup = "";
95 
96  while(parser.GetNextDocument(doc)) {
97  //cout << "Found document" << endl;
98 
99  for(it = doc.begin(); it != doc.end(); ++it) {
100  it.first() >> currentGroup;
101  //cout << "Found group: " << group << endl;
102  if(group && (currentGroup != group))
103  continue;
104 
105  const YAML::Node &paramNodes = it.second();
106  readGroup(currentGroup, paramNodes);
107  } // for
108  } // while
109 #else
110  YAML::const_iterator it;
111  string currentGroup = "";
112  std::vector<YAML::Node> doc = YAML::LoadAll(in);
113  std::vector<YAML::Node>::iterator dt = doc.begin();
114 
115  while(dt != doc.end()) {
116  //cout << "Found document" << endl;
117 
118  for(it = dt->begin(); it != dt->end(); ++it) {
119  currentGroup = it->first.as<std::string>();
120  //cout << "Found group: " << group << endl;
121  if(group && (currentGroup != group))
122  continue;
123 
124  readGroup(currentGroup, it->second);
125  } // for
126  ++dt;
127  } // while
128 #endif
129  } catch(YAML::ParserException &e) {
130  cout << e.what() << endl;
131  } catch(YAML::BadDereference &e) {
132  cout << e.what() << endl;
133  }
134  return true;
135  }

169  const std::string CFGManager::writeConfigToString(const char *group,
170  const unsigned char saveOption) const {
171  YAML::Emitter out;
172  mapStringToParam::const_iterator iter;
173  string lastGroup = "";
174  CFGParam *param = NULL;
175  out << YAML::BeginMap;
176  mutexCFGParams.lock();
177 
178  for(iter = cfgParamsByString.begin(); iter != cfgParamsByString.end(); ++iter) {
179  param = iter->second;
180  if(!group || param->getGroup() == string(group)) {
181  if( (param->getOption() & saveOption) || group) {
182  if(lastGroup != param->getGroup()) {
183  if(lastGroup != "") {
184  out << YAML::EndSeq;
185  }
186  lastGroup = param->getGroup();
187  out << YAML::Key << lastGroup;
188  out << YAML::Value << YAML::BeginSeq;
189  }
190  param->writeToYAML(out);
191  }
192  }
193 
194  } //for
195  out << YAML::EndSeq;
196  out << YAML::EndMap;
197 
198  mutexCFGParams.unlock();
199 
200  return out.c_str();
201  }

679  void CFGManager::readGroup(const string &group,
680  const YAML::Node &paramNodes) {
681  unsigned int i, j;
682  string type = "";
683  cfgParamId newId = 0;
684 
685  for(i = 0; i < paramNodes.size(); ++i) {
686 #ifdef YAML_03_API
687  paramNodes[i]["type"] >> type;
688 #else
689  type = paramNodes[i]["type"].as<std::string>();
690 #endif
691  //cout << "Found type: " << type << endl;
692 
693  for(j = 0; j < dstNrOfParamTypes; ++j) {
694  if( cfgParamTypeString[j] == type ) {
695  break;
696  }
697  } // for
698  CFGParam *param = NULL;
699  switch (j) {
700  case doubleParam :
701  newId = getNextId();
702  param = CFGParamDouble::createParam(newId, group, paramNodes[i]);
703  insertParam(param);
704  break;
705  case intParam :
706  newId = getNextId();
707  param = CFGParamInt::createParam(newId, group, paramNodes[i]);
708  insertParam(param);
709  break;
710  case boolParam :
711  newId = getNextId();
712  param = CFGParamBool::createParam(newId, group, paramNodes[i]);
713  insertParam(param);
714  break;
715  case stringParam :
716  newId = getNextId();
717  param = CFGParamString::createParam(newId, group, paramNodes[i]);
718  insertParam(param);
719  break;
720  default:
721  // do nothing
722  cout << "CFG: unknown type: " << type;
723  } // switch
724  if(newId != 0) {
725  addedCFGParam(newId);
726  newId = 0;
727  }
728  } // for
729  }

732  void CFGManager::insertParam(CFGParam *newParam) {
733  mutexCFGParams.lock();
734  cfgParamId id = newParam->getId();
735  mapStringToParam::const_iterator iter;
736  string stringID = newParam->getGroup() + ":" + newParam->getName();
737  iter = cfgParamsByString.find(stringID);
738 
739  // is there already this param?
740  if( iter != cfgParamsByString.end() ) {
741  // then get the values of the properties and override the old ones
742  CFGParam* oldParam = iter->second;
743  unsigned int i = 0;
744  for(i = 0; i < oldParam->getNrOfPropertys(); ++i) {
745  CFGProperty myProperty;
746  myProperty.setParamId(newParam->getId());
747  myProperty.setPropertyIndex(i);
748  myProperty.setPropertyType(newParam->getPropertyTypeByIndex(i));
749  if( newParam->getProperty(&myProperty) ) {
750  if( myProperty.isValueSet() ) {
751  myProperty.changeParamId(oldParam->getId());
752  oldParam->setProperty(myProperty);
753  }
754  } //if
755  } //for
756  // delete the new param which is no more needed
757  deleteParam(newParam);
758  } else {
759  string stringId = newParam->getGroup() + ":" + newParam->getName();
760  cfgParamsById.insert(pair<cfgParamId, CFGParam*>(id, newParam));
761  cfgParamsByString.insert(pair<string, CFGParam*>(stringId, newParam));
762  }
763  mutexCFGParams.unlock();
764  }

767  void CFGManager::deleteParam(CFGParam *param) {
