286  void InstancesVisitor::apply(osg::Node &searchNode){
287  // search for geometries and generate tangents for them
288  osg::Geode* geode=dynamic_cast<osg::Geode*>(&searchNode);
289  if(geode) {
290  for(unsigned int i=0; i<geode->getNumDrawables(); ++i) {
291  osg::Geometry* geom=dynamic_cast<osg::Geometry*>(geode->getDrawable(i));
292  if(geom) {
293  geom->setInitialBound(osg::BoundingBox(0, 0, 0, 2, 2, 1));
294  generateInstances(geom);
295  }
296  }
297  }
298 
299  traverse(searchNode);
300  }

302  void InstancesVisitor::generateInstances(osg::Geometry *geom) {
303  // first turn on hardware instancing for every primitive set
304  for (unsigned int i = 0; i < geom->getNumPrimitiveSets(); ++i) {
305  geom->getPrimitiveSet(i)->setNumInstances(numInstances);
306  fprintf(stderr, "set num instances: %d\n", numInstances);
307  }
308  // we need to turn off display lists for instancing to work
309  geom->setUseDisplayList(false);
310  geom->setUseVertexBufferObjects(true);
311  }
