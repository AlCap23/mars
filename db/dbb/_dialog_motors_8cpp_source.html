169  void DialogMotors::on_remove_motor() {
170  filled = false;
171 
172  for (unsigned int i = 0; i < newMotors_p.size(); i++)
173  if (pDialog->currentItem() == newMotors_p[i]) {
174  delete newDialogs[i];
175  newDialogs.erase(newDialogs.begin() + i);
176  newMotors_p.erase(newMotors_p.begin() + i);
177  break;
178  }
179 
180  for (unsigned int i = 0; i < allMotors_p.size(); i++)
181  if (allMotors_p[i] == pDialog->currentItem()) {
182  control->motors->removeMotor(allMotors[i].index);
183  allMotors.erase(allMotors.begin() + i);
184  delete allDialogs[i];
185  allDialogs.erase(allDialogs.begin() + i);
186  allMotors_p.erase(allMotors_p.begin() + i);
187  break;
188  }
189 
190  filled = true;
191 
192  if (allMotors_p.size() == 0 && newMotors_p.size() == 0) {
193  removeButton->hide();
194  addButton->hide();
195  } else if (allMotors_p.size() > 0) {
196  pDialog->setCurrentItem(allMotors_p[0]);
197  } else if (newMotors_p.size() > 0) {
198  pDialog->setCurrentItem(newMotors_p[0]);
199  }
200 
201  }

203  void DialogMotors::on_add_motor() {
204  filled = false;
205  for (unsigned int i = 0; i < newMotors_p.size(); i++)
206  if (pDialog->currentItem() == newMotors_p[i]) {
207  newDialogs[i]->accept();
208  allMotors_p.push_back(newMotors_p[i]);
209  allDialogs.push_back(newDialogs[i]);
210  newMotors_p.erase(newMotors_p.begin() + i);
211  newDialogs.erase(newDialogs.begin() + i);
212  topLevelItemChanged(allMotors_p.back());
213  break;
214  }
215 
216  control->motors->getListMotors(&allMotors);
217 
218  filled = true;
219  addButton->hide();
220  }
