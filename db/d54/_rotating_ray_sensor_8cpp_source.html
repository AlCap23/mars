57  RotatingRaySensor::RotatingRaySensor(ControlCenter *control, RotatingRayConfig config):
58  BasePolarIntersectionSensor(config.id,
59  config.name,
60  config.bands*config.lasers,
61  1,
62  config.opening_width,
63  config.opening_height),
64  SensorInterface(control),
65  config(config) {
66 
67  updateRate = config.updateRate;
68  orientation.setIdentity();
69  maxDistance = config.maxDistance;
70  turning_offset = 0.0;
71  current_pose.setIdentity();
72  num_points = 0;
73 
83  this->attached_node = config.attached_node;
84 
85  std::string groupName, dataName;
86  drawStruct draw;
87  draw_item item;
88  Vector tmp;
89  have_update = false;
90  for(int i = 0; i < 3; ++i)
91  positionIndices[i] = -1;
92  for(int i = 0; i < 4; ++i)
93  rotationIndices[i] = -1;
94 
95  bool erg = control->nodes->getDataBrokerNames(attached_node, &groupName, &dataName);
96  if(!erg) { // To remove warning.
97  assert(erg);
98  }
99  if(control->dataBroker->registerTimedReceiver(this, groupName, dataName,"mars_sim/simTimer",updateRate)) {
100  }
101 
102  position = control->nodes->getPosition(attached_node);
103  orientation = control->nodes->getRotation(attached_node);
104  orientation_offset.setIdentity();
105 
106  // Fills the direction array.
107  if(config.bands < 1) {
108  std::cerr << "Number of bands too low("<< config.bands <<"),will be set to 1" << std::endl;
109  config.bands = 1;
110  }
111  if(config.lasers < 1) {
112  std::cerr << "Number of lasers too low("<< config.lasers <<"),will be set to 1" << std::endl;
113  config.lasers = 1;
114  }
115 
116  // All bands will be turned from 'turning_offset' to 'turning_end_fullscan' in 'turning_step steps'.
117  turning_offset = 0;
118  turning_end_fullscan = config.opening_width / config.bands;
119  turning_step = config.horizontal_resolution;
120 
121  double vAngle = config.lasers <= 1 ? config.opening_height/2.0 : config.opening_height/(config.lasers-1);
122  double hAngle = config.bands <= 1 ? 0 : config.opening_width/config.bands;
123 
124  for(int b=0; b<config.bands; ++b) {
125  for(int l=0; l<config.lasers; ++l) {
126  tmp = Eigen::AngleAxisd(b*hAngle - config.opening_width / 2.0 + config.horizontal_offset, Eigen::Vector3d::UnitZ()) *
127  Eigen::AngleAxisd(l*vAngle - config.opening_height / 2.0 + config.vertical_offset, Eigen::Vector3d::UnitY()) *
128  Vector(1,0,0);
129 
130  directions.push_back(tmp);
131 
132  // Add a drawing item for each ray regarding the initial sensor orientation.
133  if(config.draw_rays) {
134  draw.ptr_draw = (DrawInterface*)this;
135  item.id = 0;
136  item.type = DRAW_LINE;
137  item.draw_state = DRAW_STATE_CREATE;
138  item.point_size = 1;
139  item.myColor.r = 1;
140  item.myColor.g = 0;
141  item.myColor.b = 0;
142  item.myColor.a = 1;
143  item.texture = "";
144  item.t_width = item.t_height = 0;
145  item.get_light = 0.0;
146 
147  // Initial vector length is set to 1.0
148  item.start = position;
149  item.end = orientation * tmp;
150  draw.drawItems.push_back(item);
151  }
152  }
153  }
154 
155  // Add sensor after everything has been initialized.
156  control->nodes->addNodeSensor(this);
157 
158  // GraphicsManager crashes if default constructor drawStruct is passed.
159  if(config.draw_rays) {
160  if(control->graphics) {
161  control->graphics->addDrawItems(&draw);
162  }
163  }
164  //assert(N == data.size());
165 
166  }

180  int RotatingRaySensor::getSensorData(double** data_) const {
181  mars::utils::MutexLocker lock(&mutex_pointcloud);
182  *data_ = (double*)malloc(pointcloud_full.size()*3*sizeof(double));
183  for(unsigned int i=0; i<pointcloud_full.size(); i++) {
184  if((pointcloud_full[i]).norm() <= config.maxDistance) {
185  int array_pos = i*3;

194  void RotatingRaySensor::receiveData(const data_broker::DataInfo &info,
195  const data_broker::DataPackage &package,
196  int callbackParam) {
197  CPP_UNUSED(info);
198  CPP_UNUSED(callbackParam);
199  long id;
200  package.get(0, &id);
201 
202  if(positionIndices[0] == -1) {
203  positionIndices[0] = package.getIndexByName("position/x");
204  positionIndices[1] = package.getIndexByName("position/y");
205  positionIndices[2] = package.getIndexByName("position/z");
206  rotationIndices[0] = package.getIndexByName("rotation/x");
207  rotationIndices[1] = package.getIndexByName("rotation/y");
208  rotationIndices[2] = package.getIndexByName("rotation/z");
209  rotationIndices[3] = package.getIndexByName("rotation/w");
210  }
211  for(int i = 0; i < 3; ++i)
212  {
213  package.get(positionIndices[i], &position[i]);
214  }
215 
216  Eigen::Vector3d position;
217  package.get(positionIndices[0], &position.x());
218  package.get(positionIndices[1], &position.y());
219  package.get(positionIndices[2], &position.z());
220  package.get(rotationIndices[0], &orientation.x());
221  package.get(rotationIndices[1], &orientation.y());
222  package.get(rotationIndices[2], &orientation.z());
223  package.get(rotationIndices[3], &orientation.w());
224 
225  current_pose.setIdentity();
226  current_pose.rotate(orientation);
227  current_pose.translation() = position;
228 
229  // Fills the pointcloud vector with (dist_m, x, y, z).
230  // data[] contains all the measured distances.
231  for(unsigned int i=0; i<data.size(); i++) {
232  if (data[i] < config.maxDistance) {
233  // Calculates the ray/vector within the sensor frame.
234  utils::Vector local_ray = orientation_offset * directions[i] * data[i];
235  // Gathers pointcloud in the world frame to prevent/reduce movement distortion.
236  // This necessitates a back-transformation (world2node) in getPointcloud().
237  utils::Vector tmpvec = current_pose * local_ray;
238  pointcloud.push_back(tmpvec); // Scale normalized vector.
239  }
240  }
241  num_points += data.size();
242 
243  have_update = true;
244  }

246  void RotatingRaySensor::update(std::vector<draw_item>* drawItems) {
247 
248  unsigned int i;
249 
250  if(have_update) {
251  control->nodes->updateRay(attached_node);
252  have_update = false;
253  }
254  if(config.draw_rays) {
255  if(!(*drawItems)[0].draw_state) {
256  for(i=0; i<data.size(); i++) {
257  (*drawItems)[i].draw_state = DRAW_STATE_UPDATE;
258  // Updates the rays using the current sensor pose.
259  (*drawItems)[i].start = position;
260  (*drawItems)[i].end = (orientation * orientation_offset * directions[i]);
261  (*drawItems)[i].end *= data[i];
262  (*drawItems)[i].end += (*drawItems)[i].start;
263  }
264  }
265  }

268  utils::Quaternion RotatingRaySensor::turn() {
269 
270  // If the scan is full the pointcloud will be copied.
271  mutex_pointcloud.lock();
272  turning_offset += turning_step;
273  if(turning_offset >= turning_end_fullscan) {
274  // Copies current full pointcloud to pointcloud_full.
275  std::list<utils::Vector>::iterator it = pointcloud.begin();
276  pointcloud_full.resize(pointcloud.size());
277  for(int i=0; it != pointcloud.end(); it++, i++) {
278  // Transforms the pointcloud back from world to current node (see receiveDate()).
279  // In addition 'transf_sensor_rot_to_sensor' is applied which describes
280  // the orientation of the sensor in the unturned sensor frame.
281  Eigen::Affine3d rot;
282  rot.setIdentity();
283  rot.rotate(config.transf_sensor_rot_to_sensor);
284  pointcloud_full[i]= rot * current_pose.inverse() * (*it);
285  }
286  pointcloud.clear();
287  turning_offset = 0;
288  }
289  orientation_offset = utils::angleAxisToQuaternion(turning_offset, utils::Vector(0.0, 0.0, 1.0));
290  mutex_pointcloud.unlock();
291 
292  return orientation_offset;
293  }

320  cfg->maxDistance = it->second[0].getDouble();
321  if((it = config->find("draw_rays")) != config->end())
322  cfg->draw_rays = it->second[0].getBool();
323  if((it = config->find("horizontal_offset")) != config->end())
324  cfg->horizontal_offset = it->second[0].getDouble();
325  if((it = config->find("vertical_offset")) != config->end())
326  cfg->vertical_offset = it->second[0].getDouble();
327  if((it = config->find("rate")) != config->end())
328  cfg->updateRate = it->second[0].getULong();
329  if((it = config->find("horizontal_resolution")) != config->end())
330  cfg->horizontal_resolution = it->second[0].getDouble();
331  cfg->attached_node = attachedNodeID;
332 
333  ConfigMap::iterator it2;
334  if((it = config->find("rotation_offset")) != config->end()) {
335  if((it2 = it->second[0].children.find("yaw")) !=
336  it->second[0].children.end()) {
337  Vector euler;
338  euler.x() = it->second[0].children["roll"][0].getDouble();
339  euler.y() = it->second[0].children["pitch"][0].getDouble();
340  euler.z() = it->second[0].children["yaw"][0].getDouble();
341  cfg->transf_sensor_rot_to_sensor = eulerToQuaternion(euler);

343  else {
344  Quaternion q;

346  q.y() = it->second[0].children["y"][0].getDouble();

348  q.w() = it->second[0].children["w"][0].getDouble();
349  cfg->transf_sensor_rot_to_sensor = q;
