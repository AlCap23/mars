108  void PythonMars::interpreteMap(ConfigItem &map) {
109  if(map.isMap()) {
110  if(map.hasKey("startSim") && (bool)map["startSim"]) {
111  control->sim->StartSimulation();
112  ConfigMap::iterator it = map.find("startSim");
113  map.erase(it);
114  }
115  if(map.hasKey("stopSim") && (bool)map["stopSim"]) {
116  control->sim->StopSimulation();
117  ConfigMap::iterator it = map.find("stopSim");
118  map.erase(it);
119  }
120  if(map.hasKey("updateTime")) {
121  updateTime = map["updateTime"];
122  ConfigMap::iterator it = map.find("updateTime");
123  map.erase(it);
124  }
125  if(map.hasKey("log")) {
126  if(map["log"].hasKey("debug")) {
127  ConfigVector::iterator it = map["log"]["debug"].begin();
128  for(; it!=map["log"]["debug"].end(); ++it) {
129  LOG_DEBUG((std::string)*it);
130  }
131  }
132  if(map["log"].hasKey("error")) {
133  ConfigVector::iterator it = map["log"]["error"].begin();
134  for(; it!=map["log"]["error"].end(); ++it) {
135  LOG_ERROR((std::string)*it);
136  }
137  }
138  ConfigMap::iterator it = map.find("log");
139  map.erase(it);
140  }
141 
142  if(map.hasKey("commands") && control->sim->isSimRunning()) {
143  ConfigMap::iterator it = map["commands"].beginMap();
144  for(; it!=map["commands"].endMap(); ++it) {
145  std::string name = it->first;
146  if(it->second.hasKey("value")) {
147  double value = it->second["value"];
148  if(motorMap.find(name) == motorMap.end()) {
149  unsigned long id = control->motors->getID(name);
150  if(id) {
151  motorMap[name] = id;
152  control->motors->setMotorValue(id, value);
153  }
154  }
155  else {
156  control->motors->setMotorValue(motorMap[name], value);
157  }
158  }
159  }
160  ConfigMap::iterator iit = map.find("commands");
161  map.erase(iit);
162  }
163 
164  if(map.hasKey("config")) {
165  ConfigMap::iterator it = map["config"].beginMap();
166  for(; it!=map["config"].endMap(); ++it) {
167  std::string group = it->first;
168  if(!it->second.isMap()) continue;
169  ConfigMap::iterator it2 = it->second.beginMap();
170  if(!it2->second.isAtom()) continue;
171  ConfigAtom &atom = it2->second;
172  std::string name = it2->first;
173  std::string value = atom.toString().c_str();
174  cfg_manager::cfgParamInfo info;
175  info = control->cfg->getParamInfo(group, name);
176  switch(info.type) {
177  case cfg_manager::boolParam:
178  control->cfg->setProperty(group, name,
179  (bool)atoi(value.c_str()));
180  break;
181  case cfg_manager::doubleParam:
182  control->cfg->setProperty(group, name, atof(value.c_str()));
183  break;
184  case cfg_manager::intParam:
185  control->cfg->setProperty(group, name, atoi(value.c_str()));
186  break;
187  case cfg_manager::stringParam:
188  control->cfg->setProperty(group, name, value);
189  break;
190  case cfg_manager::noParam:
191  switch(atom.getType()) {
192  case ConfigAtom::BOOL_TYPE:
193  control->cfg->getOrCreateProperty(group, name, (bool)atom);
194  break;
195  case ConfigAtom::INT_TYPE:
196  control->cfg->getOrCreateProperty(group, name, (int)atom);
197  break;
198  case ConfigAtom::DOUBLE_TYPE:
199  control->cfg->getOrCreateProperty(group, name, (double)atom);
200  break;
201  default:
202  control->cfg->getOrCreateProperty(group, name,
203  atom.toString());
204  break;
205  }
206  break;
207  default:
208  break;
209  }
210  }
211  ConfigMap::iterator iit = map.find("config");
212  map.erase(iit);
213  }
214 
215  if(map.hasKey("request") && map["request"].isVector()) {
216  requestMap = map["request"];
217  ConfigMap::iterator it = map.find("request");
218  map.erase(it);
219  }
220 
221  guiMapMutex.lock();
222  guiMaps.push_back(map);
223  guiMapMutex.unlock();
224 
225  }
226  nextStep = true;
227  }

229  void PythonMars::interpreteGuiMaps() {
230  guiMapMutex.lock();
231  std::vector<ConfigMap>::iterator it = guiMaps.begin();
232  for(; it!=guiMaps.end(); ++it) {
233  ConfigMap &map = *it;
234  if(map.hasKey("PointCloud") && map["PointCloud"].isMap()) {
235  mutex.lock();
236  ConfigMap::iterator it = map["PointCloud"].beginMap();
237  for(; it!=map["PointCloud"].endMap(); ++it) {
238  std::string name = it->first;
239  if(points.find(name) != points.end()) {
240  PointStruct &point = points[name];
241  control->graphics->removeOSGNode(point.p->getOSGNode());
242  delete point.data;
243  delete point.pydata;
244  points.erase(name);
245  }
246  // create point cloud
247  PointStruct point;
248  point.p = pf->createPoints();
249  point.p->setLineWidth(3.0);
250  point.size = it->second;
251  point.data = new double[point.size*3];
252  point.pydata = new double[point.size*3];
253  std::vector<osg_points::Vector> pV;
254  pV.reserve(point.size);
255  for(int i=0; i<point.size; ++i) {
256  point.data[i*3] = ((double)i)/point.size*2;
257  point.data[i*3+1] = (i%4)*0.1;
258  point.data[i*3+2] = 1;
259  pV.push_back(osg_points::Vector(point.data[i*3],
260  point.data[i*3+1],
261  point.data[i*3+2]));
262  }
263  points[name] = point;
264  point.p->setData(pV);
265  plugin->function("addPointCloudData").pass(STRING).pass(ONEDCARRAY).call(&name, point.pydata, point.size*3);
266  control->graphics->addOSGNode(point.p->getOSGNode());
267  }
268  mutex.unlock();
269  }
270  if(map.hasKey("CameraSensor") && map["CameraSensor"].isMap()) {
271  mutexCamera.lock();
272  ConfigMap::iterator it = map["CameraSensor"].beginMap();
273  for(; it!=map["CameraSensor"].endMap(); ++it) {
274  std::string name = it->first;
275  if(cameras.find(name) == cameras.end()) {
276  unsigned long id = control->sensors->getSensorID(name);
277  sReal *data;
278  int num = control->sensors->getSensorData(id, &data);
279  if(num) {
280  CameraStruct cam = {id, data, NULL, num};
281  cam.pydata = (sReal*)malloc(num*sizeof(sReal));
282  cameras[name] = cam;
283  plugin->function("addCameraData").pass(STRING).pass(ONEDCARRAY).call(&name, cam.pydata, num);
284  }
285  }
286  else {
287  CameraStruct &cam = cameras[name];
288  sReal *data;
289  int num = control->sensors->getSensorData(cam.id, &data);
290  if(num == cam.size) {
291  memcpy(cam.data, data, num*sizeof(sReal));
292  }
293  if(num) free(data);
294  }
295  }
296  mutexCamera.unlock();
297  }
298 
299  if(map.hasKey("ConfigPointCloud")) {
300  ConfigMap::iterator it = map["ConfigPointCloud"].beginMap();
301  for(; it!=map["ConfigPointCloud"].endMap(); ++it) {
302  if(points.find(it->first) == points.end()) continue;
303  points[it->first].p->setLineWidth(it->second[0]);
304  osg_points::Color c((double)it->second[1], (double)it->second[2],
305  (double)it->second[3], 1.0);
306  points[it->first].p->setColor(c);
307  }
308  }
309 
310  if(map.hasKey("Lines")) {
311  ConfigMap::iterator it = map["Lines"].beginMap();
312  for(; it!=map["Lines"].endMap(); ++it) {
313  const std::string &name = it->first;
314  ConfigVector::iterator it2 = it->second.begin();
315  for(; it2!=it->second.end(); ++it2) {
316  if(it2->isAtom()) {
317  if(lines.find(name) == lines.end()) continue;
318  std::string cmd = *it2;
319  if(cmd == "clear") {
320  lines[name].l->setData(std::list<osg_lines::Vector>());
321  }
322  else if(cmd == "remove") {
323  control->graphics->removeOSGNode(lines[name].l->getOSGNode());
324  lines.erase(name);
325  }
326  }
327  else {
328  ConfigMap &cmd = *it2;
329  if(lines.find(name) == lines.end()) {
330  LineStruct line;
331  line.l = lf->createLines();
332  line.l->setLineWidth(3.0);
333  line.l->drawStrip(false);
334  lines[name] = line;
335  control->graphics->addOSGNode(line.l->getOSGNode());
336  }
337  if(cmd.hasKey("append")) {
338  LineStruct &ls = lines[name];
339  osg_lines::Vector v((double)cmd["append"][0],
340  (double)cmd["append"][1],
341  (double)cmd["append"][2]);
342  ls.toAppend.push_back(v);
343  }
344  if(cmd.hasKey("config")) {
345  lines[name].l->setLineWidth((double)cmd["config"][0]);
346  osg_lines::Color c((double)cmd["config"][1],
347  (double)cmd["config"][2],
348  (double)cmd["config"][3], 1.0);
349  lines[name].l->setColor(c);
350  }
351  }
352  }
353  }
354  }
355  if(map.hasKey("ToDataBroker")) {
356  for(auto it:map["ToDataBroker"]) {
357  data_broker::DataInfo di;
358  data_broker::DataPackage dp;
359  std::string g, n, d;
360  double v = it["v"];
361  g << it["g"];
362  n << it["n"];
363  d << it["d"];
364  di = control->dataBroker->getDataInfo(g, n);
365  if(di.dataId > 0) {
366  dp = control->dataBroker->getDataPackage(di.dataId);
367  }
368  long index = dp.getIndexByName(d);
369  if(index == -1) {
370  dp.add(d, v);
371  }
372  else {
373  dp.set(index, v);
374  }
375  control->dataBroker->pushData(g, n, dp, 0,
376  data_broker::DATA_PACKAGE_READ_FLAG);
377  }
378  }
379  }
380  nextStep = true;
381  guiMaps.clear();
382  guiMapMutex.unlock();
383  }

390  void PythonMars::update(sReal time_ms) {
391  static double nextUpdate = 0.0;
392  if(time_ms > 0) {
393  gpMutex.lock();
394  if(updateTime > 0) {
395  nextUpdate += time_ms;
396  if(nextUpdate > updateTime) {
397  nextUpdate = fmod(nextUpdate, updateTime);
398  }
399  else {
400  gpMutex.unlock();
401  return;
402  }
403  }
404  if(pythonException) {
405  gpMutex.unlock();
406  return;
407  }
408  while(!nextStep) utils::msleep(2);
409  ConfigMap sendMap;
410 
411  ConfigVector::iterator it = requestMap.begin();
412  for(; it!=requestMap.end(); ++it) {
413  if(!it->hasKey("type")) continue;
414  if(!it->hasKey("name")) continue;
415  std::string type = (*it)["type"];
416  std::string name = (*it)["name"];
417 
418  if(type == "Node") {
419  unsigned long id = control->nodes->getID(name);
420  Vector pos = control->nodes->getPosition(id);
421  Quaternion rot = control->nodes->getRotation(id);
422  sendMap["Nodes"][name]["pos"]["x"] = pos.x();
423  sendMap["Nodes"][name]["pos"]["y"] = pos.y();
424  sendMap["Nodes"][name]["pos"]["z"] = pos.z();
425  sendMap["Nodes"][name]["rot"]["x"] = rot.x();
426  sendMap["Nodes"][name]["rot"]["y"] = rot.y();
427  sendMap["Nodes"][name]["rot"]["z"] = rot.z();
428  sendMap["Nodes"][name]["rot"]["w"] = rot.w();
429  }
430 
431  if(type == "Sensor") {
432  unsigned long id = control->sensors->getSensorID(name);
433  sReal *data;
434  int num = control->sensors->getSensorData(id, &data);
435  for(int i=0; i<num; ++i) {
436  sendMap["Sensors"][name][i] = data[i];
437  }
438  if(num) free(data);
439  }
440 
441  if(type == "Config") {
442  if(!it->hasKey("group")) continue;
443  std::string group = (*it)["group"];
444  cfg_manager::cfgParamInfo info;
445  info = control->cfg->getParamInfo(group, name);
446  switch(info.type) {
447  case cfg_manager::boolParam:
448  {
449  bool v;
450  control->cfg->getPropertyValue(group, name, "value", &v);
451  sendMap["Config"][group][name] = v;
452  break;
453  }
454  case cfg_manager::doubleParam:
455  {
456  double v;
457  control->cfg->getPropertyValue(group, name, "value", &v);
458  sendMap["Config"][group][name] = v;
459  break;
460  }
461  case cfg_manager::intParam:
462  {
463  int v;
464  control->cfg->getPropertyValue(group, name, "value", &v);
465  sendMap["Config"][group][name] = v;
466  break;
467  }
468  case cfg_manager::stringParam:
469  {
470  std::string v;
471  control->cfg->getPropertyValue(group, name, "value", &v);
472  sendMap["Config"][group][name] = v;
473  break;
474  }
475  default:
476  break;
477  }
478  }
479 
480  }
481  try {
482  iMap = ConfigItem();
483  mutexCamera.lock();
484  { // udpate point clouds
485  std::map<std::string, CameraStruct>::iterator it = cameras.begin();
486  for(; it!=cameras.end(); ++it) {
487  memcpy(it->second.pydata, it->second.data,
488  it->second.size*sizeof(sReal));
489  }
490  }
491  mutexCamera.unlock();
492  mutex.lock();
493  toConfigMap(plugin->function("update").pass(MAP).call(&sendMap).returnObject(), iMap);
494  nextStep = true;
495  mutex.unlock();
496  mutexPoints.lock();
497  { // udpate point clouds
498  std::map<std::string, PointStruct>::iterator it = points.begin();
499  for(; it!=points.end(); ++it) {
500  memcpy(it->second.data, it->second.pydata,
501  it->second.size*3*sizeof(double));
502  }
503  }
504  mutexPoints.unlock();
505  interpreteMap(iMap);
506  }
507  catch(const std::exception &e) {
508  LOG_FATAL("Error: %s", e.what());
509  pythonException = true;
510  mutex.unlock();
511  }
512  updateGraphics = true;
513  gpMutex.unlock();
514  }
515  else {
516  if(updateGraphics) {
517  interpreteGuiMaps();
518  mutexPoints.lock();
519  { // udpate point clouds
520  std::map<std::string, PointStruct>::iterator it = points.begin();
521  for(; it!=points.end(); ++it) {
522  std::vector<osg_points::Vector> pV;
523  pV.reserve(it->second.size);
524  for(int i=0; i<it->second.size; ++i) {
525  pV.push_back(osg_points::Vector(it->second.data[i*3],
526  it->second.data[i*3+1],
527  it->second.data[i*3+2]));
528  }
529  it->second.p->setData(pV);
530  }
531  }
532  mutexPoints.unlock();
533 
534  { // update lines
535  std::map<std::string, LineStruct>::iterator it = lines.begin();
536  for(; it!=lines.end(); ++it) {
537  std::vector<osg_lines::Vector>::iterator vt;
538  for(vt=it->second.toAppend.begin();
539  vt!=it->second.toAppend.end(); ++vt) {
540  it->second.l->appendData(*vt);
541  }
542  it->second.toAppend.clear();
543  }
544  }
545  updateGraphics = false;
546  }
547  }
