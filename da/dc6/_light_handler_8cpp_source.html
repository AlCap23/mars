32  LightHandler::LightHandler(QtVariantProperty *property, int ind,
33  main_gui::PropertyDialog *pd,
34  interfaces::ControlCenter *c) {
35 
36  control = c;
37  topLevel = property;
38  recompileShader = false;
39  myLightIndex = ind;
40  pDialog = pd;
41  filled = false;
42  if (myLightIndex < 0) {
43  interfaces::LightData light;
44  light.pos.z() = 3.00;
45  light.lookAt.z() = -1.00;
46  light.name = "NewLight" + QString::number(local_index).toStdString();
47  light.constantAttenuation = 1.0000;
48  light.quadraticAttenuation = 0.00002;
49  light.angle = 180;
50  light.directional = false;
51  light.ambient = to_my_color(QColor(50,50,50,255));
52  light.diffuse = to_my_color(QColor(255,255,255,255));
53  light.specular = to_my_color(QColor(255,255,255,255));
54  control->graphics->addLight(light);
55  vector<interfaces::LightData*> tmp;
56  control->graphics->getLights(&tmp);
57  for (unsigned int i = 0; i <tmp.size(); i++)
58  if (tmp[i]->name == "NewLight" + QString::number(local_index).toStdString()) {
59  myLight = tmp[i];
60  break;
61  }
62  local_index++;
63  } else {
64  vector<interfaces::LightData*> tmp;
65  control->graphics->getLights(&tmp);
66  for (unsigned int i = 0; i <tmp.size(); i++)
67  if (tmp[i]->index == myLightIndex) {
68  myLight = tmp[i];
69  break;
70  }
71  }
72 
73  topLevel->setPropertyName(QString::number(myLight->index) + ":" + QString::fromStdString(myLight->name));
74  propName = topLevel->propertyName().toStdString();
75  fill();
76  on_switched_type();
77  filled = true;
78  control->graphics->updateLight(myLight->index);
79  }

171  void LightHandler::update(interfaces::LightData* light) {
172  light->pos.x() = pos_x->value().toDouble();
173  light->pos.y() = pos_y->value().toDouble();
174  light->pos.z() = pos_z->value().toDouble();
175  light->lookAt.x() = look_x->value().toDouble();
176  light->lookAt.y() = look_y->value().toDouble();
177  light->lookAt.z() = look_z->value().toDouble();
178  light->linearAttenuation = linear->value().toDouble();
179  light->constantAttenuation = constant->value().toDouble();
180  light->quadraticAttenuation = quadratic->value().toDouble();
181  light->type = type->value().toInt() + 1;
182  light->exponent = exponent->value().toDouble();
183  light->angle = cutoff->value().toDouble();
184  if(light->directional != directional->value().toBool()) recompileShader = true;
185  light->directional = directional->value().toBool();
186  light->ambient = to_my_color(ambient->value().value<QColor>());
187  light->diffuse = to_my_color(diffuse->value().value<QColor>());
188  light->specular = to_my_color(specular->value().value<QColor>());
189  light->name = name->value().toString().toStdString();
190  }
