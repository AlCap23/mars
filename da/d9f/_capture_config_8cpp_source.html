96  // init capture

101

103  logFolder.append("_win_");

105  logFolder.append(num);
106 
107  // create dir
108  if(QDir().mkdir(logFolder)) {
109  fprintf(stderr, "created folder: %s\n",
110  qPrintable(logFolder));
111  if(gw) gw->setGrabFrames(true);
112  imageProcess = new ImageProcess(logFolder, frame_rate);
113  }
114  else {
115  fprintf(stderr, "error while creating folder\n");
116  capture = false;
117  }
118 
119  // update capture
120  while(capture) {
121  if(gw) {
122  data = 0;
123  gw->getImageData(&data, width, height);
124  if(data) {
125  newImage.data = data;
126  newImage.width = width;
127  newImage.height = height;
128  imageProcess->addImage(newImage);
129  }
130  }
131  msleep(1000/frame_rate);
132  }
133 
134  // deinit capture
135  if(gw) gw->setGrabFrames(false);
136  delete imageProcess;
137  imageProcess = 0;
138  capturing = false;
139  }
