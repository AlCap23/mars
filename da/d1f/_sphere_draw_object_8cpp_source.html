53  std::vector<SphereFace>* makeSphere(unsigned int levelOfDetail) {
54  vector<SphereFace> *faces = new vector<SphereFace>(pow(4.0,(int)levelOfDetail) * 8);
55  vector<SphereFace> &f = *faces;
56  unsigned int i, j, numFaces=0, numNewFaces;
57  osg::Vec3 pa,pb,pc;
58 
59  { // setup initial level
60  double a = 1.0 / sqrt(2.0) + 0.001;
61  osg::Vec3 p[6] = {
62  osg::Vec3(0,0,1), osg::Vec3(0,0,-1), osg::Vec3(-a,-a,0),
63  osg::Vec3(a,-a,0), osg::Vec3(a,a,0), osg::Vec3(-a,a,0)
64  };
65  f[0] = (SphereFace) { p[0], p[3], p[4] };
66  f[1] = (SphereFace) { p[0], p[4], p[5] };
67  f[2] = (SphereFace) { p[0], p[5], p[2] };
68  f[3] = (SphereFace) { p[0], p[2], p[3] };
69  f[4] = (SphereFace) { p[1], p[4], p[3] };
70  f[5] = (SphereFace) { p[1], p[5], p[4] };
71  f[6] = (SphereFace) { p[1], p[2], p[5] };
72  f[7] = (SphereFace) { p[1], p[3], p[2] };
73  numFaces = 8;
74  }
75 
76  for (j=0; j<levelOfDetail; ++j) {
77  numNewFaces = numFaces;
78  for (i=0; i<numNewFaces; ++i) {
79  pa = (f[i].p1 + f[i].p2)*0.5; pa.normalize();
80  pb = (f[i].p2 + f[i].p3)*0.5; pb.normalize();
81  pc = (f[i].p3 + f[i].p1)*0.5; pc.normalize();
82 
83  f[numFaces] = (SphereFace) { f[i].p1, pa, pc }; ++numFaces;
84  f[numFaces] = (SphereFace) { pa, f[i].p2, pb }; ++numFaces;
85  f[numFaces] = (SphereFace) { pb, f[i].p3, pc }; ++numFaces;
86 
87  f[i] = (SphereFace) { pa, pb, pc };
88  }
89  }
90 
91  return faces;
92  }

94  static void sphereUV(const osg::Vec3 &p,
95  float *s, float *t) {
96  float x = p.x();
97  float y = p.y();
98  float length = sqrt(x*x+y*y);
99  x /= length;
100  y /= length;
101  if(x > 1.0) x = 1.0;
102  else if(x<-1.0) x = -1.0;
103  *s = (asin(-x)/(M_PI*2))+0.25;
104  if(y<0) *s = 1-*s;
105  *t = (asin(p.z()) / (M_PI))+0.5;
106  /*
107  *s = atan2(p.x(), p.y()) / (2.0 * M_PI) + 0.5;
108  *t = asin(p.z()) / M_PI;
109  */
110  }

112  void SphereDrawObject::createGeometry(
113  osg::Vec3Array *vertices,
114  osg::Vec3Array *normals,
115  osg::Vec2Array *uv,
116  float radius,
117  const osg::Vec3 &topOffset,
118  const osg::Vec3 &bottomOffset,
119  bool backfaces,
120  unsigned int levelOfDetail) {
121 
122  osg::Vec3 zero(0.0f, 0.0f, 0.0f);
123  vector<SphereFace> *faces = makeSphere(levelOfDetail);
124  float s1, s2, s3, t1, t2, t3;
125  vector<osg::Vec3> verts(3);
126  vector<osg::Vec3> nors(3);
127  vector<osg::Vec2> uvs(3);
128  osg::Vec3 v;
129  const osg::Vec3 *offset;
130  osg::Vec3 halfOffset = (bottomOffset-topOffset)*0.5;
131  float factor;
132  int i;
133 
134  vertices->clear();
135  normals->clear();
136  uv->clear();
137 
138  for(vector<SphereFace>::iterator it = faces->begin();
139  it != faces->end(); ++it) {
140 
141  // add triangle face vertices
142  if(it->p1.z() >= 0.0 && it->p2.z() >= 0.0 && it->p3.z() >= 0.0) {
143  offset = &topOffset;
144  factor = 1.0;
145  } else {
146  offset = &bottomOffset;
147  factor = -1.0;
148  }
149 
150  verts[0] = it->p1 * radius + *offset;
151  verts[1] = it->p2 * radius + *offset;
152  verts[2] = it->p3 * radius + *offset;
153  // normals are easy, sphere center is origin
154  nors[0] = it->p1;
155  nors[1] = it->p2;
156  nors[2] = it->p3;
157 
158  // calculate uv coordinates for vertices
159  sphereUV(it->p1, &s1, &t1);
160  sphereUV(it->p2, &s2, &t2);
161  if(s2 < 0.15 && s1 > 0.85) s2 += 1.0;
162  else if(s2 > 0.85 && s1 < 0.15) s2 -= 1.0;
163  sphereUV(it->p3, &s3, &t3);
164  if(s3 < 0.15 && s2 > 0.85 && s1 > 0.85) s3 += 1.0;
165  else if(s3 > 0.85 && s2 < 0.15 && s1 < 0.15) s3 -= 1.0;
166 
167  /*
168  uvs[0] = osg::Vec2(s1*6.0, t1*3.0);
169  uvs[1] = osg::Vec2(s2*6.0, t2*3.0);
170  uvs[2] = osg::Vec2(s3*6.0, t3*3.0);
171  */
172  uvs[0] = osg::Vec2(s1, t1);
173  uvs[1] = osg::Vec2(s2, t2);
174  uvs[2] = osg::Vec2(s3, t3);
175 
176  // make capsule
177  // FIXME: uv coordinates have some glitches
178  float buf; //, dz1, dz2, dz3;
179  if(it->p1.z() == 0.0) {
180  buf = verts[0].z();
181  verts[0] += halfOffset*factor;
182  uvs[0].y() += 2.0*(buf-verts[0].z())*factor;
183  }
184  if(it->p2.z() == 0.0) {
185  buf = verts[1].z();
186  verts[1] += halfOffset*factor;
187  uvs[1].y() += 2.0*(buf-verts[1].z())*factor;
188  }
189  if(it->p3.z() == 0.0) {
190  buf = verts[2].z();
191  verts[2] += halfOffset*factor;
192  uvs[2].y() += 2.0*(buf-verts[2].z())*factor;
193  }
194 
195  if(backfaces) {
196  for(i=3; i>0; --i) {
197  vertices->push_back(verts[i]);
198  normals->push_back(nors[i]);
199  uv->push_back(uvs[i]);
200  }
201  } else {
202  for(i=0; i<3; ++i) {
203  vertices->push_back(verts[i]);
204  normals->push_back(nors[i]);
205  uv->push_back(uvs[i]);
206  }
207  }
208  }
209  delete faces;
210  }
