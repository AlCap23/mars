178  if(plugin_config) {
179  fclose(plugin_config);
180  libManager->loadConfigFile(coreConfigFile);
181  } else {
182  fprintf(stderr, "Loading default core libraries...\n");
183  libManager->loadLibrary("data_broker");

193  }
194  }

196  // then get the simulation

198  marsSim = libManager->getLibraryAs<mars::interfaces::SimulatorInterface>("mars_sim");

209  marsGui->setupGui();

233  }
234

237  plugin_config = fopen(otherConfigFile.c_str() , "r");
238  if(plugin_config) {
239  fclose(plugin_config);

245  libManager->loadLibrary("connexion_plugin", NULL, true);
246  libManager->loadLibrary("data_broker_gui", NULL, true);
247  libManager->loadLibrary("cfg_manager_gui", NULL, true);
248  libManager->loadLibrary("lib_manager_gui", NULL, true);
249  }
250  }
251 
252 
253  // if we have a main gui, show it
254  if(mainGui) mainGui->show();
255 
256  control->sim->runSimulation(startThread);
257 
258  if(needQApp) {
259  graphicsTimer = new mars::app::GraphicsTimer(marsGraphics,

262  }

266  void MARS::readArguments(int argc, char **argv) {
267  int c;
268  int option_index = 0;
269 
270  int i;
271  char** argv_copy = NULL;
272  unsigned int arg_len = 0;
273 
274  //remember how many arguments were already processed by getopt()
275  const int old_opterr = opterr;
276 
277  //do not print error messages for unkown arguments
278  //(may be a valid option for mars_sim).
279  opterr = 0;
280 
281  static struct option long_options[] = {
282  {"config_dir", required_argument, 0, 'C'},
283  {"no-gui",no_argument,0,'G'},
284  {"noQApp",no_argument,0,'Q'},
285  {0, 0, 0, 0}
286  };
287 
288  // copy the argument vector in order to prevent messing around with the order
289  // of the arguments by getopt
290  argv_copy = (char**) malloc(argc*sizeof(char*));
291  for (i = 0; i < argc; i++) {
292  arg_len = strlen(argv[i]);
293  argv_copy[i] = (char*) malloc((arg_len+1)*sizeof(char));
294  memcpy(argv_copy[i],argv[i],arg_len);
295  argv_copy[i][arg_len] = '\0';
296  }
297 
298  // here just work with the copied argument vector ...
299  while (1) {
300 
301 #ifdef __linux__
302  c = getopt_long(argc, argv, "GC:Q", long_options, &option_index);
303 #else
304  c = getopt_long(argc, argv_copy, "GC:Q", long_options, &option_index);
305 #endif
306  if (c == -1)
307  break;
308  switch (c) {
309  case 'C':
310  if( QDir(QString(optarg)).exists() ) {
311  configDir = optarg;
312  argConfDir = true;
313  }
314  else
315  printf("The given configuration Directory does not exists: %s\n",
316  optarg);
317  break;
318  case 'Q':
319  needQApp = false;
320  break;
321  case 'G':
322  noGUI = true;
323  break;
324  }
325  }
326 
327  // clean up the copied argument vector
328  for (i = 0; i < argc; i++) {
329  free(argv_copy[i]);
330  }
331  free(argv_copy);
332 
333  //reset error message printing to original setting
334  opterr = old_opterr;
335 
336  //reset index to read arguments again in other libraries (mars_sim).
337  optind = 1;
338  optarg = NULL;
339 
340  return;
341  }
