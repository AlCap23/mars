128  void DialogDistance::selectObjects() {
129 
130  static bool select_allowed = true;
131 
132  if (select_allowed == false)
133  return;
134 
135  QList<QListWidgetItem*> selectedItems = objectList->selectedItems();
136 
137  if (selectedItems.size() < 2)
138  return;
139 
140  select_allowed = false;
141 
142  while (selectedItems.size() > 2) {
143  selectedItems.front()->setSelected(false);
144  selectedItems.pop_front();
145  }
146 
147  selectedNodes.clear();
148  selectedJoints.clear();
149 
150  for (unsigned int i = 0 ; i < (unsigned int)selectedItems.size(); i++) {
151  int m = selectedItems[i]->text().indexOf(" ");
152  int n = selectedItems[i]->text().indexOf(":");
153  unsigned long object_id = selectedItems[i]->text().mid(m, n-m).toULong();
154  if (selectedItems[i]->text().left(m) == "Node")
155  selectedNodes.push_back(object_id);
156  else
157  selectedJoints.push_back(object_id);
158  }
159 
160  updateProperties();
161 
162  select_allowed = true;
163  }

172  void DialogDistance::updateProperties(void) {
173 
174  if (objectList->selectedItems().size() != 2) {
175  if (selection != NULL) {
176  pDialog->removeGenericProperty(selection);
177  pDialog->removeGenericProperty(distance);
178  pDialog->removeGenericProperty(ap1);

182  }
183  return;
184  }
185 
186  filled = false;
187  QString label = objectList->selectedItems()[0]->text() + " - " +
188  objectList->selectedItems()[1]->text();
189 
190  switch (selectedNodes.size()) {
191  case 2:
192  control->nodes->getNodeExchange(selectedNodes[0], &first);
193  control->nodes->getNodeExchange(selectedNodes[1], &second);
194  break;
195  case 1:
196  control->nodes->getNodeExchange(selectedNodes[0], &first);
197  control->joints->getJointExchange(selectedJoints[0], &second);
198  break;
199  case 0:
200  control->joints->getJointExchange(selectedJoints[0], &first);
201  control->joints->getJointExchange(selectedJoints[1], &second);
202  break;
203  }
204 
205  utils::Vector p1 = first.pos;
206  utils::Vector p2 = second.pos;
207  //Vector d1 = (~second.rot).VRotate(first.pos-second.pos);
208  //Vector d2 = (~first.rot).VRotate(second.pos-first.pos);
209  utils::Vector d1 = (second.rot).inverse() * (first.pos-second.pos);
210  utils::Vector d2 = (first.rot).inverse() * (second.pos-first.pos);
211  //original_distance = first.pos.distanceTo(second.pos);
212  original_distance = (first.pos - second.pos).norm();
213 
214  direction = p2;
215  origin = p1;

222  ", " + QString::number(d1.z()) + ")";
223  QString rel2 = "Vector(" + QString::number(d2.x()) + ", " + QString::number(d2.y()) +
224  ", " + QString::number(d2.z()) + ")";
225 
226  if (selection == NULL) {
227  map<QString, QVariant> attr;
228  attr["decimals"] = 9;
229  attr["singleStep"] = 0.1;
230  attr["minimum"] = 0.0;
231  selection = pDialog->addGenericProperty("../Current selection", QVariant::String, label);
232  distance = pDialog->addGenericProperty("../Distance", QVariant::Double, original_distance, &attr);
233  ap1 = pDialog->addGenericProperty("../Absolute position of " + first.name, QVariant::String, abs1);

237  } else {

239  distance->setValue(original_distance);
240  ap1->setValue(abs1);
241  ap1->setPropertyName("Absolute position of " + QString::fromStdString(first.name));
242  ap2->setValue(abs2);
243  ap2->setPropertyName("Absolute position of " + QString::fromStdString(second.name));
244  rp1->setValue(rel1);
245  rp1->setPropertyName(QString::fromStdString(first.name) + " relative to " + QString::fromStdString(second.name));
246  rp2->setValue(rel2);

261  void DialogDistance::changeDistance(double new_distance)
262  {
263  //Vector delta = (direction-origin).getNormalized()*(new_distance-original_distance);
264  utils::Vector delta = (direction-origin).normalized()*(new_distance-original_distance);
265 
266  switch (selectedNodes.size()) {
267  case 0:
268  case 1:
269  control->joints->setReloadAnchor(second.index, second.pos + delta);
270  break;
271 
272  case 2: // node - node
273  {
274  interfaces::NodeData moveNode = control->nodes->getFullNode(second.index);
275  moveNode.pos += delta;
276  control->nodes->editNode(&moveNode, interfaces::EDIT_NODE_POS);
277  }
278  }
279 
280  updateProperties();
281  }
