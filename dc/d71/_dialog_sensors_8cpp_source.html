159  void DialogSensors::on_remove_sensor() {
160  filled = false;
161 
162  for (unsigned int i = 0; i < newSensors_p.size(); i++)
163  if (pDialog->currentItem() == newSensors_p[i]) {
164  delete newDialogs[i];
165  newDialogs.erase(newDialogs.begin() + i);
166  newSensors_p.erase(newSensors_p.begin() + i);
167  break;
168  }
169 
170  for (unsigned int i = 0; i < allSensors_p.size(); i++)
171  if (allSensors_p[i] == pDialog->currentItem()) {
172  control->sensors->removeSensor(allSensors[i].index);
173  allSensors.erase(allSensors.begin() + i);
174  delete allDialogs[i];
175  allDialogs.erase(allDialogs.begin() + i);
176  allSensors_p.erase(allSensors_p.begin() + i);
177  break;
178  }
179 
180  filled = true;
181 
182  if (allSensors_p.size() == 0 && newSensors_p.size() == 0) {
183  removeButton->hide();
184  addButton->hide();
185  } else if (allSensors_p.size() > 0) {
186  pDialog->setCurrentItem(allSensors_p[0]);
187  } else if (newSensors_p.size() > 0) {
188  pDialog->setCurrentItem(newSensors_p[0]);
189  }
190

193  void DialogSensors::on_add_sensor() {
194  filled = false;
195  for (unsigned int i = 0; i < newSensors_p.size(); i++)
196  if (pDialog->currentItem() == newSensors_p[i]) {
197  newDialogs[i]->accept();
198  allSensors_p.push_back(newSensors_p[i]);
199  allDialogs.push_back(newDialogs[i]);
200  newSensors_p.erase(newSensors_p.begin() + i);
201  newDialogs.erase(newDialogs.begin() + i);
202  topLevelItemChanged(allSensors_p.back());
203  break;
204  }
205 
206  control->sensors->getListSensors(&allSensors);
207 
208  filled = true;
209  addButton->hide();
210  }
