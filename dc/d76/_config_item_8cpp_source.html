581  void ConfigItem::recursiveLoad(ConfigItem &item, std::string &path) {
582 
583  if(item.isMap()) {
584  ConfigMap &map = item;
585  ConfigMap::iterator it = item.beginMap();
586  std::list<ConfigMap::iterator> eraseList;
587  std::list<ConfigMap::iterator>::iterator eraseIt;
588  for(; it!=item.endMap(); ++it) {
589  if(it->first == "URI") {
590 
591  std::string file = path + (std::string)it->second;
592  std::string subPath = getPathOfFile(file);
593  ConfigItem m2 = fromYamlFile(file, true);
594  recursiveLoad(m2, subPath);
595  map.append(m2);
596  eraseList.push_back(it);
597  }else {
598  recursiveLoad(it->second, path);
599  }
600  }
601  for(eraseIt=eraseList.begin(); eraseIt!=eraseList.end(); ++eraseIt) {
602  map.erase(*eraseIt);
603  }
604  } else {
605  if(item.isVector()) {
606  std::vector<ConfigItem>::iterator it = ((ConfigVector&)item).begin();
607  for(; it!= ((ConfigVector&)item).end(); ++it) {
608  recursiveLoad(*it, path);
609  }
610  }
611  }
612  }
