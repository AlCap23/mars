42  string ShaderGenerator::generateSource(ShaderType shaderType) {
43  map<ShaderType,ShaderFunc*>::iterator it = functions.find(shaderType);
44  if(it == functions.end()) {
45  return "";
46  }
47  ShaderFunc *u = it->second;
48 
49  stringstream code;
50  if(u->getMinVersion() != 0)
51  code << "#version " << u->getMinVersion() << endl;
52  for(set<string>::iterator it = u->getEnabledExtensions().begin();
53  it != u->getEnabledExtensions().end(); ++it)
54  code << "#extension " << *it << " : enable" << endl;
55  for(set<string>::iterator it = u->getDisabledExtensions().begin();
56  it != u->getDisabledExtensions().end(); ++it)
57  code << "#extension " << *it << " : disable" << endl;
58  code << endl;
59 
60  for(set<GLSLUniform>::iterator it = u->getUniforms().begin();
61  it != u->getUniforms().end(); ++it)
62  code << "uniform " << *it << ";" << endl;
63 
64  for(set<GLSLConstant>::iterator it = u->getConstants().begin();
65  it != u->getConstants().end(); ++it)
66  code << "const " << *it << ";" << endl;
67 
68  switch(shaderType)
69  {
70  case SHADER_TYPE_VERTEX:
71  for(set<GLSLVarying>::iterator it = u->getVaryings().begin();
72  it != u->getVaryings().end(); ++it)
73  code << "varying " << *it << ";" << endl;
74  for(set<GLSLAttribute>::iterator it = u->getAttributes().begin();
75  it != u->getAttributes().end(); ++it)
76  code << "attribute " << *it << ";" << endl;
77  break;
78  case SHADER_TYPE_FRAGMENT:
79  for(set<GLSLVarying>::iterator it = u->getVaryings().begin();
80  it != u->getVaryings().end(); ++it)
81  code << "varying " << *it << ";" << endl;
82  break;
83  default: break;
84  }
85  code << endl;
86 
87 
88  std::vector< std::pair<std::string,std::string> > uDeps = u->getDeps();
89  for(vector< pair<string,string> >::iterator it = uDeps.begin();
90  it != uDeps.end(); ++it)
91  code << it->second;
92 
93  code << u->generateFunctionCode() << endl;
94  code << "void main()" << endl;
95  code << "{" << endl;
96 
97  for(list<GLSLVariable>::const_iterator it = u->getMainVars().begin();
98  it != u->getMainVars().end(); ++it)
99  code << " " << *it << ";" << endl;
100 
101  vector<string> calls = u->generateFunctionCall();
102  for(vector<string>::iterator it = calls.begin(); it != calls.end(); ++it)
103  code << " " << *it << endl;
104 
105  for(vector<GLSLExport>::const_iterator it = u->getExports().begin();
106  it != u->getExports().end(); ++it)
107  code << " " << *it << ";" << endl;
108  for(set<GLSLSuffix>::iterator it = u->getSuffixes().begin();
109  it != u->getSuffixes().end(); ++it)
110  code << " " << *it << ";" << endl;
111 
112  code << "}" << endl;
113 
114  return code.str();
115  }
