45  string ShaderGenerator::generateSource(ShaderType shaderType)
46  {
47  map<ShaderType,ShaderFunc*>::iterator it = functions.find(shaderType);
48  if(it == functions.end()) {
49  return "";
50  }
51  ShaderFunc *u = it->second;
52 
53  stringstream code;
54  if(u->getMinVersion() != 0)
55  code << "#version " << u->getMinVersion() << endl;
56  for(set<string>::iterator it = u->getEnabledExtensions().begin();
57  it != u->getEnabledExtensions().end(); ++it)
58  code << "#extension " << *it << " : enable" << endl;
59  for(set<string>::iterator it = u->getDisabledExtensions().begin();
60  it != u->getDisabledExtensions().end(); ++it)
61  code << "#extension " << *it << " : disable" << endl;
62  code << endl;
63 
64  for(set<GLSLUniform>::iterator it = u->getUniforms().begin();
65  it != u->getUniforms().end(); ++it)
66  code << "uniform " << *it << ";" << endl;
67 
68  for(set<GLSLConstant>::iterator it = u->getConstants().begin();
69  it != u->getConstants().end(); ++it)
70  code << "const " << *it << ";" << endl;
71 
72  switch(shaderType)
73  {
74  case mars::interfaces::SHADER_TYPE_VERTEX:
75  for(set<GLSLVarying>::iterator it = u->getVaryings().begin();
76  it != u->getVaryings().end(); ++it)
77  code << "varying " << *it << ";" << endl;
78  for(set<GLSLAttribute>::iterator it = u->getAttributes().begin();
79  it != u->getAttributes().end(); ++it)
80  code << "attribute " << *it << ";" << endl;
81  break;
82  case mars::interfaces::SHADER_TYPE_FRAGMENT:
83  for(set<GLSLVarying>::iterator it = u->getVaryings().begin();
84  it != u->getVaryings().end(); ++it)
85  code << "varying " << *it << ";" << endl;
86  break;
87  default: break;
88  }
89  code << endl;
90 
91 
92  std::vector< std::pair<std::string,std::string> > uDeps = u->getDeps();
93  for(vector< pair<string,string> >::iterator it = uDeps.begin();
94  it != uDeps.end(); ++it)
95  code << it->second;
96 
97  code << u->generateFunctionCode() << endl;
98  code << "void main()" << endl;
99  code << "{" << endl;
100 
101  for(list<GLSLVariable>::const_iterator it = u->getMainVars().begin();
102  it != u->getMainVars().end(); ++it)
103  code << " " << *it << ";" << endl;
104 
105  vector<string> calls = u->generateFunctionCall();
106  for(vector<string>::iterator it = calls.begin(); it != calls.end(); ++it)
107  code << " " << *it << endl;
108 
109  for(set<GLSLExport>::iterator it = u->getExports().begin();
110  it != u->getExports().end(); ++it)
111  code << " " << *it << ";" << endl;
112  for(set<GLSLSuffix>::iterator it = u->getSuffixes().begin();
113  it != u->getSuffixes().end(); ++it)
114  code << " " << *it << ";" << endl;
115 
116  code << "}" << endl;
117 
118  return code.str();
119  }
