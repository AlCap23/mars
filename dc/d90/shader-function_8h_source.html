115  void addMainVar(GLSLVariable var, int pos = -1) {
116  std::list<GLSLVariable>::iterator it = mainVars.begin();
117  if(var.type != "") {
118  for(; it!=mainVars.end(); ++it) {
119  if(it->name == var.name) {
120  it->value = var.value;
121  return;
122  }
123  }
124  }
125  if(pos > -1) {
126  it = mainVars.begin();
127  for(int i=0; i<pos; ++i, ++it) ;
128  mainVars.insert(it, var);
129  }
130  else {
131  mainVars.push_back(var);
132  }
133  }
