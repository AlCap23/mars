113  void PathDrawer::addVectorsFromObjFile(string file_name){
114  ifstream file;
115  file.open (file_name.c_str());
116  if (file.is_open()){
117  string line;
118  while ( getline (file, line) ) {
119  //printf("%s\n", line.c_str());
120  if(int start = line.find("v ", 0, 2) != string::npos){
121  double v[3];
122  start += 1;
123  int end_first = line.find(" ", start) + 1;
124  int end_second = line.find(" ", end_first) + 1;
125  v[0] = (sReal) atof((line.substr(start, end_first-start)).c_str());
126  v[1] = (sReal) atof((line.substr(end_first, end_second-end_first)).c_str());
127  v[2] = (sReal) atof((line.substr(end_second).c_str()));
128  l->appendData(osg_lines::Vector(v[0], v[1], v[2]));
129  }
130  }
131  file.close();
132  }else{
133  fprintf(stderr, "[PathDrawer] Warning: Unable to open file '%s'\n", file_name.c_str());
134  }
135 
136  l->setColor(osg_lines::Color(0.0, 1.0, 0.0, 1.0));
137  l->setLineWidth(4);
138  control->graphics->addOSGNode(l->getOSGNode());
139  }

141  void PathDrawer::addVectorsFromSvgFile(string file_name){
142  printf("[PathDrawer] parsing svg file '%s'\n", file_name.c_str());
143  sReal m_above_ground = 0.1;
144  ifstream file;
145  file.open (file_name.c_str());
146  if (file.is_open()){
147  string line, line_to_parse;
148  size_t start = 0;
149  size_t points = 0;
150  size_t label = 0;
151  size_t end = 0;
152  size_t line_cnt = 1;
153  bool in_path = false;
154  bool search_for_size = true;
155  sReal size[2] = {0.0, 0.0};
156  bool path_found = false;
157  while ( getline (file, line) ) {
158  //printf("%s\n", line.c_str());
159 
160  if(search_for_size){
161  if( line.find("width=") != string::npos){
162  size_t s = line.find_first_of("\"");
163  size_t e = line.find_last_of("\"");
164  size[0] = (sReal) atof(line.substr(s+1, e).c_str());
165  }
166 
167  if( line.find("height=") != string::npos){
168  size_t s = line.find_first_of("\"");
169  size_t e = line.find_last_of("\"");
170  size[1] = (sReal) atof(line.substr(s+1, e).c_str());
171  }
172 
173  if(size[0] > 0.001 && size[1] > 0.001){
174  search_for_size = false;
175  //printf("[PathDrawer] svg width %g, height %g\n", size[0], size[1]);
176  }
177  }
178 
179  if( line.find("<path") != string::npos){
180  start = line_cnt;
181  in_path = true;
182  }
183 
184  if(in_path){
185  if( line.find(" d=") != string::npos){
186  points = line_cnt;
187  line_to_parse = line;
188  }
189 
190  if( line.find("path_to_follow") != string::npos){
191  label = line_cnt;
192  }
193 
194  if( line.find("</path>") != string::npos){
195  end = line_cnt;
196  in_path = false;
197  //printf("found vectors in line %d (start %d, d= %d, end %d)\n", points, start, label, end);
198 
199  if( min( start, min( points, label ) ) == start
200  && max( end, max( points, label ) ) == end){
201  // we found what we were looking for
202  path_found = true;
203  break;
204  }
205  }
206  }
207 
208  if(start == string::npos || end == string::npos){
209  fprintf(stderr, "[PathDrawer] Error: there is no path specified\n");
210  }
211 
212  ++line_cnt;
213  }
214 
215  if(!path_found){
216  fprintf(stderr, "[PathDrawer] Error: no path 'path_to_follow' in svg file\n");
217  return;
218  }
219 
220  // get the path information
221  sReal v[3], offset[2];
222  bool absolute_coordinates = false;
223  size_t offset_pos_start = line_to_parse.find("d=\"m") + 5;
224  if(offset_pos_start < 6 || offset_pos_start > 15){ // valid range
225  //printf("searching for M\n");
226  offset_pos_start = line_to_parse.find("d=\"M") + 5;
227  absolute_coordinates = true;
228  }
229 
230  size_t offset_pos_end = line_to_parse.find(" ", offset_pos_start);
231  size_t offset_delimiter = line_to_parse.find(",", offset_pos_start);
232  //printf("[PathDrawer] SVG line offset pos start %d, delimiter %d, end %d\n", offset_pos_start, offset_delimiter, offset_pos_end);
233  //printf("[PathDrawer] SVG OffsetX = %s\n", line_to_parse.substr(offset_pos_start, offset_delimiter-offset_pos_start).c_str());
234  //printf("[PathDrawer] SVG OffsetY = %s\n", line_to_parse.substr(offset_delimiter+1, offset_pos_end-offset_delimiter-1).c_str());
235 
236  offset[0] = (sReal) atof(line_to_parse.substr(offset_pos_start, offset_delimiter-offset_pos_start).c_str());
237  offset[1] = (sReal) atof(line_to_parse.substr(offset_delimiter+1, offset_pos_end-offset_delimiter-1).c_str());
238 
239  // draw the first point
240  v[0] = offset[0];
241  v[1] = size[1]- offset[1];
242  v[2] = getHeightFromScene(v[0], v[1]) + m_above_ground;
243  l->appendData(osg_lines::Vector(v[0], v[1], v[2]));
244 
245  string val_str = line_to_parse.substr(offset_pos_end);
246  //printf("%s\n", val_str.c_str());
247  bool x_read = false;
248  bool y_read = false;
249  string x_str = "";
250  string y_str = "";
251  string* p = &x_str;
252  bool ignore = false;
253  // go through the line and get all points
254  for(string::iterator it = val_str.begin() + 1; it != val_str.end(); ++it){
255  //printf("%c\n", *it);
256  if((isdigit(*it) || *it == '.' || *it == '-') && !ignore){
257  // add the char to the corresponding string
258  p->append(1, *it);
259  }else{
260  // handle the non digits (changing string to write char to and skip other stuff)
261  if(*it == ' ' || *it == '\"'){
262  //printf("found ' '\n");
263  y_read = true;
264  if(ignore){
265  ignore = false;
266  *p = 0.0;
267  }
268  p = &x_str;
269  }else if(*it == ','){
270  //printf("found ','\n");
271  x_read = true;
272  y_read = false;
273  if(ignore){
274  ignore = false;
275  *p = 0.0;
276  }
277  p = &y_str;
278  }else if(ignore){
279  //printf("ignoring '%c'\n", *it);
280  }else if(*it == 'e'){
281  ignore = true; // this indicates a very small number, so ignore the rest and handle it
282  }else if(*it == 'L'){
283  absolute_coordinates = true;
284  }else if(*it == 'l'){
285  absolute_coordinates = false;
286  }else{
287  printf("[PathDrawer] Error: no handle for '%c'\n", *it);
288  }
289 
290  // add the relative x and y information and draw the point
291  if(x_read && y_read){
292  sReal x = (sReal) atof(x_str.c_str());
293  sReal y = (sReal) atof(y_str.c_str());
294  if(absolute_coordinates){
295  v[0] = x;
296  v[1] = size[1] - y;
297  }else{
298  v[0] += x;
299  v[1] -= y;
300  }
301  v[2] = getHeightFromScene(v[0], v[1]) + m_above_ground;
302  //printf("adding point %g / %g\n", v[0], v[1]);
303  l->appendData(osg_lines::Vector(v[0], v[1], v[2]));
304 
305  x_read = false;
306  x_str.clear();
307  y_read = false;
308  y_str.clear();
309  }
310  }
311  }
312 
313  file.close();
314  }else{
315  fprintf(stderr, "[PathDrawer] Error: Unable to open file '%s'\n", file_name.c_str());
316  }
317 
318  l->setColor(osg_lines::Color(0.0, 1.0, 0.0, 1.0));
319  l->setLineWidth(4);
320  control->graphics->addOSGNode(l->getOSGNode());
321  }
