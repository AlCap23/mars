189  double ConstraintsPlugin::getNodeAttribute(interfaces::NodeId nodeId,
190  AttributeType attr) {
191  double attrValue;
192  interfaces::NodeData n;
193  n = control->nodes->getFullNode(nodeId);
194  if(n.relative_id) {
195  interfaces::NodeData rel;
196  rel = control->nodes->getFullNode(n.relative_id);
197  interfaces::getRelFromAbs(rel, &n);
198  }
199  utils::sRotation rot = utils::quaternionTosRotation(n.rot);
200  switch(attr) {
201  case ATTRIBUTE_POSITION_X:
202  attrValue = n.pos.x();
203  break;
204  case ATTRIBUTE_POSITION_Y:
205  attrValue = n.pos.y();
206  break;
207  case ATTRIBUTE_POSITION_Z:
208  attrValue = n.pos.z();
209  break;
210  case ATTRIBUTE_ROTATION_X:
211  attrValue = rot.alpha;
212  break;
213  case ATTRIBUTE_ROTATION_Y:
214  attrValue = rot.beta;
215  break;
216  case ATTRIBUTE_ROTATION_Z:
217  attrValue = rot.gamma;
218  break;
219  case ATTRIBUTE_SIZE_X:
220  attrValue = n.ext.x();
221  break;
222  case ATTRIBUTE_SIZE_Y:
223  attrValue = n.ext.y();
224  break;
225  case ATTRIBUTE_SIZE_Z:
226  attrValue = n.ext.z();
227  break;
228  default:
229  LOG_ERROR("NodeConstraint: attribute %d not supported.", attr);
230  }
231  return attrValue;
232  }

234  void ConstraintsPlugin::parseNodeConstraints(const std::string &paramName,
235  const std::string &s) {
236  std::string nodeName, nodeAttr;
237  interfaces::NodeId nodeId = 0;
238  AttributeType attr = ATTRIBUTE_UNDEFINED;
239  ParseResult result = PARSE_SUCCESS;
240  double factor, initialValue, offset;
241  size_t pos1 = 0;
242 
243  // create new Property to control the constraint
244  cfg_manager::cfgPropertyStruct newProp;
245  newProp = control->cfg->getOrCreateProperty("Constraints", paramName,
246  (double)0.);
247  // first entry is the initial nodeName.nodeAttr
248  result = parseIdentifier(s, &pos1, &nodeId, &attr, &offset, NULL);
249  if(result != PARSE_SUCCESS)
250  return;
251 
252  initialValue = getNodeAttribute(nodeId, attr) + offset;
253  control->cfg->setPropertyValue("Constraints", paramName, "value",
254  initialValue);
255  // add constraint for initial Node with factor 1
256  BaseConstraint *c = new NodeConstraint(control, newProp.paramId,
257  nodeId, attr,
258  offset, 1, initialValue);
259  constraints[newProp.paramId].push_back(c);
260 
261  while(result == PARSE_SUCCESS) {
262  // next there is a list of NodeName.NodeAttr#factor
263  result = parseIdentifier(s, &pos1, &nodeId, &attr, &offset, &factor);
264  if((result != PARSE_SUCCESS) && (result != PARSE_SUCCESS_EOS)) {
265  continue;
266  }
267  BaseConstraint *c = new NodeConstraint(control, newProp.paramId,
268  nodeId, attr, offset, factor,
269  initialValue);
270  constraints[newProp.paramId].push_back(c);
271  }
272  }

274  ParseResult ConstraintsPlugin::parseIdentifier(const std::string &s,
275  size_t *pos,
276  interfaces::NodeId *nodeId,
277  AttributeType *attr,
278  double *offset,
279  double *factor) {
280  // expect the format "nodeName.nodeAttr#factor" where the factor is optional
281  size_t startPos=*pos, splitPos, endPos;
282  // find end of identifier
283  endPos = s.find('#', startPos);
284  // find the nodeAttribute. start from rear because nodeName may contain "."
285  splitPos = s.rfind("::", endPos);
286  std::string nodeName = s.substr(startPos, splitPos - startPos);
287  std::string nodeAttr = s.substr(splitPos + 2, endPos - splitPos - 2);
288  startPos = splitPos + 1;
289  // look for an offset
290  splitPos = nodeAttr.find('[');
291  if(splitPos == std::string::npos) {
292  *offset = 0.;
293  } else {
294  size_t tmpEndPos = nodeAttr.find(']', splitPos);
295  if(tmpEndPos == std::string::npos) {
296  *offset = 0.;
297  } else {
298  std::string offsetStr = nodeAttr.substr(splitPos + 1,
299  tmpEndPos - splitPos - 1);
300  *offset = strtod(offsetStr.c_str(), NULL);
301  nodeAttr = nodeAttr.substr(0, splitPos);
302  }
303  }
304 
305  if(factor) {
306  if(endPos == std::string::npos) {
307  *factor = 0;
308  LOG_WARN("ConstraintsPlugin::parseIdentifier: factor is missing");
309  } else {
310  startPos = endPos + 1;
311  endPos = s.find('#', startPos);
312  *factor = strtod(s.substr(startPos, endPos - startPos).c_str(),
313  NULL);
314  }
315  }
316 
317  // get the actual values from the strings and write the results back
318  *nodeId = control->nodes->getID(nodeName);
319  *attr = parseAttribute(nodeAttr);
320  *pos = endPos + 1;
321 
322  // do error checking
323  if(*nodeId == 0) {
324  LOG_ERROR("Invalid NodeName \"%s\" in NodeConstraint: \"%s\"",
325  nodeName.c_str(), s.c_str());
326  return PARSE_ERROR_NODEID;
327  }
328  if(*attr == ATTRIBUTE_UNDEFINED) {
329  LOG_ERROR("Invalid NodeAttribute \"%s\" in NodeConstraint: \"%s\"",
330  nodeAttr.c_str(), s.c_str());
331  return PARSE_ERROR_NODEATTR;
332  }
333  // signal whether we reached the End-Of-String or not
334  if(endPos == std::string::npos)
335  return PARSE_SUCCESS_EOS;
336  else
337  return PARSE_SUCCESS;
338  }

340  AttributeType ConstraintsPlugin::parseAttribute(const std::string &s) {
341  if(s == "posX")
342  return ATTRIBUTE_POSITION_X;
343  else if(s == "posY")
344  return ATTRIBUTE_POSITION_Y;
345  else if(s == "posZ")
346  return ATTRIBUTE_POSITION_Z;
347  else if(s == "rotX")
348  return ATTRIBUTE_ROTATION_X;
349  else if(s == "rotY")
350  return ATTRIBUTE_ROTATION_Y;
351  else if(s == "rotZ")
352  return ATTRIBUTE_ROTATION_Z;
353  else if(s == "sizeX")
354  return ATTRIBUTE_SIZE_X;
355  else if(s == "sizeY")
356  return ATTRIBUTE_SIZE_Y;
357  else if(s == "sizeZ")
358  return ATTRIBUTE_SIZE_Z;
359  else {
360  LOG_ERROR("unsupported Node attribute in constraint string.");
361  return ATTRIBUTE_UNDEFINED;
362  }
