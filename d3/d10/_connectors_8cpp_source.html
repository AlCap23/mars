71  void Connectors::connect(std::string male, std::string female) {
72  fprintf(stderr, "Create connection: %s, %s\n", male.c_str(), female.c_str());
73  interfaces::JointData jointdata;
74  jointdata.nodeIndex1 = control->nodes->getID((std::string)(maleconnectors[male]["link"]));
75  jointdata.nodeIndex2 = control->nodes->getID((std::string)(femaleconnectors[female]["link"]));
76  jointdata.type = interfaces::JOINT_TYPE_FIXED;
77  jointdata.name = "connector_"+male+"_"+female;
78  unsigned long jointid = control->joints->addJoint(&jointdata);
79  if (jointid > 0) {
80  // TODO: maybe update NodeData here and set collision group differently to avoid collision problems
81  maleconnectors[male]["jointid"] = jointid;
82  maleconnectors[male]["partner"] = female;
83  femaleconnectors[female]["jointid"] = jointid;
84  femaleconnectors[female]["partner"] = male;
85  connections[male] = female;
86  }
87  }

89  void Connectors::disconnect(std::string connector) {
90  configmaps::ConfigMap* conmap = NULL;
91  unsigned long jointid = 0;
92  std::string partner = "";
93  bool male = false;
94  std::map<std::string, configmaps::ConfigMap>::iterator it = maleconnectors.find(connector);
95  if (it != maleconnectors.end()) {
96  conmap = &(it->second);
97  jointid = (unsigned long)((*conmap)["jointid"]);
98  partner = (std::string)((*conmap)["partner"]);
99  male = true;
100  connections.erase(connector);
101  } else {
102  it = femaleconnectors.find(connector);
103  if (it != maleconnectors.end()) {
104  conmap = &(it->second);
105  jointid = (unsigned long)((*conmap)["jointid"]);
106  partner = (std::string)((*conmap)["partner"]);
107  male = false;
108  connections.erase(partner);
109  }
110  }
111  if (jointid != 0) {
112  control->joints->removeJoint(jointid); // if no match is found, this tries to delete joint id 0
113  (it->second)["jointid"] = 0;
114  (it->second)["partner"] = "";
115  if (male) {
116  it = femaleconnectors.find(partner);
117  } else {
118  it = maleconnectors.find(partner);
119  }
120  // now reset partner
121  (it->second)["jointid"] = 0;
122  (it->second)["partner"] = "";
123  }

125  void Connectors::registerEntity(sim::SimEntity* entity) {
126  configmaps::ConfigMap entitymap = entity->getConfig();
127  if (entitymap.hasKey("connectors")) {
128  configmaps::ConfigMap tmpmap;
129  // gather types
130  configmaps::ConfigVector typevec = entitymap["connectors"]["types"];
131  for (configmaps::ConfigVector::iterator it = typevec.begin(); it!= typevec.end(); ++it) {

134  }
135  // gather connectors
136  configmaps::ConfigVector convec = entitymap["connectors"]["connectors"];
137  for (configmaps::ConfigVector::iterator it = convec.begin(); it!= convec.end(); ++it) {
138  tmpmap = (*it);
139  tmpmap["nodeid"] = control->nodes->getID((*it)["link"]);
140  if (((std::string)((*it)["gender"])).compare("male") == 0) { // male
141  maleconnectors[(std::string)(tmpmap["name"])] = tmpmap;
142  fprintf(stderr, "Adding male connector: %s\n", ((std::string)(tmpmap["name"])).c_str(),
143  (unsigned long)(tmpmap["nodeid"]));

145  femaleconnectors[(std::string)(tmpmap["name"])] = tmpmap;
146  fprintf(stderr, "Adding female connector: %s\n", ((std::string)(tmpmap["name"])).c_str());

158  bool Connectors::mated(std::string malename, std::string femalename) {
159  unsigned long maleid = maleconnectors[malename]["nodeid"];
160  unsigned long femaleid = femaleconnectors[femalename]["nodeid"];
161  utils::Vector malerot = control->nodes->getRotation(maleid)*Vector(1.0, 0.0, 0.0);
162  utils::Vector femalerot = control->nodes->getRotation(femaleid)*Vector(1.0, 0.0, 0.0);
163  sReal angle = utils::angleBetween(malerot, femalerot);
164 
165  //TODO: this is a hard-coded hack, should be defined in type
166  sReal distance = utils::distanceBetween(control->nodes->getPosition(maleid),
167  control->nodes->getPosition(femaleid));
168  //fprintf(stderr, "distance/angle: %i, %i, %g/%g\n", maleid, femaleid, distance, angle);
169  return (distance <= ((double)(connectortypes[maleconnectors[malename]["type"]]["distance"]))
170  && angle < ((double)(connectortypes[maleconnectors[malename]["type"]]["angle"])));
171 
172  }

174  void Connectors::checkForPossibleConnections() {
175  std::string malename, femalename, maletype, femaletype;
176  for (std::map<std::string, configmaps::ConfigMap>::iterator mit= maleconnectors.begin();
177  mit!=maleconnectors.end(); ++mit) {
178  malename = (std::string)mit->second["name"];
179  maletype = (std::string)mit->second["type"];
180  for (std::map<std::string, configmaps::ConfigMap>::iterator fit= femaleconnectors.begin();
181  fit!=femaleconnectors.end(); ++fit) {
182  femalename = (std::string)fit->second["name"];
183  femaletype = (std::string)fit->second["type"];
184  if (maletype.compare(femaletype) == 0 && mated(malename, femalename)
185  && ((std::string)mit->second["partner"]).empty() && ((std::string)fit->second["partner"]).empty()) {
186  connect(malename, femalename);
187  }
188  }
189  }
