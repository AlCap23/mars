58  OSGDrawItem::OSGDrawItem(GraphicsWidget *gw, const draw_item &di,
59  std::string fontPath)
60  : osg::Group(), type_(di.type), gw_(gw), texture_(NULL) {
61 
62  osg::StateSet *states = getOrCreateStateSet();
63 
64  // create geometry and set related states
65  switch (type_) {
66  case DRAW_LINE: {
67  osg::ref_ptr<osg::LineWidth> linew = new osg::LineWidth(di.point_size);
68 
69  createLine(this, di.start, di.end, di.myColor);
70  states->setAttributeAndModes(linew.get(), osg::StateAttribute::ON);
71  break;
72  }
73  case DRAW_LINES: {
74  osg::ref_ptr<osg::LineWidth> linew = new osg::LineWidth(di.point_size);
75 
76  createLines(this, di.vertices, di.myColor);
77  states->setAttributeAndModes(linew.get(), osg::StateAttribute::ON);
78  break;
79  }
80  case DRAW_TEXT:
81  createLabel(this, di.pos, di.point_size, di.label, fontPath,
82  di.myColor.r, di.myColor.g,
83  di.myColor.b, di.myColor.a,
84  di.align_to_view);
85  break;
86  case DRAW_HUD_RECT:
87  createHUDRectangle(this,
88  di.width, di.height,
89  di.offset, di.myColor);
90  states->setRenderingHint(osg::StateSet::TRANSPARENT_BIN);
91  states->setMode(GL_BLEND,osg::StateAttribute::ON);
92  break;
93  case DRAW_TRIANGLE_SET: {
94  createTriangleSet(this, di.vertices, di.myColor);
95  osg::ref_ptr<osg::CullFace> cull = new osg::CullFace();
96  cull->setMode(osg::CullFace::BACK);
97  states->setAttributeAndModes(cull.get(), osg::StateAttribute::OFF); }
98  break;
99  case DRAW_POINTS: {
100  osg::ref_ptr<osg::Point> point = new osg::Point();
101 
102  createPoints(this, di.vertices, di.myColor);
103  point->setSize(di.point_size);
104  states->setAttribute(point.get());
105  break;
106  }
107  case DRAW_ARROW:
108  createLine(this, di.start, di.end, di.myColor);
109  break;
110  case DRAW_RECTANGLE: {
111  osg::ref_ptr<osg::LineWidth> linew = new osg::LineWidth(di.point_size);
112 
113  createRectangle(this,
114  di.start, di.end, di.pos,
115  di.myColor, di.borderColor,
116  di.align_to_view);
117  states->setRenderingHint(osg::StateSet::TRANSPARENT_BIN);
118  states->setMode(GL_BLEND,osg::StateAttribute::ON);
119  states->setAttributeAndModes(linew.get(), osg::StateAttribute::ON);
120  osg::ref_ptr<osg::CullFace> cull = new osg::CullFace();
121  cull->setMode(osg::CullFace::BACK);
122  states->setAttributeAndModes(cull.get(), osg::StateAttribute::OFF);
123  break;
124  }
125  case DRAW_CIRCLE: {
126  osg::ref_ptr<osg::LineWidth> linew = new osg::LineWidth(di.point_size);
127 
128  createCircle(this,
129  di.pos, di.start,
130  di.axis, di.myColor,
131  di.borderColor, di.resolution);
132  states->setRenderingHint(osg::StateSet::TRANSPARENT_BIN);
133  states->setMode(GL_BLEND,osg::StateAttribute::ON);
134  states->setAttributeAndModes(linew.get(), osg::StateAttribute::ON);
135  break;
136  }
137  case DRAW_POINT:
138  case DRAW_LINE_STRIP:
139  // FIXME: howto create....
140  break;
141  case DRAW_UNKNOWN:
142  case DRAW_LAST:
143  break;
144  }
145 
146  // turn off lighting if requested
147  if(di.get_light == 0) {
148  states->setMode(GL_LIGHTING,
149  osg::StateAttribute::OFF | osg::StateAttribute::PROTECTED);
150  states->setMode(GL_FOG, osg::StateAttribute::OFF);
151  }
152 
153  // create texture for this draw item
154  if (di.texture != "") {
155  texture_ = GuiHelper::loadTexture(di.texture).get();
156  states->setTextureAttributeAndModes(0, texture_,
157  osg::StateAttribute::ON | osg::StateAttribute::PROTECTED);
158  } else if (di.t_height > 0 && di.t_width > 0) {
159  osg::Image* image = new osg::Image();
160  image->setImage(di.t_height, di.t_height, 1, GL_RGBA,
161  GL_BGRA, GL_UNSIGNED_INT_8_8_8_8_REV,
162  (unsigned char*)di.t_data, osg::Image::NO_DELETE);
163  texture_ = new osg::Texture2D();
164  texture_->setDataVariance(osg::Object::DYNAMIC);
165  texture_->setImage(image);
166  states->setTextureAttributeAndModes(0, texture_,
167  osg::StateAttribute::ON | osg::StateAttribute::PROTECTED);
168  }
169  }

171  void OSGDrawItem::update(draw_item &di) {
172  osg::StateSet *states = getOrCreateStateSet();
173 
174  switch (type_) {
175  case DRAW_LINE: {
176  removeChildren(0, getNumChildren());
177  createLine(this, di.start, di.end, di.myColor);
178  break;
179  }
180  case DRAW_TEXT:
181  updateLabel(this,
182  di.pos, di.point_size, di.label,
183  di.myColor.r, di.myColor.g,
184  di.myColor.b, di.myColor.a,
185  di.align_to_view);
186  break;
187  case DRAW_HUD_RECT:
188  updateHUDRectangle(this,
189  di.width, di.height,
190  di.offset, di.myColor);
191  if (di.t_width > 1 && di.t_height > 0) {
192  di.t_width = 1;
193  memcpy(texture_->getImage()->data(), di.t_data,
194  di.t_height*di.t_height*4);
195  texture_->getImage()->dirty();
196  } else if (di.t_width == 0 && di.t_height == 0) {
197  states->setTextureAttributeAndModes(0,
198  GuiHelper::loadTexture(di.texture).get(),
199  osg::StateAttribute::ON | osg::StateAttribute::PROTECTED);
200  }
201  break;
202  case DRAW_TRIANGLE_SET:
203  addToTriangleSet(this, di.vertices, di.myColor);
204  break;
205  case DRAW_POINTS:
206  updatePoints(this, di.vertices);
207  break;
208  case DRAW_POINT:
209  case DRAW_ARROW:
210  case DRAW_RECTANGLE:
211  case DRAW_CIRCLE:
212  case DRAW_LINE_STRIP:
213  // TODO: howto update....
214  break;
215  case DRAW_UNKNOWN:
216  case DRAW_LINES:
217  case DRAW_LAST:
218  break;
219  }

416  void OSGDrawItem::createRectangle(OSGDrawItem *osgNode,
417  Vector a, Vector b,
418  Vector q, Color myColor,
419  Color borderColor,
420  int align_to_view) {

422  osg::ref_ptr<osg::Geometry> linesGeom = new osg::Geometry;
423  osg::ref_ptr<osg::Geode> geode = new osg::Geode;
424  Vector n;
425 
426  a*=0.5;
427  b*=0.5;
428 
429  osg::Vec3Array* corners = new osg::Vec3Array(5);
430  (*corners)[0].set(-(a.x()+b.x()), -(a.y()+b.y()), -(a.z()+b.z()));
431  (*corners)[1].set(a.x()-b.x(), a.y()-b.y(), a.z()-b.z());
432  (*corners)[3].set(b.x()-a.x(), b.y()-a.y(), b.z()-a.z());
433  (*corners)[2].set(a.x()+b.x(), a.y()+b.y(), a.z()+b.z());
434  (*corners)[4].set(-(a.x()+b.x()), -(a.y()+b.y()), -(a.z()+b.z()));
435 
436  osg::ref_ptr<osg::Vec2Array> texcoords(new osg::Vec2Array());
437  texcoords->push_back(osg::Vec2(0.0f, 0.0f));
438  texcoords->push_back(osg::Vec2(1.0f, 0.0f));
439  texcoords->push_back(osg::Vec2(1.0f, 1.0f));
440  texcoords->push_back(osg::Vec2(0.0f, 1.0f));
441

448 
449  osg::Vec3Array* normals = new osg::Vec3Array(1);
450  n = a.cross(b);
451  (*normals)[0].set(n.x(), n.y(), n.z());
452  (*normals)[0].normalize();
453 
454  quadGeom->setVertexArray(corners);
455  quadGeom->setTexCoordArray(0,texcoords.get());
456  quadGeom->setColorArray(quadColors);
457  quadGeom->setColorBinding(osg::Geometry::BIND_OVERALL);
458  quadGeom->setNormalArray(normals);
459  quadGeom->setNormalBinding(osg::Geometry::BIND_OVERALL);
460  quadGeom->addPrimitiveSet(new osg::DrawArrays(osg::PrimitiveSet::QUADS, 0, 4));
461 
462  linesGeom->setVertexArray(corners);
463  linesGeom->setColorArray(lineColors);
464  linesGeom->setColorBinding(osg::Geometry::BIND_OVERALL);
465  linesGeom->setNormalArray(normals);

467  linesGeom->addPrimitiveSet(new osg::DrawArrays(osg::PrimitiveSet::LINE_STRIP, 0, 5));
468 
469  geode->addDrawable(quadGeom.get());
470  geode->addDrawable(linesGeom.get());
471 
472  if(align_to_view == 1) {
473  osg::ref_ptr<osg::AutoTransform> at = new osg::AutoTransform;
474  at->setPosition(osg::Vec3(q.x(), q.y(), q.z()));
475  at->setAutoRotateMode(osg::AutoTransform::ROTATE_TO_SCREEN);
476  at->addChild(geode.get());
477  osgNode->addChild(at.get());
478  } else {
479  osg::ref_ptr<osg::PositionAttitudeTransform> at = new osg::PositionAttitudeTransform;
480  at->setPosition(osg::Vec3(q.x(),q.y(),q.z()));
481  at->addChild(geode.get());
482  osgNode->addChild(at.get());
483  }

489  Color borderColor,

493  osg::ref_ptr<osg::Geometry> linesGeom = new osg::Geometry;
494  osg::ref_ptr<osg::Geode> geode = new osg::Geode;
495  Quaternion rot = mars::utils::angleAxisToQuaternion(2.0*M_PI/resolution, q);
496  //Quaternion rot(q, 6.2831853/((double)resolution));
497  Vector tmp = b;
498 
499  // create points
500  osg::Vec3Array* vertices = new osg::Vec3Array(resolution+2);
501  (*vertices)[0].set(a.x(), a.y(), a.z());
502  (*vertices)[1].set(a.x()+b.x(), a.y()+b.y(), a.z()+b.z());
503  for(int i=0; i<resolution-1; ++i) {
504  tmp = rot *tmp;
505  (*vertices)[i+2].set(a.x()+tmp.x(), a.y()+tmp.y(), a.z()+tmp.z());
506  }

513  (*lineColours)[0].set(borderColor.r, borderColor.g,
514  borderColor.b, borderColor.a);
515 
516  osg::Vec3Array* normals = new osg::Vec3Array(1);
517  (*normals)[0].set(q.x(), q.y(), q.z());
518  (*normals)[0].set(0.0, 0.0, 1.0);
519  (*normals)[0].normalize();
520 
521  circleGeom->setVertexArray(vertices);
522  circleGeom->setColorArray(circleColours);

689  void OSGDrawItem::createPlane(OSGDrawItem *osgNode,
690  double w, double l, double h, double res) {
691  osg::ref_ptr<osg::Geode> geode = new osg::Geode;
692  osg::ref_ptr<osg::Geometry> geometry(new osg::Geometry());
693  double iStep = w/res;
694  double jStep = l/res;
695 
696  // create plane vertices
697  osg::ref_ptr<osg::Vec3Array> vertices(new osg::Vec3Array());
698  osg::ref_ptr<osg::Vec2Array> texcoords(new osg::Vec2Array());
699  for (double i =-(w/2); i<(w/2);) {
700  for (double j=-(l/2); j<(l/2);) {
701  vertices->push_back(osg::Vec3(i, j, h));
702  texcoords->push_back(osg::Vec2(0.0f, 0.0f));
703  vertices->push_back(osg::Vec3(i+iStep, j, h));
704  texcoords->push_back(osg::Vec2(0.5f, 0.0f));
705  vertices->push_back(osg::Vec3(i+iStep, j+jStep, h));
706  texcoords->push_back(osg::Vec2(0.5f, 0.5f));
707  vertices->push_back(osg::Vec3(i, j, h));
708  texcoords->push_back(osg::Vec2(0.0f, 0.0f));
709  vertices->push_back(osg::Vec3(i+iStep, j+jStep, h));
710  texcoords->push_back(osg::Vec2(0.5f, 0.5f));
711  vertices->push_back(osg::Vec3(i, j+jStep, h));
712  texcoords->push_back(osg::Vec2(0.0f, 0.5f));
713  j+=jStep;
714  }
715  i+=iStep;
716  }
717 
718  osg::ref_ptr<osg::Vec3Array> normals(new osg::Vec3Array());
719  normals->push_back(osg::Vec3(0.0f, 0.0f, 1.0f));
720 
721  geometry->setVertexArray(vertices.get());
722  geometry->setTexCoordArray(0,texcoords.get());
723  geometry->setNormalArray(normals.get());
724  geometry->setNormalBinding(osg::Geometry::BIND_OVERALL);
725  geometry->addPrimitiveSet(new osg::DrawArrays(
726  osg::PrimitiveSet::TRIANGLES, 0, vertices->size()));
727 
728  geode->addDrawable(geometry.get());
729  osgNode->addChild(geode.get());
