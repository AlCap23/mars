108  void MainGUI::addGenericMenuAction(const std::string &path, int action,
109  MenuInterface *menu, int qtKey,
110  const std::string &icon,
111  bool toolbar, int checkable) {
112  vector<string> menuPath;
113  QAction *csMenu = 0;
114  QMenu *qmenu = 0;
115  QToolBar *qtoolbar = 0;
116  genericMenu g_menu;
117  menuStruct new_menu;
118  unsigned int j = 0;
119  unsigned int i;
120  bool menu_exists = false;
121 
122  g_menu.menu = menu;
123  g_menu.action = action;
124 
125  for(i = 0; i < path.length()+1; i++) {
126  if(path[i] == '/') {
127  //copy part until '/'
128  menuPath.push_back(path.substr(j, i-j));
129  j = i + 1;
130  }
131  //save last entry part
132  if(i == path.length() && menuPath.size() > 0) {
133  menuPath.push_back(path.substr(j, i-j));
134  }
135  }
136 
137  //if a '/' is in the parentName it seems to be a menu structure
138  i = 0; //extract the .. if its in the filename
139  if(menuPath.size() > 0) {
140  //build main menu entry if first element is ".."
141  if(menuPath[0] == "..") {
142  i = 1;
143  vector<menuStruct>::iterator iter;
144  for(iter = v_qmenu.begin(); iter != v_qmenu.end(); iter++) {
145  if((*iter).label == menuPath[i]) {
146  qmenu = (*iter).menu;
147  qtoolbar = (*iter).toolbar;
148  if(toolbar && !qtoolbar) {
149  qtoolbar = mainWindow->addToolBar(QString::fromStdString(menuPath[i]));
150  (*iter).toolbar = qtoolbar;
151  }
152  menu_exists = true;
153  break;
154  }
155  }
156  if(!menu_exists) {
157  qmenu = new QMenu(QString::fromStdString(menuPath[i]));
158  menuBar->insertMenu(helpMenu->menuAction(), qmenu);
159  new_menu.menu = qmenu;
160  new_menu.label = menuPath[i];
161  if(toolbar && allow_toolbar) {
162  qtoolbar = mainWindow->addToolBar(QString::fromStdString(menuPath[i]));
163  new_menu.toolbar = qtoolbar;
164  } else {
165  new_menu.toolbar = 0;
166  }
167  v_qmenu.push_back(new_menu);
168  }
169  i++;
170  }
171  //else use the menuWindows
172  else {
173  //qmenu = mw_form.menuWindows;
174  }
175 
176 
177  //build menu structure
178  for(; i < menuPath.size(); i++) {
179  //if last entry
180  if(i == menuPath.size()-1) {
181  if(checkable < 0) { // this is only a separator
182  csMenu = qmenu->addSeparator();
183  return;
184  }
185  if(menuPath[i].empty()) {
186  return;
187  }
188 
189  if(icon != "") {
190  csMenu = qmenu->addAction(QIcon(icon.data()),
191  QString::fromStdString(menuPath[i]));
192  } else {
193  csMenu = qmenu->addAction(QString::fromStdString(menuPath[i]));
194  }
195 
196  if(toolbar && qtoolbar) {
197  qtoolbar->addAction(csMenu);
198  }
199 
200  if(qtKey != 0) {
201  csMenu->setShortcut(QKeySequence((Qt::Key)qtKey));
202  csMenu->setShortcutContext(Qt::ApplicationShortcut);
203  //shortcut = new QShortcut((Qt::Key)qtKey, mainWindow, 0, 0, Qt::ApplicationShortcut);
204  //connect(shortcut, SIGNAL(activated()), csMenu, SLOT(trigger()));
205  }
206  if(checkable > 0) {
207  csMenu->setCheckable(true);
208  if(checkable > 1) {
209  csMenu->setChecked(true);
210  } else {
211  csMenu->setChecked(false);
212  }
213  connect(csMenu, SIGNAL(toggled(bool)), this, SLOT(genericAction(bool)));
214  } else {
215  connect(csMenu, SIGNAL(triggered()), this, SLOT(genericAction()));
216  }
217  }
218  //else continue building menu groups
219  else {
220  menu_exists = false;
221  vector<menuStruct>::iterator iter;
222  for(iter = v_qmenu.begin(); iter != v_qmenu.end(); iter++) {
223  if((*iter).label == menuPath[i]) {
224  qmenu = (*iter).menu;
225  qtoolbar = (*iter).toolbar;
226  if(toolbar && !qtoolbar) {
227  qtoolbar = mainWindow->addToolBar(QString::fromStdString(menuPath[i]));
228  (*iter).toolbar = qtoolbar;
229  }
230  menu_exists = true;
231  break;
232  }
233  }
234  if(!menu_exists) {
235  QMenu *qmenu2 = new QMenu(QString::fromStdString(menuPath[i]));
236  qmenu->addMenu(qmenu2);
237  qmenu = qmenu2;
238  new_menu.menu = qmenu;
239  new_menu.label = menuPath[i];
240  if (toolbar && allow_toolbar) {
241  qtoolbar = mainWindow->addToolBar(QString::fromStdString(menuPath[i]));
242  new_menu.toolbar = qtoolbar;
243  } else {
244  new_menu.toolbar = 0;
245  }
246  v_qmenu.push_back(new_menu);
247  }
248  }
249  }
250  } else {
251  //if no menu structure is given just create a menu in the window menu
252  //csMenu = mw_form.menuWindows->addAction(QString::fromStdString(path));
253  if(qtKey != 0) {
254  csMenu->setShortcut(QKeySequence((Qt::Key)qtKey));
255  csMenu->setShortcutContext(Qt::ApplicationShortcut);
256  //shortcut = new QShortcut((Qt::Key)qtKey, mainWindow, 0, 0, Qt::ApplicationShortcut);
257  //connect(shortcut, SIGNAL(activated()), csMenu, SLOT(trigger()));
258  }
259  if(checkable > 0) {
260  csMenu->setCheckable(true);
261  if(checkable > 1) {
262  csMenu->setChecked(true);
263  } else {
264  csMenu->setChecked(false);
265  }
266  connect(csMenu, SIGNAL(toggled(bool)), this, SLOT(genericAction(bool)));
267  } else {
268  connect(csMenu, SIGNAL(triggered()), this, SLOT(genericAction()));
269  }
270  }
271  g_menu.genericAction = csMenu;
272  g_menu.path = path;
273  genericMenus.push_back(g_menu);
274  }
