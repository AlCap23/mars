101  void DbToCfg::update(sReal time_ms) {
102  static sReal leftTime = 0.0;
103  static bool updateCFG = true;
104 
105  if(time_ms > 0.0) {
106  leftTime += time_ms;
107  if(leftTime > 100.0) {
108  leftTime -= 100.0;
109  mutex.lock();
110  updateCFG = true;
111  mutex.unlock();

113  }

117  updateCFG = false;
