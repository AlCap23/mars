189  void HUDTerminal::addText(std::string text, double color[4]) {
190  double pos;
191  osg::Vec3 t_pos;
192  osg::ref_ptr<osg::Geode> node;
193  osg::ref_ptr<osg::PositionAttitudeTransform> transform;
194  unsigned int max_row_index = (height - 10) / (font_size*line_spacing);
195 
196  while(text.length() > (unsigned int)max_caracters) {
197  addText(text.substr(0, max_caracters), color);
198  text = text.substr(max_caracters);
199  }
200 
201  transform = new osg::PositionAttitudeTransform();
202 
203  pos = height - row_index*(font_size*line_spacing) - 5;
204  node = createLabel(pos, text, color);
205  transform->setPosition(osg::Vec3(posx + 3, posy + pos, 0.0f));
206  transform->addChild(node.get());
207  hudBox->addChild(transform.get());
208  hudTerminalList->addChild(transform.get());
209 
210 
211  if(hudTerminalList->getNumChildren() > max_row_index) {
212  osg::Node* toRemove = hudTerminalList->getChild(0);
213  hudBox->removeChild(toRemove);
214  hudTerminalList->removeChild(toRemove);
215  }
216  if(row_index >= (unsigned int)max_row_index) {
217  for(unsigned int i=0; i<hudTerminalList->getNumChildren(); i++) {
218  transform = dynamic_cast<osg::PositionAttitudeTransform*>(hudTerminalList->getChild(i));
219  t_pos = transform->getPosition();
220  t_pos[1] += font_size*line_spacing;
221  transform->setPosition(t_pos);
222  }
223  }
224  else row_index++;
225  }
