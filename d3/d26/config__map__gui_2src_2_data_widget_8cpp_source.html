139  void DataWidget::addConfigAtom(const std::string &name,
140  ConfigAtom &v) {
141  QtVariantProperty *guiElem;
142  std::map<QString, QVariant> attr;
143  attr["singleStep"] = 0.01;
144  attr["decimals"] = 7;

146  if(propMap.find(name) == propMap.end()) {

152  editable = true;
153  break;
154  }
155  }
156  }
157  if(type == ConfigAtom::UNDEFINED_TYPE) {
158  guiElem = pDialog->addGenericProperty(name,
159  QVariant::String,
160  QString::fromStdString(v.getUnparsedString()),
161  &attr);
162  }
163  else if(type == ConfigAtom::STRING_TYPE) {
164  guiElem = pDialog->addGenericProperty(name,
165  QVariant::String,
166  QString::fromStdString(v.getString()),
167  &attr);
168  }
169  else if(type == ConfigAtom::INT_TYPE) {
170  guiElem = pDialog->addGenericProperty(name,
171  QVariant::Int,
172  v.getInt(),
173  &attr);
174  }
175  else if(type == ConfigAtom::UINT_TYPE) {
176  guiElem = pDialog->addGenericProperty(name,
177  QVariant::Int,
178  (int)v.getUInt(),
179  &attr);
180  }
181  else if(type == ConfigAtom::DOUBLE_TYPE) {
182  guiElem = pDialog->addGenericProperty(name,
183  QVariant::Double,
184  v.getDouble(),
185  &attr);
186  }
187  else if(type == ConfigAtom::ULONG_TYPE) {
188  guiElem = pDialog->addGenericProperty(name,
189  QVariant::Int,
190  (int)v.getULong(),
191  &attr);
192  }
193  else if(type == ConfigAtom::BOOL_TYPE) {
194  guiElem = pDialog->addGenericProperty(name,
195  QVariant::Bool,
196  v.getBool(),
197  &attr);
198  }
199  guiElem->setEnabled(editable);
200  dataMap[guiElem] = &v;
201  propMap[name] = guiElem;
202  nameMap[guiElem] = name;
203  }

263  void DataWidget::updateConfigAtomI(const std::string &name,
264  ConfigAtom &v) {
265 
266  if(propMap.find(name) != propMap.end()) {
267  QtVariantProperty *guiElem;
268  std::map<QString, QVariant> attr;
269  attr["singleStep"] = 0.01;
270  attr["decimals"] = 7;
271  ConfigAtom atom = v;

273  guiElem = propMap[name];
274  *(dataMap[guiElem]) = v;
275  if(type == ConfigAtom::UNDEFINED_TYPE) {
276  guiElem->setValue(QVariant(QString::fromStdString(atom.getUnparsedString())));
277  }
278  else if(type == ConfigAtom::STRING_TYPE) {
279  guiElem->setValue(QVariant(QString::fromStdString(atom.getString())));
280  }
281  else if(type == ConfigAtom::INT_TYPE) {
282  guiElem->setValue(QVariant(atom.getInt()));
283  }
284  else if(type == ConfigAtom::UINT_TYPE) {
285  guiElem->setValue(QVariant(atom.getUInt()));
286  }
287  else if(type == ConfigAtom::DOUBLE_TYPE) {
288  guiElem->setValue(QVariant(atom.getDouble()));
289  }
290  else if(type == ConfigAtom::ULONG_TYPE) {
291  guiElem->setValue(QVariant((int)atom.getULong()));
292  }
293  else if(type == ConfigAtom::BOOL_TYPE) {
294  guiElem->setValue(QVariant(atom.getBool()));
295  }
296  }
297  else { // add the element
298  std::vector<std::string> arrPath = utils::explodeString('/', name);
299  ConfigItem *item = config[arrPath[1]];
300  for(size_t i=2; i<arrPath.size(); ++i) {
301  item = ((*item)[arrPath[i]]);
302  }
303  *item = v;
304  addConfigAtom(name, *item);
305  }

332  void DataWidget::valueChanged(QtProperty *property, const QVariant &value) {

350  it->second->setUInt(value.toUInt());
351  }
352  else if(type == ConfigAtom::DOUBLE_TYPE) {
353  it->second->setDouble(value.toDouble());
354  }
355  else if(type == ConfigAtom::ULONG_TYPE) {
356  it->second->setULong(value.toULongLong());
357  }
358  else if(type == ConfigAtom::BOOL_TYPE) {
359  it->second->setBool(value.toBool());
360  }
361  emit valueChanged(nameMap[(QtVariantProperty*)property], it->second->toString());
362  }
363  }
364  {
365  map<QtVariantProperty*, ConfigMap*>::iterator it;
366  it = addMap.find((QtVariantProperty*)property);
367  if(it != addMap.end()) {
368  addKeyStr = value.toString().toStdString();
369  addProperty = it->first;
370  }
371  }
372  emit mapChanged();
373  }
