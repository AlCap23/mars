69  void DataWidget::addConfigMap(const std::string &name,
70  ConfigMap &map) {
71  QtVariantProperty *guiElem;
72  ConfigMap::iterator it = map.begin();
73  for(;it!=map.end(); ++it) {
74  for(size_t i=0; i<it->second.size(); ++i) {
75  char iText[64];
76  iText[0] = '\0';
77  if(it->second.size() > 1) {
78  sprintf(iText, "/%d", (int)i);
79  }
80  if(it->second[i].children.size() > 0) {
81  addConfigMap(name+"/"+it->first+iText, it->second[i].children);
82  }
83  else {
84  std::map<QString, QVariant> attr;

90

94  QVariant::String,

99  guiElem = pDialog->addGenericProperty(propName,
100  QVariant::String,

104  else if(type == ConfigItem::INT_TYPE) {
105  guiElem = pDialog->addGenericProperty(propName,
106  QVariant::Int,

229  void DataWidget::valueChanged(QtProperty *property, const QVariant &value) {

244  it->second->setInt(value.toInt());
245  }
246  else if(type == ConfigItem::UINT_TYPE) {
247  it->second->setUInt(value.toUInt());
248  }
249  else if(type == ConfigItem::DOUBLE_TYPE) {
250  it->second->setDouble(value.toDouble());
251  }
252  else if(type == ConfigItem::ULONG_TYPE) {
253  it->second->setULong(value.toULongLong());
254  }
255  else if(type == ConfigItem::BOOL_TYPE) {
256  it->second->setBool(value.toBool());
257  }
258  }
259  }

262  it = addMap.find((QtVariantProperty*)property);
263  if(it != addMap.end()) {
