146  void Widget_Node_State::connect(unsigned long index){
147  std::string groupName, dataName;
148  if (actualNode != 0){
149  control->nodes->getDataBrokerNames(actualNode, &groupName, &dataName);
150  control->dataBroker->unregisterAsyncReceiver(this, groupName, dataName);
151  }
152  control->nodes->getDataBrokerNames(index, &groupName, &dataName);
153  control->dataBroker->registerAsyncReceiver(this, groupName, dataName);
154  actualNode = index;
155  connected = true;
