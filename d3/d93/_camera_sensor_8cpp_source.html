225  void CameraSensor::preGraphicsUpdate(void) {
226  mutex.lock();
227  if(gc) {
228  Vector p = control->graphics->getDrawObjectPosition(draw_id);
229  Quaternion qcorrect = Quaternion(0.5, 0.5, -0.5, -0.5);
230  Quaternion q = control->graphics->getDrawObjectQuaternion(draw_id) * qcorrect;
231  gc->updateViewportQuat(p.x(), p.y(), p.z(),
232  q.x(), q.y(),
233  q.z(), q.w());
234  if(config.enabled) {
235  if(renderCam > 2) --renderCam;
236  else if(renderCam == 2) {
237  control->graphics->activate3DWindow(cam_window_id);
238  renderCam = 1;
239  }
240  else if(renderCam == 1) {
241  control->graphics->deactivate3DWindow(cam_window_id);
242  renderCam = 0;
243  }
244  }
245  }
246  mutex.unlock();
247  }

249  void CameraSensor::receiveData(const data_broker::DataInfo &info,
250  const data_broker::DataPackage &package,
251  int callbackParam) {
252  CPP_UNUSED(info);
253  mutex.lock();
254  renderCam = 2+config.frameOffset;
255  if(dbPosIndices[0] == -1) {
256  dbPosIndices[0] = package.getIndexByName("position/x");
257  dbPosIndices[1] = package.getIndexByName("position/y");
258  dbPosIndices[2] = package.getIndexByName("position/z");
259  dbRotIndices[0] = package.getIndexByName("rotation/x");
260  dbRotIndices[1] = package.getIndexByName("rotation/y");
261  dbRotIndices[2] = package.getIndexByName("rotation/z");
262  dbRotIndices[3] = package.getIndexByName("rotation/w");
263  }
264  package.get(dbPosIndices[0], &position.x());
265  package.get(dbPosIndices[1], &position.y());
266  package.get(dbPosIndices[2], &position.z());
267  package.get(dbRotIndices[0], &orientation.x());
268  package.get(dbRotIndices[1], &orientation.y());
269  package.get(dbRotIndices[2], &orientation.z());
270  package.get(dbRotIndices[3], &orientation.w());
271  position += (orientation * config.pos_offset);
272  orientation= orientation * config.ori_offset;
273  mutex.unlock();

304 
305  if((it = config->find("width")) != config->end())

316

318  cfg->opening_height = it->second;
319 
320  if((it = config->find("opening_angle2")) != config->end())
321  cfg->opening_height = it->second;
322 
323  // correct opening_height

325  cfg->opening_height = cfg->opening_width * ((double)cfg->height / (double)cfg->width);
326

334  cfg->show_cam = false;
335  }
336 
337  if((it = config->find("enabled")) != config->end()){
338  cfg->enabled = it->second;

340  cfg->enabled = true;
341  }
342 
343  if((it = config->find("hud_size")) != config->end()) {
344  cfg->hud_width = it->second["x"];
345  cfg->hud_height = it->second["y"];
346  }
347 
348  // if hud_height and hud_width are present, overwrite parameters
349  if((it = config->find("hud_width")) != config->end())
350  cfg->hud_width = it->second;
351  if((it = config->find("hud_height")) != config->end())
352  cfg->hud_height = it->second;
353  else
354  cfg->hud_height = cfg->hud_width * ((double)cfg->height / (double)cfg->width);
355 
356  if((it = config->find("position_offset")) != config->end()) {
357  cfg->pos_offset[0] = it->second["x"];
358  cfg->pos_offset[1] = it->second["y"];
359  cfg->pos_offset[2] = it->second["z"];

361  cfg->pos_offset[1], cfg->pos_offset[2]);

365  Vector euler;

367  euler.y() = it->second["pitch"];

372  cfg->ori_offset.x() = it->second["x"];
373  cfg->ori_offset.y() = it->second["y"];
374  cfg->ori_offset.z() = it->second["z"];
375  cfg->ori_offset.w() = it->second["w"];
376  }
