75  unsigned long MotorManager::addMotor(MotorData *motorS, bool reload)
76  {
77  iMutex.lock();
78  motorS->index = next_motor_id;
79  next_motor_id++;
80  iMutex.unlock();
81 
82  if (!reload) {
83  iMutex.lock();
84  simMotorsReload.push_back(*motorS);
85  iMutex.unlock();
86  }
87 
88  SimMotor* newMotor = new SimMotor(control, *motorS);
89  newMotor->attachJoint(control->joints->getSimJoint(motorS->jointIndex));
90 
91  if(motorS->jointIndex2)
92  newMotor->attachPlayJoint(control->joints->getSimJoint(motorS->jointIndex2));
93 
94  newMotor->setSMotor(*motorS);
95  iMutex.lock();
96  simMotors[newMotor->getIndex()] = newMotor;
97  iMutex.unlock();
98  control->sim->sceneHasChanged(false);
99  return motorS->index;
100  }
