77  unsigned long MotorManager::addMotor(MotorData *motorS, bool reload)
78  {
79  iMutex.lock();
80  motorS->index = next_motor_id;
81  next_motor_id++;
82  iMutex.unlock();
83 
84  if (!reload) {
85  iMutex.lock();
86  simMotorsReload.push_back(*motorS);
87  iMutex.unlock();
88  }
89 
90  SimMotor* newMotor = new SimMotor(control, *motorS);
91  newMotor->attachJoint(control->joints->getSimJoint(motorS->jointIndex));
92 
93  if(motorS->jointIndex2)
94  newMotor->attachPlayJoint(control->joints->getSimJoint(motorS->jointIndex2));
95 
96  newMotor->setSMotor(*motorS);
97  iMutex.lock();
98  simMotors[newMotor->getIndex()] = newMotor;
99  iMutex.unlock();
100  control->sim->sceneHasChanged(false);
101 
102  configmaps::ConfigMap &config = motorS->config;
103 
104  // set motor mimics
105  if (config.find("mimic_motor") != config.end()) {
106  mimicmotors[motorS->index] = (std::string)config["mimic_motor"];
107  newMotor->setMimic(
108  (sReal)config["mimic_multiplier"], (sReal)config["mimic_offset"]);
109  }
110 
111  // set approximation functions
112  if (config.find("maxeffort_approximation") != config.end()) {
113  std::vector<sReal>* maxeffort_coefficients = new std::vector<sReal>;
114  ConfigVector::iterator vIt = config["maxeffort_coefficients"].begin();
115  for (; vIt != config["maxeffort_coefficients"].end(); ++vIt) {
116  maxeffort_coefficients->push_back((double)(*vIt));
117  newMotor->setMaxEffortApproximation(
118  utils::getApproximationFunctionFromString((std::string)config["maxeffort_approximation"]),
119  maxeffort_coefficients);
120  }
121  }
122  if (config.find("maxspeed_approximation") != config.end()) {
123  std::vector<sReal>* maxspeed_coefficients = new std::vector<sReal>;
124  ConfigVector::iterator vIt = config["maxspeed_coefficients"].begin();
125  for (; vIt != config["maxspeed_coefficients"].end(); ++vIt) {
126  maxspeed_coefficients->push_back((double)(*vIt));
127  newMotor->setMaxSpeedApproximation(
128  utils::getApproximationFunctionFromString((std::string)config["maxspeed_approximation"]),
129  maxspeed_coefficients);
130  }
131  }
132  if (config.find("current_approximation") != config.end()) {
133  std::vector<sReal>* current_coefficients = new std::vector<sReal>;
134  ConfigVector::iterator vIt = config["current_coefficients"].begin();
135  for (; vIt != config["current_coefficients"].end(); ++vIt) {
136  current_coefficients->push_back((double)(*vIt));
137  newMotor->setCurrentApproximation(
138  utils::getApproximationFunction2DFromString((std::string)config["current_approximation"]),
139  current_coefficients);
140  }
141  }
142 
143  return motorS->index;
144  }

545  void MotorManager::edit(interfaces::MotorId id, const std::string &key,
546  const std::string &value) {

548  map<unsigned long, SimMotor*>::iterator iter = simMotors.find(id);

554  iter->second->setI(atof(value.c_str()));
555  }
556  if(matchPattern("*/d", key)) {
557  iter->second->setI(atof(value.c_str()));
558  }
559  if(matchPattern("*/maxSpeed", key)) {
560  iter->second->setMaxSpeed(atof(value.c_str()));
561  }
562  if(matchPattern("*/maxEffort", key)) {
563  iter->second->setMaxEffort(atof(value.c_str()));
564  }
565  if(matchPattern("*/minValue", key)) {
566  iter->second->setMinValue(atof(value.c_str()));
567  }
568  if(matchPattern("*/maxValue", key)) {
569  iter->second->setMaxValue(atof(value.c_str()));
570  }
571  if(matchPattern("*/type", key)) {
572  if(value == "DC" || value == "2") {
573  iter->second->setType(MOTOR_TYPE_VELOCITY);
574  }
575  else {
576  iter->second->setType(MOTOR_TYPE_POSITION);
