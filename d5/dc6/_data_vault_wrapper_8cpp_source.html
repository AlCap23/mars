132  void DataVaultWrapper::receiveData(const data_broker::DataInfo& info,
133  const data_broker::DataPackage& package,
134  int callbackParam) {
135  receiverMutex.lock();
136  if(callbackParam == CALLBACK_NEW_STREAM) {
137  DataInfo newInfo;
138  package.get("groupName", &newInfo.groupName);
139  package.get("dataName", &newInfo.dataName);
140  package.get("dataId", (long*)&newInfo.dataId);
141  package.get("flags", (int*)&newInfo.flags);
142  control->dataBroker->registerTimedReceiver(this,
143  newInfo.groupName,
144  newInfo.dataName,
145  "_REALTIME_",
146  DVW_UPDATE_MS,
147  dataWrapperStructs.size());
148  DataWrapperStruct newS = {false, 0, NULL, NULL};
149  dataWrapperStructs.push_back(newS);
150  fprintf(stderr, "register\n");
151  } else {
152  assert(callbackParam >= 0 &&
153  callbackParam < dataWrapperStructs.size());
154  const DataItem *item;
155  DataWrapperStruct *data = &dataWrapperStructs[callbackParam];
156  if(!data->isInitialised) {
157  data->numItems = package.size();
158  data->text = new QString[data->numItems];
159  data->data = new double[data->numItems];
160 
161  std::string text;
162  for(int i=0; i<data->numItems; ++i) {
163  text = info.groupName+":"+info.dataName;
164  item = &(package[i]);
165  if(package.size() > 1) {
166  text += ":"+item->getName();
167  }
168  std::replace(text.begin(), text.end(), '/', ':');
169  data->text[i] = QString::fromStdString(text);
170  }
171  data->isInitialised = true;
172  }
173  for(int i=0; i<data->numItems; ++i) {
174  data->data[i] = package[i].d;
175  mpDataVaultInterface->addValue(data->text[i],
176  data->data[i]);
177  }
178  }
179  receiverMutex.unlock();
180  }
