42  OSGHudElementStruct::OSGHudElementStruct(const hudElementStruct &he,
43  const std::string &config_path,
44  unsigned int id, osg::Node* node)
45  : elem_(NULL)
46  {
47  switch (he.type) {
48  case HUD_ELEMENT_LABEL:
49  {
50  HUDLabel *myLabel = new HUDLabel(this);
51  myLabel->setConfigPath(config_path);
52  myLabel->setID(id);
53  myLabel->setPos(he.posx, he.posy);
54  myLabel->setBackgroundColor(he.background_color[0],
55  he.background_color[1],
56  he.background_color[2],
57  he.background_color[3]);
58  myLabel->setBorderColor(he.border_color[0], he.border_color[1],
59  he.border_color[2], he.border_color[3]);
60  myLabel->setBorderWidth(he.border_width);
61  myLabel->setFontSize(he.font_size);
62  myLabel->setPadding(he.padding[0], he.padding[1],
63  he.padding[2], he.padding[3]);
64  myLabel->setDirection(he.direction);
65  elem_ = myLabel;
66  break;
67  }
68  case HUD_ELEMENT_LINES:
69  {
70  HUDLines *myLines = new HUDLines(this);
71  myLines->setID(id);
72  myLines->setPos(he.posx, he.posy);
73  myLines->setBackgroundColor(he.background_color[0],
74  he.background_color[1],
75  he.background_color[2],
76  he.background_color[3]);
77  myLines->setBorderColor(he.border_color[0], he.border_color[1],
78  he.border_color[2], he.border_color[3]);
79  myLines->setBorderWidth(he.border_width);
80  myLines->setPadding(he.padding[0], he.padding[1],
81  he.padding[2], he.padding[3]);
82  myLines->setPointSize(he.font_size);
83  if(he.direction > 0) myLines->setRenderOrder(he.direction);
84  elem_ = myLines;
85  break;
86  }
87  case HUD_ELEMENT_TERMINAL:
88  break;
89  case HUD_ELEMENT_TEXTURE:
90  {
91  HUDTexture *myTexture = new HUDTexture(this);
92  myTexture->setID(id);
93  myTexture->setSize(he.width, he.height);
94  myTexture->setTextureSize(he.texture_width, he.texture_height);
95  myTexture->setPos(he.posx, he.posy);
96  myTexture->setBorderColor(he.border_color[0], he.border_color[1],
97  he.border_color[2], he.border_color[3]);
98  myTexture->setBorderWidth(he.border_width);
99  myTexture->createBox();
100  elem_ = myTexture;
101  break;
102  }
103  case HUD_ELEMENT_OSGNODE:
104  {
105  HUDOSGNode *myNode = new HUDOSGNode(this);
106  myNode->setID(id);
107  myNode->setOSGNode(node);
108  elem_ = myNode;
109  break;
110  }
111  }
112  osg::StateSet *state = elem_->getNode()->getOrCreateStateSet();
113  state->setRenderBinDetails(HUDElement::elemCount++, "RenderBin");
114  }
