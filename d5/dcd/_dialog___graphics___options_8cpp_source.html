159  void Dialog_Graphics_Options::on_change_viewport(int index) {
160  for (unsigned int i = 0; i < winIds.size(); i++)
161  if (winIds[i]->value().toBool() == true) {
162  interfaces::GraphicsCameraInterface *g_cam = cam_backup[i].g_cam;
163  switch (index) {
164  case 1: g_cam->context_setCamPredefLeft(); break;
165  case 2: g_cam->context_setCamPredefRight(); break;
166  case 3: g_cam->context_setCamPredefFront(); break;
167  case 4: g_cam->context_setCamPredefRear(); break;
168  case 5: g_cam->context_setCamPredefTop(); break;
169  case 6: g_cam->context_setCamPredefBottom(); break;
170  default: break;
171  }
172 
173  if (index) {
174  viewport->removeSubProperty(rot);
175  viewport->removeSubProperty(pos);
176  } else {
177  viewport->addSubProperty(pos);
178  viewport->addSubProperty(rot);
179  double rx, ry, tx, ty, tz, rz;
180  g_cam->getViewport(&rx, &ry, &tx, &ty, &tz, &rz);
181  rot_x->setValue(rx); rot_y->setValue(ry); rot_z->setValue(rz);
182  pos_x->setValue(tx); pos_y->setValue(ty); pos_z->setValue(tz);
183  }
184  }
185  }

223  void Dialog_Graphics_Options::valueChanged(QtProperty* property, const QVariant& value) {
224 
225  if (filled == false)
226  return;
227 
228  for (unsigned int i = 0; i < winIds.size(); i++)
229  if (property == winIds[i]) {
230  on_change_window(value.toInt());
231  update();
232  return;
233  }
234 
235  if (property == fogEnabled) {
236  on_change_fog(value.toBool());
237  update();
238  }
239 
240  else if (property == coords)
241  value.toBool() ? control->graphics->showCoords() : control->graphics->hideCoords();
242 
243  else if (property == clouds)
244  value.toBool() ? control->graphics->showClouds() : control->graphics->hideClouds();
245 
246  else if (property == grid)
247  value.toBool() ? control->graphics->showGrid() : control->graphics->hideGrid();
248 
249  else if (property == camera) {
250  for (unsigned int i = 0; i < winIds.size(); i++)
251  if (winIds[i]->value().toBool() == true)
252  cam_backup[i].g_cam->setCamera(value.toInt()+1);
253  }
254 
255  else if (property == viewport)
256  on_change_viewport(value.toInt());
257 
258  else if (property == cam_type)
259  on_change_camera_type(value.toInt());
260 
261  else if (property == rot_x || property == rot_y || property == rot_z ||
262  property == pos_x || property == pos_y || property == pos_z) {
263  for (unsigned int i = 0; i < winIds.size(); i++)
264  if (winIds[i]->value().toBool() == true)
265  cam_backup[i].g_cam->updateViewport(rot_x->value().toDouble(), rot_y->value().toDouble(),
266  pos_x->value().toDouble(), pos_y->value().toDouble(),
267  pos_z->value().toDouble(), rot_z->value().toDouble(), 1);
268  }
269 
270  else
271  update();
272  }
