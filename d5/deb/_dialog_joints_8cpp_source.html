185  void DialogJoints::on_remove_joint() {
186  filled = false;
187 
188  for (unsigned int i = 0; i < newJoints_p.size(); i++)
189  if (pDialog->currentItem() == newJoints_p[i]) {
190  delete newDialogs[i];
191  newDialogs.erase(newDialogs.begin() + i);
192  newJoints_p.erase(newJoints_p.begin() + i);
193  break;
194  }
195 
196  for (unsigned int i = 0; i < allJoints_p.size(); i++)
197  if (allJoints_p[i] == pDialog->currentItem()) {
198  control->joints->removeJoint(allJoints[i].index);
199  allJoints.erase(allJoints.begin() + i);
200  delete allDialogs[i];
201  allDialogs.erase(allDialogs.begin() + i);
202  allJoints_p.erase(allJoints_p.begin() + i);
203  break;
204  }
205 
206  filled = true;
207 
208  if (allJoints_p.size() == 0 && newJoints_p.size() == 0) {
209  stateButton->hide();
210  removeButton->hide();
211  addButton->hide();
212  previewButton->hide();
213  } else if (allJoints_p.size() > 0) {
214  pDialog->setCurrentItem(allJoints_p[0]);
215  } else if (newJoints_p.size() > 0) {
216  pDialog->setCurrentItem(newJoints_p[0]);
217  }
218 
219  }

221  void DialogJoints::on_add_joint() {
222  filled = false;
223 
224  for (unsigned int i = 0; i < newJoints_p.size(); i++)
225  if (pDialog->currentItem() == newJoints_p[i]) {
226  if ((newDialogs[i]->accept()) != 0) {
227  return;
228  } else {
229  allJoints_p.push_back(newJoints_p[i]);
230  allDialogs.push_back(newDialogs[i]);
231  oldFocus = NULL;
232  newJoints_p.erase(newJoints_p.begin() + i);
233  newDialogs.erase(newDialogs.begin() + i);
234  topLevelItemChanged(allJoints_p.back());
235  }
236  break;
237  }
238  control->joints->getListJoints(&allJoints);
239 
240  filled = true;
241  addButton->hide();
242  previewButton->hide();
243 
244  }
