45  void CylinderDrawObject::createShellGeometry(osg::Vec3Array *vertices,
46  osg::Vec3Array *normals,
47  osg::Vec2Array *uv,
48  float radius,
49  float height,
50  float theta) {
51  osg::Vec3 heightOffset1(0.0f, 0.0f, -0.5*height);
52  osg::Vec3 heightOffset2(0.0f, 0.0f, 0.5*height);
53  osg::Vec3 currPoint, v;
54  osg::Vec2 v2;
55  bool done = false;
56 
57 #define CYLINDER_POINT_(p,n,u) \
58  v = p; v2.set(u, v.z()); \
59  vertices->push_back(v); \
60  v = n; v.normalize(); \
61  normals->push_back(v); \
62  uv->push_back(v2);
63 
64  for(float alpha=0.0f; !done; alpha+=theta) {
65  if(alpha>=2.0*M_PI) {
66  alpha = 2.0*M_PI;
67  done = true;
68  }
69 
70  float sa = radius*sin(alpha);
71  float ca = radius*cos(alpha);
72 
73  currPoint.set( sa, ca, 0.0f );
74 
75  CYLINDER_POINT_(currPoint + heightOffset1, currPoint, alpha);
76  CYLINDER_POINT_(currPoint + heightOffset2, currPoint, alpha);
77  }
78 #undef CYLINDER_POINT_
79  }

81  void createCircleGeometry(osg::Vec3Array *vertices,
82  osg::Vec3Array *normals,
83  osg::Vec2Array *uv,
84  float radius,
85  float theta,
86  osg::Vec3 offset,
87  bool switchFace) {
88  osg::Vec3 topVec(0.0f, 0.0f, 1.0f);
89  osg::Vec3 p(0.0f, 0.0f, 0.0f);
90  osg::Vec2 v2;
91  bool done = false;
92 
93  for(float alpha=0.0f; !done; alpha+=theta) {
94  if(alpha>=2.0*M_PI) {
95  alpha = 2.0*M_PI;
96  done = true;
97  }
98 
99  if(!switchFace) {
100  p.set(0.0f, 0.0f, 0.0f);
101  normals->push_back(-topVec);
102  } else {
103  p.set(radius*sin(alpha), radius*cos(alpha), 0.0f);
104  normals->push_back(topVec);
105  }
106  vertices->push_back(p + offset);
107  v2.set(p.x(), p.y());
108  uv->push_back(v2);
109 
110  if(!switchFace) {
111  p.set(radius*sin(alpha), radius*cos(alpha), 0.0f);
112  normals->push_back(-topVec);
113  } else {
114  p.set(0.0f, 0.0f, 0.0f);
115  normals->push_back(topVec);
116  }
117  vertices->push_back(p + offset);
118  v2.set(p.x(), p.y());
119  uv->push_back(v2);
120  }
121  }

124  osg::ref_ptr<osg::Vec3Array> vertices(new osg::Vec3Array());
125  osg::ref_ptr<osg::Vec3Array> normals(new osg::Vec3Array());

127  osg::Geometry *geom = new osg::Geometry();
128  std::list< osg::ref_ptr< osg::Geode > > geodes;
129 
130  double step = 0.4;
131  // top cap geom
132  createCircleGeometry(
133  vertices, normals, uv,

135  osg::Vec3(0.0f, 0.0f, 0.5*height_),
136  true);
137  // cylinder shell geom
138  CylinderDrawObject::createShellGeometry(

140  radius_, height_, step);
141  // bottom cap geom
142  createCircleGeometry(
143  vertices, normals, uv,
144  radius_, step,
145  osg::Vec3(0.0f, 0.0f, -0.5*height_),
146  false);
147 
148  geom->setVertexArray(vertices.get());
149  geom->setNormalArray(normals.get());
150  geom->setNormalBinding(osg::Geometry::BIND_PER_VERTEX);
151  geom->setTexCoordArray(DEFAULT_UV_UNIT,uv.get());
152  geom->addPrimitiveSet(new osg::DrawArrays(
153  osg::PrimitiveSet::TRIANGLE_STRIP,
154  0, // index of first vertex
155  vertices->size()));
156 
157  geom->setUseDisplayList(false);
158  geom->setUseVertexBufferObjects(true);
159  osg::ref_ptr<osg::Geode> geode = new osg::Geode;
160  geode->addDrawable(geom);
161  geodes.push_back(geode);
162 
163  return geodes;
164  }
