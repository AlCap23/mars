62  void CameraConfig::updateCamera(void) {
63  if(lock_camera) {
64  utils::Quaternion q;
65  q.setIdentity();
66  utils::Vector p;
67  dataMutex.lock();
68 
69  unsigned long id = control->nodes->getDrawID(node_id);
70  node_pos = control->graphics->getDrawObjectPosition(id);
71  node_rot = control->graphics->getDrawObjectQuaternion(id);
72  if(lock_rotation) {
73  q = offset_rot;
74  p = offset_pos;
75  }
76  else {
77  q = node_rot*offset_rot;
78  //p = QVRotate(node_rot, offset_pos);
79  p = (node_rot * offset_pos);
80  }
81  p += node_pos;
82  dataMutex.unlock();
83  gc->updateViewportQuat(p.x(), p.y(), p.z(),
84  q.x(), q.y(), q.z(), q.w());
85  if(++update_ticks > 10) setLockID(node_id);
86  }
87  }
