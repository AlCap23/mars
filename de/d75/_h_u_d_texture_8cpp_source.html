114  void HUDTexture::createBox(void) {
115  osg::ref_ptr<osg::Geometry> quadGeom = new osg::Geometry;
116  osg::Vec3Array* corners = new osg::Vec3Array(4);
117  osg::Vec2Array* texcoords = new osg::Vec2Array(4);
118 
119  geode = new osg::Geode;
120 
121  (*corners)[0].set(0, 0, -1.5f);
122  (*corners)[1].set(0 + width, 0, -1.5f);
123  (*corners)[2].set(0 + width, 0 + height, -1.5f);
124  (*corners)[3].set(0, 0 + height, -1.5f);
125 
126  (*texcoords)[0].set(0.0f, 0.0f);
127  (*texcoords)[1].set(1.0f, 0.0f);
128  (*texcoords)[2].set(1.0f, 1.0f);
129  (*texcoords)[3].set(0.0f, 1.0f);
130 
131  quadGeom->setVertexArray(corners);
132  quadGeom->setTexCoordArray(0,texcoords);
133 
134 
135  osg::Vec4Array* colours = new osg::Vec4Array(1);
136 
137  (*colours)[0].set(1.0, 1.0, 1.0, 1.0);
138 
139  quadGeom->setColorArray(colours);
140  quadGeom->setColorBinding(osg::Geometry::BIND_OVERALL);
141 
142  osg::Vec3Array* normals = new osg::Vec3Array(1);
143 
144  (*normals)[0].set(0, 0, 1);
145  (*normals)[0].normalize();
146 
147  quadGeom->setNormalArray(normals);
148  quadGeom->setNormalBinding(osg::Geometry::BIND_OVERALL);
149  //quadGeom->setDataVariance(osg::Object::DYNAMIC);
150 
151  quadGeom->addPrimitiveSet(new osg::DrawArrays(osg::PrimitiveSet::QUADS,0,4));
152 
153  geode->addDrawable(quadGeom.get());
154 
155  image = new osg::Image();
156  image->setImage(t_width, t_height, 1, GL_RGBA,
157  GL_BGRA, GL_UNSIGNED_INT_8_8_8_8_REV,
158  (unsigned char*)malloc(t_width*t_height*4),
159  osg::Image::NO_DELETE);
160  texture = new osg::Texture2D();
161  texture->setDataVariance(osg::Object::DYNAMIC);
162  texture->setImage(image.get());
163  /*
164  geode->getOrCreateStateSet()->setTextureAttributeAndModes(0, texture.get(),
165  osg::StateAttribute::ON);
166 */
167  posTransform->addChild(geode.get());
168 
169  if(border_width > 0.0) {
170  osg::ref_ptr<osg::Geometry> linesGeom = new osg::Geometry;
171  osg::ref_ptr<osg::Geode> linesGeode = new osg::Geode;
172  //nodemanager tempnode;
173 
174  osg::Vec3Array* vertices = new osg::Vec3Array(8);
175  (*vertices)[0].set(0, 0, -2.0f);
176  (*vertices)[1].set(0+width, 0, -2.0f);
177  (*vertices)[2].set(0+width, 0, -2.0f);
178  (*vertices)[3].set(0+width, 0+height, -2.0f);
179  (*vertices)[4].set(0+width, 0+height, -2.0f);
180  (*vertices)[5].set(0, 0+height, -2.0f);
181  (*vertices)[6].set(0, 0+height, -2.0f);
182  (*vertices)[7].set(0, 0, -2.0f);
183 
184  // pass the created vertex array to the points geometry object.
185  linesGeom->setVertexArray(vertices);
186 
187  // set the colors as before, plus using the above
188  osg::Vec4Array* colors2 = new osg::Vec4Array;
189  colors2->push_back(osg::Vec4(border_color[0], border_color[1],
190  border_color[2], border_color[3]));
191  linesGeom->setColorArray(colors2);
192  linesGeom->setColorBinding(osg::Geometry::BIND_OVERALL);
193 
194  // set the normal in the same way color.
195  linesGeom->setNormalArray(normals);
196  linesGeom->setNormalBinding(osg::Geometry::BIND_OVERALL);
197 
198  // This time we simply use primitive, and hardwire the number of coords to use
199  // since we know up front,
200  linesGeom->addPrimitiveSet(new osg::DrawArrays(osg::PrimitiveSet::LINES,0,8));
201 
202  osg::ref_ptr<osg::LineWidth> linew = new osg::LineWidth(border_width);
203  linesGeom->getOrCreateStateSet()->setAttributeAndModes(linew.get(),
204  osg::StateAttribute::ON);
205 
206  // add the points geometry to the geode.
207  linesGeode->addDrawable(linesGeom.get());
208  posTransform->addChild(linesGeode.get());
209  }
210  osg::StateSet* stateset = parent->getOrCreateStateSet();
211 
212  stateset->setMode(GL_LIGHTING,osg::StateAttribute::OFF);
213  stateset->setMode(GL_BLEND,osg::StateAttribute::ON);
214  stateset->setRenderingHint(osg::StateSet::TRANSPARENT_BIN);
215  stateset->setMode(GL_DEPTH_TEST, osg::StateAttribute::OFF);
216  //stateset->setRenderBinDetails(30, "RenderBin");
217  parent->addChild(scaleTransform.get());
218  }
