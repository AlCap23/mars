87  bool CFGParamInt::setProperty(const CFGProperty &_property) {
88  unsigned int state = _property.getState();
89  bool rValue = false;
90  if( state & CFGProperty::allSet ) {
91  switch ( _property.getPropertyIndex() ) {
92  case value :
93  rValue = setPropertyValue(_property);
94  break;
95  case min :
96  rValue = setPropertyMin(_property);
97  break;
98  case max :
99  rValue = setPropertyMax(_property);
100  break;
101  case stepsize :
102  rValue = setPropertyStepsize(_property);
103  break;
104  default :
105  rValue = false;
106  } // switch
107  if(rValue) {
108  updateClients(_property);
109  }
110  }
111  return rValue;
112  }
