195  int ScanningSonar::getSensorData(double ** data) const {
196  if(!gw) return 0;
197 
198  SimMotor *motor = control->motors->getSimMotor(motorID);
199  //Quaternion q = motor->getJoint()->getAttachedNode2()->getRotation().inverse() * motor->getJoint()->getAttachedNode1()->getRotation();
200  Quaternion q = motor->getJoint()->getAttachedNode()->getRotation().inverse() * motor->getJoint()->getAttachedNode(2)->getRotation();
201 
202 
203  double bearing = mars::utils::getYaw(q);
204 
205  if(raySensor){
206  std::vector<double> res = raySensor->getSensorData();
207  (*data) = new double[res.size()+1];
208  memcpy((*data)+1,&res[0],sizeof(double)*res.size());
209  (*data)[0] = bearing;
210  return res.size()+1;
211  }
212 
213  (*data) = new double[(int)(config.maxDist/config.resolution)+1];
214  double *res = (*data);
215  int width, height;
216  float *img_data;
217  gw->getRTTDepthData(&img_data, width, height);
218 
219 
220  res[0] = bearing;
221  for(int i=0;i<config.maxDist/config.resolution;i++){
222  res[i+1] = 0;
223  }
224 
225  for(int x = 0; x < width; x++){
226  for(int y = 0;y < height; y++){
227  double dist = img_data[y+(x*height)];
228  if((int)(dist/config.resolution)+1 < (int)(config.maxDist/config.resolution)+1)
229  res[(int)(dist/config.resolution)+1]++;
230  }
231  }
232 
233  static int wth = width*height;
234  for(int i=0;i<config.maxDist/config.resolution;i++){
235  res[i+1]= std::min((res[i+1]/wth)*255.0*config.gain,255.0);
236  }
237  free(img_data);
238  return config.maxDist/config.resolution;
239  }

244  void ScanningSonar::receiveData(const data_broker::DataInfo &info,
245  const data_broker::DataPackage &package,
246  int callbackParam) {
247  CPP_UNUSED(info);
248  if(dbPosIndices[0] == -1) {
249  dbPosIndices[0] = package.getIndexByName("position/x");
250  dbPosIndices[1] = package.getIndexByName("position/y");
251  dbPosIndices[2] = package.getIndexByName("position/z");
252  dbRotIndices[0] = package.getIndexByName("rotation/x");
253  dbRotIndices[1] = package.getIndexByName("rotation/y");
254  dbRotIndices[2] = package.getIndexByName("rotation/z");
255  dbRotIndices[3] = package.getIndexByName("rotation/w");
256  }
257  package.get(dbPosIndices[0], &head_position.x());
258  package.get(dbPosIndices[1], &head_position.y());
259  package.get(dbPosIndices[2], &head_position.z());
260  package.get(dbRotIndices[0], &head_orientation.x());
261  package.get(dbRotIndices[1], &head_orientation.y());
262  package.get(dbRotIndices[2], &head_orientation.z());
263  package.get(dbRotIndices[3], &head_orientation.w());
264  head_position +=config.pos_offset;
265  head_orientation= head_orientation * config.ori_offset ;
266 
267  if(gc) {
268  gc->updateViewportQuat(head_position.x(), head_position.y(), head_position.z(),head_orientation.x(), head_orientation.y(), head_orientation.z(), head_orientation.w());
269  }
270 
271  SimMotor *motor = control->motors->getSimMotor(motorID);
272  if(motor && config.ping_pong_mode)
273  {
274  Quaternion q = motor->getJoint()->getAttachedNode()->getRotation().inverse() * motor->getJoint()->getAttachedNode(2)->getRotation();
275  double bearing = mars::utils::getYaw(q);
276 
277  bool range_switch = false;
278  if(config.right_limit - config.left_limit >= 0)
279  range_switch = true;
280  if((range_switch && (bearing <= config.left_limit || bearing >= config.right_limit)) ||
281  (!range_switch && (bearing <= config.left_limit && bearing >= config.right_limit)))
282  {
283  switch_motor_direction = true;
284  }
285  else if(switch_motor_direction)
286  {
287  switch_motor_direction = false;
288  motor->setVelocity(motor->getVelocity() * -1.0);
289  }
290  }

309 
310  cfg->show_cam = config->get("show_cam", false);
311  cfg->updateRate = config->get("rate", 0UL);
312 
313  if(cfg->show_cam) {
314  if((it2 = it->second.find("hud_idx")) !=
315  it->second.endMap())
316  cfg->hud_pos = it2->second;
317  }
318 
319  if((it = config->find("only_ray")) != config->end())
320  cfg->only_ray = it->second;
321 
322  if((it = config->find("resolution")) != config->end())
323  cfg->resolution = it->second;
324 
325  if((it = config->find("gain")) != config->end())
326  cfg->gain = it->second;
327 
328  if((it = config->find("maxDistance")) != config->end())
329  cfg->maxDist = it->second;
330 
331  if((it = config->find("internal_width")) != config->end())
332  cfg->width = it->second;
333 
334  if((it = config->find("internal_height")) != config->end())
335  cfg->height = it->second;

341  }
342 
343  if((it = config->find("position_offset")) != config->end()) {

347  }
348 
349  if((it = config->find("extension")) != config->end()) {
350  cfg->extension[0] = it->second["x"];
351  cfg->extension[1] = it->second["y"];
352  cfg->extension[2] = it->second["z"];
353  }
354 
355  if((it = config->find("orientation_offset")) != config->end()) {
356  if(it->second.hasKey("yaw")) {
357  Vector euler;
358  euler.x() = it->second["roll"];
359  euler.y() = it->second["pitch"];
360  euler.z() = it->second["yaw"];
361  cfg->ori_offset *= eulerToQuaternion(euler);
362  }
363  else {
364  Quaternion q;
365  q.x() = it->second["x"];
366  q.y() = it->second["y"];
367  q.z() = it->second["z"];
368  q.w() = it->second["w"];
369  cfg->ori_offset *= q;
370  }
371  }
372 
373  if((it = config->find("orientation")) != config->end()) {
374  if(it->second.hasKey("yaw")) {
375  Vector euler;
376  euler.x() = it->second["roll"];
377  euler.y() = it->second["pitch"];
378  euler.z() = it->second["yaw"];
379  cfg->orientation = eulerToQuaternion(euler);
380  }
381  else {
382  cfg->orientation.x() = it->second["x"];
383  cfg->orientation.y() = it->second["y"];
384  cfg->orientation.z() = it->second["z"];
385  cfg->orientation.w() = it->second["w"];
386  }
387  }
388 
389  return cfg;
390  }
