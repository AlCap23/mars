296  void NodeHandler::on_type_changed() {
297  // adjust the properies in the gui according to the selected type
298  if (mode == NodeTree::EditMode) {
299  if (node.name == "") { // terrain
300  primitive_type = interfaces::NODE_TYPE_TERRAIN;
301  topLevelNode->setPropertyName("Terrain");
302  node_name->setValue("Terrain");
303  } else if (node.filename == "PRIMITIVE")
304  primitive_type = atoi(node.origName.c_str());
305  else
306  primitive_type = interfaces::NODE_TYPE_BOX;
307  node_type->setEnabled(false);
308  topLevelNode->removeSubProperty(create_material);
309  } else { // preview mode
310 #warning cleaned up, why is here an +1 ??? this might confuse if someone looks to the enums and enters this value in the config?!?!? (matthias Goldhoorn 15.03.2011)
311  primitive_type = node_type->value().toInt() + 1;
312  geometry->removeSubProperty(move_all);
313  topLevelNode->removeSubProperty(edit_material);
314  }
315 
316  switch (primitive_type) {
317  case interfaces::NODE_TYPE_TERRAIN:
318  previewOff();
319  general->removeSubProperty(group_id);
320  general->addSubProperty(image);
321  for (unsigned int i = 0; i < (unsigned int)physics_props.size(); ++i)
322  physics->removeSubProperty(physics_props[i]);
323  physics->setValue(false);
324  physics->setEnabled(false);
325  pDialog->removeAllSubProperties(geometry);
326  geometry->addSubProperty(width);
327  geometry->addSubProperty(length);
328  geometry->addSubProperty(height);
329  topLevelNode->removeSubProperty(ode);
330  break;
331 
332  case interfaces::NODE_TYPE_PLANE:
333  general->removeSubProperty(image);
334  general->addSubProperty(group_id);
335  for (unsigned int i = 0; i < (unsigned int)physics_props.size(); ++i)
336  physics->removeSubProperty(physics_props[i]);
337  physics->setValue(true);
338  physics->setEnabled(false);
339  geometry->removeSubProperty(size);
340  geometry->removeSubProperty(rotation);
341  position->removeSubProperty(relative_pos);
342  position->removeSubProperty(pos_x);
343  position->removeSubProperty(pos_y);
344  geometry->removeSubProperty(width);
345  geometry->removeSubProperty(length);
346  geometry->removeSubProperty(height);
347  topLevelNode->addSubProperty(ode);
348  physics_model->setValue(interfaces::NODE_TYPE_PLANE-1);
349  break;
350 
351  case interfaces::NODE_TYPE_BOX:
352  size->removeSubProperty(size_d);
353  size->removeSubProperty(size_r);
354  size->removeSubProperty(size_h);
355  size->addSubProperty(size_x);
356  size->addSubProperty(size_y);
357  size->addSubProperty(size_z);
358  physics_model->setValue(interfaces::NODE_TYPE_BOX-1);
359  break;
360 
361  case interfaces::NODE_TYPE_SPHERE:
362  size->addSubProperty(size_d);
363  size->removeSubProperty(size_r);
364  size->removeSubProperty(size_h);
365  size->removeSubProperty(size_x);
366  size->removeSubProperty(size_y);
367  size->removeSubProperty(size_z);
368  physics_model->setValue(interfaces::NODE_TYPE_SPHERE-1);
369  break;
370 
371  case interfaces::NODE_TYPE_CYLINDER:
372  size->addSubProperty(size_r);
373  size->addSubProperty(size_h);
374  size->removeSubProperty(size_d);
375  size->removeSubProperty(size_x);
376  size->removeSubProperty(size_y);
377  size->removeSubProperty(size_z);
378  physics_model->setValue(interfaces::NODE_TYPE_CYLINDER-1);
379  break;
380 
381  case interfaces::NODE_TYPE_CAPSULE:
382  size->addSubProperty(size_r);
383  size->addSubProperty(size_h);
384  size->removeSubProperty(size_d);
385  size->removeSubProperty(size_x);
386  size->removeSubProperty(size_y);
387  size->removeSubProperty(size_z);
388  physics_model->setValue(interfaces::NODE_TYPE_CAPSULE-1);
389  break;
390  }
391 
392  if (primitive_type != interfaces::NODE_TYPE_TERRAIN && primitive_type != interfaces::NODE_TYPE_PLANE) {
393  general->removeSubProperty(image);
394  general->addSubProperty(group_id);
395  physics->setEnabled(true);
396  physics->setValue(!node.noPhysical);
397  for (unsigned int i = 0; i < (unsigned int)physics_props.size(); ++i)
398  if (node.noPhysical)
399  physics->removeSubProperty(physics_props[i]);
400  else
401  physics->addSubProperty(physics_props[i]);
402  geometry->addSubProperty(position);
403  geometry->addSubProperty(rotation);
404  geometry->addSubProperty(size);
405  geometry->removeSubProperty(width);
406  geometry->removeSubProperty(length);
407  geometry->removeSubProperty(height);
408  position->insertSubProperty(pos_y, 0);
409  position->insertSubProperty(pos_x, 0);
410  position->insertSubProperty(relative_pos, 0);
411  }
412 
413  if (mode == NodeTree::PreviewMode)
414  update();
415 
416  }

428  // create the node and transit from Preview to Edit mode
429  int NodeHandler::accept() {
430  if (mode == NodeTree::EditMode)
431  return 1;
432 
433  if (node_type->value().toInt() != interfaces::NODE_TYPE_TERRAIN-1) { // not terrain
434  if ((control->nodes->addNode(&node)) == 0) {
435  QMessageBox::information( pDialog, "Error",
436  "Primitive Creation Failed!",
437  "OK", 0); // ok == button
438  return 1;
439  } else {
440  myNodeIndex = (int) node.index;
441  actualName = node.name;
442  nodeName = QString::number(myNodeIndex).toStdString() + ":" + actualName;
443  topLevelNode->setPropertyName(QString::fromStdString(nodeName));
444  dem->setMaterial(&(node.material));
445  pDialog->addGenericProperty(topLevelNode, edit_material);
446  geometry->insertSubProperty(move_all, 0);
447  topLevelNode->removeSubProperty(create_material);
448  node_type->setEnabled(false);
449  }
450  }
451 
452  else { // terrain
453  // editing terrain is not implemented yet, so just ignore it after creation
454  if (image->value().toString() != "") {
455  control->nodes->addTerrain(&terrain);
456  // a terrain cannot be edited currently
457  pDialog->removeAllSubProperties(topLevelNode);
458  } else {
459  QMessageBox::information(pDialog, "Add Terrain",
460  "No image selected!\nTerrain creation failed",
461  "OK", 0);
462  return 1;
463  }
464  }
465  previewOff(); // close the preview
466 
467  pDialog->setPropertyColor(topLevelNode, editColor);
468 
469  mode = NodeTree::EditMode;
470  on_type_changed();
471  return 0;
472  }

553  void NodeHandler::update() {
554 
555  //set mass/density
556  if (mass->value().toDouble() != 0) {
557  node.mass = mass->value().toDouble();
558  node.density = 0;
559  }
560  else if (density->value().toDouble() != 0) {
561  node.density = density->value().toDouble();
562  node.mass = 0;
563  }
564  else if (physics->value().toBool()) {
565  fprintf(stderr, "Invalid mass/density values! Setting default values.\n");
566  node.density = 1.0;
567  node.mass = 0.0;
568  density->setValue(1.0);
569  }
570 
571 
572  utils::Vector pos;
573  pos.x() = pos_x->value().toDouble();
574  pos.y() = pos_y->value().toDouble();
575  pos.z() = pos_z->value().toDouble();
576  utils::sRotation rot;
577  rot.alpha = rot_alpha->value().toDouble();
578  rot.beta = rot_beta->value().toDouble();
579  rot.gamma = rot_gamma->value().toDouble();
580  utils::Quaternion qrot = utils::eulerToQuaternion(rot);
581 
582  node.relative_id = relative_pos->value().toInt();
583 
584  if (node.relative_id == 0){
585  node.pos = pos;
586  node.rot = qrot;
587  }
588  else{
589  myRelNode = control->nodes->getFullNode(allNodes[relative_pos->
590  value().toInt()-1].index);
591  node.pos = pos;
592  node.rot = qrot;
593  getAbsFromRel(myRelNode, &node);
594  }
595  actualName = node.name = node_name->value().toString().toStdString();
596  nodeName = QString::number(myNodeIndex).toStdString() + ":" + actualName;
597 
598  utils::Vector ext;
599 #warning why +1 again here? *strongly confused*
600  switch (node_type->value().toInt()+1) {
601  case interfaces::NODE_TYPE_BOX:
602  case interfaces::NODE_TYPE_PLANE:
603  ext.x() = size_x->value().toDouble();
604  ext.y() = size_y->value().toDouble();
605  ext.z() = size_z->value().toDouble();
606  break;
607  case interfaces::NODE_TYPE_SPHERE:
608  ext.x() = size_d->value().toDouble();
609  ext.y() = size_y->value().toDouble();
610  ext.z() = size_z->value().toDouble();
611  break;
612  case interfaces::NODE_TYPE_CYLINDER:
613  case interfaces::NODE_TYPE_CAPSULE:
614  ext.x() = size_r->value().toDouble();
615  ext.y() = size_h->value().toDouble();
616  ext.z() = size_z->value().toDouble();
617  break;
618  case interfaces::NODE_TYPE_TERRAIN: // Terrain
619  // terrain.name = node_name->value().toString().toStdString();
620  terrain.srcname = image->value().toString().toStdString();
621  terrain.scale = height->value().toDouble();
622  terrain.targetWidth = width->value().toDouble();
623  terrain.targetHeight = length->value().toDouble();
624  dcm->accept();
625  terrain.material = material;
626  control->loadCenter->loadHeightmap->readPixelData(&terrain);
627  break;
628  }
629  node.ext = ext;
630 
631  node.physicMode = static_cast<interfaces::NodeType>(physics_model->value().toInt()+1);
632  node.groupID = group_id->value().toInt();
633  node.movable = movable->value().toBool();
634  node.noPhysical = !(physics->value().toBool());
635  for (unsigned int i = 0; i < (unsigned int)physics_props.size(); ++i)
636  if (node.noPhysical)
637  physics->removeSubProperty(physics_props[i]);
638  else
639  physics->addSubProperty(physics_props[i]);
640  node.filename = "PRIMITIVE";
641  node.origName = (QString::number(node_type->value().toInt()+1)).toStdString();
642

679  changes = changes | interfaces::EDIT_NODE_TYPE;

683  changes = changes | interfaces::EDIT_NODE_MOVABLE;
684  if (node.groupID != group_id->value().toInt())
685  changes = changes | interfaces::EDIT_NODE_GROUP;
686  if (node.noPhysical == physics->value().toBool())
687  changes = changes | interfaces::EDIT_NODE_PHYSICS;
688

690  node.density = density->value().toDouble();
691  actualName = node.name = node_name->value().toString().toStdString();
692  nodeName = QString::number(myNodeIndex).toStdString() + ":" + actualName;
693  node.physicMode = static_cast<interfaces::NodeType>(physics_model->value().toInt()+1);
694  node.movable = movable->value().toBool();
695  node.groupID = group_id->value().toInt();
696  node.noPhysical = !(physics->value().toBool());
