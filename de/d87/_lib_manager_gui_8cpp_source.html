97  void LibManagerGui::update() {
98  if(!widget || !widget->isVisible())
99  return;
100  std::list<std::string> libNameList;
101  libManager->getAllLibraryNames(&libNameList);
102  for(std::list<std::string>::iterator it = libNameList.begin();
103  it != libNameList.end(); ++it) {
104  lib_manager::LibInfo info = libManager->getLibraryInfo(*it);
105  widget->updateLibInfo(info);
106  }
