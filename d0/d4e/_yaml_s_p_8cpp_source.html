64  string YamlSP::generateMainSource() {
65  if (function.get() == nullptr) {
66  return "";
67  }
68  ShaderFunc *u = function.get();
69 
70  stringstream code;
71  code << "void main()" << endl;
72  code << "{" << endl;
73 
74  for (set<GLSLAttribute>::const_iterator it = u->getMainVarDecs().begin();
75  it != u->getMainVarDecs().end(); ++it)
76  code << " " << *it << ";" << endl;
77 
78  std::priority_queue<PrioritizedLine> lines;
79  for (list<MainVar>::const_iterator it = u->getMainVars().begin();
80  it != u->getMainVars().end(); ++it)
81  lines.push(PrioritizedLine((*it).toString(), (*it).priority, lines.size()));
82 
83  for (vector<FunctionCall>::const_iterator it = u->getFunctionCalls().begin();
84  it != u->getFunctionCalls().end(); ++it)
85  lines.push(PrioritizedLine((*it).toString(), (*it).priority, lines.size()));
86 
87  for (vector<PrioritizedLine>::const_iterator it = u->getSnippets().begin(); it != u->getSnippets().end(); ++it) {
88  lines.push(*it);
89  }
90 
91  while (!lines.empty()) {
92  code << " " << lines.top().line << ";" << " //Priority: " << lines.top().priority << endl;
93  lines.pop();
94  }
95 
96  for (vector<GLSLExport>::const_iterator it = u->getExports().begin();
97  it != u->getExports().end(); ++it)
98  code << " " << *it << ";" << endl;
99  for (set<GLSLSuffix>::iterator it = u->getSuffixes().begin();
100  it != u->getSuffixes().end(); ++it)
101  code << " " << *it << ";" << endl;
102 
103  code << "}" << endl;
104 
105  return code.str();
106  }

157  -1);
158  vertexShader->addMainVar((GLSLVariable)
159  {"vec4", "fPos",
160  "vec4(gl_Vertex.xy + offset.xy, gl_Vertex.z/**(0.5+offset.z)*/, gl_Vertex.w)"},
161  -1);

163  {"vec3", "sc",
164  "vec3(normalize(vec2(fPos.x*0.1+0.5*rnd(float(gl_InstanceIDARB)*0.16, float(gl_InstanceIDARB)*0.8), fPos.y*0.1+0.5*rnd(float(gl_InstanceIDARB)*0.8, float(gl_InstanceIDARB)*0.16))), rnd(float(gl_InstanceIDARB)*0.4, float(gl_InstanceIDARB)*0.4))-0.5"},
165  -1);
166  vertexShader->addMainVar((GLSLVariable)
167  {"vec4", "vWorldPos",
168  "fPos + vec4(0.1*sc.z*(sin_*gl_Vertex.z*sc.x + cos_*gl_Vertex.z*sc.y), 0.1*sc.z*(sin_*gl_Vertex.z*sc.y + cos_*gl_Vertex.z*sc.x), 0, 0)"},
169  -1);

172  vertexShader->addMainVar((GLSLVariable)
