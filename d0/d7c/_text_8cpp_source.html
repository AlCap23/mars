184  void Text::updateBoundingBox() {
185  osg::BoundingBox bb;
186  labelText->dirtyBound();
187  bb = labelText->COMPUTE_BOUND();
188  //bb.expandBy(labelText->getBound());
189  //d = bb.zMin()-2.0;
190  width = resolutionCorrectionX*(bb.xMax() - bb.xMin()) + pl + pr;
191  height = resolutionCorrectionY*(bb.yMax() - bb.yMin()) + pt + pb;

195  void Text::updatePosition() {
196  if(fixedWidth > 0.0) {
197  w = fixedWidth;
198  }
199  else {
200  w = width;
201  }
202  if(fixedHeight > 0.0) {
203  h = fixedHeight;
204  }
205  else {
206  h = height;
207  }
208  if(textAlign == ALIGN_LEFT) {
209  posXB = posXI = 0.0;
210  }
211  else if (textAlign == ALIGN_CENTER) {
212  posXI = - width*0.5;
213  posXB = - w*0.5;
214  }
215  else {
216  posXI = - width;
217  posXB = - w;
218  }
219  resolutionCorrection->setMatrix(osg::Matrix::translate(posXI, 0.0, 0.0f));
220  resolutionCorrection->preMult(osg::Matrix::scale(resolutionCorrectionX,
221  resolutionCorrectionY, 1.));
222  //labelText->setPosition(osg::Vec3(posXI, 0.0, 0.0f));
223  transform->setPosition(osg::Vec3(posX, posY, -1.501f));
224  }

226  void Text::createBackground(Color bgColor) {
227  backgroundGeom = new osg::Geometry;
228  backgroundVertices = new osg::Vec3Array;
229  backgroundVertices->push_back(osg::Vec3(posXB-pl, -h+pt, 0.));
230  backgroundVertices->push_back(osg::Vec3(posXB-pl, pt, 0.));
231  backgroundVertices->push_back(osg::Vec3(posXB+w-pl, pt, 0.));
232  backgroundVertices->push_back(osg::Vec3(posXB+w-pl, -h+pt, 0.));
233  backgroundGeom->setVertexArray(backgroundVertices);
234 
235  osg::Vec3Array* normals = new osg::Vec3Array;
236  normals->push_back(osg::Vec3(0.0f,0.0f,1.0f));
237  backgroundGeom->setNormalArray(normals);
238  backgroundGeom->setNormalBinding(osg::Geometry::BIND_OVERALL);
239 
240  backgroundColor = new osg::Vec4Array;
241  backgroundColor->push_back(osg::Vec4(bgColor.r, bgColor.g,
242  bgColor.b, bgColor.a));
243  backgroundGeom->setColorArray(backgroundColor);
244  backgroundGeom->setColorBinding(osg::Geometry::BIND_OVERALL);
245 
246  backgroundGeom->addPrimitiveSet(new osg::DrawArrays(GL_QUADS,0,4));
247 
248  osg::StateSet* stateset = backgroundGeom->getOrCreateStateSet();
249  stateset->setMode(GL_LIGHTING,osg::StateAttribute::OFF);
250  stateset->setMode(GL_BLEND,osg::StateAttribute::ON);
251  stateset->setRenderingHint(osg::StateSet::TRANSPARENT_BIN);
252  stateset->setMode(GL_DEPTH_TEST, osg::StateAttribute::OFF);
253  stateset->setRenderBinDetails(21, "RenderBin");
254 
255  backgroundGeode = new osg::Geode;
256  backgroundGeode->addDrawable(backgroundGeom.get());
257  if((*backgroundColor)[0][0] > 0.0001) {
258  transform->addChild(backgroundGeode.get());
259  }
260  }

338  void Text::setFontResolution(int x, int y) {
