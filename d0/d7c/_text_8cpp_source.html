178  void Text::createBackground(Color bgColor) {
179  backgroundGeom = new osg::Geometry;
180  backgroundVertices = new osg::Vec3Array;
181  backgroundVertices->push_back(osg::Vec3(posXB-pl, -h+pt, 0.));
182  backgroundVertices->push_back(osg::Vec3(posXB-pl, pt, 0.));
183  backgroundVertices->push_back(osg::Vec3(posXB+w-pl, pt, 0.));
184  backgroundVertices->push_back(osg::Vec3(posXB+w-pl, -h+pt, 0.));
185  backgroundGeom->setVertexArray(backgroundVertices);
186 
187  osg::Vec3Array* normals = new osg::Vec3Array;
188  normals->push_back(osg::Vec3(0.0f,0.0f,1.0f));
189  backgroundGeom->setNormalArray(normals);
190  backgroundGeom->setNormalBinding(osg::Geometry::BIND_OVERALL);
191 
192  backgroundColor = new osg::Vec4Array;
193  backgroundColor->push_back(osg::Vec4(bgColor.r, bgColor.g,
194  bgColor.b, bgColor.a));
195  backgroundGeom->setColorArray(backgroundColor);
196  backgroundGeom->setColorBinding(osg::Geometry::BIND_OVERALL);
197 
198  backgroundGeom->addPrimitiveSet(new osg::DrawArrays(GL_QUADS,0,4));
199 
200  osg::StateSet* stateset = backgroundGeom->getOrCreateStateSet();
201  stateset->setMode(GL_LIGHTING,osg::StateAttribute::OFF);
202  stateset->setMode(GL_BLEND,osg::StateAttribute::ON);
203  stateset->setRenderingHint(osg::StateSet::TRANSPARENT_BIN);
204  stateset->setMode(GL_DEPTH_TEST, osg::StateAttribute::OFF);
205  stateset->setRenderBinDetails(21, "RenderBin");
206 
207  backgroundGeode = new osg::Geode;
208  backgroundGeode->addDrawable(backgroundGeom.get());
209  if((*backgroundColor)[0][0] > 0.0001) {
210  transform->addChild(backgroundGeode.get());
211  }
212  }
