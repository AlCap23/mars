185  void MenuWindow::menuAction(int action, bool checked)
186  {
187  switch (action) {
188  case GUI_ACTION_ADD_WINDOW:
189  menu_addWindow();
190  break;
191  case GUI_ACTION_DOCK_WINDOWS:
192  menu_dock(checked);
193  break;
194  case GUI_ACTION_BLENDER_EXPORT:
195  menu_blender();
196  break;
197  case GUI_ACTION_CONTROLLER_CONFIG:
198  menu_controller_config();
199  break;
200  case MENU_SHOW_VISUAL:
201  if(updateProp) {
202  if(checked)
203  cfgVisRep.iValue |= 1;
204  else
205  cfgVisRep.iValue ^= 1;
206  updateProp = false;
207  control->cfg->setProperty(cfgVisRep);
208  updateProp = true;
209  }
210  break;
211  case MENU_SHOW_PHYSICAL:
212  if(updateProp) {
213  if(checked)
214  cfgVisRep.iValue |= 2;
215  else
216  cfgVisRep.iValue ^= 2;
217  updateProp = false;
218  control->cfg->setProperty(cfgVisRep);
219  updateProp = true;
220  }
221  break;
222  case MENU_SHOW_COORDS:
223  if(updateProp) {
224  cfgShowCoords.bValue = checked;
225  updateProp = false;
226  control->cfg->setProperty(cfgShowCoords);
227  updateProp = true;
228  }
229  break;
230  case MENU_SHOW_GRID:
231  if(updateProp) {
232  cfgShowGrid.bValue = checked;
233  updateProp = false;
234  control->cfg->setProperty(cfgShowGrid);
235  updateProp = true;
236  }
237  break;
238  case MENU_SHOW_CONTACTS:
239  if(updateProp) {
240  cfgShowContacts.bValue = checked;
241  updateProp = false;
242  control->cfg->setProperty(cfgShowContacts);
243  updateProp = true;
244  }
245  break;
246  case MENU_SHOW_SELECTION:
247  if(updateProp) {
248  cfgShowSelection.bValue = checked;
249  updateProp = false;
250  control->cfg->setProperty(cfgShowSelection);
251  updateProp = true;
252  }
253  break;
254  case MENU_VIEW_TOP:
255  case MENU_VIEW_FRONT:
256  case MENU_VIEW_RIGHT:
257  case MENU_VIEW_BACK:
258  case MENU_VIEW_LEFT:
259  case MENU_VIEW_BOTTOM:
260  control->graphics->setCameraDefaultView(-action-4);
261  break;
262  }
263  }
