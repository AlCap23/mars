359  void Simulator::step(bool setState) {
360  std::vector<pluginStruct>::iterator p_iter;
361  long time;
362 
363  Status oldState;
364 
365  physicsThreadLock();
366 
367  if(setState) {
368  oldState = simulationStatus;
369  simulationStatus = STEPPING;
370  }
371 
372 #ifdef DEBUG_TIME
373  long startTime = utils::getTime();
374 
375 #endif
376  if(control->dataBroker) {
377  control->dataBroker->trigger("mars_sim/prePhysicsUpdate");
378  }
379  physics->stepTheWorld();
380 #ifdef DEBUG_TIME
381  LOG_DEBUG("Step World: %ld", getTimeDiff(startTime));
382 #endif
383 
384  control->nodes->updateDynamicNodes(calc_ms); //Moved update to here, otherwise RaySensor is one step behind the world every time
385  control->joints->updateJoints(calc_ms);
386  control->motors->updateMotors(calc_ms);
387  control->controllers->updateControllers(calc_ms);
388 
389  if(show_time)
390  time = utils::getTime();
391 
392 
393  getTimeMutex.lock();
394  dbSimTimePackage[0].d += calc_ms;
395  getTimeMutex.unlock();
396  if(control->dataBroker) {
397  control->dataBroker->pushData(dbSimTimeId,
398  dbSimTimePackage);
399  control->dataBroker->stepTimer("mars_sim/simTimer", calc_ms);
400  }
401 
402  if(show_time) {
403  avg_log_time += getTimeDiff(time);
404  if(++count > 100) {
405  avg_log_time /= count;
406  count = 0;
407  fprintf(stderr, "debug_log_time: %g\n", avg_log_time);
408  avg_log_time = 0.0;
409  }
410  }
411 
412  pluginLocker.lockForRead();
413 
414  // It is possible for plugins to call switchPluginUpdateMode during
415  // the update call and get removed from the activePlugins list there.
416  // We use erased_active to notify this loop about an erasure.
417  for(unsigned int i = 0; i < activePlugins.size();) {
418  erased_active = false;
419  if(show_time)
420  time = utils::getTime();
421 
422  activePlugins[i].p_interface->update(calc_ms);
423 
424  if(!erased_active) {
425  if(show_time) {
426  time = getTimeDiff(time);
427  activePlugins[i].timer += time;
428  activePlugins[i].t_count++;
429  if(activePlugins[i].t_count > 20) {
430  activePlugins[i].timer /= activePlugins[i].t_count;
431  activePlugins[i].t_count = 0;
432  fprintf(stderr, "debug_time: %s: %g\n",
433  activePlugins[i].name.c_str(),
434  activePlugins[i].timer);
435  activePlugins[i].timer = 0.0;
436  }
437  }
438  ++i;
439  }
440  }
441  pluginLocker.unlock();
442  if (sync_graphics) {
443  calc_time += calc_ms;
444  if (calc_time >= sync_time) {
445  sync_count = 0;
446  if(control->graphics)
447  this->allowDraw();
448  calc_time = 0;
449  }
450  }
451  if(control->dataBroker) {
452  control->dataBroker->trigger("mars_sim/postPhysicsUpdate");
453  }
454 
455  if(setState) {
456  simulationStatus = oldState;
457  }
458 
459  physicsThreadUnlock();
460  }

510  void Simulator::myRealTime() {
511 #ifdef __linux__ //__unix__, wenn Darwin das mitmacht.
512  //used to remember last time this function was called
513  //and as absolute (minimum) wake-up time.
514  static struct timespec ts;
515  static bool tsNeedsInit = true;
516  if (tsNeedsInit)
517  {
518  int retval = clock_gettime(CLOCK_MONOTONIC, &ts);
519  if (retval != 0)
520  {
521  throw std::runtime_error("clock_gettime(CLOCK_MONOTONIC, ...) failed");
522  }
523  tsNeedsInit = false;
524  }
525 
526  //schedule minimum sleep time
527  ts.tv_nsec += calc_ms * 1000000;
528 
529  //the nsec value may not exceed one second
530  while (ts.tv_nsec > 1000000000)
531  {
532  ts.tv_nsec -= 1000000000;
533  ts.tv_sec += 1;
534  }
535 
536  //sleep...
537  int retval = clock_nanosleep(CLOCK_MONOTONIC, TIMER_ABSTIME, &ts, 0);
538  if (retval != 0)
539  {
540  std::cerr << "WARNING: Your system is too slow!" << std::endl;
541  }
542 
543  //remember time
544  retval = clock_gettime(CLOCK_MONOTONIC, &ts);
545  if (retval != 0)
546  {
547  throw std::runtime_error("clock_gettime(CLOCK_MONOTONIC, ...) failed");
548  }
549 #else
550  static long myTime = utils::getTime();
551  long timeDiff = getTimeDiff(myTime);
552 
553  if(show_time) {
554  fprintf(stderr, "timeDiff: %ld\n", timeDiff);
555  }
556  if(timeDiff < calc_ms) {
557  long valSleep = calc_ms - timeDiff;
558  msleep(valSleep);
559  myTime = utils::getTime();
560  if(show_time) {
561  fprintf(stderr, "sleep time: %ld\n", valSleep);
562  }
563  } else {
564  myTime += timeDiff;
565  }
566 #endif
567  }
