1084  void GraphicsWidget::getRTTDepthData(float* buffer, int& width, int& height)
1085  {
1086  if(isRTTWidget) {
1087  GLuint* data2 = (GLuint *)rttDepthImage->data();
1088  width = rttDepthImage->s();
1089  height = rttDepthImage->t();
1090 
1091  double fovy, aspectRatio, Zn, Zf;
1092  graphicsCamera->getOSGCamera()->getProjectionMatrixAsPerspective( fovy, aspectRatio, Zn, Zf );
1093  int d = 0;
1094  for(int i=height-1; i>=0; --i) {
1095  for(int k=0; k<width; ++k) {
1096  GLuint di = data2[i*width+k];
1097 
1098  const float dv = ((float) di) / std::numeric_limits< GLuint >::max() ;
1099  // 1.0 is the max depth in the depth buffer, and
1100  // is represented as a nan in the distance image
1101  if( dv >= 1.0 )
1102  buffer[d++] = std::numeric_limits<float>::quiet_NaN();
1103  else
1104  buffer[d++] = Zn*Zf/(Zf-dv*(Zf-Zn));
1105  }
1106  }
1107  } else {
1108  throw std::runtime_error("Depth image not supported on non RTT Widges");
1109  }
1110  }
