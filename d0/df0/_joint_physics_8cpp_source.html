76  void JointPhysics::calculateCfmErp(const JointData *jointS) {
77  // how to set erp and cfm
78  // ERP = h kp / (h kp + kd)
79  // CFM = 1 / (h kp + kd)
80  damping = (dReal)jointS->damping_const_constraint_axis1;
81  spring = (dReal)jointS->spring_const_constraint_axis1;
82  dReal h = theWorld->getWorldStep();
83  cfm = damping;
84  erp1 = h*(dReal)jointS->spring_const_constraint_axis1
85  +(dReal)jointS->damping_const_constraint_axis1;
86  cfm1 = h*(dReal)jointS->spring_const_constraint_axis1
87  +(dReal)jointS->damping_const_constraint_axis1;
88  erp2 = h*(dReal)jointS->spring_const_constraint_axis2
89  +(dReal)jointS->damping_const_constraint_axis2;
90  cfm2 = h*(dReal)jointS->spring_const_constraint_axis2
91  +(dReal)jointS->damping_const_constraint_axis2;
92 
93  lo1 = jointS->lowStopAxis1;
94  hi1 = jointS->highStopAxis1;
95  lo2 = jointS->lowStopAxis2;
96  hi2 = jointS->highStopAxis2;
97  // we don't want to run in the trap where kp and kd are set to zero
98  cfm = (cfm>0)?1/cfm:0.000000000001;
99  erp1 = (erp1>0)?h*(dReal)jointS->spring_const_constraint_axis1/erp1:0;
100  cfm1 = (cfm1>0)?1/cfm1:0.00000000001;
101  erp2 = (erp2>0)?h*(dReal)jointS->spring_const_constraint_axis2/erp2:0;
102  cfm2 = (cfm2>0)?1/cfm2:0.000000000001;
103  }

109  bool JointPhysics::createJoint(JointData *jointS, const NodeInterface *node1,
110  const NodeInterface *node2) {
111 #ifdef _VERIFY_WORLD_
112  fprintf(stderr, "joint %d ; %d ; %d ; %.4f, %.4f, %.4f ; %.4f, %.4f, %.4f\n",
113  jointS->index, jointS->nodeIndex1, jointS->nodeIndex2,
114  jointS->anchor.x(), jointS->anchor.y(), jointS->anchor.z(),
115  jointS->axis1.x(), jointS->axis1.y(), jointS->axis1.z());
116 #endif
117  MutexLocker locker(&(theWorld->iMutex));
118  if ( theWorld && theWorld->existsWorld() ) {
119  //get the bodies from the interfaces nodes
120  //here we have to make some verifications
121  const NodePhysics *n1= static_cast<const NodePhysics*>(node1);
122  const NodePhysics *n2= static_cast<const NodePhysics*>(node2);
123  dBodyID b1 = 0, b2 = 0;
124 
125  calculateCfmErp(jointS);
126 
127  joint_type = jointS->type;
128  if(n1) b1 = n1->getBody();
129  if(n2) b2 = n2->getBody();
130  body1 = b1;
131  body2 = b2;
132  switch(jointS->type) {
133  case JOINT_TYPE_HINGE:
134  createHinge(jointS, b1, b2);
135  break;
136  case JOINT_TYPE_HINGE2:
137  if(b1 || b2) createHinge2(jointS, b1, b2);
138  else return 0;
139  break;
140  case JOINT_TYPE_SLIDER:
141  createSlider(jointS, b1, b2);
142  break;
143  case JOINT_TYPE_BALL:
144  createBall(jointS, b1, b2);
145  break;
146  case JOINT_TYPE_UNIVERSAL:
147  if(b1 || b2) createUniversal(jointS, b1, b2);
148  else return 0;
149  break;
150  case JOINT_TYPE_FIXED:
151  if(b1 || b2) createFixed(jointS, b1, b2);
152  else return 0;
153  break;
154  default:
155  // Invalid type. No physically node will be created.
156  std::cerr << __FILE__":" << __LINE__<< " invalid joint type for joint \""
157  << jointS->name << "\". No joint created." << std::endl;
158  return 0;
159  break;
160  }
161  // we have created a joint. To get the information
162  // of the forces the joint attached to the bodies
163  // we need to set a feedback pointer for the joint (ode stuff)
164  dJointSetFeedback(jointId, &feedback);
165  return 1;
166  }
167  return 0;
168  }

893  void JointPhysics::update(void) {
894  const dReal *b1_pos, *b2_pos;
895  dReal anchor[4], axis[4], axis2[4];
896  int calc1 = 0, calc2 = 0;
897  dReal radius, dot, torque;
898  dReal v1[3], normal[3], load[3], tmp1[3], axis_force[3];
899  MutexLocker locker(&(theWorld->iMutex));
900 
901  switch(joint_type) {
902  case JOINT_TYPE_HINGE:
903  dJointGetHingeAnchor(jointId, anchor);
904  dJointGetHingeAxis(jointId, axis);
905  calc1 = 1;
906  break;
907  case JOINT_TYPE_HINGE2:
908  dJointGetHinge2Anchor(jointId, anchor);
909  dJointGetHinge2Axis1(jointId, axis);
910  dJointGetHinge2Axis2(jointId, axis2);
911  calc1 = 1;
912  calc2 = 0;
913  break;
914  case JOINT_TYPE_SLIDER:
915  dJointGetSliderAxis(jointId, axis);
916  calc1 = 2;
917  break;
918  case JOINT_TYPE_BALL:
919  // no axis
920  break;
921  case JOINT_TYPE_UNIVERSAL:
922  dJointGetUniversalAnchor(jointId, anchor);
923  dJointGetUniversalAxis1(jointId, axis);
924  dJointGetUniversalAxis2(jointId, axis2);
925  calc1 = 1;
926  calc2 = 0;
927  break;
928  default:
929  // no correct type is spezified, so no physically node will be created
930  break;
931  }
932  motor_torque = feedback.lambda;
933  axis1_torque.x() = axis1_torque.y() = axis1_torque.z() = 0;
934  axis2_torque.x() = axis2_torque.y() = axis2_torque.z() = 0;
935  joint_load.x() = joint_load.y() = joint_load.z() = 0;
936  if(calc1 == 1) {
937  if(body1) {
938  b1_pos = dBodyGetPosition(body1);
939  dOP(v1, -, b1_pos, anchor);
940  //radius = dLENGTH(v1);
941  dCROSS(normal, =, axis, v1);
942  dot = dDOT(normal, feedback.f1);
943  dOPEC(normal, *=, dot);
944  dOP(load, -, feedback.f1, normal);
945  dCROSS(tmp1, =, v1, normal);
946  axis1_torque.x() = (sReal)tmp1[0];
947  axis1_torque.y() = (sReal)tmp1[1];
948  axis1_torque.z() = (sReal)tmp1[2];
949  dCROSS(tmp1, =, v1, load);
950  joint_load.x() = (sReal)tmp1[0];
951  joint_load.y() = (sReal)tmp1[1];
952  joint_load.z() = (sReal)tmp1[2];
953  // now nearly the same for the torque
954  dot = dDOT(axis, feedback.t1);
955  dOPC(tmp1, *, axis, dot);
956  dOP(load, -, feedback.t1, tmp1);
957  axis1_torque.x() += (sReal)tmp1[0];
958  axis1_torque.y() += (sReal)tmp1[1];
959  axis1_torque.z() += (sReal)tmp1[2];
960  joint_load.x() += (sReal)load[0];
961  joint_load.y() += (sReal)load[1];
962  joint_load.z() += (sReal)load[2];
963  }
964  else if(body2) {
965  // now we do it correct
966  // first get the position vector v1
967  b2_pos = dBodyGetPosition(body2);
968  dOP(v1, -, b2_pos, anchor);
969 
970  // then differentiate the torque from the force
971  dot = dDOT(feedback.f2, v1) / dDOT(v1, v1);
972  dOPC(axis_force, *, v1, dot);
973 
974  // the difference is the torque
975  dOP(tmp1, -, feedback.f2, axis_force);
976 
977  // the torque value is given by:
978  torque = dLENGTH(tmp1) / dLENGTH(v1);
979  // then get the normal to v1 and the torque vector
980  dCROSS(normal, =, v1, tmp1);
981  // and scale the normal to represent the correct torque length
982  dOPEC(normal, *=, torque / dLENGTH(normal));
983  // now the normal represents the torque vector at the anchor
984  // and we make a projection to the axis
985  dot = dDOT(normal, axis);
986  dOPC(tmp1, *, axis, dot);
987  dOP(load, -, normal, tmp1);
988  //dCROSS(tmp1, =, v1, normal);
989  axis1_torque.x() = (sReal)tmp1[0];
990  axis1_torque.y() = (sReal)tmp1[1];
991  axis1_torque.z() = (sReal)tmp1[2];
992  //dCROSS(tmp1, =, v1, load);
993  joint_load.x() = (sReal)load[0];
994  joint_load.y() = (sReal)load[1];
995  joint_load.z() = (sReal)load[2];
996  // now nearly the same for the torque
997  dot = dDOT(feedback.t2, axis);
998  dOPC(tmp1, *, axis, dot);
999  dOP(load, -, feedback.t2, tmp1);
1000  //axis1_torque.x() += (sReal)tmp1[0];
1001  //axis1_torque.y() += (sReal)tmp1[1];
1002  //axis1_torque.z() += (sReal)tmp1[2];
1003  joint_load.x() += (sReal)load[0];
1004  joint_load.y() += (sReal)load[1];
1005  joint_load.z() += (sReal)load[2];
1006  }
1007  }
1008  else if(calc1 == 2) {
1009  // this is for the slider
1010  }
1011  if(calc2 == 1) {
1012  if(body1) {
1013  b1_pos = dBodyGetPosition(body1);
1014  dOP(v1, -, b1_pos, anchor);
1015  radius = dLENGTH(v1);
1016  dCROSS(normal, =, axis2, v1);
1017  dot = dDOT(normal, feedback.f1);
1018  dOPEC(normal, *=, dot*radius);
1019  axis2_torque.x() = (sReal)normal[0];
1020  axis2_torque.y() = (sReal)normal[1];
1021  axis2_torque.z() = (sReal)normal[2];
1022  // now nearly the same for the torque
1023  dot = dDOT(axis2, feedback.t1);
1024  axis2_torque.x() += (sReal)(axis2[0]*dot);
1025  axis2_torque.y() += (sReal)(axis2[1]*dot);
1026  axis2_torque.z() += (sReal)(axis2[2]*dot);
1027  }
1028  if(body2) {
1029  b2_pos = dBodyGetPosition(body2);
1030  dOP(v1, -, b2_pos, anchor);
1031  radius = dLENGTH(v1);
1032  dCROSS(normal, =, axis2, v1);
1033  dot = dDOT(normal, feedback.f2);
1034  dOPEC(normal, *=, dot*radius);
1035  axis2_torque.x() += (sReal)normal[0];
1036  axis2_torque.y() += (sReal)normal[1];
1037  axis2_torque.z() += (sReal)normal[2];
1038  // now nearly the same for the torque
1039  dot = dDOT(axis2, feedback.t2);
1040  axis2_torque.x() += (sReal)(axis2[0]*dot);
1041  axis2_torque.y() += (sReal)(axis2[1]*dot);
1042  axis2_torque.z() += (sReal)(axis2[2]*dot);
1043  }
1044  }
