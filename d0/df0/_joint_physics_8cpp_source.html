77  void JointPhysics::calculateCfmErp(const JointData *jointS) {
78  // how to set erp and cfm
79  // ERP = h kp / (h kp + kd)
80  // CFM = 1 / (h kp + kd)
81  damping = (dReal)jointS->damping_const_constraint_axis1;
82  spring = (dReal)jointS->spring_const_constraint_axis1;
83  dReal h = theWorld->getWorldStep();
84  cfm = damping;
85  erp1 = h*(dReal)jointS->spring_const_constraint_axis1
86  +(dReal)jointS->damping_const_constraint_axis1;
87  cfm1 = h*(dReal)jointS->spring_const_constraint_axis1
88  +(dReal)jointS->damping_const_constraint_axis1;
89  erp2 = h*(dReal)jointS->spring_const_constraint_axis2
90  +(dReal)jointS->damping_const_constraint_axis2;
91  cfm2 = h*(dReal)jointS->spring_const_constraint_axis2
92  +(dReal)jointS->damping_const_constraint_axis2;
93 
94  lo1 = jointS->lowStopAxis1;
95  hi1 = jointS->highStopAxis1;
96  lo2 = jointS->lowStopAxis2;
97  hi2 = jointS->highStopAxis2;
98  // we don't want to run in the trap where kp and kd are set to zero
99  cfm = (cfm>0)?1/cfm:0.000000000001;
100  erp1 = (erp1>0)?h*(dReal)jointS->spring_const_constraint_axis1/erp1:0;
101  cfm1 = (cfm1>0)?1/cfm1:0.00000000001;
102  erp2 = (erp2>0)?h*(dReal)jointS->spring_const_constraint_axis2/erp2:0;
103  cfm2 = (cfm2>0)?1/cfm2:0.000000000001;
104  }

110  bool JointPhysics::createJoint(JointData *jointS, const NodeInterface *node1,
111  const NodeInterface *node2) {
112 #ifdef _VERIFY_WORLD_
113  fprintf(stderr, "joint %d ; %d ; %d ; %.4f, %.4f, %.4f ; %.4f, %.4f, %.4f\n",
114  jointS->index, jointS->nodeIndex1, jointS->nodeIndex2,
115  jointS->anchor.x(), jointS->anchor.y(), jointS->anchor.z(),
116  jointS->axis1.x(), jointS->axis1.y(), jointS->axis1.z());
117 #endif
118  MutexLocker locker(&(theWorld->iMutex));
119  if ( theWorld && theWorld->existsWorld() ) {
120  //get the bodies from the interfaces nodes
121  //here we have to make some verifications
122  const NodePhysics *n1= static_cast<const NodePhysics*>(node1);
123  const NodePhysics *n2= static_cast<const NodePhysics*>(node2);
124  dBodyID b1 = 0, b2 = 0;
125 
126  calculateCfmErp(jointS);
127  if(jointS->config.hasKey("jointCFM")) {
128  jointCFM = jointS->config["jointCFM"];
129  }
130 
131  joint_type = jointS->type;
132  if(n1) b1 = n1->getBody();
133  if(n2) b2 = n2->getBody();
134  body1 = b1;
135  body2 = b2;
136  switch(jointS->type) {
137  case JOINT_TYPE_HINGE:
138  createHinge(jointS, b1, b2);
139  break;
140  case JOINT_TYPE_HINGE2:
141  if(b1 || b2) createHinge2(jointS, b1, b2);
142  else return 0;
143  break;
144  case JOINT_TYPE_SLIDER:
145  createSlider(jointS, b1, b2);
146  break;
147  case JOINT_TYPE_BALL:
148  createBall(jointS, b1, b2);
149  break;
150  case JOINT_TYPE_UNIVERSAL:
151  if(b1 || b2) createUniversal(jointS, b1, b2);
152  else return 0;
153  break;
154  case JOINT_TYPE_FIXED:
155  if(b1 || b2) createFixed(jointS, b1, b2);
156  else return 0;
157  break;
158  default:
159  // Invalid type. No physically node will be created.
160  std::cerr << __FILE__":" << __LINE__<< " invalid joint type for joint \""
161  << jointS->name << "\". No joint created." << std::endl;
162  return 0;
163  break;
164  }
165  // we have created a joint. To get the information
166  // of the forces the joint attached to the bodies
167  // we need to set a feedback pointer for the joint (ode stuff)
168  dJointSetFeedback(jointId, &feedback);
169  return 1;
170  }
171  return 0;
172  }

913  void JointPhysics::update(void) {
914  const dReal *b1_pos, *b2_pos;
915  dReal anchor[4], axis[4], axis2[4];
916  int calc1 = 0, calc2 = 0;
917  dReal radius, dot, torque;
918  dReal v1[3], normal[3], load[3], tmp1[3], axis_force[3];
919  MutexLocker locker(&(theWorld->iMutex));
920 
921  switch(joint_type) {
922  case JOINT_TYPE_HINGE:
923  dJointGetHingeAnchor(jointId, anchor);
924  dJointGetHingeAxis(jointId, axis);
925  calc1 = 1;
926  break;
927  case JOINT_TYPE_HINGE2:
928  dJointGetHinge2Anchor(jointId, anchor);
929  dJointGetHinge2Axis1(jointId, axis);
930  dJointGetHinge2Axis2(jointId, axis2);
931  calc1 = 1;
932  calc2 = 0;
933  break;
934  case JOINT_TYPE_SLIDER:
935  dJointGetSliderAxis(jointId, axis);
936  calc1 = 2;
937  break;
938  case JOINT_TYPE_BALL:
939  // no axis
940  break;
941  case JOINT_TYPE_UNIVERSAL:
942  dJointGetUniversalAnchor(jointId, anchor);
943  dJointGetUniversalAxis1(jointId, axis);
944  dJointGetUniversalAxis2(jointId, axis2);
945  calc1 = 1;
946  calc2 = 0;
947  break;
948  default:
949  // no correct type is spezified, so no physically node will be created
950  break;
951  }
952  motor_torque = feedback.lambda;
953  axis1_torque.x() = axis1_torque.y() = axis1_torque.z() = 0;
954  axis2_torque.x() = axis2_torque.y() = axis2_torque.z() = 0;
955  joint_load.x() = joint_load.y() = joint_load.z() = 0;
956  if(calc1 == 1) {
957  if(body1) {
958  b1_pos = dBodyGetPosition(body1);
959  dOP(v1, -, b1_pos, anchor);
960  //radius = dLENGTH(v1);
961  dCROSS(normal, =, axis, v1);
962  dot = dDOT(normal, feedback.f1);
963  dOPEC(normal, *=, dot);
964  dOP(load, -, feedback.f1, normal);
965  dCROSS(tmp1, =, v1, normal);
966  axis1_torque.x() = (sReal)tmp1[0];
967  axis1_torque.y() = (sReal)tmp1[1];
968  axis1_torque.z() = (sReal)tmp1[2];
969  dCROSS(tmp1, =, v1, load);
970  joint_load.x() = (sReal)tmp1[0];
971  joint_load.y() = (sReal)tmp1[1];
972  joint_load.z() = (sReal)tmp1[2];
973  // now nearly the same for the torque
974  dot = dDOT(axis, feedback.t1);
975  dOPC(tmp1, *, axis, dot);
976  dOP(load, -, feedback.t1, tmp1);
977  axis1_torque.x() += (sReal)tmp1[0];
978  axis1_torque.y() += (sReal)tmp1[1];
979  axis1_torque.z() += (sReal)tmp1[2];
980  joint_load.x() += (sReal)load[0];
981  joint_load.y() += (sReal)load[1];
982  joint_load.z() += (sReal)load[2];
983  }
984  else if(body2) {
985  // now we do it correct
986  // first get the position vector v1
987  b2_pos = dBodyGetPosition(body2);
988  dOP(v1, -, b2_pos, anchor);
989 
990  // then differentiate the torque from the force

992  dOPC(axis_force, *, v1, dot);
993 
994  // the difference is the torque
995  dOP(tmp1, -, feedback.f2, axis_force);
996 
997  // the torque value is given by:
998  torque = dLENGTH(tmp1) / dLENGTH(v1);
999  // then get the normal to v1 and the torque vector
1000  dCROSS(normal, =, v1, tmp1);
1001  // and scale the normal to represent the correct torque length
1002  dOPEC(normal, *=, torque / dLENGTH(normal));
1003  // now the normal represents the torque vector at the anchor
1004  // and we make a projection to the axis
1005  dot = dDOT(normal, axis);
1006  dOPC(tmp1, *, axis, dot);
1007  dOP(load, -, normal, tmp1);
1008  //dCROSS(tmp1, =, v1, normal);
1009  axis1_torque.x() = (sReal)tmp1[0];
1010  axis1_torque.y() = (sReal)tmp1[1];
1011  axis1_torque.z() = (sReal)tmp1[2];
1012  //dCROSS(tmp1, =, v1, load);

1014  joint_load.y() = (sReal)load[1];
1015  joint_load.z() = (sReal)load[2];
1016  // now nearly the same for the torque
1017  dot = dDOT(feedback.t2, axis);
1018  dOPC(tmp1, *, axis, dot);
1019  dOP(load, -, feedback.t2, tmp1);
1020  //axis1_torque.x() += (sReal)tmp1[0];
1021  //axis1_torque.y() += (sReal)tmp1[1];
1022  //axis1_torque.z() += (sReal)tmp1[2];
1023  joint_load.x() += (sReal)load[0];
1024  joint_load.y() += (sReal)load[1];
1025  joint_load.z() += (sReal)load[2];
1026  }
1027  }
1028  else if(calc1 == 2) {
1029  // this is for the slider
1030  }
1031  if(calc2 == 1) {
1032  if(body1) {
1033  b1_pos = dBodyGetPosition(body1);
1034  dOP(v1, -, b1_pos, anchor);
1035  radius = dLENGTH(v1);
1036  dCROSS(normal, =, axis2, v1);
1037  dot = dDOT(normal, feedback.f1);
1038  dOPEC(normal, *=, dot*radius);
1039  axis2_torque.x() = (sReal)normal[0];
1040  axis2_torque.y() = (sReal)normal[1];
1041  axis2_torque.z() = (sReal)normal[2];

1045  axis2_torque.y() += (sReal)(axis2[1]*dot);
1046  axis2_torque.z() += (sReal)(axis2[2]*dot);
1047  }
1048  if(body2) {
1049  b2_pos = dBodyGetPosition(body2);
1050  dOP(v1, -, b2_pos, anchor);
1051  radius = dLENGTH(v1);
1052  dCROSS(normal, =, axis2, v1);
1053  dot = dDOT(normal, feedback.f2);
1054  dOPEC(normal, *=, dot*radius);
1055  axis2_torque.x() += (sReal)normal[0];
1056  axis2_torque.y() += (sReal)normal[1];
1057  axis2_torque.z() += (sReal)normal[2];
1058  // now nearly the same for the torque
