71  osg::ref_ptr<osg::GraphicsContext> QtOsgMixGraphicsWidget::createWidgetContext(
72  void* parent,
73  osg::ref_ptr<osg::GraphicsContext::Traits> traits) {
74 #if (QT_VERSION < QT_VERSION_CHECK(5, 2, 0))
75  retinaScale = 1.0;
76 #else
77  retinaScale = devicePixelRatio();
78 #endif
79  fprintf(stderr, "retina scale: %g\n", retinaScale);
80  traits->windowDecoration = true;
81 
82  osg::DisplaySettings* ds = osg::DisplaySettings::instance();
83  if (ds->getStereo()) {
84  switch(ds->getStereoMode()) {
85  case(osg::DisplaySettings::QUAD_BUFFER):
86  traits->quadBufferStereo = true;
87  break;
88  case(osg::DisplaySettings::VERTICAL_INTERLACE):
89  case(osg::DisplaySettings::CHECKERBOARD):
90  case(osg::DisplaySettings::HORIZONTAL_INTERLACE):
91  traits->stencil = 8;
92  break;
93  default: break;
94  }
95  }
96 
97  osgQt::GraphicsWindowQt *gc = new osgQt::GraphicsWindowQt(traits.get(), (QWidget*)parent);
98  childWidget = (QWidget*)gc->getGLWidget();
99  eventChild = gc->getGLWidget();
100  eventChild->installEventFilter(this);
101 
102  if (parent) {
103  traits->x = ((QWidget*)parent)->x()*retinaScale;
104  traits->y = ((QWidget*)parent)->y()*retinaScale;
105  traits->width = ((QWidget*)parent)->width()*retinaScale;
106  traits->height = ((QWidget*)parent)->height()*retinaScale;
107  }
108  else {
109  traits->x = childWidget->x()*retinaScale;
110  traits->y = childWidget->y()*retinaScale;
111  traits->width = childWidget->width()*retinaScale;
112  traits->height = childWidget->height()*retinaScale;
113  }
114  widgetWidth = traits->width;
115  widgetHeight = traits->height;
116  return gc;
117  }
