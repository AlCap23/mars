73  osg::ref_ptr<osg::GraphicsContext> QtOsgMixGraphicsWidget::createWidgetContext(
74  void* parent,
75  osg::ref_ptr<osg::GraphicsContext::Traits> traits) {
76  traits->windowDecoration = false;
77 
78 #if defined(__APPLE__)
79  wdata = new WindowData(WindowData::CreateOnlyView);
80  traits->inheritedWindowData = wdata;
81  haveNSView = false;
82 #elif defined(WIN32) && !defined(__CYGWIN__)
83  traits->inheritedWindowData = new WindowData((HWND)winId());
84 #else // all others
85  traits->inheritedWindowData = new WindowData(winId());
86 #endif // __APPLE__
87 
88  osg::DisplaySettings* ds = osg::DisplaySettings::instance();
89  if (ds->getStereo()) {
90  switch(ds->getStereoMode()) {
91  case(osg::DisplaySettings::QUAD_BUFFER):
92  traits->quadBufferStereo = true;
93  break;
94  case(osg::DisplaySettings::VERTICAL_INTERLACE):
95  case(osg::DisplaySettings::CHECKERBOARD):
96  case(osg::DisplaySettings::HORIZONTAL_INTERLACE):
97  traits->stencil = 8;
98  break;
99  default: break;
100  }
101  }
102 
103  osg::ref_ptr<osg::GraphicsContext> gc =
104  osg::GraphicsContext::createGraphicsContext(traits.get());
105 
106  if (parent) {
107  traits->x = ((QWidget*)parent)->x();
108  traits->y = ((QWidget*)parent)->y();
109  traits->width = ((QWidget*)parent)->width();
110  traits->height = ((QWidget*)parent)->height();
111  }
112  else {
113  traits->x = x();
114  traits->y = y();
115  traits->width = width();
116  traits->height = height();
117  }
118 
119  return gc;
120  }

202  static int qtToOsgKey(QKeyEvent* e) {
203  switch(e->key()) {
204  case Qt::Key_Shift:
205  return osgGA::GUIEventAdapter::KEY_Shift_L;
206  case Qt::Key_Alt:
207  return osgGA::GUIEventAdapter::KEY_Alt_L;
208  case Qt::Key_Meta:
209  return osgGA::GUIEventAdapter::KEY_Meta_L;
210  case Qt::Key_Control:
211  return osgGA::GUIEventAdapter::KEY_Control_L;
212  case Qt::Key_Up:
213  return osgGA::GUIEventAdapter::KEY_Up;
214  case Qt::Key_Down:
215  return osgGA::GUIEventAdapter::KEY_Down;
216  case Qt::Key_Left:
217  return osgGA::GUIEventAdapter::KEY_Left;
218  case Qt::Key_Right:
219  return osgGA::GUIEventAdapter::KEY_Right;
220  case Qt::Key_Delete:
221  return osgGA::GUIEventAdapter::KEY_Delete;
222  default:
223  return (osgGA::GUIEventAdapter::KeySymbol)*e->text().toLatin1().data();
224  }
225  }

324  bool QtOsgMixGraphicsWidget::eventFilter(QObject *obj, QEvent *event) {
325  if (obj != parent()) {
326  return false;
327  }
328  else if (event->type() == QEvent::KeyPress) {
329  QKeyEvent *ke = static_cast<QKeyEvent *>(event);
330 
331  if (ke->key() == Qt::Key_Tab || ke->key() == Qt::Key_Backtab) {
332  /*
333  qDebug() << "QOSGWidget::eventFilter: TAB Press on "
334  << qPrintable(objectName());
335  */
336  // Empirically have found that it's not necessary to call
337  // keyPressEvent on tab press ... my guess is that OSG ignores it.
338  keyPressEvent( ke );
339 
340  // Return false so that the parent QWidget will process the tab.
341  return false;
342  }
343  else {
344  /*
345  qDebug() << "QOSGWidget::eventFilter: KeyPress on "
346  << qPrintable(objectName());
347  */
348  keyPressEvent( ke );
349  // event handled, return true because parent does not have to see
350  // this event
351  return false;
352  }
353  }
354  else if (event->type() == QEvent::KeyRelease) {
355  QKeyEvent *ke = static_cast<QKeyEvent *>(event);
356 
357  if (ke->key() == Qt::Key_Tab || ke->key() == Qt::Key_Backtab) {
358  /*
359  qDebug() << "QOSGWidget::eventFilter: TAB Release on "
360  << qPrintable(objectName());
361  */
362  keyReleaseEvent( ke );
363  // Return false so that the parent QWidget will process the tab..
364  return false;
365  }
366  else {
367  /*
368  qDebug() << "QOSGWidget::eventFilter: KeyRelease on "
369  << qPrintable(objectName());
370  */
371  keyReleaseEvent( ke );
372  // event handled, return true because parent does not have to see
373  // this event
374  return false;
375  }
376  }
377  else if (event->type() == QEvent::Move) {
378  QMoveEvent *re = static_cast<QMoveEvent *>(event);
379  moveEvent(re);
380  return false;
381  } else if (event->type() == QEvent::Resize) {
382  QResizeEvent *re = static_cast<QResizeEvent *>(event);
383  setGeometry(0, 0, re->size().width(), re->size().height());
384  // if(re) resizeEvent(re);
385 
386  // event handled, return true because parent does not have to see
387  // this event ???
388  return false;
389  }
390  else if (event->type() == QEvent::Close) {
391  QCloseEvent *ce = static_cast<QCloseEvent *>(event);
392  closeEvent( ce );
393  }
394 
395  return false;
396  }
