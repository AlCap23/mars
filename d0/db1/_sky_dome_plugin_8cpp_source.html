49  class SkyTransform : public osg::Transform {
50  public:
51  virtual bool computeLocalToWorldMatrix(osg::Matrix& matrix,
52  osg::NodeVisitor* nv) const {
53  osgUtil::CullVisitor* cv = dynamic_cast<osgUtil::CullVisitor*>(nv);
54  if (cv) {
55  osg::Vec3 eyePointLocal = cv->getEyeLocal();
56  matrix.preMultTranslate(eyePointLocal);
57  }
58  return true;
59  }
60 
61  virtual bool computeWorldToLocalMatrix(osg::Matrix& matrix,
62  osg::NodeVisitor* nv) const {
63  osgUtil::CullVisitor* cv = dynamic_cast<osgUtil::CullVisitor*>(nv);
64  if (cv) {
65  osg::Vec3 eyePointLocal = cv->getEyeLocal();
66  matrix.postMultTranslate(-eyePointLocal);
67  }
68  return true;
69  }
70  };

77  void SkyDomePlugin::init() {
78  if(!control->graphics) return;
79  posTransform = new SkyTransform();
80  posTransform->setCullingActive(false);
81  updateProp = true;
82  scene = static_cast<osg::Group*>(control->graphics->getScene());
83 
84  resPath = ".";
85  if(control->cfg) {
86  cfgProp = control->cfg->getOrCreateProperty("Graphics",
87  "resources_path",
88  ".");
89  resPath = cfgProp.sValue;
90  cfgProp = control->cfg->getOrCreateProperty("Scene",
91  "skydome_path",
92  "cubemap",
93  this);
94  folder = cfgProp.sValue;
95  cfgEnableSD = control->cfg->getOrCreateProperty("Scene",
96  "skydome_enabled",
97  false, this);
98  }
99  else {
100  cfgEnableSD.bValue = false;
101  }
102 
103  if(cfgEnableSD.bValue) {
104  scene->addChild(posTransform);
105  }
106 
107  if(pathExists(resPath+"/Textures/"+folder)) {
108  folder = resPath+"/Textures/"+folder;
109  }
110 
111  // todo: handle wrong path
112  osg::ref_ptr<osg::TextureCubeMap> _cubemap = loadCubeMapTextures();
113 
114  _skyDome = new SkyDome( 1.9f, 24, 24, _cubemap.get() );
115  _skyDome->setCullingActive(false);
116  posTransform->addChild(_skyDome.get());
117 
118  osg::StateSet *states = _skyDome->getOrCreateStateSet();
119  states->setMode(GL_LIGHTING,
120  osg::StateAttribute::OFF | osg::StateAttribute::PROTECTED);
121  //states->setMode(GL_BLEND,osg::StateAttribute::OFF);
122  states->setMode(GL_FOG, osg::StateAttribute::OFF);
123  states->setRenderBinDetails( -1, "RenderBin");
124  states->setMode(GL_DEPTH_TEST, osg::StateAttribute::OFF);
125  //states->setRenderingHint(osg::StateSet::TRANSPARENT_BIN);
126  //states->setMode(GL_BLEND,osg::StateAttribute::ON);
127 
128  control->sim->switchPluginUpdateMode(0, this);
129  gui->addGenericMenuAction("../View/", 0, NULL, 0, "", 0, -1); // separator
130  gui->addGenericMenuAction("../View/Sky dome", 1, this, 0, "", 0,
131  1 + cfgEnableSD.bValue);
132  }

141  osg::ref_ptr<osg::TextureCubeMap> SkyDomePlugin::loadCubeMapTextures() {
142  enum {POS_X, NEG_X, POS_Y, NEG_Y, POS_Z, NEG_Z};
143  std::string filenames[6];
144  filenames[POS_X] = folder + "/east.png";
145  filenames[NEG_X] = folder + "/west.png";
146  filenames[POS_Z] = folder + "/north.png";
147  filenames[NEG_Z] = folder + "/south.png";
148  filenames[POS_Y] = folder + "/down.png";
149  filenames[NEG_Y] = folder + "/up.png";
150 
151  osg::ref_ptr<osg::TextureCubeMap> cubeMap = new osg::TextureCubeMap;
152  cubeMap->setInternalFormat(GL_RGBA);
153 
154  cubeMap->setFilter( osg::Texture::MIN_FILTER,
155  osg::Texture::LINEAR_MIPMAP_LINEAR);
156  cubeMap->setFilter( osg::Texture::MAG_FILTER,
157  osg::Texture::LINEAR);
158  cubeMap->setWrap ( osg::Texture::WRAP_S,
159  osg::Texture::CLAMP_TO_EDGE);
160  cubeMap->setWrap ( osg::Texture::WRAP_T,
161  osg::Texture::CLAMP_TO_EDGE);
162 
163  cubeMap->setImage(osg::TextureCubeMap::NEGATIVE_X,
164  osgDB::readImageFile( filenames[NEG_X] ) );
165  cubeMap->setImage(osg::TextureCubeMap::POSITIVE_X,
166  osgDB::readImageFile( filenames[POS_X] ) );
167  cubeMap->setImage(osg::TextureCubeMap::NEGATIVE_Y,
168  osgDB::readImageFile( filenames[NEG_Y] ) );
169  cubeMap->setImage(osg::TextureCubeMap::POSITIVE_Y,
170  osgDB::readImageFile( filenames[POS_Y] ) );
171  cubeMap->setImage(osg::TextureCubeMap::NEGATIVE_Z,
172  osgDB::readImageFile( filenames[NEG_Z] ) );
173  cubeMap->setImage(osg::TextureCubeMap::POSITIVE_Z,
174  osgDB::readImageFile( filenames[POS_Z] ) );
175 
176  return cubeMap;
177  }

190  void SkyDomePlugin::cfgUpdateProperty(cfg_manager::cfgPropertyStruct _property) {
191  if(cfgEnableSD.paramId == _property.paramId) {
192  if(updateProp) {
193  cfgEnableSD.bValue = _property.bValue;
194  updateProp = false;
195  gui->setMenuActionSelected("../View/Sky dome", cfgEnableSD.bValue);
196  scene->removeChild(posTransform.get());
197  if(cfgEnableSD.bValue) {
198  scene->addChild(posTransform.get());
199  }
200  updateProp = true;
201  }
202  }
203  else if(cfgProp.paramId == _property.paramId) {
204  folder = cfgProp.sValue = _property.sValue;
205  if(pathExists(resPath + "/Textures/"+folder)) {
206  folder = resPath + "/Textures/"+folder;
207  }
208  if(pathExists(folder)) {
209  _skyDome->setCubeMap(loadCubeMapTextures().get());
210  }
211  }
