180  void DialogNodes::on_remove_node() {
181  filled = false;
182 
183  for (unsigned int i = 0; i < newNodes_p.size(); i++)
184  if (pDialog->currentItem() == newNodes_p[i]) {
185  delete newDialogs[i];
186  newDialogs.erase(newDialogs.begin() + i);
187  newNodes_p.erase(newNodes_p.begin() + i);
188  break;
189  }
190 
191  for (unsigned int i = 0; i < allNodes_p.size(); i++)
192  if (allNodes_p[i] == pDialog->currentItem()) {
193  control->nodes->removeNode(allNodes[i].index);
194  allNodes.erase(allNodes.begin() + i);
195  delete allDialogs[i];
196  allDialogs.erase(allDialogs.begin() + i);
197  allNodes_p.erase(allNodes_p.begin() + i);
198  break;
199  }
200 
201  filled = true;
202 
203  if (allNodes_p.size() == 0 && newNodes_p.size() == 0) {
204  removeButton->hide();
205  addButton->hide();
206  stateButton->hide();
207  } else if (allNodes_p.size() > 0) {
208  pDialog->setCurrentItem(allNodes_p[0]);
209  } else if (newNodes_p.size() > 0) {
210  pDialog->setCurrentItem(newNodes_p[0]);

215  void DialogNodes::on_add_node() {
216  filled = false;
217  for (unsigned int i = 0; i < newNodes_p.size(); i++)
218  if (pDialog->currentItem() == newNodes_p[i]) {
219  if ((newDialogs[i]->accept()) != 0) {
220  on_remove_node();
221  } else {
222  allNodes_p.push_back(newNodes_p[i]);
223  allDialogs.push_back(newDialogs[i]);
224  oldFocus = NULL;
225  newNodes_p.erase(newNodes_p.begin() + i);
226  newDialogs.erase(newDialogs.begin() + i);
227  topLevelItemChanged(allNodes_p.back());
228  }
229  break;
230  }
231  control->nodes->getListNodes(&allNodes);
232 
233  filled = true;
234  addButton->hide();
235  stateButton->show();
236  }
