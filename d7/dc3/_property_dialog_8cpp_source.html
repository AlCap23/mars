106  int type,
107  const QVariant& value,
108  std::map<QString, QVariant> *attributes,
109  QStringList *options) {

145  QtVariantProperty* PropertyDialog::addGenericProperty(const string &path,
146  int type,
147  const QVariant &value,
148  map<QString, QVariant> *attributes,
149  QStringList *options) {
150  QList<string> propertyPath;
151  string propertyName;
152  bool propertyExists = false;
153  bool parentExists = false;
154  QtVariantProperty *parent = NULL;
155  QtVariantProperty *item = NULL;
156  QList<QtProperty*>::iterator it;
157  QList<QtProperty*> current = myProperties;
158  // tokenize the path
159  unsigned int j = 0;
160  for(unsigned int i = 0; i < path.length()+1; i++) {
161  // ignore // for path
162  if(path[i] == '/' && path[i+1] != '/' && path[i-1]!='/') {
163  propertyPath.push_back(path.substr(j, i-j));
164  j = i + 1;
165  }
166  if (i == path.length() && propertyPath.size() > 0) {
167  propertyPath.push_back(path.substr(j, i-j));
168  }
169  }
170 
171  if(propertyPath.size() <= 0) { // empty or invalid path
172  return NULL;
173  }
174  // if the path does not begin with "../" then insert a tab
175  if(propertyPath[0] != "..") {
176  return addTabbedProperty(path, type, value, attributes, options);
177  }
178  // remove double //
179  for(int i = 0; i < propertyPath.size(); i++) {
180  for(size_t k=0; k<propertyPath[i].size(); k++) {
181  if(propertyPath[i][k] == '/') {
182  propertyPath[i].erase(propertyPath[i].begin()+k+1);
183  }
184  }
185  }
186  propertyPath.pop_front(); // get rid of the "../"
187  propertyName = propertyPath.back(); // save the name of the actual property
188  propertyPath.pop_back(); // leave only the path
189  for(int i = 0; i < propertyPath.size(); i++) {
190  parentExists = false;
191  // check if this part of the path exists
192  for(it = current.begin(); it != current.end(); it++) {
193  if((*it)->propertyName().toStdString() == propertyPath[i]) {
194  item = (QtVariantProperty*)(*it);
195  parentExists = true;
196  break;
197  }
198  }
199  if(parentExists) { // start iterating over its subproperties
200  current = item->subProperties();
201  } else { // create it as a group property
202  item = variantManager->addProperty(QtVariantPropertyManager::groupTypeId(),
203  QString::fromStdString(propertyPath[i]));
204  if(i == 0) { // add it to the views and myProperties
205  variantEditorButton->addProperty(item);
206  variantEditorTree->addProperty(item);
207  collapseTree(variantEditorTree->items(item));
208  myProperties.push_back(item);
209  } else { // add it to the parent's subproperties
210  parent->addSubProperty(item);
211  }
212  }
213  parent = item;
214  }
215 
216  if(parent != NULL) { // now parent points to the direct parent of the new property
217  current = parent->subProperties();
218  } else { // no parent means that it is created directly in the browsers
219  current = myProperties;
220  }
221 
222  propertyExists = false;
223  for(it = current.begin(); it != current.end(); it++) {
224  if (propertyName == (*it)->propertyName().toStdString()) { // it already exists;
225  propertyExists = true;
226  item = (QtVariantProperty*)(*it);
227  break;
228  }
229  }
230  if(!propertyExists) { // add the new property to the manager
231  item = variantManager->addProperty(type, QString::fromStdString(propertyName));
232  }
233 
234  if(parent != NULL) { // add the property to its parent's subproperties
235  parent->addSubProperty(item);
236  collapseTree(variantEditorTree->items(parent));
237  } else { // add the property directly to the browsers
238  myProperties.push_back(item);
239  variantEditorTree->addProperty(item);
240  variantEditorButton->addProperty(item);
241  collapseTree(variantEditorTree->items(item));
242  }
243 
244  customizeItem(item, type, value, attributes, options);
245 
246  return item;
247  }

300  void PropertyDialog::removeGenericProperty(QtVariantProperty *property) {
301  QtProperty *parent = getParent(NULL, property);
302  if(parent) { // a parent was found, remove the property from its subproperties
303  parent->removeSubProperty(property);
304  // remove all upward properties if they become empty after removing <property>
305  if(parent->subProperties().size() == 0) {
306  this->removeGenericProperty((QtVariantProperty*)parent);
307  }
308  } else { // no parent, remove the property directly from the browsers and <myProperties>
309  variantEditorButton->removeProperty(property);
310  variantEditorTree->removeProperty(property);
311  QList<QtProperty*>::iterator it;
312  for(it = myProperties.begin(); it != myProperties.end(); ) {
313  if(*it == property) {
314  // delete *it;
315  // the property should not be deleted, only removed (hidden)
316  it = myProperties.erase(it);
317  } else {
318  ++it;
319  }
320  }
321  }
322  // propagate the remove to the underlying TabPropertyDialogs, if there are any
323  map<QString, PropertyDialog*>::iterator it;
324  for(it = myTabs.begin(); it != myTabs.end(); ) {
325  it->second->removeGenericProperty(property);
326  if(it->second->myProperties.size() == 0) {
327  tabWidget->removeTab(tabWidget->indexOf(it->second));
328  //delete it->second;
329  myTabs.erase(it++);
330  } else {
331  ++it;
332  }
333  }
334  }

369  QtProperty* PropertyDialog::getParent(QtProperty *parent,
370  QtVariantProperty *property) const {
371  QList<QtProperty*> current;
372  if(parent) { // a parent is passed, iterate over its subproperties
373  current = parent->subProperties();
374  } else { // no parent means initial iteration
375  current = myProperties;
376  }
377 
378  for(int i = 0; i < current.size(); i++) {
379  if(current[i] == property) { // the property is found
380  return parent;
381  }
382  QtProperty *tmp = getParent(current[i], property);
383  if(tmp) { // the property is found
384  return tmp;
385  }
386  }
387  return NULL; // no such property in this branch
