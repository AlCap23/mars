393  void GraphicsManager::update(){
394  //update drawElements
395  for (unsigned int i=0; i<draws.size(); i++) {
396  drawMapper &draw = draws[i];
397  vector<draw_item> tmp_ditem;
398  vector<osg::Node*> tmp_nodes;
399  //update draws
400  draw.ds.ptr_draw->update(&(draw.ds.drawItems));
401 
402  for (unsigned int j=0; j<draw.ds.drawItems.size(); j++) {
403  draw_item &di = draw.ds.drawItems[j];
404 
405  if(di.draw_state == DRAW_STATE_ERASE) {
406  scene->removeChild(draw.nodes[j]);
407  }
408  else if (di.draw_state == DRAW_STATE_CREATE) {
409  std::string font_path = resources_path.sValue;
410  font_path.append("/Fonts");
411  osg::ref_ptr<osg::Group> osgNode = new OSGDrawItem(osgWidget, di,
412  font_path);
413  scene->addChild(osgNode.get());
414 
415  di.draw_state = DRAW_UNKNOWN;
416  tmp_ditem.push_back(di);
417  tmp_nodes.push_back(osgNode.get());
418  }
419  else if (di.draw_state == DRAW_STATE_UPDATE) {
420  assert(draws[i].nodes.size() > j);
421  osg::Node *n = draws[i].nodes[j];
422  OSGDrawItem *diWrapper = dynamic_cast<OSGDrawItem*>(n->asGroup()); // TODO: asGroup unneeded?
423  assert(diWrapper != NULL); // TODO: handle this case better
424 
425  diWrapper->update(di);
426 
427  di.draw_state = DRAW_UNKNOWN;
428  tmp_ditem.push_back(di);
429  tmp_nodes.push_back(n);
430  }
431  else { // invalid draw state!
432  di.draw_state = DRAW_UNKNOWN;
433  tmp_ditem.push_back(di);
434  tmp_nodes.push_back(draws[i].nodes[j]);
435  }
436  }
437 
438  draws[i].ds.drawItems.clear();
439  draws[i].ds.drawItems = tmp_ditem;
440  draws[i].nodes.clear();
441  draws[i].nodes = tmp_nodes;
442  }
443  }

741  unsigned long GraphicsManager::addDrawObject(const mars::interfaces::NodeData &snode,
742  bool activated) {
743  unsigned long id = next_draw_object_id++;

745  int mask = 0;
746 
747  getLights(&lightList);
748  if(lightList.size() == 0) lightList.push_back(&defaultLight.lStruct);
749  osg::ref_ptr<OSGNodeStruct> drawObject = new OSGNodeStruct(lightList, snode, false, id, marsShader.bValue, useFog, useNoise, drawLineLaser, marsShadow.bValue);
750  osg::PositionAttitudeTransform *transform = drawObject->object()->getPosTransform();
751 
752  DrawCoreIds.insert(pair<unsigned long int, unsigned long int>(id, snode.index));
753  drawObjects_[id] = drawObject;
754 
755  if(snode.isShadowCaster) {
756  mask |= CastsShadowTraversalMask;
757  }
758  if(snode.isShadowReceiver) {
759  mask |= ReceivesShadowTraversalMask;
760  }
761  transform->setNodeMask(transform->getNodeMask() | mask);
762 
763  // import an .STL file : we have to insert an additional transformation
764  // in order to add the additional rotation by 90 degrees around the
765  // x-axis (adding the rotation to "transform" does not help at all,
766  // because the values of "transform" are constantly resetted by MARS
767  // itself)
768  if((snode.filename.substr(snode.filename.size()-4, 4) == ".STL") ||
769  (snode.filename.substr(snode.filename.size()-4, 4) == ".stl")) {

771  osg::ref_ptr<osg::PositionAttitudeTransform> transformSTL =
772  new osg::PositionAttitudeTransform();
773 
774  // remove all child nodes from "transform" and add them to
775  // "transformSTL"
776  osg::Node* node = NULL;
777  while (transform->getNumChildren() > 0) {
778  node = transform->getChild(0);
779  transformSTL->addChild(node);
780  transform->removeChild(node);
781  }
782 
783  // add "transformSTL" as child to "transform"
784  transform->addChild(transformSTL);
785 
786  // calulate the quaternion for the rotation of 90 degrees around the
787  // x-axis
788  mars::utils::Quaternion offset =
789  mars::utils::eulerToQuaternion(mars::utils::Vector(90.0, 0.0, 0.0));
790 
791  // set the orientation to the newly added transformation
792  transformSTL->setAttitude(osg::Quat(offset.x(),
793  offset.y(),
794  offset.z(),
795  offset.w()));
796  }
797 
798  if(activated) {
799  if(mask != 0) {
800  shadowedScene->addChild(transform);
801  }
802  else {
803  shadowedScene->addChild(transform);
804  }
805  if(shadowMap.valid() && snode.map.find("shadowCenterRadius") != snode.map.end()) {
806  shadowMap->setCenterObject(drawObject->object());
807  configmaps::ConfigMap m = snode.map;
808  shadowMap->setRadius(m["shadowCenterRadius"][0]);

811  //osgUtil::Optimizer optimizer;
812  //optimizer.optimize(shadowedScene.get());
813
