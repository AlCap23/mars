455  void GraphicsManager::update(){
456  //update drawElements
457  for (unsigned int i=0; i<draws.size(); i++) {
458  drawMapper &draw = draws[i];
459  vector<draw_item> tmp_ditem;
460  vector<osg::Node*> tmp_nodes;
461  //update draws
462  draw.ds.ptr_draw->update(&(draw.ds.drawItems));
463 
464  for (unsigned int j=0; j<draw.ds.drawItems.size(); j++) {
465  draw_item &di = draw.ds.drawItems[j];
466 
467  if(di.draw_state == DRAW_STATE_ERASE) {
468  scene->removeChild(draw.nodes[j]);
469  }
470  else if (di.draw_state == DRAW_STATE_CREATE) {
471  std::string font_path = resources_path.sValue;
472  font_path.append("/Fonts");
473  osg::ref_ptr<osg::Group> osgNode = new OSGDrawItem(osgWidget, di,
474  font_path);
475  scene->addChild(osgNode.get());
476 
477  di.draw_state = DRAW_UNKNOWN;
478  tmp_ditem.push_back(di);
479  tmp_nodes.push_back(osgNode.get());
480  }
481  else if (di.draw_state == DRAW_STATE_UPDATE) {
482  assert(draws[i].nodes.size() > j);
483  osg::Node *n = draws[i].nodes[j];
484  OSGDrawItem *diWrapper = dynamic_cast<OSGDrawItem*>(n->asGroup()); // TODO: asGroup unneeded?
485  assert(diWrapper != NULL); // TODO: handle this case better
486 
487  diWrapper->update(di);
488 
489  di.draw_state = DRAW_UNKNOWN;
490  tmp_ditem.push_back(di);
491  tmp_nodes.push_back(n);
492  }
493  else { // invalid draw state!
494  di.draw_state = DRAW_UNKNOWN;
495  tmp_ditem.push_back(di);
496  tmp_nodes.push_back(draws[i].nodes[j]);
497  }
498  }
499 
500  draws[i].ds.drawItems.clear();
501  draws[i].ds.drawItems = tmp_ditem;
502  draws[i].nodes.clear();
503  draws[i].nodes = tmp_nodes;
504  }
505  }

868  mask |= ReceivesShadowTraversalMask;

870  transform->setNodeMask(transform->getNodeMask() | mask);

873  // in order to add the additional rotation by 90 degrees around the
874  // x-axis (adding the rotation to "transform" does not help at all,

888  transformSTL->addChild(node);

902  offset.y(),
903  offset.z(),
904  offset.w()));
905  }
