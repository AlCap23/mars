321  unsigned long ControllerManager::addController(const ControllerData &controller) {
322  if (do_not_load_controller)
323  return 0;
324  std::vector<SimMotor*> vmotor;
325  std::vector<BaseSensor*> vsensor;
326  std::vector<NodeData*> nodes;
327  std::vector<unsigned long>::const_iterator nter;
328  Controller *newController;
329  unsigned long id = next_controller_id++;
330 
331  for (nter = controller.motors.begin(); nter != controller.motors.end();
332  nter++) {
333  SimMotor* retval = control->motors->getSimMotor(*nter);
334  if (retval)
335  vmotor.push_back(retval);
336  }
337 
338  for (nter = controller.sensors.begin(); nter != controller.sensors.end();
339  nter++) {
340  BaseSensor* retval = control->sensors->getSimSensor(*nter);
341  if(retval) {
342  vsensor.push_back(retval);
343  }
344  }
345 
346  if (!controller.sNodes.empty()) {
347  LOG_WARN("ControllerManager::addController: sNodes are not implemented yet and are currently ignored!");
348  }
349 
350  newController = new Controller(controller.rate, vmotor, vsensor, nodes,
351  control, std_port);
352  newController->setDylibPath(controller.dylib_path);
353  newController->setID(id);
354  iMutex.lock();
355  simController[id] = newController;
356  iMutex.unlock();
357  return id;
358  }
