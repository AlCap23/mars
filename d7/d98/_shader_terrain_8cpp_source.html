85  std::string file = map["file"];

87  posTransform->zOffset = map["pos/z"];

90  posTransform->zOffset = 0.;
91  }
92  posTransform->overridePos = false;
93  if(mars::utils::getFilenameSuffix(file) == ".bobj") {
94  node = readBobjFromFile(file);
95  }
96  else {
97  node = osgDB::readNodeFile(file);
98  }
99  node->setNodeMask(0xff | 0x1000);
100 
101  // handle bounding box
102  boundCallback = new ShaderTerrain::BoundCallback();
103  osg::ComputeBoundsVisitor cbbv;
104  node->accept(cbbv);
105  osg::BoundingBox bb = cbbv.getBoundingBox();
106  boundCallback->size[0] = fabs(bb.xMax() - bb.xMin());
107  boundCallback->size[1] = fabs(bb.yMax() - bb.yMin());
108  boundCallback->size[2] = map["size/z"];;
109  fprintf(stderr, "size: %g %g %g %g\n", boundCallback->size[0],
110  boundCallback->size[1], boundCallback->size[2], posTransform->zOffset);
111  ShaderTerrain::BoundVisitor visitor;
112  visitor.s = boundCallback.get();
113  node->accept(visitor);
114 
115  osg::PolygonMode *polyModeObj = new osg::PolygonMode;
116  polyModeObj->setMode(osg::PolygonMode::FRONT_AND_BACK, osg::PolygonMode::LINE);
117  bool wireframe = map["wireframe"];
118  if(wireframe) {
119  osg::ref_ptr<osg::LineWidth> linew = new osg::LineWidth(3.0);
120  node->getOrCreateStateSet()->setAttributeAndModes(linew.get(),
121  osg::StateAttribute::ON);
122  node->getOrCreateStateSet()->setAttribute( polyModeObj );
123  }
124  posTransform->addChild(node.get());
125  this->addChild(posTransform.get());
126  }

147  void ShaderTerrain::BoundVisitor::apply(osg::Node &searchNode){
148  // search for geometries and generate tangents for them
149  osg::Geode* geode=dynamic_cast<osg::Geode*>(&searchNode);
150  if(geode) {
151  for(unsigned int i=0; i<geode->getNumDrawables(); ++i) {
152  osg::Geometry* geom=dynamic_cast<osg::Geometry*>(geode->getDrawable(i));

154  geom->setComputeBoundingBoxCallback(s);
155  geom->setUseDisplayList(false);

157  geom->dirtyBound();
158  }
159  }
160  }
161 
162  traverse(searchNode);
163  }
