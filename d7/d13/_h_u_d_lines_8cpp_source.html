129  void HUDLines::setLines(std::vector<double> *_vertices, double color[4]) {
130  if(init) {
131  init = false;
132 
133  osg::ref_ptr<osg::Geode> node = new osg::Geode;
134  osg::ref_ptr<osg::Geometry> linesGeom = new osg::Geometry;
135 
136 
137  osg::Vec3Array* ver = new osg::Vec3Array((*_vertices).size()/3);
138  ver->setDataVariance(osg::Object::DYNAMIC);
139  for(size_t i=0; i<_vertices->size()/3; ++i)
140  (*ver)[i].set((*_vertices)[i*3],
141  (*_vertices)[(i*3)+1],
142  (*_vertices)[(i*3)+2]-1.0);
143 
144  linesGeom->setDataVariance(osg::Object::DYNAMIC);
145  linesGeom->setVertexArray(ver);
146 
147  osg::Vec4Array* colors = new osg::Vec4Array;
148  colors->push_back(osg::Vec4(color[0], color[1], color[2], color[3]));
149  linesGeom->setColorArray(colors);
150  linesGeom->setColorBinding(osg::Geometry::BIND_OVERALL);
151 
152  // set the normal in the same way color.
153  osg::Vec3Array* normals = new osg::Vec3Array;
154  normals->push_back(osg::Vec3(0.0f,0.0f,1.0f));
155  linesGeom->setNormalArray(normals);
156  linesGeom->setNormalBinding(osg::Geometry::BIND_OVERALL);
157 
158  linesGeom->addPrimitiveSet(new osg::DrawArrays(osg::PrimitiveSet::LINES,
159  0, (*_vertices).size()/3));
160 
161 
162  node->addDrawable(linesGeom.get());
163 
164  osg::ref_ptr<osg::LineWidth> linew;
165  linew = new osg::LineWidth(point_size);
166  node->getOrCreateStateSet()->setAttributeAndModes(linew.get(),
167  osg::StateAttribute::ON);
168 
169  node->getOrCreateStateSet()->setMode(GL_DEPTH_TEST, osg::StateAttribute::OFF);
170  node->getOrCreateStateSet()->setRenderBinDetails(render_order, "RenderBin");
171 
172  scaleTransform->addChild(node.get());
173 
174  osg::BoundingBox bb;
175  bb.expandBy(node->getDrawable(0)->getBound());
176  float d = bb.zMin()-2.0;
177  float w = bb.xMax() - bb.xMin();
178  float h = bb.yMax() - bb.yMin();
179 
180  // now get the bounding box
181  if(background_color[3] > 0.0001) {
182  osg::ref_ptr<osg::Geode> geode = new osg::Geode;
183  osg::Geometry* geom = new osg::Geometry;
184 
185  osg::Vec3Array* vertices = new osg::Vec3Array;
186  vertices->push_back(osg::Vec3(posx-pl, posy-h+pt, d));
187  vertices->push_back(osg::Vec3(posx-pl, posy+pt, d));
188  vertices->push_back(osg::Vec3(posx+w-pl, posy+pt, d));
189  vertices->push_back(osg::Vec3(posx+w-pl, posy-h+pt, d));
190  geom->setVertexArray(vertices);
191 
192  osg::Vec3Array* normals = new osg::Vec3Array;
193  normals->push_back(osg::Vec3(0.0f,0.0f,1.0f));
194  geom->setNormalArray(normals);
195  geom->setNormalBinding(osg::Geometry::BIND_OVERALL);
196 
197  osg::Vec4Array* colors = new osg::Vec4Array;
198  colors->push_back(osg::Vec4(background_color[0], background_color[1],
199  background_color[2], background_color[3]));
200  geom->setColorArray(colors);
201  geom->setColorBinding(osg::Geometry::BIND_OVERALL);
202 
203  geom->addPrimitiveSet(new osg::DrawArrays(GL_QUADS,0,4));
204 
205  //osg::StateSet* stateset = geom->getOrCreateStateSet();
206  //stateset->setMode(GL_LIGHTING,osg::StateAttribute::OFF);
207  //stateset->setMode(GL_BLEND,osg::StateAttribute::ON);
208  //stateset->setRenderingHint(osg::StateSet::TRANSPARENT_BIN);
209  //stateset->setMode(GL_DEPTH_TEST, osg::StateAttribute::OFF);
210  geode->addDrawable(geom);
211  scaleTransform->addChild(geode.get());
212  }
213 
214  if(border_width > 0.0) {
215  osg::ref_ptr<osg::Geometry> linesGeom = new osg::Geometry;
216  osg::ref_ptr<osg::Geode> linesGeode = new osg::Geode;
217  //nodemanager tempnode;
218 
219  osg::Vec3Array* vertices = new osg::Vec3Array(8);
220  (*vertices)[0].set(posx-pl, posy-h+pt, d);
221  (*vertices)[1].set(posx-pl, posy+pt, d);
222  (*vertices)[2].set(posx-pl, posy+pt, d);
223  (*vertices)[3].set(posx+w-pl, posy+pt, d);
224  (*vertices)[4].set(posx+w-pl, posy+pt, d);
225  (*vertices)[5].set(posx+w-pl, posy-h+pt, d);
226  (*vertices)[6].set(posx+w-pl, posy-h+pt, d);
227  (*vertices)[7].set(posx-pl, posy-h+pt, d);
228 
229  // pass the created vertex array to the points geometry object.
230  linesGeom->setVertexArray(vertices);
231 
232  // set the colors as before, plus using the above
233  osg::Vec4Array* colors2 = new osg::Vec4Array;
234  colors2->push_back(osg::Vec4(border_color[0], border_color[1],
235  border_color[2], border_color[3]));
236  linesGeom->setColorArray(colors2);
237  linesGeom->setColorBinding(osg::Geometry::BIND_OVERALL);
238 
239  // set the normal in the same way color.
240  osg::Vec3Array* normals = new osg::Vec3Array;
241  normals->push_back(osg::Vec3(0.0f,0.0f,1.0f));
242  linesGeom->setNormalArray(normals);
243  linesGeom->setNormalBinding(osg::Geometry::BIND_OVERALL);
244 
245  // since we know up front,
246  linesGeom->addPrimitiveSet(new osg::DrawArrays(osg::PrimitiveSet::LINES,
247  0,8));
248 
249  osg::ref_ptr<osg::LineWidth> linew = new osg::LineWidth(border_width);
250  linesGeom->getOrCreateStateSet()->setAttributeAndModes(linew.get(),
251  osg::StateAttribute::ON);
252 
253  // add the points geometry to the geode.
254  linesGeode->addDrawable(linesGeom.get());
255  scaleTransform->addChild(linesGeode.get());
256  }
257  }
258  else {
259  osg::Geode* geode;
260  osg::Drawable* drawable;
261  osg::Geometry* geometry;
262  osg::ref_ptr<osg::Vec3Array> v(new osg::Vec3Array());
263 
264  if((geode = dynamic_cast<osg::Geode*>(scaleTransform->getChild(0)))) {
265  if((drawable = geode->getDrawable(0))) {
266  if((geometry = drawable->asGeometry())) {
267  for(unsigned int i=0; i<(*_vertices).size()/3; ++i) {
268  v->push_back(osg::Vec3((*_vertices)[i*3],
269  (*_vertices)[(i*3)+1],
270  (*_vertices)[(i*3)+2]-1.0));
271  }
272  geometry->setVertexArray(v.get());
273  osg::Vec4Array* colors = new osg::Vec4Array;
274  colors->push_back(osg::Vec4(color[0], color[1], color[2], color[3]));
275  geometry->setColorArray(colors);
276  geometry->dirtyDisplayList();
277  geometry->dirtyBound();
278  (dynamic_cast<osg::DrawArrays*>(geometry->getPrimitiveSet(0)))->setCount((*_vertices).size()/3);
279 
280  }
281  }
282  }
283 
284  }
