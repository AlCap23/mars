56  bool matchPattern(const std::string &pattern, const std::string &str) {
57  size_t pos1 = 0, pos2 = 0;
58 
59  // chop up pattern
60  std::vector<std::string> patternList;
61  while(1) {
62  pos2 = pattern.find("*", pos1);
63  if(pos2 == pattern.npos) {
64  if(pos1 != pattern.length())
65  patternList.push_back(pattern.substr(pos1));
66  break;
67  }
68  patternList.push_back(pattern.substr(pos1, pos2 - pos1));
69  pos1 = pos2 + 1;
70  }
71 
72  // no wildcards. do direct test
73  if(patternList.empty())
74  return pattern == str;
75 
76  // special case the first pattern because it must match at pos == 0
77  std::vector<std::string>::iterator patternListIt = patternList.begin();
78  int result = str.compare(0, patternListIt->length(), *patternListIt);
79  if(result != 0)
80  return false;
81  pos1 = patternListIt->length();
82  ++patternListIt;
83  // do the matching
84  for(/*nothing*/; patternListIt != patternList.end(); ++patternListIt) {
85  pos1 = str.find(*patternListIt, pos1);
86  if(pos1 == str.npos)
87  return false;
88  pos1 += patternListIt->length();
89  }
90  return true;
91  }

93  std::string trim(const std::string& str) {
94 
95  int front_idx, back_idx, len;
96 
97  front_idx = 0;
98  back_idx = ( len = str.size() ) - 1;
99 
100  while (isspace(str[front_idx]) && front_idx < len ) front_idx++;
101  while (isspace(str[back_idx]) && back_idx > 0 ) back_idx--;
102 
103  if ( front_idx >= back_idx ) return "";
104  else return str.substr(front_idx, back_idx-front_idx+1);
105  }
