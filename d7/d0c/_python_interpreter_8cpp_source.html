60 void throwPythonException()
61 {
62  PyObject* error = PyErr_Occurred(); // Borrowed reference
63  if(error != NULL)
64  {
65  PyObject* ptype, * pvalue, * ptraceback;
66  PyErr_Fetch(&ptype, &pvalue, &ptraceback);
67 
68  // Exception type
69  PyObjectPtr pyexception = makePyObjectPtr(PyObject_GetAttrString(
70  ptype, (char*)"__name__"));
71  std::string type = PyString_AsString(pyexception.get());
72 
73  // Message
74  PyObjectPtr pymessage = makePyObjectPtr(PyObject_Str(pvalue));
75  std::string message = PyString_AsString(pymessage.get());
76 
77  // Traceback
78  PyObjectPtr tracebackModule = makePyObjectPtr(PyImport_ImportModule("traceback"));
79  std::string traceback;
80  if (tracebackModule) {
81  PyObjectPtr tbList = makePyObjectPtr(
82  PyObject_CallMethod(
83  tracebackModule.get(), (char*)"format_exception",
84  (char*)"OOO", ptype, pvalue == NULL ? Py_None : pvalue,
85  ptraceback == NULL ? Py_None : ptraceback));
86 
87  PyObjectPtr emptyString = makePyObjectPtr(PyString_FromString(""));
88  PyObjectPtr strRetval = makePyObjectPtr(
89  PyObject_CallMethod(emptyString.get(), (char*)"join", (char*)"O",
90  tbList.get()));
91 
92  traceback = PyString_AsString(strRetval.get());
93  }
94  else
95  {
96  traceback = "Empty traceback";
97  }
98 
99  Py_XDECREF(ptype);
100  Py_XDECREF(pvalue);
101  Py_XDECREF(ptraceback);
102 
103  std::string error_message = "Python exception (" + type + "): " + message;
104  if(traceback != (type + ": " + message + "\n"))
105  error_message += "\n" + traceback;
106  throw std::runtime_error(error_message);
107  }
108 }

254  else if(PyFloat_Check(obj)) {

286  PyObject* mapToPyObjectPtr_(ConfigItem *map) {
287  if(map->isMap()) {
288  PyObject *dict = PyDict_New();
289  ConfigMap::iterator it = map->beginMap();
290  for(; it!=map->endMap(); ++it) {
291  PyObject* obj = mapToPyObjectPtr_(it->second);
292  if(obj) {
293  PyDict_SetItemString(dict, it->first.c_str(), obj);
294  Py_XDECREF(obj);

296  }
297  return dict;
298  }
299  else if(map->isVector()) {
300  PyObject *list = PyList_New(0);
301  ConfigVector::iterator it = map->begin();
302  for(; it!=map->end(); ++it) {
303  PyObject* obj = mapToPyObjectPtr_(*it);

305  PyList_Append(list, obj);
306  Py_XDECREF(obj);
307  }
308  }
309  return list;
310  }

312  ConfigAtom &atom = *map;
313  switch(atom.getType()) {
314  case ConfigAtom::DOUBLE_TYPE:
315  return PyFloat_FromDouble((double)atom);

318  return PyFloat_FromDouble((int)atom);

320  case ConfigAtom::BOOL_TYPE:
321  return PyFloat_FromDouble((bool)atom);
322  break;
323  default:
324  return PyString_FromString(atom.toString().c_str());

326  }

336  void toPyObjects(std::va_list& cppArgs, const std::list<CppType>& types, std::vector<PyObjectPtr>& args)
337 {
338  for(std::list<CppType>::const_iterator t = types.begin(); t != types.end();
339  t++)
340  {
341  switch(*t)
342  {
343  case INT:
344  {
345  const int i = va_arg(cppArgs, int);
346  args.push_back(Int::make(i).obj);
347  break;
348  }
349  case DOUBLE:
350  {
351  const double d = va_arg(cppArgs, double);
352  args.push_back(Double::make(d).obj);
353  break;
354  }
355  case BOOL:
356  {
357  // bool is promoted to int when passed through "..."
358  const int b = va_arg(cppArgs, int);
359  args.push_back(Bool::make((bool)b).obj);
360  break;
361  }
362  case STRING:
363  {
364  std::string* str = va_arg(cppArgs, std::string*);
365  args.push_back(String::make(*str).obj);
366  break;
367  }
368  case OBJECT:
369  {
370  Object* object = va_arg(cppArgs, Object*);
371  args.push_back(object->state->objectPtr);
372  break;
373  }
374  case MAP:
375  {
376  ConfigMap* map = va_arg(cppArgs, ConfigMap*);
377  args.push_back(mapToPyObjectPtr(map));
378  break;
379  }
380  case ONEDCARRAY:
381  {
382  double* array = va_arg(cppArgs, double*);
383  int size = va_arg(cppArgs, int);
384  npy_intp dims[1] = {(npy_intp) size};
385  PyObject *obj = PyArray_SimpleNewFromData(1, dims, NPY_DOUBLE,
386  (void*)(array));
387  args.push_back(makePyObjectPtr(obj));
388  break;
389  }
390  default:
391  throw std::runtime_error("Unknown function argument type");
392  }
393  throwPythonException();
394  }
395 }
