122 
123  QLocale::setDefault(QLocale::C);
124 
125  LOG_INFO("Load: loading scene: %s", sceneFilename.c_str());
126 
127  //test to open the xmlfile
128  if (!file.open(QIODevice::ReadOnly)) {
129  std::cout<<"Error while opening scene file content "

137 
138  //test to pass the content from the xmlfile to the DOM-Object
139  QDomDocument doc;
140  if (!doc.setContent(&file, false, &xmlErrorMsg,

151  //the XML-file has been opend and is ready for parsing it's content

156

158  if (!xmlnodelist.isEmpty()) {
159  tmpElement=xmlnodelist.at(0).toElement();

161  }

163  LOG_WARN("Load: old SceneFile version. Take care of the following changes:");

219  unsigned int Load::loadScene() {
220  for(unsigned int i=0; i<materialList.size(); ++i) if(!loadMaterial(materialList[i])) return 0;
221  for(unsigned int i=0; i<nodeList.size(); ++i) if(!loadNode(nodeList[i])) return 0;
222  for(unsigned int i=0; i<jointList.size(); ++i) if(!loadJoint(jointList[i])) return 0;
223  for(unsigned int i=0; i<motorList.size(); ++i) if(!loadMotor(motorList[i])) return 0;
224  for(unsigned int i=0; i<sensorList.size(); ++i) if(!loadSensor(sensorList[i])) return 0;
225  for(unsigned int i=0; i<controllerList.size(); ++i) if(!loadController(controllerList[i])) return 0;
226  for(unsigned int i=0; i<graphicList.size(); ++i) if(!loadGraphic(graphicList[i])) return 0;
227  for(unsigned int i=0; i<lightList.size(); ++i) if(!loadLight(lightList[i])) return 0;
228 
229  return 1;
230  }

245  NodeData node;
246  config["mapIndex"].push_back(configmaps::ConfigItem(mapIndex));
247  int valid = node.fromConfigMap(&config, tmpPath, control->loadCenter);
248  if(!valid) return 0;
249 
250  // handle material
251  configmaps::ConfigMap::iterator it;
252  if((it = config.find("material_id")) != config.end()) {
253  unsigned long id = it->second[0].getULong();
254  if(id) {
255  std::map<unsigned long, MaterialData>::iterator it = materials.find(id);
256  if(it != materials.end())
257  node.material = it->second;
258  }
259  }
260 
261  // the group ids could be also handled in the NodeData by the mapIndex
262  if(node.groupID)
263  node.groupID += groupIDOffset;
264 
265  NodeId oldId = node.index;
266  NodeId newId = control->nodes->addNode(&node);
267  if(!newId) {
268  LOG_ERROR("addNode returned 0");
269  return 0;
270  }
271  control->loadCenter->setMappedID(oldId, newId, MAP_TYPE_NODE, mapIndex);
272 
273  if(mRobotName != "") {
274  control->entities->addNode(mRobotName, node.index, node.name);
275  }
276  return 1;
