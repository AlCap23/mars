92  CPP_UNUSED(filenamePrefix);
93  CPP_UNUSED(loadCenter);
94  ConfigMap::iterator it;
95  unsigned int mapIndex;
96  GET_VALUE("mapIndex", mapIndex, UInt);
97 
98  GET_VALUE("name", name, String);
99  if((it = config->find("type")) != config->end()) {
100  type = getJointType(it->second[0].getString());
101  if(type == JOINT_TYPE_UNDEFINED) {
102  LOG_ERROR("JointData: type given is undefined");
103  }
104  }
105  else {
106  LOG_ERROR("JointData: no type given for joint");
107  }
108  GET_VALUE("index", index, ULong);
109  GET_VALUE("nodeindex1", nodeIndex1, ULong);
110  if(!nodeIndex1) {
111  LOG_ERROR("JointData: no first node attached to joint");
112  }
113  else if(mapIndex && loadCenter) {
114  nodeIndex1 = loadCenter->getMappedID(nodeIndex1, MAP_TYPE_NODE,
115  mapIndex);
116  }
117 
118  GET_VALUE("nodeindex2", nodeIndex2, ULong);
119  if(mapIndex && loadCenter) {
120  nodeIndex2 = loadCenter->getMappedID(nodeIndex2, MAP_TYPE_NODE,
121  mapIndex);
122  }
123 
124  { // handle axis 1
125  GET_OBJECT("axis1", axis1, vector);
126  GET_VALUE("lowStopAxis1", lowStopAxis1, Double);
127  GET_VALUE("highStopAxis1", highStopAxis1, Double);
128  GET_VALUE("damping_const_constraint_axis1",
129  damping_const_constraint_axis1, Double);
130  GET_VALUE("spring_const_constraint_axis1",
131  spring_const_constraint_axis1, Double);
132  GET_VALUE("angle1_offset", angle1_offset, Double);
133  }
134 
135  { // handle axis 2
136  GET_OBJECT("axis2", axis2, vector);
137  GET_VALUE("lowStopAxis2", lowStopAxis2, Double);
138  GET_VALUE("highStopAxis2", highStopAxis2, Double);
139  GET_VALUE("damping_const_constraint_axis2",
140  damping_const_constraint_axis2, Double);
141  GET_VALUE("spring_const_constraint_axis2",
142  spring_const_constraint_axis2, Double);
143  GET_VALUE("angle2_offset", angle2_offset, Double);
144  }
145 
146  GET_VALUE("anchorpos", anchorPos, Int);
147  GET_OBJECT("anchor", anchor, vector);
148  GET_VALUE("invertAxis", invertAxis, Bool);
