119  void Sonar::update() {
120  if(beams.size() < 2) return;
121 
122  //Cleaning up internal structures
123  vertices->clear();
124  color->clear();
125 
126  std::list<Beam>::iterator it = beams.begin();
127  osg::Quat lastQ, curQ;
128  double v;
129  lastQ = osg::Quat(invert*it->rad, osg::Vec3(0.0, 0.0, 1.0));
130  ++it;
131  for(; it!=beams.end(); ++it) {
132  curQ = osg::Quat(invert*it->rad, osg::Vec3(0.0, 0.0, 1.0));
133  for(unsigned int i=0; i<it->echos.size(); ++i) {
134  osg::Vec3d currentPoint(beamEchoStepDist * i, 0,0);
135  osg::Vec3d nextPoint(beamEchoStepDist * (i+1),0,0);
136 
137  v = it->echos[i];
138  if(v != 0.0){
139  color->push_back(osg::Vec4f(0,0.5+0.5*v,0,1.0));
140  color->push_back(osg::Vec4f(0,0.5+0.5*v,0,1.0));
141  color->push_back(osg::Vec4f(0,0.5+0.5*v,0,1.0));
142  color->push_back(osg::Vec4f(0,0.5+0.5*v,0,1.0));
143  color->push_back(osg::Vec4f(0,0.5+0.5*v,0,1.0));
144  color->push_back(osg::Vec4f(0,0.5+0.5*v,0,1.0));
145  }else{
146  color->push_back(osg::Vec4f(0.0,0.5,0.0,0.5));
147  color->push_back(osg::Vec4f(0.0,0.5,0.0,0.5));
148  color->push_back(osg::Vec4f(0.0,0.5,0.0,0.5));
149  color->push_back(osg::Vec4f(0.0,0.5,0.0,0.5));
150  color->push_back(osg::Vec4f(0.0,0.5,0.0,0.5));
151  color->push_back(osg::Vec4f(0.0,0.5,0.0,0.5));
152  }
153 
154  //First triangle
155  vertices->push_back(curQ*currentPoint);
156  vertices->push_back(curQ*nextPoint);
157  vertices->push_back(lastQ*currentPoint);
158  //Second Triangle
159  vertices->push_back(lastQ*currentPoint);
160  vertices->push_back(lastQ*nextPoint);
161  vertices->push_back(curQ*nextPoint);
162  }
163  lastQ = curQ;
164  }
165  drawArrays->setCount(vertices->size());
166  geom->setVertexArray(vertices);
167  geom->setColorArray(color);
168  }
