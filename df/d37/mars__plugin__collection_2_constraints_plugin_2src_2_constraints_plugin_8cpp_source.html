169  void ConstraintsPlugin::parseNodeConstraints(const std::string &paramName,
170  const std::string &s) {
171  std::string nodeName, nodeAttr;
172  NodeId nodeId = 0;
173  AttributeType attr = ATTRIBUTE_UNDEFINED;
174  ParseResult result = PARSE_SUCCESS;
175  double factor, initialValue;
176  size_t pos1 = 0;
177 
178  // first entry is the initial nodeName.nodeAttr
179  result = parseIdentifier(s, &pos1, &nodeId, &attr, NULL);
180  if(result != PARSE_SUCCESS)
181  return;
182  initialValue = getNodeAttribute(nodeId, attr);
183  // create new Property to control the constraint
184  cfgPropertyStruct newProp;
185  newProp = control->cfg->getOrCreateProperty("Constraints", paramName,
186  initialValue);
187  // add constraint for initial Node with factor 1
188  BaseConstraint *c = new NodeConstraint(control, nodeId, attr,
189  1, initialValue);
190  constraints[newProp.paramId].push_back(c);
191  control->cfg->registerToParam(newProp.paramId, c);
192 
193  while(result == PARSE_SUCCESS) {
194  // next there is a list of NodeName.NodeAttr#factor
195  result = parseIdentifier(s, &pos1, &nodeId, &attr, &factor);
196  if((result != PARSE_SUCCESS) && (result != PARSE_SUCCESS_EOS)) {
197  continue;
198  }
199  BaseConstraint *c = new NodeConstraint(control, nodeId, attr, factor,
200  initialValue);
201  constraints[newProp.paramId].push_back(c);
202  control->cfg->registerToParam(newProp.paramId, c);
203  }

206  ParseResult ConstraintsPlugin::parseIdentifier(const std::string &s,
207  size_t *pos,
208  NodeId *nodeId,
209  AttributeType *attr,
210  double *factor) {
211  // expect the format "nodeName.nodeAttr#factor" where the factor is optional
212  size_t startPos=*pos, splitPos, endPos;
213  // find end of identifier
214  endPos = s.find('#', startPos);
215  // find the nodeAttribute. start from rear because nodeName may contain "."
216  splitPos = s.rfind('.', endPos);
217  std::string nodeName = s.substr(startPos, splitPos - startPos);
218  std::string nodeAttr = s.substr(splitPos+1, endPos - splitPos - 1);
219 
220  if(factor) {
221  if(endPos == std::string::npos) {
222  *factor = 0;
223  LOG_WARN("ConstraintsPlugin::parseIdentifier: factor is missing");
224  } else {
225  startPos = endPos + 1;
226  endPos = s.find('#', startPos);
227  *factor = strtod(s.substr(startPos, endPos - startPos).c_str(), NULL);
228  }
229  }
230 
231  // get the actual values from the strings and write the results back
232  *nodeId = control->nodes->getID(nodeName);
233  *attr = parseAttribute(nodeAttr);
234  *pos = endPos + 1;
235 
236  // do error checking
237  if(*nodeId == 0) {
238  LOG_ERROR("Invalid NodeName \"%s\" in NodeConstraint: \"%s\"",
239  nodeName.c_str(), s.c_str());
240  return PARSE_ERROR_NODEID;
241  }
242  if(*attr == ATTRIBUTE_UNDEFINED) {
243  LOG_ERROR("Invalid NodeAttribute \"%s\" in NodeConstraint: \"%s\"",
244  nodeAttr.c_str(), s.c_str());
245  return PARSE_ERROR_NODEATTR;
246  }
247  // signal whether we reached the End-Of-String or not
248  if(endPos == std::string::npos)
249  return PARSE_SUCCESS_EOS;
250  else
251  return PARSE_SUCCESS;
252  }
