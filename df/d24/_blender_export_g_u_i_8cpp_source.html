85  void BlenderExportGUI::startButton(void) {
86  if(state == 1) {
87  if(fileHandle) fclose(fileHandle);
88  std::string filename_str = filename->value().toString().toStdString() +std::string("/export.dat");

90  status->setValue("Cannot open file. Aborted.");

117  void BlenderExportGUI::writeGenericData(unsigned long id,
118  void* data) const {
119  (void)id;
120  (void)data;
121  if(state == 3) {
122  std::vector<interfaces::core_objects_exchange> objectList;
123  std::vector<interfaces::core_objects_exchange>::iterator iter;
124  interfaces::NodeData theNode;
125  utils::Quaternion q;
126  utils::Vector pos;
127  control->nodes->getListNodes(&objectList);
128 
129  fprintf(fileHandle, "frame %lu\n", frame++);
130 
131  for(iter=objectList.begin(); iter!=objectList.end(); ++iter) {
132  theNode = control->nodes->getFullNode((*iter).index);
133  q = (*iter).rot * theNode.visual_offset_rot;
134 
135  //pos = QVRotate((*iter).rot, theNode.visual_offset_pos);
136  pos = ((*iter).rot * theNode.visual_offset_pos);
137  pos += (*iter).pos;
138  fprintf(fileHandle, "node%lu %g %g %g %g %g %g %g\n",
139  (*iter).index, pos.x(), pos.y(),
140  pos.z(), q.w(), q.x(), q.y(), q.z());
141  }
142  }
