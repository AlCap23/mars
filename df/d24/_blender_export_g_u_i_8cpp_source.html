88  void BlenderExportGUI::startButton(void) {
89  if(state == 1) {
90  if(fileHandle) fclose(fileHandle);
91  std::string path = filename->value().toString().toStdString();
92  std::string filename_str = path +std::string("/export.dat");
93  if ((fileHandle = fopen(filename_str.data(), "w")) == 0) {
94  status->setValue("Cannot open file. Aborted.");
95  status = 0;
96  }
97  control->nodes->exportGraphicNodesByID(path);
98  std::string cmd = "cp "+rPath+"/blender/export.blend " + path;
99  system(cmd.c_str());
100  state = 2;
101  }
102  if(state == 2) {
103  state = 3;
104  writeGenericData(0, NULL);
105  status->setValue(tr("Exporting started..."));
106  control->dataBroker->registerTimedReceiver(this, "mars_sim", "simTime",
107  "mars_sim/simTimer", 40);
108  }
109  }

132  void BlenderExportGUI::writeGenericData(unsigned long id,
133  void* data) const {
134  (void)id;
135  (void)data;
136  if(state == 3) {
137  std::vector<interfaces::core_objects_exchange> objectList;
138  std::vector<interfaces::core_objects_exchange>::iterator iter;
139  interfaces::NodeData theNode;
140  utils::Quaternion q;
141  utils::Vector pos;
142  control->nodes->getListNodes(&objectList);
143 
144  fprintf(fileHandle, "frame %lu\n", frame++);
145 
146  for(iter=objectList.begin(); iter!=objectList.end(); ++iter) {
147  theNode = control->nodes->getFullNode((*iter).index);
148  q = (*iter).rot * theNode.visual_offset_rot;
149 
150  //pos = QVRotate((*iter).rot, theNode.visual_offset_pos);
151  pos = ((*iter).rot * theNode.visual_offset_pos);
152  pos += (*iter).pos;
153  fprintf(fileHandle, "node%lu %g %g %g %g %g %g %g\n",
154  (*iter).index, pos.x(), pos.y(),
155  pos.z(), q.w(), q.x(), q.y(), q.z());
156  }
157  }
158  }
