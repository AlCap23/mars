55  OSGNodeStruct::OSGNodeStruct(GraphicsManager *g,
56  const NodeData &node, bool isPreview,
57  unsigned long id)
58  : osg::Group(), drawObject_(NULL), id_(id), isPreview_(isPreview) {
59  configmaps::ConfigMap map = node.map;
60  unsigned long sharedID = 0;
61  std::string filename, origname;
62  std::string visualType;
63  if(map.find("sharedDrawID") != map.end()) {
64  sharedID = map["sharedDrawID"];
65  }
66  if(map.find("visualType") != map.end()) {
67  visualType = (std::string)map["visualType"];
68  if(visualType == "mesh") {
69  filename = node.filename;
70  origname = node.origName;
71  }
72  else {
73  filename = "PRIMITIVE";
74  origname = visualType;
75  }
76  }
77  else {
78  filename = node.filename;
79  origname = node.origName;
80  }
81  if (filename.compare("PRIMITIVE") == 0) {
82  Vector vizSize = node.ext;
83  switch(NodeData::typeFromString(origname.c_str())) {
84  case mars::interfaces::NODE_TYPE_BOX: {
85  drawObject_ = new CubeDrawObject(g);
86  break;
87  }
88  case mars::interfaces::NODE_TYPE_SPHERE: {
89  vizSize.x() *= 2;
90  vizSize.y() = vizSize.z() = vizSize.x();
91  drawObject_ = new SphereDrawObject(g);
92  break;
93  }
94  case mars::interfaces::NODE_TYPE_REFERENCE: {
95 #warning add here an coordinate system item
96  //For now until we have an real coordinate system
97  drawObject_ = new SphereDrawObject(g);
98  break;
99  }
100  case mars::interfaces::NODE_TYPE_MESH:
101  case mars::interfaces::NODE_TYPE_CYLINDER:
102  vizSize.x() *= 2;
103  vizSize.z() = vizSize.y();
104  vizSize.y() = vizSize.x();
105  drawObject_ = new CylinderDrawObject(g, 1, 1);
106  break;
107  case mars::interfaces::NODE_TYPE_CAPSULE: {
108  vizSize.x() *= 2;
109  vizSize.z() = vizSize.y();
110  vizSize.y() = vizSize.x();
111  drawObject_ = new CapsuleDrawObject(g);
112  break;
113  }
114  case mars::interfaces::NODE_TYPE_PLANE: {
115  drawObject_ = new PlaneDrawObject(g, vizSize);
116  break;
117  }
118  case mars::interfaces::NODE_TYPE_EMPTY: {
119  drawObject_ = new EmptyDrawObject(g);
120  break;
121  }
122  default:
123  fprintf(stderr,"Cannot find primitive type: %i(%s), at %s:%i\n",
124  NodeData::typeFromString(origname.c_str()),
125  origname.c_str(), __FILE__, __LINE__);
126  throw std::runtime_error("unknown primitive type");
127  }
128  if(map.find("maxNumLights") != map.end()) {
129  drawObject_->setMaxNumLights(map["maxNumLights"]);
130  }
131  drawObject_->createObject(id, Vector(0.0, 0.0, 0.0), sharedID);
132  if(node.visual_size != Vector(0.0, 0.0, 0.0)) {
133  vizSize = node.visual_size;
134  }
135  drawObject_->setScaledSize(vizSize);
136  } else if (node.physicMode == mars::interfaces::NODE_TYPE_TERRAIN) {
137  // we have a heightfield
138  if (!node.terrain->pixelData) {
139  node.terrain->pixelData = (double*)calloc(
140  (node.terrain->width*node.terrain->height), sizeof(double));
141  //QImage image(QString::fromStdString(snode->filename));
142  int r = 0, g = 0, b = 0;
143  int count = 0;
144  for (int y=0; y<node.terrain->height; y++) {
145  for (int x=0; x<node.terrain->width; x++) {
146  //convert to greyscale by common used scale
147  node.terrain->pixelData[count++] = ((r*0.3+g*0.59+b*0.11)/255.0);
148  }
149  }
150  }
151  drawObject_ = new TerrainDrawObject(g, node.terrain);
152  if(map.find("maxNumLights") != map.end()) {
153  drawObject_->setMaxNumLights(map["maxNumLights"]);
154  }
155  drawObject_->createObject(id, Vector(node.terrain->targetWidth*0.5,
156  node.terrain->targetHeight*0.5,
157  0.0),
158  sharedID);
159  } else { // we have to load the node from an import file
160  if(map.find("filename") == map.end()) {
161  map["filename"] = filename;
162  }
163  if(map.find("origname") == map.end()) {
164  map["origname"] = origname;
165  }
166  drawObject_ = new LoadDrawObject(g, map, node.ext);
167  if(map.find("maxNumLights") != map.end()) {
168  drawObject_->setMaxNumLights(map["maxNumLights"]);
169  }
170  drawObject_->createObject(id, node.pivot, sharedID);
171  drawObject_->setScale(node.visual_scale);
172  if(node.visual_size != Vector(0.0, 0.0, 0.0)) {
173  drawObject_->setScaledSize(node.visual_size);
174  }
175  }
176 
177  drawObject_->setPosition(node.pos + node.rot * node.visual_offset_pos);
178  drawObject_->setQuaternion(node.rot * node.visual_offset_rot);
179  }

181  void OSGNodeStruct::edit(const NodeData &node, bool resize) {
182  osg::ComputeBoundsVisitor cbbv;
183  osg::BoundingBox bb = cbbv.getBoundingBox();
184  osg::Quat oquat;
185  Vector ex;
186 
187  // get old size, create bounding box to get center point
188  drawObject_->getObject()->accept(cbbv);
189  // compute bounding box has to be done in this way
190  if(fabs(bb.xMax()) > fabs(bb.xMin())) {
191  ex.x() = fabs(bb.xMax() - bb.xMin());
192  } else {
193  ex.x() = fabs(bb.xMin() - bb.xMax());
194  }
195  if(fabs(bb.yMax()) > fabs(bb.yMin())) {
196  ex.y() = fabs(bb.yMax() - bb.yMin());
197  } else {
198  ex.y() = fabs(bb.yMin() - bb.yMax());
199  }
200  if(fabs(bb.zMax()) > fabs(bb.zMin())) {
201  ex.z() = fabs(bb.zMax() - bb.zMin());
202  } else {
203  ex.z() = fabs(bb.zMin() - bb.zMax());
204  }
205 
206  //compute scale factor
207  double scaleX = (ex.x() != 0) ? node.visual_size.x() / ex.x() : 1;
208  double scaleY = (ex.y() != 0) ? node.visual_size.y() / ex.y() : 1;
209  double scaleZ = (ex.z() != 0) ? node.visual_size.z() / ex.z() : 1;
210 
211  //resize
212  if (resize) {
213  drawObject_->getScaleMatrix()->setMatrix(osg::Matrix::scale(scaleX, scaleY, scaleZ));
214  drawObject_->getTransform()->setPivotPoint(osg::Vec3(
215  node.pivot.x()*scaleX, node.pivot.y()*scaleY, node.pivot.z()*scaleZ));
216  }
217  drawObject_->getTransform()->setPosition(osg::Vec3(
218  node.pos.x()+node.visual_offset_pos.x(),
219  node.pos.y()+node.visual_offset_pos.y(),
220  node.pos.z()+node.visual_offset_pos.z()));
221  oquat.set(node.rot.x(), node.rot.y(), node.rot.z(), node.rot.w());
222  drawObject_->getTransform()->setAttitude(oquat);
223  }
