136  void Dialog_Motor_Control::sliderValueChanged(int value) {
137  if (not radboxactive and not degboxactive) {
138  slideractive = true;
139  QObject* obj = sender();
140  for (uint i = 0; i < sliders.size(); ++i)
141  if (obj == sliders[i]) {
142  motors[i].value = (double)value/scale;
143  radspinboxes[i]->setValue(motors[i].value);
144  degspinboxes[i]->setValue(motors[i].value/3.141593*180);
145  control->motors->moveMotor(motors[i].index, motors[i].value);
146  break;
147  }
148  slideractive = false;
149  }
