35  void PhobosGraphSP::parseGraph() {
36  vector<GLSLAttribute> vars;
37  vector<string> function_calls;
38  ConfigVector::iterator vit;
39  ConfigMap::iterator mit;
40  ConfigMap functionInfo;
41  string customPath = options["loadPath"].getString() + options["customPath"].getString() + "/";
42  string defaultPath = resPath + "/graph_shader/";
43  string legacyPath = resPath + "/shader/";
44  string actualPath = "";
45  for (vit = model["sort"].begin(); vit != model["sort"].end(); ++vit) { // iterate over sorted node names
46  string nodeName = vit.base()->getString();
47  string nodeType = model["nodes"][nodeName]["type"].getString();
48  if (nodeType == "uniform") {
49  if (model["nodes"][nodeName]["uniform_builtin"].getString() == "false") { // TODO: Should actually get/read a bool right away..
50  uniforms.insert({model["nodes"][nodeName]["uniform_type"].getString(),
51  model["nodes"][nodeName]["uniform_name"].getString()});
52  }
53  continue;
54  } else if (nodeType == "varying_fragment") {
55  varyings.insert({model["nodes"][nodeName]["varying_type"].getString(),
56  model["nodes"][nodeName]["varying_name"].getString()});
57  continue;
58  } else if (nodeType == "varying_vertex") {
59  varyings.insert({model["nodes"][nodeName]["varying_type"].getString(),
60  model["nodes"][nodeName]["varying_name"].getString()});
61  stringstream call;
62  call.clear();
63  call << model["nodes"][nodeName]["varying_name"].getString() << " = "
64  << model["nodes"][nodeName]["incoming"]["input"].getString() << ";" << endl;
65  function_calls.push_back(call.str());
66  continue;
67  }
68  if (file_exists(customPath + nodeType + ".yaml")) {
69  actualPath = customPath + nodeType + ".yaml";
70  } else if (file_exists(defaultPath + nodeType + ".yaml")) {
71  actualPath = defaultPath + nodeType + ".yaml";
72  } else if (file_exists(legacyPath + nodeType + ".yaml")) {
73  actualPath = legacyPath + nodeType + ".yaml";
74  } else {
75  cout << "ERROR: Could not find node definition for " << nodeType << "!" << endl;
76  // TODO: How to handle an error like that?
77  return;
78  }
79  functionInfo = ConfigMap::fromYamlFile(actualPath);
80  parse_functionInfo(nodeType, functionInfo);
81  // CREATE VAR DEFINITIONS
82  for (mit = model["nodes"][nodeName]["outgoing"].beginMap();
83  mit != model["nodes"][nodeName]["outgoing"].endMap(); ++mit) {
84  vars.push_back({(mit->second)["type"], (mit->second)["name"]});
85  }
86  //Create function calls
87  priority_queue<PrioritizedLine> incoming, outgoing;
88  if (functionInfo["params"].hasKey("in")) {
89  for (mit = functionInfo["params"]["in"].beginMap(); mit != functionInfo["params"]["in"].endMap(); ++mit) {
90  incoming.push((PrioritizedLine) {model["nodes"][nodeName]["incoming"][mit->first],
91  (int) functionInfo["params"]["in"][mit->first]["index"], 0});
92  }
93  }
94  if (functionInfo["params"].hasKey("out")) {
95  for (mit = functionInfo["params"]["out"].beginMap(); mit != functionInfo["params"]["out"].endMap(); ++mit) {
96  outgoing.push((PrioritizedLine) {model["nodes"][nodeName]["outgoing"][mit->first]["name"].getString(),
97  (int) functionInfo["params"]["out"][mit->first]["index"], 0});
98  }
99  }
100  stringstream call;
101  bool first = true;
102  call.clear();
103  call << nodeType << "(";
104  while (!incoming.empty()) {
105  if (!first) {
106  call << ", ";
107  }
108  first = false;
109  call << incoming.top().line;
110  incoming.pop();
111  }
112  while (!outgoing.empty()) {
113  if (!first) {
114  call << ", ";
115  }
116  first = false;
117  call << outgoing.top().line;
118  outgoing.pop();
119  }
120  call << ");" << endl;
121  function_calls.push_back(call.str());
122  } // end iteration over sorted node names
123  // Create main function code
124  stringstream code;
125  code.clear();
126  code << "void main() {" << endl;
127  for (size_t i = 0; i < vars.size(); ++i) {
128  code << " " << vars[i] << ";" << endl;
129  }
130  code << endl;
131  for (size_t i = 0; i < function_calls.size(); ++i) {
132  code << function_calls[i] << endl;
133  }
134  code << "}" << endl;
135  main_source = code.str();
136  }

186  void PhobosGraphSP::parse_functionInfo(string functionName, ConfigMap functionInfo) {
187  ConfigMap::iterator mit;
188  ConfigVector::iterator vit;
189  if (source_files.count(functionName) == 0) {
190  string suffix = "";
191  if (functionInfo.hasKey("source")) {
192  suffix = functionInfo["source"].getString();
193  } else {
194  suffix = functionName + ".c";
195  }
196  if (file_exists(options["loadPath"].getString() + options["customPath"].getString() + "/" + suffix)) {
197  source_files[functionName] = options["loadPath"].getString() + options["customPath"].getString() + "/" + suffix;
198  } else if (file_exists(resPath + "/" + suffix)) {
199  source_files[functionName] = resPath + "/" + suffix;
200  }
201  }
202 
203  if (functionInfo.hasKey("minVersion")) {
204  int version = functionInfo["minVersion"];
205  if (version > minVersion) {
206  minVersion = version;
207  }
208  }
209 
210  if (functionInfo.hasKey("extensions")) {
211  ConfigItem list = functionInfo["extensions"];
212  if (list.hasKey("enabled")) {
213  std::cout << "Having enabled extensions" <<
214  std::endl;
215  for (vit = list["enabled"].begin(); vit != list["enabled"].end(); vit++) {
216  enabledExtensions.insert(vit.base()->getString());
217  }
218  }
219  if (list.hasKey("disabled")) {
220  for (vit = list["disabled"].begin(); vit != list["disabled"].end(); vit++) {
221  disabledExtensions.insert(vit.base()->getString());
222  }
223  }
224  }
225 
226  if (functionInfo.hasKey("uniforms")) {
227  ConfigItem list = functionInfo["uniforms"];
228  for (mit = list.beginMap(); mit != list.endMap(); ++mit) {
229  string type = mit->first;
230  string type_name = "";
231  std::size_t a_pos = type.find("[]");
232  bool is_array = false;
233  if (a_pos != string::npos) {
234  type_name = type.substr(0, a_pos);
235  is_array = true;
236  } else {
237  type_name = type;
238  }
239  for (vit = list[type].begin(); vit != list[type].end(); ++vit) {
240  ConfigItem &item = *vit;
241  string name = item["name"].getString();
242  if (is_array) {
243  stringstream s;
244  int num = 1;
245  if (item.hasKey("arraySize")) {
246  num = (int) options[item["arraySize"].getString()];
247  }
248  s << "[" << num << "]";
249  name.append(s.str());
250  }
251  uniforms.insert((GLSLUniform) {type_name, name});
252  }
253  }
254  }
255 
256  if (functionInfo.hasKey("varyings")) {
257  ConfigItem list = functionInfo["varyings"];
258  for (mit = list.beginMap(); mit != list.endMap(); ++mit) {
259  string type = mit->first;
260  string type_name = "";
261  std::size_t a_pos = type.find("[]");
262  bool is_array = false;
263  if (a_pos != string::npos) {
264  type_name = type.substr(0, a_pos);
265  is_array = true;
266  } else {
267  type_name = type;
268  }
269  for (vit = list[type].begin(); vit != list[type].end(); ++vit) {
270  ConfigItem &item = *vit;
271  string name = item["name"].getString();
272  if (is_array) {
273  stringstream s;
274  int num = 1;
275  if (item.hasKey("arraySize")) {
276  num = (int) options[item["arraySize"].getString()];
277  }
278  s << "[" << num << "]";
279  name.append(s.str());
280  }
281  varyings.insert((GLSLAttribute) {type_name, name});
282  }
283  }
284  }
285 
286  if (functionInfo.hasKey("attributes")) {
287  ConfigItem list = functionInfo["attributes"];
288  for (mit = list.beginMap(); mit != list.endMap(); ++mit) {
289  string type = mit->first;
290  string type_name = "";
291  std::size_t a_pos = type.find("[]");
