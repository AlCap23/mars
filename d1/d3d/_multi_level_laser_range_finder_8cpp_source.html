203 void MultiLevelLaserRangeFinder::calculateSamplingPixels()
204 {
205  const double verticalStartAngle = -config.verticalOpeningAngle / 2.0;
206  const double scansVertical = config.numRaysVertical;
207  const double scansHorizontal = config.numRaysHorizontal;
208 
209  double stepHorizontal = config.horizontalOpeningAngle / (scansHorizontal - 1);
210  double stepVertical = config.verticalOpeningAngle / (scansVertical - 1);
211 
212 
213  double curHorAngle = 0;
214 
215  std::vector<RaySubSensor>::iterator it = subSensors.begin();
216 
217  const double b_x = (config.rttResolutionX / 2.0) / tan(M_PI/4.0);
218  const double b_y = (config.rttResolutionY / 2.0) / tan(M_PI/4.0);
219 
220  int counter = 0;
221 
222  lookups.resize(config.numRaysHorizontal * config.numRaysVertical);
223 
224  //Fill all distance images with 1.0.
225  //This need to be done, so that we can precompute the directionVector
226  for(std::vector<RaySubSensor>::iterator it = subSensors.begin(); it != subSensors.end();it++)
227  {
228  std::fill(it->distImage.data.begin(), it->distImage.data.end(), 1.0);
229  }
230 
231 
232  // Run through the image horizontally.
233  for(int h = 0; h < config.numRaysHorizontal; h++)
234  {
235  double horAngle = h * stepHorizontal;
236  assert(fabs(curHorAngle - (horAngle - counter * M_PI / 2)) < 0.001);
237 
238  int x = tan(curHorAngle) * b_x + (config.rttResolutionX / 2.0);
239 
240  for(int v = 0; v < config.numRaysVertical; v++)
241  {
242  double verAngle = verticalStartAngle + v * stepVertical;
243 
244  int y = tan(verAngle) / cos(curHorAngle) * b_y + (config.rttResolutionY / 2.0);
245 
246  Lookup &lookup(lookups[v + (config.numRaysHorizontal - h - 1) * config.numRaysVertical]);
247  lookup.x = x;
248  lookup.y = y;
249  lookup.sensor = &(*it);
250 
251  Eigen::Vector3d dirVec;
252  bool result = it->distImage.getScenePoint(x, y, dirVec);
253  assert(result);
254  lookup.directionVector = dirVec;
255  }
256 
257  curHorAngle += stepHorizontal;
258 
259  if(curHorAngle > M_PI/4.0)
260  {
261  counter++;
262  std::cout << "Camera " << counter << std::endl;
263  curHorAngle -= M_PI/2.0;
264  it++;
265  if(it == subSensors.end())
