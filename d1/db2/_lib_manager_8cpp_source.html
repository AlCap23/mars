163  void *config, bool optional) {

178 #endif
179 #ifdef DEBUG

181 #endif

183  //try to locate the library somewhere by checking at various path positions
184  FILE *testFile = fopen(libPath.c_str(), "r");
185  std::string filepath;
186  if(!testFile) {
187  filepath = prefix;
188  filepath.append(libPath);
189  filepath.append(suffix);
190  char* lib_path = getenv(env);
191  char* lib_path2 = getenv(env2);
192  if(lib_path || lib_path2) {
193  // try to first find library
194  std::string lib_path_s;
195  if(lib_path) {
196  lib_path_s = lib_path;
197  if(lib_path2) {
198  lib_path_s.append(":");
199  }
200  }
201  if(lib_path2) {
202  lib_path_s.append(lib_path2);
203  }
204  size_t next_path_pos = 0;
205  size_t actual_path_pos = 0;
206  while(next_path_pos != string::npos) {
207  next_path_pos = lib_path_s.find(sep, actual_path_pos);
208  string actual_path = lib_path_s.substr(actual_path_pos, (next_path_pos != string::npos) ? next_path_pos - actual_path_pos : lib_path_s.size() - actual_path_pos);
209  actual_path.append("/");
210  std::string actual_lib_path = actual_path;
211  actual_lib_path.append(filepath);
212  testFile = fopen(actual_lib_path.c_str(), "r");
213  if(testFile) {
214  fclose(testFile);
215  filepath = actual_lib_path;
216 #ifdef DEBUG
217  printf("lib_manager: found plugin at: %s\n",
218  filepath.c_str());
219 #endif
220  break;
221  }
222  actual_path_pos = next_path_pos + 1;
223  }
224  }
225  }
226  else {
227  filepath = libPath;
228  fclose(testFile);
229  }

340  void LibManager::loadConfigFile(const std::string &config_file) {
341  char plugin_chars[255];
342  std::string plugin_path;
343  FILE *plugin_config;
344 
345  plugin_config = fopen(config_file.c_str() , "r");
346  if(!plugin_config) {
347  fprintf(stderr, "LibManager::loadConfigFile: file \"%s\" not found.\n",
348  config_file.c_str());
349  return;
350  }
351 
352  //check every input line for a library to load and if one is found, try to load it
353  while(fgets(plugin_chars, 255, plugin_config)) {
354  plugin_path = plugin_chars;
355  // strip whitespaces from start and end of line
356  size_t pos1 = plugin_path.find_first_not_of(" \t\n\r");
357  size_t pos2 = plugin_path.find_last_not_of(" \t\n\r");
358  if(pos1 == string::npos || pos2 == string::npos) {
359  continue;
360  }
361  plugin_path = plugin_path.substr(pos1, pos2 - pos1 + 1);
362  // ignore lines that start with #
363  if(plugin_path[0] != '#') {
364  loadLibrary(plugin_path);
365  }
366  }
367  fclose(plugin_config);
368  }
