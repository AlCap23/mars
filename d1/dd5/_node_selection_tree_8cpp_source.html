124  void NodeSelectionTree::fill(unsigned long id, QTreeWidgetItem *current) {
125  QStringList temp;
126 
127  if (current == NULL) // this is a top level node
128  for (unsigned int i = 0; i < simNodes.size(); i++)
129  if (simNodes[i].index == id) {
130  bool found = false;
131  for (unsigned int k = 0; k < present.size() && !found; k++)
132  if (present[k] == id)
133  found = true;
134  if (found == false) {
135  temp << QString::number(id) + ":" + QString::fromStdString(simNodes[i].name);
136  current = new QTreeWidgetItem(temp);
137  treeWidget->addTopLevelItem(current);
138  present.push_back(id);
139  }
140  }
141 
142  vector<unsigned long> children = control->nodes->getConnectedNodes(id);
143  vector<unsigned long> newNodes;
144 
145  for (unsigned int i = 0; i < children.size(); i++) {
146  bool found = false;
147  for (unsigned int k = 0; k < present.size() && !found; k++)
148  if (present[k] == children[i])
149  found = true;
150  if (found == false) {
151  present.push_back(children[i]);
152  newNodes.push_back(children[i]);
153  }
154  }
155 
156  for (unsigned int i = 0; i < newNodes.size(); i++)
157  for (unsigned int j = 0; j < simNodes.size(); j++)
158  if (newNodes[i] == simNodes[j].index) {
159  temp.clear();
160  temp << QString::number(newNodes[i]) + ":" + QString::fromStdString(simNodes[j].name);
161  QTreeWidgetItem *next = new QTreeWidgetItem(temp);
162  current->addChild(next);
163  fill(newNodes[i], next);
164  }
165  }
