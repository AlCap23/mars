185  void Viz::loadScene(std::string filename, std::string robotname) {
186 
187  std::string tmpPath = "./tmp/";
188  scene_loader::Load load(filename, control, tmpPath, "");
189  load.prepareLoad();
190  load.parseScene();
191 
192  std::map<std::string, ForwardTransform> jointsI;
193  std::map<unsigned long, NodeData> nodeMapI;
194  std::map<unsigned long, NodeData> nodeMapReady;
195  std::map<unsigned long, NodeData>::iterator it1;
196  std::map<unsigned long, NodeData>::iterator it2;
197  std::map<unsigned long, NodeData>::iterator it3;
198  std::list<JointData> jointList;
199  std::list<JointData>::iterator jointIt;
200 
201  for(unsigned int i=0; i<load.jointList.size(); ++i) {
202  JointData joint;
203  joint.fromConfigMap(&load.jointList[i], tmpPath, NULL);
204  jointList.push_back(joint);
205  }
206 
207  for(unsigned int i=0; i<load.materialList.size(); ++i) {
208  MaterialData material;
209  unsigned long id;
210  material.fromConfigMap(&load.materialList[i], tmpPath);
211  if((id = load.materialList[i]["id"])) {
212  load.materials[id] = material;
213  }
214  }
215 
216  for(unsigned int i=0; i<load.nodeList.size(); ++i) {
217  NodeData node;
218  node.fromConfigMap(&load.nodeList[i], tmpPath, NULL);
219 
220  if(node.terrain) {
221  graphics->getLoadHeightmapInterface()->readPixelData(node.terrain);
222  }
223 
224  // handle material
225  configmaps::ConfigMap::iterator it;
226  if((it = load.nodeList[i].find("material_id")) != load.nodeList[i].end()) {
227  unsigned long id = it->second;
228  if(id) {
229  std::map<unsigned long, MaterialData>::iterator it = load.materials.find(id);
230  if(it != load.materials.end())
231  node.material = it->second;
232  }
233  }
234  nodeMapI[node.index] = node;
235  nodeMapI[node.index].index = graphics->addDrawObject(node);
236  if(node.terrain) {
237  nodeMapI[node.index].pivot = utils::Vector(node.terrain->targetWidth*0.5,
238  node.terrain->targetHeight*0.5,
239  node.terrain->scale*0.0);
240  }
241  }
242 
243  bool done = false;
244  while(!done) {
245  done = true;
246  for(it1=nodeMapI.begin(); it1!=nodeMapI.end(); ++it1) {
247  if(it1->second.relative_id) {
248  it2 = nodeMapReady.find(it1->second.relative_id);
249  if(it2 != nodeMapReady.end()) {
250  getAbsFromRel(it2->second, &it1->second);
251  nodeMapReady[it1->first] = it1->second;
252  nodeMapI.erase(it1);
253  done = false;
254  break;
255  }
256  }
257  else {
258  nodeMapReady[it1->first] = it1->second;
259  nodeMapI.erase(it1);
260  done = false;
261  break;
262  }
263  }
264  }
265 
266  assert(nodeMapI.empty());
267 
268  nodeMapReady.swap(nodeMapI);
269  for(it1=nodeMapI.begin(); it1!=nodeMapI.end(); ++it1) {
270  utils::Vector offset = it1->second.rot * it1->second.visual_offset_pos;
271  it1->second.pos += offset;
272  it1->second.rot = it1->second.rot * it1->second.visual_offset_rot;
273 
274  graphics->setDrawObjectPos(it1->second.index, it1->second.pos);
275  graphics->setDrawObjectRot(it1->second.index, it1->second.rot);
276  }
277 
278  // handle relations
279  nodeMapReady.clear();
280  it1 = nodeMapI.find(1);
281  if(it1!=nodeMapI.end()) {
282  nodeMapReady[it1->first] = it1->second;
283  nodeMapI.erase(it1);
284 
285  it1 = nodeMapReady.begin();
286  done = false;
287  while(!done && !nodeMapI.empty()) {
288  done = true;
289  for(it1=nodeMapReady.begin(); it1!=nodeMapReady.end(); ++it1) {
290  it2 = nodeMapI.begin();
291  while(it2 != nodeMapI.end()) {
292  if(it2->second.groupID == it1->second.groupID) {
293  NodeData node = it2->second;
294  utils::Vector v = it1->second.pos - it1->second.rot*it1->second.pivot;
295  node.pos = it1->second.rot.inverse() * (node.pos - v);
296  node.rot = it1->second.rot.inverse() * node.rot;
297 
298  graphics->makeChild(it1->second.index, it2->second.index);
299  graphics->setDrawObjectPos(node.index, node.pos);
300  graphics->setDrawObjectRot(node.index, node.rot);
301 
302  nodeMapReady[it2->first] = it2->second;
303  nodeMapI.erase(it2++);
304  done = false;
305  }
306  else {
307  bool connected = false;
308  bool invert = false;
309  for(jointIt=jointList.begin(); jointIt!=jointList.end();
310  ++jointIt) {
311  if((jointIt->nodeIndex1 == it1->first &&
312  jointIt->nodeIndex2 == it2->first)) {
313  connected = true;
314  done = false;
315  invert = true;
316  break;
317  }
318  else if((jointIt->nodeIndex2 == it1->first &&
319  jointIt->nodeIndex1 == it2->first)) {
320  connected = true;
321  done = false;
322  break;
323  }
324  }
325  if(connected) {
326  NodeData node = it2->second;
327  utils::Vector v = it1->second.pos - it1->second.rot*it1->second.pivot;
328  node.pos = it1->second.rot.inverse() * (node.pos - v);
329  node.rot = it1->second.rot.inverse() * node.rot;
330 
331  // ToDo: - handle second axis for hing2 and universal
332  // - handle if we have to invert the axis depending on
333  // on the node order
334  ForwardTransform ft;
335  ft.name = jointIt->name;
336  if(jointIt->anchorPos == ANCHOR_NODE1) {
337  ft.anchor = it1->second.pos;
338  }
339  else if(jointIt->anchorPos == ANCHOR_NODE2) {
340  ft.anchor = it2->second.pos;
341  }
342  else if(jointIt->anchorPos == ANCHOR_CENTER) {
343  ft.anchor = (it1->second.pos + it2->second.pos) * 0.5;
344  }
345  else {
346  ft.anchor = jointIt->anchor;
347  }
348 
349  ft.anchor = it1->second.rot.inverse() * (ft.anchor - v);
350  ft.relPos = node.pos - ft.anchor;
351  ft.axis = it1->second.rot.inverse() * jointIt->axis1;
352  ft.axis.normalize();
353  ft.q = node.rot;
354  if(invert) {
355  ft.axis *= -1;
356  ft.value = -jointIt->angle1_offset;
357  ft.offset = -jointIt->angle1_offset;
358  }
359  else {
360  ft.value = jointIt->angle1_offset;
361  ft.offset = jointIt->angle1_offset;
362  }
363  ft.id = node.index;
364  ft.jointId = jointIt->index;
365  if(jointIt->type == JOINT_TYPE_SLIDER) {
366  ft.linear = true;
367  }
368  else {
369  ft.linear = false;
370  }
371 
372  jointMapByName[jointIt->name] = ft;
373  jointMapById[ft.jointId] = &jointMapByName[jointIt->name];
374  jointMapByNodeId[ft.id] = &jointMapByName[jointIt->name];
375  graphics->makeChild(it1->second.index, it2->second.index);
376  graphics->setDrawObjectPos(node.index, node.pos);
377  graphics->setDrawObjectRot(node.index, node.rot);
378  if(jointIt->type != JOINT_TYPE_FIXED) {
379  std::string packageName;
380  if(robotname.empty()) {
381  packageName = jointIt->name;
382  }
383  else {
384  packageName = robotname+"/"+jointIt->name;
385  }
386  data_broker::DataPackage dbPackage;
387  dbPackage.add("value", 0.0);
388  control->dataBroker->pushData("viz", packageName,
389  dbPackage, NULL,
390  data_broker::DATA_PACKAGE_READ_WRITE_FLAG);
391  control->dataBroker->registerSyncReceiver(this, "viz",
392  packageName,
393  ft.id);
394  }
395 
396  nodeMapReady[it2->first] = it2->second;
397  nodeMapI.erase(it2++);
398  }
399  else ++it2;
400  }
401  }
402  if(!done) break;
403  }
404  }
405  std::map<unsigned long, MotorData> motorMapById;
406  for(unsigned int i=0; i<load.motorList.size(); ++i) {
407  MotorData motor;
408  motor.fromConfigMap(&load.motorList[i], tmpPath, NULL);
409  motorMapById[motor.index] = motor;
410  }
411  if(!load.controllerList.empty()) {
412  ControllerData controller;
413  controller.fromConfigMap(&load.controllerList[0], tmpPath, NULL);
414  for(unsigned int i=0; i<controller.motors.size(); ++i) {
415  jointByControllerIdx.push_back(jointMapById[motorMapById[controller.motors[i]].jointIndex]);
416  }
417  }
418  }
419 
420  for(it1=nodeMapI.begin(); it1!=nodeMapI.end(); ++it1) {
421  nodeMapById[it1->second.index] = it1->second;
422  nodeMapByName[it1->second.name] = it1->second;
423  }
424  }

440  void Viz::setJointValue(ForwardTransform *joint, double value) {
441  joint->value = value;
442  if(joint->linear) {
443  utils::Vector v = joint->axis*joint->value + joint->relPos;
444  graphics->setDrawObjectPos(joint->id, joint->anchor+v);
445  }
446  else {
447  utils::Quaternion q = utils::angleAxisToQuaternion(joint->value+joint->offset,
448  joint->axis);
449  utils::Vector v = q * joint->relPos;
450  graphics->setDrawObjectPos(joint->id, joint->anchor+v);
451  graphics->setDrawObjectRot(joint->id, q * joint->q);
452  }
453  }
