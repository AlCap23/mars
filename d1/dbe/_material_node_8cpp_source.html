124  void MaterialNode::setTransparency(float t) {
125  if(!isCreated) return;
126  // handle transparency
127  transparencyUniform->set(1-t);
128  osg::StateSet *state = getOrCreateStateSet();
129  if (t >= 0.00001) {
130  //state->setRenderingHint(osg::StateSet::TRANSPARENT_BIN);
131  state->setMode(GL_ALPHA_TEST, osg::StateAttribute::ON);
132  state->setMode(GL_BLEND,osg::StateAttribute::ON);
133  //state->setRenderBinDetails(1, "TransparentBin");
134  state->setRenderBinDetails(1, "DepthSortedBin");
135  /*
136  depth = new osg::Depth;
137  depth->setWriteMask( false );
138  state->setAttributeAndModes(depth.get(),
139  osg::StateAttribute::ON | osg::StateAttribute::PROTECTED);
140  */
141  }
142  else {
143  if(depth.valid()) {
144  state->removeAttribute(depth.get());
145  }
146  state->setRenderingHint(osg::StateSet::DEFAULT_BIN);
147  state->setMode(GL_ALPHA_TEST, osg::StateAttribute::OFF);
148  state->setMode(GL_BLEND,osg::StateAttribute::OFF);
149  state->setRenderBinDetails(0, "RenderBin");
150  }

214  void MaterialNode::updateLights(std::vector<mars::interfaces::LightData*> &lightList) {
215  if(!isCreated) return;
216  std::vector<mars::interfaces::LightData*>::iterator it;
217  std::list<mars::interfaces::LightData*> lightList_;
218  std::list<mars::interfaces::LightData*>::iterator it2;
219  double dist1, dist2;
220  Vector position_;
221  getNodePosition(&position_);
222  for(it=lightList.begin(); it!=lightList.end(); ++it) {
223  dist1 = Vector((*it)->pos - position_).norm();
224  for(it2=lightList_.begin(); it2!=lightList_.end(); ++it2) {
225  dist2 = Vector((*it2)->pos - position_).norm();
226  if(dist1 < dist2) {
227  lightList_.insert(it2, *it);
228  break;
229  }
230  }
231  if(it2 == lightList_.end()) {
232  lightList_.push_back(*it);
233  }
234  }
235  int i=0;
236  for(it2=lightList_.begin(); it2!=lightList_.end() && i<maxNumLights; ++it2, ++i) {
237  /*
238  fprintf(stderr, "set light: %d\n", i);
239  fprintf(stderr, " pos: %g %g %g\n", (*it2)->pos.x(), (*it2)->pos.y(), (*it2)->pos.z());
240  fprintf(stderr, " ambient: %g %g %g %g\n", (*it2)->ambient.r, (*it2)->ambient.g,
241  (*it2)->ambient.b, (*it2)->ambient.a);
242  fprintf(stderr, " diffuse: %g %g %g %g\n", (*it2)->diffuse.r, (*it2)->diffuse.g,
243  (*it2)->diffuse.b, (*it2)->diffuse.a);
244  */
245  lightPosUniform->setElement(i, osg::Vec3f((*it2)->pos.x(),
246  (*it2)->pos.y(),
247  (*it2)->pos.z()));
248  lightAmbientUniform->setElement(i, osg::Vec4f((*it2)->ambient.r,
249  (*it2)->ambient.g,
250  (*it2)->ambient.b,
251  (*it2)->ambient.a));
252  lightDiffuseUniform->setElement(i, osg::Vec4f((*it2)->diffuse.r,
253  (*it2)->diffuse.g,
254  (*it2)->diffuse.b,
255  (*it2)->diffuse.a));
256  lightSpecularUniform->setElement(i, osg::Vec4f((*it2)->specular.r,
257  (*it2)->specular.g,
258  (*it2)->specular.b,
259  (*it2)->specular.a));
260  lightIsSpotUniform->setElement(i, (*it2)->type-1);
261  Vector v = (*it2)->lookAt;// - (*it2)->pos;
262  lightSpotDirUniform->setElement(i, osg::Vec3f(v.x(), v.y(), v.z()));
263  lightIsDirectionalUniform->setElement(i, (*it2)->directional);
264  lightConstantAttUniform->setElement(i, (float)(*it2)->constantAttenuation);
265  lightLinearAttUniform->setElement(i, (float)(*it2)->linearAttenuation);
266  lightQuadraticAttUniform->setElement(i, (float)(*it2)->quadraticAttenuation);
267  lightIsSetUniform->setElement(i, 1);
268  lightCosCutoffUniform->setElement(i, (float)cos((*it2)->angle*0.017453292519943));
269  lightSpotExponentUniform->setElement(i, (float)(*it2)->exponent);
270  }
271  for(; i<maxNumLights; ++i) {
272  lightIsSetUniform->setElement(i, 0);
273  }
274  }

303  void TangentVisitor::apply(osg::Node &searchNode){
304  // search for geometries and generate tangents for them
305  osg::Geode* geode=dynamic_cast<osg::Geode*>(&searchNode);
306  if(geode) {
307  for(unsigned int i=0; i<geode->getNumDrawables(); ++i) {
308  osg::Geometry* geom=dynamic_cast<osg::Geometry*>(geode->getDrawable(i));
309  if(geom) {
310  generateTangents(geom);
311  }
312  }
313  }
314 
315  traverse(searchNode);
316  }

318  void TangentVisitor::generateTangents(osg::Geometry *geom) {
319  osg::ref_ptr< osgUtil::TangentSpaceGenerator > tsg = new osgUtil::TangentSpaceGenerator;
320  // color map unit for texture coordinates
321  tsg->generate( geom, DEFAULT_UV_UNIT );
322  osg::Vec4Array *tangents = tsg->getTangentArray();
323  if(tangents==NULL || tangents->size()==0) {
324  std::cerr << "Failed to generate tangents for plane!" << std::endl;
325  }
326  else {
327 #if (OPENSCENEGRAPH_MAJOR_VERSION < 3 || ( OPENSCENEGRAPH_MAJOR_VERSION == 3 && OPENSCENEGRAPH_MINOR_VERSION < 2))
328  geom->setVertexAttribData( TANGENT_UNIT, osg::Geometry::ArrayData( tangents, osg::Geometry::BIND_PER_VERTEX ) );
329 #elif (OPENSCENEGRAPH_MAJOR_VERSION > 3 || (OPENSCENEGRAPH_MAJOR_VERSION == 3 && OPENSCENEGRAPH_MINOR_VERSION >= 2))
330  geom->setVertexAttribArray( TANGENT_UNIT,
331  tangents, osg::Array::BIND_PER_VERTEX );
332 #else
333 #error Unknown OSG Version
334 #endif
335  }
336  }

347  else {
348  traverse(searchNode);
