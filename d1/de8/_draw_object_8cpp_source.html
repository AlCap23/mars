112  void DrawObject::createObject(unsigned long id,
113  const Vector &pivot) {
114  id_ = id;
115  pivot_ = pivot;
116 
117  brightnessUniform = new osg::Uniform("brightness", 1.0f);
118  transparencyUniform = new osg::Uniform("alpha", 1.0f);
119  texScaleUniform = new osg::Uniform("texScale", 1.0f);
120  lineLaserPosUniform = new osg::Uniform("lineLaserPos", osg::Vec3f(0.0f, 0.0f, 0.0f));
121  lineLaserNormalUniform = new osg::Uniform("lineLaserNormal", osg::Vec3f(1.0f, 0.0f, 0.0f));
122  lineLaserColor = new osg::Uniform("lineLaserColor", osg::Vec4f(1.0f, 0.0f, 0.0f, 1.0f));
123  lineLaserDirection = new osg::Uniform("lineLaserDirection", osg::Vec3f(0.0f, 0.0f, 1.0f));
124  lineLaserOpeningAngle = new osg::Uniform("lineLaserOpeningAngle", (float)M_PI * 2.0f);
125 
126  scaleTransform_ = new osg::MatrixTransform();
127  scaleTransform_->setMatrix(osg::Matrix::scale(1.0, 1.0, 1.0));
128  scaleTransform_->setDataVariance(osg::Node::STATIC);
129 
130  posTransform_ = new osg::PositionAttitudeTransform();
131  posTransform_->setPivotPoint(osg::Vec3(pivot_.x(), pivot_.y(), pivot_.z()));
132  posTransform_->setPosition(osg::Vec3(0.0, 0.0, 0.0));
133  posTransform_->addChild(scaleTransform_.get());
134  posTransform_->setNodeMask(nodeMask_);
135 
136  group_ = new osg::Group;
137 
138  std::list< osg::ref_ptr< osg::Geode > > geodes = createGeometry();
139  for(std::list< osg::ref_ptr< osg::Geode > >::iterator it = geodes.begin();
140  it != geodes.end(); ++it) {
141  group_->addChild(it->get());
142  for(unsigned int i=0; i<it->get()->getNumDrawables(); ++i) {
143  osg::Drawable *draw = it->get()->getDrawable(i);
144  geometry_.push_back(draw->asGeometry());
145  }
146  }
147 
148  // get the size of the object
149  osg::ComputeBoundsVisitor cbbv;
150  group_->accept(cbbv);
151  osg::BoundingBox bb = cbbv.getBoundingBox();
152  if(fabs(bb.xMax()) > fabs(bb.xMin())) {
153  geometrySize_.x() = fabs(bb.xMax() - bb.xMin());
154  } else {
155  geometrySize_.x() = fabs(bb.xMin() - bb.xMax());
156  }
157  if(fabs(bb.yMax()) > fabs(bb.yMin())) {
158  geometrySize_.y() = fabs(bb.yMax() - bb.yMin());
159  } else {
160  geometrySize_.y() = fabs(bb.yMin() - bb.yMax());
161  }
162  if(fabs(bb.zMax()) > fabs(bb.zMin())) {
163  geometrySize_.z() = fabs(bb.zMax() - bb.zMin());
164  } else {
165  geometrySize_.z() = fabs(bb.zMin() - bb.zMax());
166  }
167 
168  scaleTransform_->addChild(group_.get());
169  }

377  void DrawObject::generateTangents()
378  {
379  osg::ref_ptr< osgUtil::TangentSpaceGenerator > tsg = new osgUtil::TangentSpaceGenerator;
380  for(std::list< osg::ref_ptr<osg::Geometry> >::iterator it = geometry_.begin();
381  it != geometry_.end(); ++it) {
382  osg::ref_ptr<osg::Geometry> geom = *it;
383  if(geom->empty()) {
384  cerr << "WARNING: empty geometry for DrawObject " << id_ << "!" << endl;
385  }
386  tsg->generate( geom.get(), DEFAULT_UV_UNIT );
387  osg::Vec4Array *tangents = tsg->getTangentArray();
388  if(tangents==NULL || tangents->size()==0) {
389  cerr << "Failed to generate tangents for DrawObject " << id_ << "!" << endl;
390  return;
391  }
392 
393 
394 #if (OPENSCENEGRAPH_MAJOR_VERSION < 3 || ( OPENSCENEGRAPH_MAJOR_VERSION == 3 && OPENSCENEGRAPH_MINOR_VERSION < 2))
395  geom->setVertexAttribData( TANGENT_UNIT, osg::Geometry::ArrayData( tangents, osg::Geometry::BIND_PER_VERTEX ) );
396 #elif (OPENSCENEGRAPH_MAJOR_VERSION > 3 || (OPENSCENEGRAPH_MAJOR_VERSION == 3 && OPENSCENEGRAPH_MINOR_VERSION >= 2))
397  geom->setVertexAttribArray( TANGENT_UNIT,
398  tangents, osg::Array::BIND_PER_VERTEX );
399 #else
400  #error Unknown OSG Version
401 #endif
402  }
403  }

466  void DrawObject::updateShader(
467  vector<mars::interfaces::LightData*> &lightList,
468  bool reload,
469  const std::map<mars::interfaces::ShaderType, std::string> &shaderSources) {
470 
471  //return;
472  if(!useMARSShader || !getLight) return;
473  if(shaderSources.empty()) {
474  if(reload && hasShaderSources) {
475  // no need to regenerate, shader source did not changed
476  return;
477  }
478  }
479  else {
480  hasShaderSources = true;
481  }
482 
483  if(shaderSources.empty()) { // generate shader!
484  ShaderGenerator shaderGenerator;
485  vector<string> args;
486 
487  bool hasTexture = (colorMap_.valid()
488  || normalMap_.valid());
489 
490  {
491  ShaderFunc *vertexShader = new ShaderFunc;
492  vertexShader->addUniform( (GLSLUniform)
493  { "mat4", "osg_ViewMatrix" } );
494  vertexShader->addExport( (GLSLExport)
495  {"gl_Position", "gl_ModelViewProjectionMatrix * gl_Vertex"} );
496  vertexShader->addExport( (GLSLExport) {"gl_ClipVertex", "v"} );
497  if(hasTexture) {
498  vertexShader->addExport( (GLSLExport)
499  { "gl_TexCoord[0].xy", "gl_MultiTexCoord0.xy" });
500  }
501  shaderGenerator.addShaderFunction(vertexShader, mars::interfaces::SHADER_TYPE_VERTEX);
502  }
503 
504  {
505  ShaderFunc *fragmentShader = new ShaderFunc;
506  if(colorMap_.valid()) {
507  fragmentShader->addUniform( (GLSLUniform)
508  { "float", "texScale" } );
509  fragmentShader->addUniform( (GLSLUniform)
510  { "sampler2D", "BaseImage" } );
511  if(bumpMap_.valid()) {
512  fragmentShader->addUniform( (GLSLUniform)
513  { "sampler2D", "BumpMap" } );
514  fragmentShader->addMainVar( (GLSLVariable)
515  { "vec2", "texCoord", "gl_TexCoord[0].xy*texScale + texture2D(BumpMap, gl_TexCoord[0].xy*texScale).x*0.01*eyeVec.xy" });
516 
517  }
518  else {
519  fragmentShader->addMainVar( (GLSLVariable)
520  { "vec2", "texCoord", "gl_TexCoord[0].xy*texScale" });
521  }
522 
523  fragmentShader->addMainVar( (GLSLVariable)
524  { "vec4", "col", "texture2D( BaseImage, texCoord)" });
525 
526  } else {
527  fragmentShader->addMainVar( (GLSLVariable)
528  { "vec4", "col", "vec4(1.0)" });
529  }
530  fragmentShader->addExport( (GLSLExport) {"gl_FragColor", "col"} );
531  shaderGenerator.addShaderFunction(fragmentShader, mars::interfaces::SHADER_TYPE_FRAGMENT);
532  }
533 
534  args.clear();
535  args.push_back("v");
536  PixelLightVert *plightVert = new PixelLightVert(args, lightList, drawLineLaser, marsShadow);
537  plightVert->addMainVar( (GLSLVariable)
538  { "vec4", "v", "gl_ModelViewMatrix * gl_Vertex" } );
539  plightVert->addExport( (GLSLExport)
540  { "normalVarying", "normalize( gl_NormalMatrix * gl_Normal )" } );
541  plightVert->addVarying( (GLSLVarying)
542  { "vec3", "normalVarying" } );
543  shaderGenerator.addShaderFunction(plightVert, mars::interfaces::SHADER_TYPE_VERTEX);
544 
545  if(normalMap_.valid()) {
546  args.clear();
547  args.push_back("gl_NormalMatrix * gl_Normal");
548  BumpMapVert *bumpVert = new BumpMapVert(args, lightList);
549  shaderGenerator.addShaderFunction(bumpVert, mars::interfaces::SHADER_TYPE_VERTEX);
550 
551  args.clear();
552  args.push_back("texture2D( NormalMap, texCoord )");
553  args.push_back("n");
554  BumpMapFrag *bumpFrag = new BumpMapFrag(args);
555  bumpFrag->addUniform( (GLSLUniform) { "sampler2D", "NormalMap" } );
556  shaderGenerator.addShaderFunction(bumpFrag, mars::interfaces::SHADER_TYPE_FRAGMENT);
557  }
558 
559  args.clear();
560  args.push_back("col");
561  args.push_back("n");
562  args.push_back("col");
563  PixelLightFrag *plightFrag = new PixelLightFrag(args, useFog,
564  useNoise, drawLineLaser,
565  marsShadow,
566  lightList);
567  // invert the normal if gl_FrontFacing=true to handle back faces
568  // correctly.
569  // TODO: check not needed if backfaces not processed.
570  plightFrag->addMainVar( (GLSLVariable)
571  { "vec3", "n", "normalize( gl_FrontFacing ? normalVarying : -normalVarying )"} );
572  plightFrag->addVarying( (GLSLVarying)
573  { "vec3", "normalVarying" } );
