97  void DrawObject::createObject(unsigned long id,
98  const Vector &pivot,
99  unsigned long sharedID) {
100  id_ = id;
101  pivot_ = pivot;
102 
103  scaleTransform_ = new osg::MatrixTransform();
104  scaleTransform_->setMatrix(osg::Matrix::scale(1.0, 1.0, 1.0));
105  scaleTransform_->setDataVariance(osg::Node::STATIC);
106 
107  posTransform_ = new osg::PositionAttitudeTransform();
108  posTransform_->setPivotPoint(osg::Vec3(pivot_.x(), pivot_.y(), pivot_.z()));
109  posTransform_->setPosition(osg::Vec3(0.0, 0.0, 0.0));
110  posTransform_->addChild(scaleTransform_.get());
111  posTransform_->setNodeMask(nodeMask_);
112 
113  group_ = new osg::Group;
114  if(sharedID) {
115  materialNode = g->getSharedStateGroup(sharedID);
116  if(materialNode.valid()) {
117  sharedStateGroup = true;
118  }
119  else {
120  sharedStateGroup = false;
121  }
122  }
123 
124  std::list< osg::ref_ptr< osg::Geode > > geodes = createGeometry();
125  for(std::list< osg::ref_ptr< osg::Geode > >::iterator it = geodes.begin();
126  it != geodes.end(); ++it) {
127  group_->addChild(it->get());
128  for(unsigned int i=0; i<it->get()->getNumDrawables(); ++i) {
129  osg::Drawable *draw = it->get()->getDrawable(i);
130  geometry_.push_back(draw->asGeometry());
131  }
132  }
133 
134  // get the size of the object
135  osg::ComputeBoundsVisitor cbbv;
136  group_->accept(cbbv);
137  osg::BoundingBox bb = cbbv.getBoundingBox();
138  if(fabs(bb.xMax()) > fabs(bb.xMin())) {
139  geometrySize_.x() = fabs(bb.xMax() - bb.xMin());
140  } else {
141  geometrySize_.x() = fabs(bb.xMin() - bb.xMax());
142  }
143  if(fabs(bb.yMax()) > fabs(bb.yMin())) {
144  geometrySize_.y() = fabs(bb.yMax() - bb.yMin());
145  } else {
146  geometrySize_.y() = fabs(bb.yMin() - bb.yMax());
147  }
148  if(fabs(bb.zMax()) > fabs(bb.zMin())) {
149  geometrySize_.z() = fabs(bb.zMax() - bb.zMin());
150  } else {
151  geometrySize_.z() = fabs(bb.zMin() - bb.zMax());
152  }
153 
154  if(lod.valid()) {
155  scaleTransform_->addChild(lod.get());
156  }
157  else {
158  scaleTransform_->addChild(group_.get());
159  }
160  }

162  void DrawObject::addLODGeodes(std::list< osg::ref_ptr< osg::Geode > > geodes,
163  float start, float end) {
164  std::list< osg::ref_ptr< osg::Geode > >::iterator it;
165  if(!lod.valid()) {
166  lod = new osg::LOD();
167  }
168  for(it=geodes.begin(); it!=geodes.end(); ++it) {
169  lod->addChild(it->get(), start, end);
170  for(unsigned int i=0; i<it->get()->getNumDrawables(); ++i) {
171  osg::Drawable *draw = it->get()->getDrawable(i);
172  geometry_.push_back(draw->asGeometry());
173  }
174  }

194  void DrawObject::exportModel(const std::string &filename) {
195  // note obj export ignores stateset's of groups
196  if(materialNode.valid()) {
197  osg::ref_ptr<osg::Group> tmpNode = new osg::Group;
198  osg::ref_ptr<osg_material_manager::OsgMaterial> material = materialNode->getMaterial();
199  if(material.valid()) {
200  osg::StateSet *set = material->getStateSet();
201  tmpNode->setStateSet(set);
202  }
203  tmpNode->getOrCreateStateSet()->merge(*(materialNode->getStateSet()));
204  tmpNode->addChild(posTransform_.get());
205  osgDB::writeNodeFile(*(tmpNode.get()), filename.data());
206  }

272  void DrawObject::setSelected(bool val) {
273  selected_ = val;
274  if(selectable_) {
275  osg::PolygonMode *polyModeObj;
276 
277  osg::StateSet *state = group_->getOrCreateStateSet();
278  polyModeObj = dynamic_cast<osg::PolygonMode*>
279  (state->getAttribute(osg::StateAttribute::POLYGONMODE));
280 
281  if (!polyModeObj){
282  polyModeObj = new osg::PolygonMode;
283  state->setAttribute( polyModeObj );
284  }
285 
286  if(val && showSelected) {
287  if(!selectShader.valid()) {
288  selectShader = new osg::Program;
289  char fragmentSource[]=
290  "void main(void)\n"
291  "{\n"
292  " gl_FragColor = vec4(0, 0.75, 0, 1);\n"
293  "}\n";
294  selectShader->addShader(new osg::Shader(osg::Shader::FRAGMENT, fragmentSource));
295  }
296  state->setAttributeAndModes(selectShader.get(), osg::StateAttribute::ON);
297  osg::LineWidth *lw = new osg::LineWidth(2);
298  state->setAttributeAndModes(lw, osg::StateAttribute::ON);
299  polyModeObj->setMode(osg::PolygonMode::FRONT_AND_BACK, osg::PolygonMode::LINE);
300  //state->setAttributeAndModes(material_.get(), osg::StateAttribute::OFF);
301 
302  state->setAttributeAndModes(selectionMaterial.get(),
303  osg::StateAttribute::ON | osg::StateAttribute::OVERRIDE);
304  state->setMode(GL_LIGHTING,
305  osg::StateAttribute::OFF | osg::StateAttribute::PROTECTED | osg::StateAttribute::OVERRIDE);
306  state->setMode(GL_FOG, osg::StateAttribute::OFF | osg::StateAttribute::OVERRIDE);
307 
308  }
309  else {
310  if(selectShader.valid()) {
311  state->removeAttribute(selectShader.get());
312  }
313  polyModeObj->setMode(osg::PolygonMode::FRONT_AND_BACK, osg::PolygonMode::FILL);
314 
315  state->removeAttribute(selectionMaterial.get());
316  //state->setAttributeAndModes(material_.get(), osg::StateAttribute::ON);
317  state->removeMode(GL_LIGHTING);
318  state->removeMode(GL_FOG);
319 
320  }
321  }
322  for(auto it:selectionChilds) {
323  it->setSelected(val);
324  }
325  }
