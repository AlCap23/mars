119  template<typename Key, typename T>
120  FIFOMap<Key, T>& FIFOMap<Key, T>::operator=(const FIFOMap<Key, T> &other) {
121  if(this == &other)
122  return *this;

124  std::map<Key, T>::operator=(other);

126  FIFOItem<Key, T> newItem(it->first,
