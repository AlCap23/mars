183  void Joint6DOFSensor::receiveData(const data_broker::DataInfo &info,
184  const data_broker::DataPackage &package,
185  int callbackParam) {
186  if(nodePackageId == info.dataId) {
187  package.get(nodeRotIndices[0], &sensor_data.body_q.x());
188  package.get(nodeRotIndices[1], &sensor_data.body_q.y());
189  package.get(nodeRotIndices[2], &sensor_data.body_q.z());
190  package.get(nodeRotIndices[3], &sensor_data.body_q.w());
191  sensor_data.body_q.x() = -sensor_data.body_q.x();
192  sensor_data.body_q.y() = -sensor_data.body_q.y();
193  sensor_data.body_q.z() = -sensor_data.body_q.z();
194  } else if(jointPackageId == info.dataId) {
195  for(int i = 0; i < 3; ++i) {
196  package.get(jointForceIndices[i], &sensor_data.force[i]);
197  package.get(jointTorqueIndices[i], &sensor_data.torque[i]);
198  }
199  } else {
200  // we don't know the dataId yet, so assign the ids...
201  switch(callbackParam) {
202  case CALLBACK_NODE:
203  nodePackageId = info.dataId;
204  nodeRotIndices[0] = package.getIndexByName("rotation/x");
205  nodeRotIndices[1] = package.getIndexByName("rotation/y");
206  nodeRotIndices[2] = package.getIndexByName("rotation/z");
207  nodeRotIndices[3] = package.getIndexByName("rotation/w");
208  break;
209  case CALLBACK_JOINT:
210  jointPackageId = info.dataId;
211  jointForceIndices[0] = package.getIndexByName("force1/x");
212  jointForceIndices[1] = package.getIndexByName("force1/y");
213  jointForceIndices[2] = package.getIndexByName("force1/z");
214  jointTorqueIndices[0] = package.getIndexByName("torque1/x");
215  jointTorqueIndices[1] = package.getIndexByName("torque1/y");
216  jointTorqueIndices[2] = package.getIndexByName("torque1/z");
217  break;
218  default:
219  // Error: We are not interested in this package. Why did we get it?
220  return;
221  }
222  // ...and call this method again.
223  receiveData(info, package, callbackParam);
224  }
225  }
