48  void MainCfgGui::setupGUI(std::string path) {

53  if(cfg) {
54  cfg->registerToCFG(dynamic_cast<CFGClient*>(this));
55 
56  cfgPropertyStruct r_path;

68  gui->addGenericMenuAction("../Windows/GuiCfg", 1,
69  dynamic_cast<main_gui::MenuInterface*>(this),0,

72  cfgWidget = new CfgWidget(cfg);

75 
76  if (cfg)
77  setupCFG();
78

189  void MainCfgGui::geometryChanged() {
190  bool update_cfg = false;
191 
192  if (ignore_next) {
193  ignore_next = false;
194  return;
195  }
196 
197  int top = cfgWidget->pDialog->geometry().y();
198  int left = cfgWidget->pDialog->geometry().x();
199  int width = cfgWidget->pDialog->geometry().width();
200  int height = cfgWidget->pDialog->geometry().height();
201 
202  if(top != cfgW_top.iValue) {
203  cfgW_top.iValue = top;
204  update_cfg = true;
205  }
206  if(left != cfgW_left.iValue) {
207  cfgW_left.iValue = left;
208  update_cfg = true;
209  }
210  if(width != cfgW_width.iValue) {
211  cfgW_width.iValue = width;
212  update_cfg = true;
213  }
214  if(height != cfgW_height.iValue) {
215  cfgW_height.iValue = height;
216  update_cfg = true;
217  }
218  if(update_cfg && cfg) {
219  set_window_prop = true;
220  cfg->setProperty(cfgW_top);
221  cfg->setProperty(cfgW_left);
222  cfg->setProperty(cfgW_width);
223  cfg->setProperty(cfgW_height);
224  set_window_prop = false;
225  }
