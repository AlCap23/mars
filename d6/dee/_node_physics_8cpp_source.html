137  bool NodePhysics::createNode(NodeData* node) {
138 #ifdef _VERIFY_WORLD_
139  sRotation euler = utils::quaternionTosRotation(node->rot);
140  fprintf(stderr, "node %d ; %.4f, %.4f, %.4f ; %.4f, %.4f, %.4f ; %.4f ; %.4f\n",
141  node->index, node->pos.x(), node->pos.y(),
142  node->pos.z(), euler.alpha, euler.beta, euler.gamma,
143  node->mass, node->density);
144 #endif
145  MutexLocker locker(&(theWorld->iMutex));
146  if(theWorld && theWorld->existsWorld()) {
147  bool ret;
148  //LOG_DEBUG("physicMode %d", node->physicMode);
149  // first we create a ode geometry for the node
150  switch(node->physicMode) {
151  case NODE_TYPE_MESH:
152  ret = createMesh(node);
153  break;
154  case NODE_TYPE_BOX:
155  ret = createBox(node);
156  break;
157  case NODE_TYPE_SPHERE:
158  ret = createSphere(node);
159  break;
160  case NODE_TYPE_CAPSULE:
161  ret = createCapsule(node);
162  break;
163  case NODE_TYPE_CYLINDER:
164  ret = createCylinder(node);
165  break;
166  case NODE_TYPE_PLANE:
167  ret = createPlane(node);
168  break;
169  case NODE_TYPE_TERRAIN:
170  ret = createHeightfield(node);
171  break;
172  default:
173  // no correct type is spezified, so no physically node will be created
174  return 0;
175  break;
176  }
177  if(ret == 0) {
178  // Error createing the physical Node
179  return 0;
180  }
181 
182  // then, if the geometry was sucsessfully build, we can create a
183  // body for the node or add the node to an existing body
184  if(node->movable) setProperties(node);
185  else if(node->physicMode != NODE_TYPE_PLANE) {
186  dQuaternion tmp, t1, t2;
187  tmp[1] = (dReal)node->rot.x();
188  tmp[2] = (dReal)node->rot.y();
189  tmp[3] = (dReal)node->rot.z();
190  tmp[0] = (dReal)node->rot.w();
191  dGeomSetPosition(nGeom, (dReal)node->pos.x(),
192  (dReal)node->pos.y(), (dReal)node->pos.z());
193  if(node->physicMode == NODE_TYPE_TERRAIN) {
194  dGeomGetQuaternion(nGeom, t1);
195  dQMultiply0(t2, tmp, t1);
196  dGeomSetQuaternion(nGeom, t2);
197  }
198  else
199  dGeomSetQuaternion(nGeom, tmp);
200  }
201  node_data.id = node->index;
202  dGeomSetData(nGeom, &node_data);
203  locker.unlock();
204  setContactParams(node->c_params);
205  return 1;
206  }
207  LOG_WARN("NodePhysics: tried to create a Node without there being a world.");
208  return 0;
209  }

509  const Quaternion NodePhysics::setRotation(const Quaternion &q, bool move_group) {
510  dQuaternion tmp, tmp2, tmp3, tmp4, tmp5;
511  const dReal *brot, *bpos, *gpos;
512  Quaternion q2;
513  dMatrix3 R;
514  dVector3 pos, new_pos, new2_pos;
515  MutexLocker locker(&(theWorld->iMutex));
516 
517  pos[0] = pos[1] = pos[2] = 0;
518  tmp[1] = (dReal)q.x();
519  tmp[2] = (dReal)q.y();
520  tmp[3] = (dReal)q.z();
521  tmp[0] = (dReal)q.w();
522  if(nBody) {
523  bpos = dBodyGetPosition(nBody);
524  pos[0] = bpos[0];
525  pos[1] = bpos[1];
526  pos[2] = bpos[2];
527  brot = dBodyGetQuaternion(nBody);
528  tmp4[0] = brot[0];
529  tmp4[1] = brot[1];
530  tmp4[2] = brot[2];
531  tmp4[3] = brot[3];
532  if(composite && !move_group) {
533  dGeomGetQuaternion(nGeom, tmp2);
534  dGeomSetOffsetWorldQuaternion(nGeom, tmp);
535  }
536  else if (composite) {
537  dGeomGetQuaternion(nGeom, tmp2);
538  dGeomGetOffsetQuaternion(nGeom, tmp3);
539  tmp3[1] *= -1;
540  tmp3[2] *= -1;
541  tmp3[3] *= -1;
542  dQMultiply0(tmp5, tmp, tmp3);
543  dBodySetQuaternion(nBody, tmp5);
544  }
545  else {
546  tmp2[0] = brot[0];
547  tmp2[1] = brot[1];
548  tmp2[2] = brot[2];
549  tmp2[3] = brot[3];
550  dBodySetQuaternion(nBody, tmp);
551  }
552  }
553  else if(nGeom) {
554  dGeomGetQuaternion(nGeom, tmp2);
555  dGeomSetQuaternion(nGeom, tmp);
556  }
557  dQMultiply2(tmp3, tmp, tmp2);
558  q2.x() = (sReal)tmp3[1];
559  q2.y() = (sReal)tmp3[2];
560  q2.z() = (sReal)tmp3[3];
561  q2.w() = (sReal)tmp3[0];
562  if(nBody && composite && move_group) {
563  gpos = dGeomGetOffsetPosition(nGeom);
564  dQtoR(tmp4, R);
565  dMULTIPLY0_331(new_pos, R, gpos);
566  pos[0] += new_pos[0];
567  pos[1] += new_pos[1];
568  pos[2] += new_pos[2];
569  new_pos[0] *= -1;
570  new_pos[1] *= -1;
571  new_pos[2] *= -1;
572  dQtoR(tmp3, R);
573  dMULTIPLY0_331(new2_pos, R, new_pos);
574  new_pos[0] = pos[0]+new2_pos[0];
575  new_pos[1] = pos[1]+new2_pos[1];
576  new_pos[2] = pos[2]+new2_pos[2];
577  dBodySetPosition(nBody, new_pos[0], new_pos[1], new_pos[2]);
578  }
579  return q2;
580  }

606  bool NodePhysics::createMesh(NodeData* node) {
607  int i;
608 
609  if (!node->inertia_set &&
610  (node->ext.x() <= 0 || node->ext.y() <= 0 || node->ext.z() <= 0)) {
611  LOG_ERROR("Cannot create Node \"%s\" (id=%lu):\n"
612  " Mesh Nodes must have ext.x(), ext.y(), and ext.z() > 0.\n"
613  " Current values are: x=%g; y=%g, z=%g",
614  node->name.c_str(), node->index,
615  node->ext.x(), node->ext.y(), node->ext.z());
616  return false;
617  }
618 
619  myVertices = (dVector3*)calloc(node->mesh.vertexcount, sizeof(dVector3));
620  myIndices = (dTriIndex*)calloc(node->mesh.indexcount, sizeof(dTriIndex));
621  //LOG_DEBUG("%d %d", node->mesh.vertexcount, node->mesh.indexcount);
622  // first we have to copy the mesh data to prevent errors in case
623  // of double to float conversion
624  for(i=0; i<node->mesh.vertexcount; i++) {
625  myVertices[i][0] = (dReal)node->mesh.vertices[i][0];
626  myVertices[i][1] = (dReal)node->mesh.vertices[i][1];
627  myVertices[i][2] = (dReal)node->mesh.vertices[i][2];
628  }
629  for(i=0; i<node->mesh.indexcount; i++) {
630  myIndices[i] = (dTriIndex)node->mesh.indices[i];
631  }
632 
633  // then we can build the ode representation
634  myTriMeshData = dGeomTriMeshDataCreate();
635  dGeomTriMeshDataBuildSimple(myTriMeshData, (dReal*)myVertices,
636  node->mesh.vertexcount,
637  myIndices, node->mesh.indexcount);
638  nGeom = dCreateTriMesh(theWorld->getSpace(), myTriMeshData, 0, 0, 0);
639 
640  // at this moment we set the mass properties as the mass of the
641  // bounding box if no mass and inertia is set by the user
642  if(node->inertia_set) {
643  setInertiaMass(node);
644  }
645  else if(node->density > 0) {
646  dMassSetBox(&nMass, (dReal)node->density, (dReal)node->ext.x(),
647  (dReal)node->ext.y(),(dReal)node->ext.z());
648  }
649  else if(node->mass > 0) {
650  dMassSetBoxTotal(&nMass, (dReal)node->mass, (dReal)node->ext.x(),
651  (dReal)node->ext.y(),(dReal)node->ext.z());
652  }
653  return true;
