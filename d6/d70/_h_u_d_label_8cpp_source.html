125  void HUDLabel::setText(std::string text, double color[4]) {
126  if(labelText == 0) {
127  osg::Vec3 t_pos;
128  osg::ref_ptr<osg::Geode> node;
129  osg::ref_ptr<osg::PositionAttitudeTransform> transform;
130 
131  node = createLabel(text, color);
132 
133  transform = new osg::PositionAttitudeTransform();
134  transform->addChild(node.get());
135 
136  scaleTransform->addChild(transform.get());
137  //scaleTransform->addChild(node.get());
138 
139  osg::BoundingBox bb;
140  bb.expandBy(node->getDrawable(0)->getBound());
141  float d = bb.zMin()-2.0;
142  float w = bb.xMax() - bb.xMin() + pl + pr;
143  float h = bb.yMax() - bb.yMin() + pt + pb;
144 
145  double pos;
146  if(direction == 1) pos = posx - w;
147  else if(direction == 2) pos = posx - w*0.5;
148  else pos = posx;
149 
150  transform->setPosition(osg::Vec3(pos, posy, -1.5f));
151 
152 
153  // now get the bounding box
154  if(background_color[3] > 0.0001) {
155  osg::ref_ptr<osg::Geode> geode = new osg::Geode;
156  osg::Geometry* geom = new osg::Geometry;
157 
158  osg::Vec3Array* vertices = new osg::Vec3Array;
159  vertices->push_back(osg::Vec3(pos-pl, posy-h+pt, d));
160  vertices->push_back(osg::Vec3(pos-pl, posy+pt, d));
161  vertices->push_back(osg::Vec3(pos+w-pl, posy+pt, d));
162  vertices->push_back(osg::Vec3(pos+w-pl, posy-h+pt, d));
163  geom->setVertexArray(vertices);
164 
165  osg::Vec3Array* normals = new osg::Vec3Array;
166  normals->push_back(osg::Vec3(0.0f,0.0f,1.0f));
167  geom->setNormalArray(normals);
168  geom->setNormalBinding(osg::Geometry::BIND_OVERALL);
169 
170  osg::Vec4Array* colors = new osg::Vec4Array;
171  colors->push_back(osg::Vec4(background_color[0], background_color[1],
172  background_color[2], background_color[3]));
173  geom->setColorArray(colors);
174  geom->setColorBinding(osg::Geometry::BIND_OVERALL);
175 
176  geom->addPrimitiveSet(new osg::DrawArrays(GL_QUADS,0,4));
177 
178  osg::StateSet* stateset = geom->getOrCreateStateSet();
179  stateset->setMode(GL_LIGHTING,osg::StateAttribute::OFF);
180  stateset->setMode(GL_BLEND,osg::StateAttribute::ON);
181  stateset->setRenderingHint(osg::StateSet::TRANSPARENT_BIN);
182  stateset->setMode(GL_DEPTH_TEST, osg::StateAttribute::OFF);
183  stateset->setRenderBinDetails(21, "RenderBin");
184  geode->addDrawable(geom);
185  scaleTransform->addChild(geode.get());
186  }
187 
188  if(border_width > 0.0) {
189  osg::ref_ptr<osg::Geometry> linesGeom = new osg::Geometry;
190  osg::ref_ptr<osg::Geode> linesGeode = new osg::Geode;
191  //nodemanager tempnode;
192 
193  osg::Vec3Array* vertices = new osg::Vec3Array(8);
194  (*vertices)[0].set(pos-pl, posy-h+pt, d);
195  (*vertices)[1].set(pos-pl, posy+pt, d);
196  (*vertices)[2].set(pos-pl, posy+pt, d);
197  (*vertices)[3].set(pos+w-pl, posy+pt, d);
198  (*vertices)[4].set(pos+w-pl, posy+pt, d);
199  (*vertices)[5].set(pos+w-pl, posy-h+pt, d);
200  (*vertices)[6].set(pos+w-pl, posy-h+pt, d);
201  (*vertices)[7].set(pos-pl, posy-h+pt, d);
202 
203  // pass the created vertex array to the points geometry object.
204  linesGeom->setVertexArray(vertices);
205 
206  // set the colors as before, plus using the above
207  osg::Vec4Array* colors2 = new osg::Vec4Array;
208  colors2->push_back(osg::Vec4(border_color[0], border_color[1],
209  border_color[2], border_color[3]));
210  linesGeom->setColorArray(colors2);
211  linesGeom->setColorBinding(osg::Geometry::BIND_OVERALL);
212 
213  // set the normal in the same way color.
214  osg::Vec3Array* normals = new osg::Vec3Array;
215  normals->push_back(osg::Vec3(0.0f,0.0f,1.0f));
216  linesGeom->setNormalArray(normals);
217  linesGeom->setNormalBinding(osg::Geometry::BIND_OVERALL);
218 
219  // since we know up front,
220  linesGeom->addPrimitiveSet(new osg::DrawArrays(osg::PrimitiveSet::LINES,
221  0,8));
222 
223  osg::ref_ptr<osg::LineWidth> linew = new osg::LineWidth(border_width);
224  linesGeom->getOrCreateStateSet()->setAttributeAndModes(linew.get(),
225  osg::StateAttribute::ON);
226  linesGeom->getOrCreateStateSet()->setRenderBinDetails(20, "RenderBin");
227 
228  // add the points geometry to the geode.
229  linesGeode->addDrawable(linesGeom.get());
230  scaleTransform->addChild(linesGeode.get());
231  }
232  }
233  else {
234  labelText->setText(text);
235  labelText->setColor(osg::Vec4(color[0], color[1], color[2], color[3]));
236  }
237  }
