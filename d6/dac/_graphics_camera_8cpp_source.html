316  void GraphicsCamera::getCameraInfo(cameraStruct *s) {
317  if (!s) return;
318  osg::Quat q = myCameraMatrix.getRotate();
319  s->rot.x() = q.x();
320  s->rot.y() = q.y();
321  s->rot.z() = q.z();
322  s->rot.w() = q.w();
323 
324  osg::Vec3d t=myCameraMatrix.getTrans();
325  s->pos[0] = t.x();
326  s->pos[1] = t.y();
327  s->pos[2] = t.z();
328 
329  // get the intrinsic parameters of the camera
330  double fovy, aspectRatio, Zn, Zf;
331  getOSGCamera()->getProjectionMatrixAsPerspective( fovy, aspectRatio, Zn, Zf );
332 
333  // center points are just the center of the image
334  s->center_x = width / 2.0;
335  s->center_y = height / 2.0;
336 
337  // the f_x and f_y paramters are calculated from
338  // the fovy parameter we got from osg.
339  // We set the scale value such that a
340  // y value of tan(fovy/2) is equal to height/2.
341  //
342  // Note that fovy is in radians.
343  double fovy_rad = fovy * M_PI / 180.0;
344  s->scale_y = height / (2.0 * tan( fovy_rad / 2.0 ) );
345  s->scale_x = width / (2.0 * aspectRatio * tan( fovy_rad / 2.0 ) );
346  }

500  void GraphicsCamera::lookAtIso(double x, double y, double z)
501  {
502  //set to iso camera if not set
503  if(camera !=ISO_CAM){
504  setCamera(ISO_CAM);
505  }
506 
507  if(z > isoMinHeight && z < isoMaxHeight){d_zp = z;}
508 
509  //calculate horizontal forward vector
510  osg::Vec3 dirTMP(0, 1, 0);
511  dirTMP = dirTMP*cameraRotation;
512  osg::Vec3 dir(dirTMP.x(), dirTMP.y(), 0);
513 
514  //calculate horizontal distance to targetPoint
515  //camera angle
516  double distance = d_zp * tan(osg::DegreesToRadians(30.0));
517  dir.normalize();
518  //go backwards
519  dir *=-distance;
520  d_xp = x + dir.x();
521  d_yp = y + dir.y();
522 
523  osg::Quat q;
524  q = cameraRotation.getRotate();
525  updateViewportQuat(d_xp, d_yp, d_zp, q.x(), q.y(), q.z(), q.w());

587  void GraphicsCamera::calcEyeSep(void) {
588  eyeSep = (f_focal*separation)/20;
589 
590  f_left[0] = ((-f_ratio*f_nearPlane* tan(f_aperture))
591  -0.5*eyeSep*f_nearPlane/f_focal);
592  f_right[0] = ((f_ratio*f_nearPlane*tan(f_aperture))
593  -0.5*eyeSep*f_nearPlane/f_focal);
594  f_left[1] = ((-f_ratio*f_nearPlane*tan(f_aperture))
595  +0.5*eyeSep*f_nearPlane/f_focal);
596  f_right[1] = ((f_ratio*f_nearPlane*tan(f_aperture))
597  +0.5*eyeSep*f_nearPlane/f_focal);
