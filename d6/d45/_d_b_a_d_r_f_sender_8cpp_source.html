54  ConfigMap map;

56  ConfigVector::iterator it, it2;

113  void DBADRFSender::initADRF() {
114  if(tryToUseMulti) {
115  // try to use adrf from multisim
116  MultiSimPlugin *multi = libManager->getLibraryAs<MultiSimPlugin>("MultiSimPlugin");
117  if(multi) {
118  if((dataSpace = multi->getDataSpace())) {
119  dpm = multi->getDPManager();
120 
121  useMulti = true;
122  initialised = true;
123  fprintf(stderr, "DBADRFSender: using adrf from multisim-plugin\n");
124  return;
125  }
126  }
127  }
128 
129  dataSpaces = new DataSpace*[dstNrOfDataSpaceTypes];
130 
131  for (int i=0; i < dstNrOfDataSpaceTypes; ++i) {
132  dataSpaces[i] = NULL;
133  }
134 
135  dataSpaces[dstDefault] = new DataSpace(dstDefault);
136  dataSpace = dataSpaces[dstDefault];
137  dataSpace->setVersion(dataSpaceVersion);
138 
139  // start all (at the moment one) dataSpaces
140  for (int i=0; i < dstNrOfDataSpaceTypes; ++i) {
141  if (dataSpaces[i]) {
142  dataSpaces[i]->start();
143  } // if
144  } // for
145

147 
148  // start the DataProcessManager thread
149  dpm->start();
