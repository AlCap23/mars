113  T* LibManager::acquireLibraryAs(const std::string &libName, bool load) {
114  T *lib = NULL;
115  LibInterface *libInterface = acquireLibrary(libName);
116  if(load && !libInterface) {
117  loadLibrary(libName);
118  libInterface = acquireLibrary(libName);
119  }
120  if(libInterface){
121  lib = dynamic_cast<T*>(libInterface);
122  if(!lib) {
123  releaseLibrary(libName);
124  }
125  }
126  return lib;
127  }
