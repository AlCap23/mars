70  unsigned long EntityFactoryManager::createEntity(configmaps::ConfigMap& config) {
71  unsigned long id = 0;
72  utils::MutexLocker locker(&iMutex);
73  std::map<std::string, EntityFactoryInterface*>::iterator it;
74  bool knowntype = false;
75  for(it=factories.begin(); it!=factories.end(); ++it) {
76  //fprintf(stderr, "Listing types: %s.\n", (it->second->getType()).c_str());
77  if((std::string)config["type"] == (std::string)(it->first))
78  knowntype = true;
79  }
80  if(knowntype) {
81  fprintf(stderr, "Loading Entity of type %s.\n", ((std::string)config["type"]).c_str());
82  sim::SimEntity* newentity = factories[config["type"]]->createEntity(config);
83  id = control->entities->addEntity(newentity);
84  }
85  else {
86  fprintf(stderr, "No EntityFactory known for type %s.\n", ((std::string)config["type"]).c_str());
87  id = 0;
88  }
89  return id;
90  }
