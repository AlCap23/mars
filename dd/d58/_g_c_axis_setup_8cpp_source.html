141 double GCAxisSetup::mapValue(double _input) const {
142  if(_input < (0.0 + deadZoneMinMax)) {
143  // we are in deadZone "Min"
144  return 0.0;
145  } else if(_input < (0.5 - deadZoneCenter)) {
146  return (m * _input) + b;
147  } else if( _input < (0.5 + deadZoneCenter)) {
148  // we are in deadZone "Center"
149  return 0.5;
150  } else if(_input < (1.0 - deadZoneMinMax)) {
151  // _input > 0.5, we have to shift the value
152  return (m * (_input - (2 * deadZoneCenter)) ) + b;
153  } else {
154  // we are in deadZone "Max"
