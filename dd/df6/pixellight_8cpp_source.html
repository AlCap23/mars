144  static std::string pssmAmount(double textureRes=2028,
145  bool filtered=false,
146  unsigned int textureOffset=1) {
147  std::stringstream sstr;
148 
149  sstr << "float pssmAmount() {" << endl;
150  sstr << " float testZ = gl_FragCoord.z*2.0-1.0;" <<std::endl;
151  sstr << " float map0 = step(testZ, zShadow0);"<< std::endl;
152  for (unsigned int i=1;i<NUM_PSSM_SPLITS;i++) {
153  sstr << " float map" << i << " = step(zShadow"<<i-1<<",testZ)*step(testZ, zShadow"<<i<<");"<< std::endl;
154  }
155  if (filtered) {
156  sstr << " float fTexelSize="<< (1.41 / textureRes ) <<";" << std::endl;
157  sstr << " float fZOffSet = -0.001954;" << std::endl; // 2^-9 good value for ATI / NVidia
158  }
159  for (unsigned int i=0;i<NUM_PSSM_SPLITS;i++) {
160  if (!filtered) {
161  sstr << " float shadow" << i <<" = step(0.25,shadow2DProj( shadowTexture" << i <<
162  ",gl_TexCoord[" << (i+textureOffset) <<"]).r);" << std::endl;
163  } else {
164  sstr << " float shadowOrg" << i <<" = shadow2DProj( shadowTexture" <<
165  i <<",gl_TexCoord[" << (i+textureOffset) <<"]+vec4(0.0,0.0,fZOffSet,0.0) ).r;" << std::endl;
166  sstr << " float shadow0" << i <<" = shadow2DProj( shadowTexture" <<
167  i <<",gl_TexCoord[" << (i+textureOffset) <<"]+vec4(-fTexelSize,-fTexelSize,fZOffSet,0.0) ).r;" << std::endl;
168  sstr << " float shadow1" << i <<" = shadow2DProj( shadowTexture" <<
169  i <<",gl_TexCoord[" << (i+textureOffset) <<"]+vec4( fTexelSize,-fTexelSize,fZOffSet,0.0) ).r;" << std::endl;
170  sstr << " float shadow2" << i <<" = shadow2DProj( shadowTexture" <<
171  i <<",gl_TexCoord[" << (i+textureOffset) <<"]+vec4( fTexelSize, fTexelSize,fZOffSet,0.0) ).r;" << std::endl;
172  sstr << " float shadow3" << i <<" = shadow2DProj( shadowTexture" <<
173  i <<",gl_TexCoord[" << (i+textureOffset) <<"]+vec4(-fTexelSize, fTexelSize,fZOffSet,0.0) ).r;" << std::endl;
174  sstr << " float shadow" << i <<" = ( 2.0*shadowOrg" << i
175  <<" + shadow0" << i <<" + shadow1" << i <<" + shadow2" << i <<" + shadow3" << i << ")/6.0;"<< std::endl;
176  }
177  }
178 
179  sstr << " float term0 = (1.0-shadow0)*map0; " << std::endl;
180  for (unsigned int i=1;i<NUM_PSSM_SPLITS;i++) {
181  sstr << " float term" << i << " = map"<< i << "*(1.0-shadow"<<i<<");"<< std::endl;
182  }
183  sstr << " return clamp(";
184  for (unsigned int i=0;i<NUM_PSSM_SPLITS;i++) {
185  sstr << "term" << i;
186  if ( i+1 < NUM_PSSM_SPLITS ){
187  sstr << "+";
188  }
189  }
190  sstr << "0.0, 1.0);" << std::endl;
191  sstr << "}" << endl;
192  cout << sstr.str() << endl;
193  return sstr.str();
194  }
