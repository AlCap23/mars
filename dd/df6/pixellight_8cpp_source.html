84  static std::string pssmAmount(double textureRes=2028,
85  bool filtered=false,
86  unsigned int textureOffset=1) {
87  std::stringstream sstr;
88 
89  sstr << "float pssmAmount() {" << endl;
90  sstr << " float testZ = gl_FragCoord.z*2.0-1.0;" <<std::endl;
91  sstr << " float map0 = step(testZ, zShadow0);"<< std::endl;
92  for (unsigned int i=1;i<NUM_PSSM_SPLITS;i++) {
93  sstr << " float map" << i << " = step(zShadow"<<i-1<<",testZ)*step(testZ, zShadow"<<i<<");"<< std::endl;
94  }
95  if (filtered) {
96  sstr << " float fTexelSize="<< (1.41 / textureRes ) <<";" << std::endl;
97  sstr << " float fZOffSet = -0.001954;" << std::endl; // 2^-9 good value for ATI / NVidia
98  }
99  for (unsigned int i=0;i<NUM_PSSM_SPLITS;i++) {
100  if (!filtered) {
101  sstr << " float shadow" << i <<" = step(0.25,shadow2DProj( shadowTexture" << i <<
102  ",gl_TexCoord[" << (i+textureOffset) <<"]).r);" << std::endl;
103  } else {
104  sstr << " float shadowOrg" << i <<" = shadow2DProj( shadowTexture" <<
105  i <<",gl_TexCoord[" << (i+textureOffset) <<"]+vec4(0.0,0.0,fZOffSet,0.0) ).r;" << std::endl;
106  sstr << " float shadow0" << i <<" = shadow2DProj( shadowTexture" <<
107  i <<",gl_TexCoord[" << (i+textureOffset) <<"]+vec4(-fTexelSize,-fTexelSize,fZOffSet,0.0) ).r;" << std::endl;
108  sstr << " float shadow1" << i <<" = shadow2DProj( shadowTexture" <<
109  i <<",gl_TexCoord[" << (i+textureOffset) <<"]+vec4( fTexelSize,-fTexelSize,fZOffSet,0.0) ).r;" << std::endl;
110  sstr << " float shadow2" << i <<" = shadow2DProj( shadowTexture" <<
111  i <<",gl_TexCoord[" << (i+textureOffset) <<"]+vec4( fTexelSize, fTexelSize,fZOffSet,0.0) ).r;" << std::endl;
112  sstr << " float shadow3" << i <<" = shadow2DProj( shadowTexture" <<
113  i <<",gl_TexCoord[" << (i+textureOffset) <<"]+vec4(-fTexelSize, fTexelSize,fZOffSet,0.0) ).r;" << std::endl;
114  sstr << " float shadow" << i <<" = ( 2.0*shadowOrg" << i
115  <<" + shadow0" << i <<" + shadow1" << i <<" + shadow2" << i <<" + shadow3" << i << ")/6.0;"<< std::endl;
116  }
117  }
118 
119  sstr << " float term0 = (1.0-shadow0)*map0; " << std::endl;
120  for (unsigned int i=1;i<NUM_PSSM_SPLITS;i++) {
121  sstr << " float term" << i << " = map"<< i << "*(1.0-shadow"<<i<<");"<< std::endl;
122  }
123  sstr << " return clamp(";
124  for (unsigned int i=0;i<NUM_PSSM_SPLITS;i++) {
125  sstr << "term" << i;
126  if ( i+1 < NUM_PSSM_SPLITS ){
127  sstr << "+";
128  }
129  }
130  sstr << "0.0, 1.0);" << std::endl;
131  sstr << "}" << endl;
132  cout << sstr.str() << endl;
133  return sstr.str();
134  }
