386  //_data.y = (_data.y > 1.0) ? 1.0 : _data.y;
387 
388  _data.z = (double)_joyState.lZ / 65534.0;
389  _data.z = axisZSetup.calcValue(_data.z);
390  //_data.z = (_data.z > 1.0) ? 1.0 : _data.z;

427 }
