109  void MotorHandler::fill() {
110 
111  std::map<QString, QVariant> attr;

113  attr.insert(std::pair<QString, QVariant>(QString("singleStep"), 0.1));

116 
117  name = pDialog->addGenericProperty("../"+motorName+"/Name", QVariant::String,

127  QVariant::Double, myMotor.maxSpeed);

132  i_value = pDialog->addGenericProperty("../"+motorName+"/I",

134  d_value = pDialog->addGenericProperty("../"+motorName+"/D",
135  QVariant::Double, myMotor.d, &attr);
136 
137  enumNames.clear();
138  int j_ind = 0;
139  for (unsigned int i =0; i< joints.size(); i++) {
140  enumNames << QString::fromStdString(joints[i].name);
141  if (joints[i].index == myMotor.jointIndex)
142  j_ind = i;
143  }
144 
145  j_options = pDialog->addGenericProperty("../"+motorName+"/Joint Options",
146  QtVariantPropertyManager::groupTypeId(), 0);
147  joint = pDialog->addGenericProperty("../"+motorName+"/Joint Options/Joint",
148  QtVariantPropertyManager::enumTypeId(),
149  j_ind, NULL, &enumNames);
150  j_type = pDialog->addGenericProperty("../"+motorName+"/Joint Options/Type", QVariant::String, "");
151  enumNames.clear();
152  enumNames << "Axis 1" << "Axis 2";
153  if (myMotor.axis == 0) myMotor.axis = 1;
154  axis = pDialog->addGenericProperty("../"+motorName+"/Joint Options/Axis",
155  QtVariantPropertyManager::enumTypeId(),
156  myMotor.axis-1, NULL, &enumNames);
157 
158  // j_type->setEnabled(false);

163  void MotorHandler::on_change_joint(int index) {
164  switch (control->joints->getFullJoint(joints[index].index).type) {
165  case interfaces::JOINT_TYPE_UNIVERSAL:
166  j_options->addSubProperty(axis);
167  j_type->setValue("Universal");
168  break;
169  case interfaces::JOINT_TYPE_HINGE2:
170  j_options->addSubProperty(axis);
171  j_type->setValue("Hinge 2");
172  break;
173  case interfaces::JOINT_TYPE_HINGE:
174  j_options->removeSubProperty(axis);
175  j_type->setValue("Hinge");
176  break;
177  case interfaces::JOINT_TYPE_SLIDER:
178  j_options->removeSubProperty(axis);
179  j_type->setValue("Slider");
180  break;
181  case interfaces::JOINT_TYPE_BALL:
182  j_options->removeSubProperty(axis);
183  j_type->setValue("Ball and Socket");
184  break;
185  default:
186  break;
187  }
188  }

191  void MotorHandler::on_change_type(int index) {
192  switch (index) {
193  case 0:
194  myMotor.type = interfaces::MOTOR_TYPE_DC;
195  topLevelMotor->removeSubProperty(p_value);
196  topLevelMotor->removeSubProperty(i_value);
197  topLevelMotor->removeSubProperty(d_value);
198  break;
199  case 1:
200  myMotor.type = interfaces::MOTOR_TYPE_PID;
201  pDialog->insertGenericProperty(topLevelMotor, d_value, force);
202  pDialog->insertGenericProperty(topLevelMotor, i_value, force);
203  pDialog->insertGenericProperty(topLevelMotor, p_value, force);
204  break;
205  default:
206  myMotor.type = interfaces::MOTOR_TYPE_UNDEFINED;
207  break;
208  }
209  }

220  myMotor.jointIndex = joints[joint->value().toInt()].index;
221 
222  if (mode == MotorTree::EditMode)
223  control->motors->editMotor(myMotor);
