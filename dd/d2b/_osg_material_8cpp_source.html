129  void OsgMaterial::initMaterial() {
130  isInit = true;
131  //map.toYamlFile("material.yml");
132  if(map.hasKey("loadPath")) {
133  loadPath << map["loadPath"];
134  if(loadPath[loadPath.size()-1] != '/') {
135  loadPath.append("/");
136  }
137  }
138  if(map.hasKey("filePrefix")) {
139  loadPath << map["filePrefix"];
140  if(loadPath[loadPath.size()-1] != '/') {
141  loadPath.append("/");
142  }
143  }
144  name << map["name"];
145  getLight = map.get("getLight", true);
146 
147  // create the osg::Material
148  material->setColorMode(osg::Material::OFF);
149 
150  material->setAmbient(osg::Material::FRONT_AND_BACK, getColor("ambientColor"));
151  material->setSpecular(osg::Material::FRONT_AND_BACK, getColor("specularColor"));
152  material->setDiffuse(osg::Material::FRONT_AND_BACK, getColor("diffuseColor"));
153  material->setEmission(osg::Material::FRONT_AND_BACK, getColor("emissionColor"));
154  material->setShininess(osg::Material::FRONT_AND_BACK, map.get("shininess", 0.0));
155  material->setTransparency(osg::Material::FRONT_AND_BACK, map.get("transparency", 0.0));
156 
157  // get the StateSet of the Object
158  osg::StateSet *state = getOrCreateStateSet();
159 
160  // set the material
161  state->setAttributeAndModes(material.get(), osg::StateAttribute::ON);
162 
163  if (map.hasKey("culling") && !map["culling"]) {
164  state->setMode(GL_CULL_FACE, osg::StateAttribute::OFF);
165  }
166 
167  if(!getLight) {
168  osg::ref_ptr<osg::CullFace> cull = new osg::CullFace();
169  cull->setMode(osg::CullFace::BACK);
170  state->setAttributeAndModes(cull.get(), osg::StateAttribute::OFF);
171  state->setMode(GL_LIGHTING,
172  osg::StateAttribute::OFF);
173  state->setMode(GL_FOG, osg::StateAttribute::OFF);
174  }
175 
176  float transparency = (float)map.get("transparency", 0.0);
177  string texturename = map.get("diffuseTexture", std::string());
178  double tex_scale = map.get("tex_scale", 1.0);
179  texScaleUniform->set((float)tex_scale);
180 
181  //disable all textures
182  std::map<std::string, TextureInfo>::iterator it = textures.begin();
183  for(; it!=textures.end(); ++it) {
184  it->second.enabled = false;
185  state->setTextureAttributeAndModes(it->second.unit,
186  it->second.texture,
187  osg::StateAttribute::OFF);
188  state->removeUniform(it->second.textureUniform);
189  }
190  if (!texturename.empty()) {
191  ConfigMap config;
192  config["name"] = "diffuseMap";
193  config["file"] = texturename;
194  config["texScale"] = tex_scale;
195  addTexture(config, map.hasKey("instancing"));
196  }
197 
198  bool generateTangents = false;
199  texturename = map.get("normalTexture", std::string());
200  if (!texturename.empty()) {
201  generateTangents = true;
202  ConfigMap config;
203  config["name"] = "normalMap";
204  config["file"] = texturename;
205  config["texScale"] = tex_scale;
206  addTexture(config, map.hasKey("instancing"));
207  }
208  bumpNorFacUniform->set((float)map.get("bumpNorFac", 1.0));
209 
210  if(map.hasKey("textures")) {
211  ConfigVector::iterator it = map["textures"].begin();
212  for(; it!=map["textures"].end(); ++it) {
213  addTexture(*it, map.hasKey("instancing"));
214  }
215  }
216  useWorldTexCoords = map.get("useWorldTexCoords", false);
217  updateShader(true);
218 
219  {
220  std::vector<osg::ref_ptr<MaterialNode> >::iterator it = materialNodeVector.begin();
221  for(; it!=materialNodeVector.end(); ++it) {
222  if(generateTangents) {
223  (*it)->setNeedTangents(true);
224  //(*it)->generateTangents();
225  }
226  (*it)->setTransparency(transparency);
227  }
228  }
229  }

231  void OsgMaterial::addTexture(ConfigMap &config, bool nearest) {
232  osg::StateSet *state = getOrCreateStateSet();
233  std::map<std::string, TextureInfo>::iterator it = textures.find((std::string)config["name"]);
234  if(it != textures.end()) {
235  TextureInfo &info = it->second;
236  // todo: handle changes in texture unit etc.
237  std::string file = config["file"];
238  if(!loadPath.empty() && file[0] != '/') {
239  file = loadPath + file;
240  }
241  info.texture = OsgMaterialManager::loadTexture(file);
242  if(!info.enabled) {
243  state->setTextureAttributeAndModes(info.unit, info.texture,
244  osg::StateAttribute::ON);
245  state->addUniform(info.textureUniform.get());
246  info.enabled = true;
247  }
248  }
249  else {
250  TextureInfo info;
251  info.name << config["name"];
252  fprintf(stderr, "load texture: %s\n", info.name.c_str());
253  std::string file = config["file"];
254  if(!loadPath.empty() && file[0] != '/') {
255  file = loadPath + file;
256  }
257  fprintf(stderr, " texture file: %s\n", file.c_str());
258  if(info.name == "terrainMap") {
259  info.texture = loadTerrainTexture(file);
260  //nearest = true;
261  }
262  else {
263  info.texture = OsgMaterialManager::loadTexture(file);
264  }
265  if(nearest) {
266  info.texture->setFilter(osg::Texture::MIN_FILTER,
267  osg::Texture::NEAREST);
268  info.texture->setFilter(osg::Texture::MAG_FILTER,
269  osg::Texture::NEAREST);
270  }
271  else {
272  info.texture->setFilter(osg::Texture::MIN_FILTER,
273  osg::Texture::LINEAR_MIPMAP_LINEAR);
274  info.texture->setFilter(osg::Texture::MAG_FILTER,
275  osg::Texture::LINEAR);
276  }
277  info.texture->setWrap( osg::Texture::WRAP_S, osg::Texture::REPEAT );
278  info.texture->setWrap( osg::Texture::WRAP_T, osg::Texture::REPEAT );
279  info.texture->setMaxAnisotropy(8);
280  info.unit = 0;
281  if(unitMap.hasKey(info.name)) {
282  info.unit = unitMap[info.name];
283  }
284  if(config.hasKey("unit")) {
285  info.unit = config["unit"];
286  }
287  info.textureUniform = new osg::Uniform(info.name.c_str(), info.unit);
288  state->setTextureAttributeAndModes(info.unit, info.texture,
289  osg::StateAttribute::ON);
290  state->addUniform(info.textureUniform.get());
291  if(config.hasKey("texScale") && (double)config["texScale"] != 1.0) {
292  osg::ref_ptr<osg::TexMat> scaleTexture = new osg::TexMat();
293  float tex_scale = (double)config["texScale"];
294  scaleTexture->setMatrix(osg::Matrix::scale(tex_scale, tex_scale,
295  tex_scale));
296  state->setTextureAttributeAndModes(info.unit, scaleTexture.get(),
297  osg::StateAttribute::ON);
298  }
299  info.enabled = true;
300  textures[info.name] = info;
301  }
302  }

345  void OsgMaterial::edit(const std::string &key, const std::string &value) {
346  if(matchPattern("*/ambientColor/*", key) ||
347  matchPattern("*/ambientFront/*", key)) {
348  setColor("ambientColor", key, value);
349  }
350  else if(matchPattern("*/diffuseColor/*", key) ||
351  matchPattern("*/diffuseFront/*", key)) {
352  setColor("diffuseColor", key, value);
353  }
354  if(matchPattern("*/specularColor/*", key) ||
355  matchPattern("*/specularFront/*", key)) {
356  setColor("specularColor", key, value);
357  }
358  if(matchPattern("*/emissionColor/*", key) ||
359  matchPattern("*/emissionFront/*", key)) {
360  setColor("emissionColor", key, value);
361  }
362 
363  if(matchPattern("*/diffuseTexture", key) ||
364  matchPattern("*/texturename", key)) {
365  if(value == "") {
366  map["diffuseTexture"] = string();
367  fprintf(stderr, "edit material: %s\n", map["diffuseTexture"].c_str());
368  }
369  else if(pathExists(value)) {
370  map["diffuseTexture"] = mars::utils::trim(value);
371  fprintf(stderr, "edit material: %s\n", map["diffuseTexture"].c_str());
372  }
373  setMaterial(map);
374  }
375  if(matchPattern("*/normalTexture", key) ||
376  matchPattern("*/bumpmap", key)) {
377  if(value == "") {
378  map["normalTexture"] = string();
379  fprintf(stderr, "edit material: %s\n", map["normalTexture"].c_str());
380  }
381  else if(pathExists(value)) {
382  map["normalTexture"] = mars::utils::trim(value);
383  fprintf(stderr, "edit material: %s\n", map["normalTexture"].c_str());
384  }
385  setMaterial(map);
386  }
387  if(matchPattern("*/displacementTexture", key) ||
388  matchPattern("*/displacementmap", key)) {
389  if(value == "") {
390  map["displacementTexture"] = string();
391  }
392  else if(pathExists(value)) {
393  map["displacementTexture"] = mars::utils::trim(value);
394  }
395  setMaterial(map);
396  }
397  if(matchPattern("*/bumpNorFac", key)) {
398  map["bumpNorFac"] = atof(value.c_str());
399  setMaterial(map);
400  }
401  if(matchPattern("*/shininess", key)) {
402  map["shininess"] = atof(value.c_str());
403  setMaterial(map);
404  }
405  if(matchPattern("*/transparency", key)) {
406  map["transparency"] = atof(value.c_str());
407  setMaterial(map);
408  }
409  if(matchPattern("*/tex_scale", key)) {
410  map["tex_scale"] = atof(value.c_str());
411  setMaterial(map);
412  }
413  if(matchPattern("*/getLight", key)) {
414  bool b = atoi(value.c_str());
415  map["getLight"] = b;
416  setMaterial(map);
417  }
418  }

452  void OsgMaterial::updateShader(bool reload) {
453  osg::StateSet* stateSet = getOrCreateStateSet();
454 
455  //return;
456  if(!useShader || !getLight) {
457  if(lastProgram.valid()) {
458  stateSet->removeAttribute(lastProgram.get());
459  lastProgram = NULL;
460  }
461  disableTexture("normalMap");
462  stateSet->setTextureAttributeAndModes(NOISE_MAP_UNIT, noiseMap,
463  osg::StateAttribute::OFF);
464  return;
465  }
466  //enableTexture("normalMap");
467 
468  if(!reload && hasShaderSources) {
469  // no need to regenerate, shader source did not changed
470  return;
471  }
472  hasShaderSources = true;
473  bool has_texture = (textures.size()>0);//checkTexture("environmentMap") || checkTexture("diffuseMap") || checkTexture("normalMap");
474  stateSet->setTextureAttributeAndModes(NOISE_MAP_UNIT, noiseMap,
475  osg::StateAttribute::ON);
476  stateSet->removeUniform(envMapSpecularUniform.get());
477  stateSet->removeUniform(envMapScaleUniform.get());
478  stateSet->removeUniform(terrainScaleZUniform.get());
479  stateSet->removeUniform(terrainDimUniform.get());
480  ShaderFactory factory;
481  osg::Program *glslProgram;
482 
483  if(!map.hasKey("shader")) {
484  map["shader"]["PixelLightVertex"] = true;
485  map["shader"]["PixelLightFragment"] = true;
486  if(checkTexture("normalMap")) {
487  map["shader"]["NormalMapVertex"] = true;
488  map["shader"]["NormalMapFragment"] = true;
489  }
490  }
491 
492  if (map["shader"].hasKey("provider")) {
493  if ((string)map["shader"]["provider"] == "DRockGraph") {
494  ConfigMap options;
495  options["numLights"] = maxNumLights;
496  string vertexPath = map["shader"]["vertex"];
497  string fragmentPath = map["shader"]["fragment"];
498  if(!loadPath.empty() && vertexPath[0] != '/') {
499  vertexPath = loadPath + vertexPath;
500  }
501  if(!loadPath.empty() && fragmentPath[0] != '/') {
502  fragmentPath = loadPath + fragmentPath;
503  }
504  ConfigMap vertexModel = ConfigMap::fromYamlFile(vertexPath);
505  ConfigMap fragmentModel = ConfigMap::fromYamlFile(fragmentPath);
506  DRockGraphSP *vertexProvider = new DRockGraphSP(resPath, vertexModel, options);
507  DRockGraphSP *fragmentProvider = new DRockGraphSP(resPath, fragmentModel, options);
508  factory.setShaderProvider(vertexProvider, SHADER_TYPE_VERTEX);
509  factory.setShaderProvider(fragmentProvider, SHADER_TYPE_FRAGMENT);
510  if(textures.find("terrainMap") != textures.end()) {
511  stateSet->addUniform(terrainScaleZUniform.get());
512  stateSet->addUniform(terrainDimUniform.get());
513  terrainScaleZUniform->set((float)(double)map["scaleZ"]);
514  }
515  } else if ((string)map["shader"]["provider"] == "PhobosGraph") {
516  ConfigMap options;
517  options["numLights"] = maxNumLights;
518  options["loadPath"] = loadPath;
519  options["customPath"] = "";
520  if (map["shader"].hasKey("custom")) {
521  options["customPath"] = (string)map["shader"]["custom"];
522  }
523  string vertexPath = map["shader"]["vertex"];
524  string fragmentPath = map["shader"]["fragment"];
525  if(!loadPath.empty() && vertexPath[0] != '/') {
526  vertexPath = loadPath + vertexPath;
527  }
528  if(!loadPath.empty() && fragmentPath[0] != '/') {
529  fragmentPath = loadPath + fragmentPath;
530  }
531  ConfigMap vertexModel = ConfigMap::fromYamlFile(vertexPath);
532  ConfigMap fragmentModel = ConfigMap::fromYamlFile(fragmentPath);
533  PhobosGraphSP *vertexProvider = new PhobosGraphSP(resPath, vertexModel, options);
534  PhobosGraphSP *fragmentProvider = new PhobosGraphSP(resPath, fragmentModel, options);
535  factory.setShaderProvider(vertexProvider, SHADER_TYPE_VERTEX);
536  factory.setShaderProvider(fragmentProvider, SHADER_TYPE_FRAGMENT);
537  }
538  } else {
539  vector<string> args;
540  stringstream s;
541  s << maxNumLights;
542  YamlSP *vertexShader = new YamlSP(resPath);
543  YamlSP *fragmentShader = new YamlSP(resPath);
544  if(map["shader"].hasKey("TerrainMapVertex")) {
545  ConfigMap map2 = ConfigMap::fromYamlFile(resPath+"/shader/terrainMap_vert.yml");
546  YamlShader *terrainMapVert = new YamlShader((string)map2["name"], args, map2, resPath);
547  vertexShader->addShaderFunction(terrainMapVert);
548  stateSet->addUniform(terrainScaleZUniform.get());
549  stateSet->addUniform(terrainDimUniform.get());
550  terrainScaleZUniform->set((float)(double)map["scaleZ"]);
551  }
552  if (map["shader"].hasKey("PixelLightVertex")) {
553  ConfigMap map2 = ConfigMap::fromYamlFile(resPath+"/shader/plight_vert.yaml");
554  map2["mappings"]["numLights"] = s.str();
555  YamlShader *plightVert = new YamlShader((string)map2["name"], args, map2, resPath);
556  vertexShader->addShaderFunction(plightVert);
557  }
558  if(map["shader"].hasKey("NormalMapVertex")) {
559  ConfigMap map2 = ConfigMap::fromYamlFile(resPath+"/shader/bumpmapping_vert.yaml");
560  YamlShader *bumpVert = new YamlShader((string)map2["name"], args, map2, resPath);
561  vertexShader->addShaderFunction(bumpVert);
562  }
563  if(map["shader"].hasKey("EnvMapVertex")) {
564  ConfigMap map2 = ConfigMap::fromYamlFile(resPath+"/shader/envMap_vert.yml");
565  YamlShader *shader = new YamlShader((string)map2["name"], args, map2, resPath);
566  vertexShader->addShaderFunction(shader);
567 
568  }
569  if (map["shader"].hasKey("PixelLightFragment")) {
570  ConfigMap map2 = ConfigMap::fromYamlFile(resPath+"/shader/plight_frag.yaml");
571  map2["mappings"]["numLights"] = s.str();
572  YamlShader *plightFrag = new YamlShader((string)map2["name"], args, map2, resPath);
573  if(checkTexture("diffuseMap")) {
574  plightFrag->addMainVar( (GLSLVariable) { "vec4", "col", "texture2D(diffuseMap, texCoord)" }, 1);
575  plightFrag->addUniform((GLSLUniform) {"sampler2D", "diffuseMap"});
576  }
577  fragmentShader->addShaderFunction(plightFrag);
578  }
579  if(map["shader"].hasKey("NormalMapFragment")) {
580  ConfigMap map2 = ConfigMap::fromYamlFile(resPath+"/shader/bumpmapping_frag.yaml");
