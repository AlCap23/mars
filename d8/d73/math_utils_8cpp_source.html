42  double angleBetween(const Vector &v1, const Vector &v2, Vector *axis) {
43  double squared_length1 = v1.squaredNorm();
44  double squared_length2 = v2.squaredNorm();
45  double tmp = sqrt(squared_length1 * squared_length2);
46  double angle = acos(v1.dot(v2) / tmp);
47  if (axis != NULL) {
48  // special case 0 and 180 degrees where the axis is ambiguous
49  // Note that acos returns a value in [0, M_PI] so angle is in that range
50  if ((angle <= EPSILON) || (M_PI - angle <= EPSILON)) {
51  // find an axis that isn't zero and switch it with another one
52  // make sure to keep the length intact
53  Vector dummy;
54  if (v1(0) > EPSILON) {
55  dummy = Vector(v1.y(), -v1.x(), v1.z());
56  } else {
57  dummy = Vector(v1.x(), v1.z(), -v1.y());
58  }
59  *axis = dummy.cross(v2) / tmp;
60  } else {
61  *axis = v1.cross(v2) / tmp;
62  }
63  }
64  return angle;
65  }

100  sRotation euler;

102  double sqx = value.x()*value.x();
103  double sqy = value.y()*value.y();
104  double sqz = value.z()*value.z();
105  // heading
106  euler.gamma = radToDeg(atan2(2.*(value.x()*value.y()+value.z()*value.w()),
107  (sqx - sqy - sqz + sqw)));
108  // bank
109  euler.alpha = radToDeg(atan2(2.*(value.y()*value.z()+value.x()*value.w()),

111  // attitude

113  test = (test > 1. ? 1. : (test < -1. ? -1. : test));

130  Vector slerp(const Vector &from, const Vector &to, double t) {
131  if (fabs(t) > 1) {
132  throw std::runtime_error("Argument t to Vector::slerp must be in range [-1, 1].");
133  }
134 
135  double length1 = from.norm();
136  double length2 = to.norm();
137  double tmp = from.dot(to) / (length1 * length2);
138  /* make sure tmp is in the range [-1:1] so acos won't return NaN */
139  tmp = (tmp < -1 ? -1 : (tmp > 1 ? 1: tmp));
140  double angle = acos(tmp);
141 
142  /* if t < 0 we take the long arch of the greate circle to the destiny */
143  if (t < 0) {
144  angle -= 2 * M_PI;
145  t = -t;
146  }
147  if (from.x() * to.y() < from.y() * to.x())
148  angle *= -1;
149 
150  Vector ret;
151  /* special case angle==0 and angle==360 */
152  if ((fabs(angle) <= EPSILON) ||
153  (fabs(fabs(angle) - 2 * M_PI) <= EPSILON)) {
154  /* approximate with lerp, because slerp diverges with 1/sin(angle) */
155  ret = lerp(from, to, t);
156  }
157  /* special case angle==180 and angle==-180 */
158  else if (fabs(fabs(angle) - M_PI) <= EPSILON) {
159  throw std::runtime_error("SLERP with 180 degrees is undefined.");
160  }
161  else {
162  double f0 = ((length2 - length1) * t + length1) / ::sin(angle);
163  double f1 = ::sin(angle * (1-t)) / length1;
164  double f2 = ::sin(angle * t) / length2;
165  ret = ((from.array() * f1 + to.array() * f2) * f0).matrix();
166  }
167  return ret;
168  }

170  Vector lerp(const Vector &from, const Vector &to, double t) {
171  if (t < 0 || t > 1) {
172  throw std::runtime_error("Argument t to Vector::lerp must be in range [0, 1]");
173  }
174  return (from.array() * (1-t) + to.array() * t).matrix();
175  }
