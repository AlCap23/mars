90  void DRockGraphSP::parseGraph() {
91  stringstream code;
92  std::map<unsigned long, ConfigMap> nodeMap;
93  std::vector<ConfigMap *> sortedNodes;
94  std::map<std::string, unsigned long> nodeNameId;
95  std::map<unsigned long, ConfigMap>::iterator nodeIt;
96  std::vector<ConfigMap *>::iterator sNodeIt;
97  std::vector<std::string> add; // lines to add after function call generation
98  std::vector<GLSLAttribute> vars; //definitions of main variables
99  std::vector<GLSLVariable> defaultVars;
100  std::vector<std::string> function_calls;
101  ConfigItem graph = model["versions"][0]["components"];
102  ConfigMap nodeConfig;
103  ConfigMap iOuts; //Map containing already created edge variables for output interfaces
104  ConfigMap filterMap;
105  filterMap["int"] = 1;
106  filterMap["float"] = 1;
107  filterMap["vec2"] = 1;
108  filterMap["vec3"] = 1;
109  filterMap["vec4"] = 1;
110  filterMap["sampler2D"] = 1;
111  filterMap["outColor"] = 1;
112 
113 
114  ConfigVector::iterator it;
115  ConfigMap::iterator mit;
116 
117  // Making default values of nodes easily accessible
118  for (it = graph["configuration"]["nodes"].begin(); it != graph["configuration"]["nodes"].end(); ++it) {
119  std::string name = replaceString((std::string)(*it)["name"], "::", "_");
120  ConfigMap data = ConfigMap::fromYamlString((*it)["data"].getString());
121  nodeConfig[name] = data["data"];
122  }
123 
124  // create node ids for tsort
125  unsigned long id = 1;
126  for (it = graph["nodes"].begin(); it != graph["nodes"].end(); ++it) {
127  string function = (*it)["model"]["name"];
128  std::string name = replaceString((std::string)(*it)["name"], "::", "_");
129  (*it)["name"] = name;
130  nodeMap[id] = (*it);
131  nodeNameId[name] = id++;
132  if (nodeConfig.hasKey(name) && nodeConfig[name].hasKey("type")) {
133  string type = nodeConfig[name]["type"].getString();
134  if (type == "uniform") {
135  uniforms.insert((GLSLUniform) {function, name});
136  } else if (type == "varying") {
137  varyings.insert((GLSLAttribute) {function, name});
138  }
139  }
140  }
141 
142  for (it = graph["edges"].begin(); it != graph["edges"].end(); ++it) {
143  string fromNodeName = replaceString((std::string)(*it)["from"]["name"], "::", "_");
144  string toNodeName = replaceString((std::string)(*it)["to"]["name"], "::", "_");
145  string fromInterfaceName = (*it)["from"]["interface"];
146  string toInterface = (*it)["to"]["interface"];
147  string fromVar = "";
148  // create relations for tsort
149  if(nodeNameId.count(fromNodeName) > 0 && nodeNameId.count(toNodeName) > 0) {
150  add_relation(nodeNameId[fromNodeName], nodeNameId[toNodeName]);
151  }
152  ConfigMap fromNode = nodeMap[nodeNameId[fromNodeName]];
153  if (filterMap.hasKey(fromNode["model"]["name"].getString())) {
154  fromVar = fromNodeName;
155  } else {
156  fromVar = fromNodeName + "_at_" + fromInterfaceName;
157  }
158  // Create Mappings from mainVar names to node input interfaces
159  nodeConfig[toNodeName]["toParams"][toInterface] = fromVar;
160  ConfigMap toNode = nodeMap[nodeNameId[toNodeName]];
161  if (filterMap.hasKey(toNode["model"]["name"]) && nodeConfig[toNodeName].hasKey("type")) {
162  if ( nodeConfig[toNodeName]["type"].getString() == "varying") {
163  // We have to output into a varying
164  add.push_back(toNodeName + " = " + fromVar + ";\n");
165  }
166  }
167  }
168 
169  tsort();
170  unsigned long *ids = get_sorted_ids();
171  while (*ids != 0) {
172  sortedNodes.push_back(&(nodeMap[*ids]));
173  ids++;
174  }
175  for (nodeIt = nodeMap.begin(); nodeIt != nodeMap.end(); ++nodeIt) {
176  if (find(sortedNodes.begin(), sortedNodes.end(), &(nodeIt->second)) == sortedNodes.end()) {
177  sortedNodes.push_back(&(nodeIt->second));
178  }
179  }
180 
181  for (sNodeIt = sortedNodes.begin(); sNodeIt != sortedNodes.end(); ++sNodeIt) {
182  ConfigMap &nodeMap = **sNodeIt;
183  string nodeFunction = nodeMap["model"]["name"];
184  string nodeName = nodeMap["name"];
185  priority_queue<PrioritizedLine> incoming, outgoing;
186  bool first = true;
187 
188  if (!filterMap.hasKey(nodeFunction)) {
189  ConfigMap functionInfo = ConfigMap::fromYamlFile(resPath + "/graph_shader/" + nodeFunction + ".yaml");
190  parse_functionInfo(nodeFunction, functionInfo);
191  stringstream call;
192  call.clear();
193  call << nodeFunction << "(";
194  if (functionInfo["params"].hasKey("out")) {
195  for (mit = functionInfo["params"]["out"].beginMap(); mit != functionInfo["params"]["out"].endMap(); mit++) {
196  string varName = nodeName + "_at_" + mit->first;
197  string varType = (mit->second)["type"];
198  outgoing.push((PrioritizedLine) {varName, (int) functionInfo["params"]["out"][mit->first]["index"], 0});
199  vars.push_back((GLSLAttribute) {varType, varName}); // Add main var
200  }
201  }
202  if (functionInfo["params"].hasKey("in")) {
203  for (mit = functionInfo["params"]["in"].beginMap(); mit != functionInfo["params"]["in"].endMap(); mit++) {
204  string varName = "";
205  if (nodeConfig[nodeName]["toParams"].hasKey(mit->first)) {
206  varName = nodeConfig[nodeName]["toParams"][mit->first].getString();
207  } else if (nodeConfig[nodeName].hasKey("inputs") && nodeConfig[nodeName]["inputs"].hasKey(mit->first)) {
208  varName = "default_" + mit->first + "_for_" + nodeName;
209  string varType = (mit->second)["type"];
210  defaultVars.push_back(GLSLVariable {varType, varName, nodeConfig[nodeName]["inputs"][mit->first]});
211  }
212  incoming.push((PrioritizedLine) {varName, (int) functionInfo["params"]["in"][mit->first]["index"], 0});
213  }
214  }
215  while (!incoming.empty()) {
216  if (!first) {
217  call << ", ";
218  }
219  first = false;
220  call << incoming.top().line;
221  incoming.pop();
222  }
223  while (!outgoing.empty()) {
224  if (!first) {
225  call << ", ";
226  }
227  first = false;
228  call << outgoing.top().line;
229  outgoing.pop();
230  }
231  call << ");" << endl;
232  function_calls.push_back(call.str());
233  }
234  }
235 
236  // Compose code
237  code << "void main() {" << endl;
238  for (size_t i = 0; i < vars.size(); ++i) {
239  code << " " << vars[i] << ";" << endl;
240  }
241  code << endl;
242  for (size_t i = 0; i < defaultVars.size(); ++i) {
243  code << " " << defaultVars[i] << ";" << endl;
244  }
245  code << endl;
246  for (size_t i = 0; i < function_calls.size(); ++i) {
247  code << function_calls[i] << endl;
248  }
249  for (size_t i = 0; i < add.size(); ++i) {
250  code << add[i];
251  }
252  code << "}" << endl;
253  main_source = code.str();
254  }

256  void DRockGraphSP::parse_functionInfo(string functionName, ConfigMap functionInfo) {
257  ConfigMap::iterator mit;
258  ConfigVector::iterator vit;
259  if (functionInfo.hasKey("source") && source_files.count(functionName) == 0) {
260  source_files[functionName] = functionInfo["source"].getString();
261  }
262 
263  if (functionInfo.hasKey("minVersion")) {
264  int version = functionInfo["minVersion"];
265  if (version > minVersion) {
266  minVersion = version;
267  }
268  }
269 
270  if (functionInfo.hasKey("extensions")) {
271  ConfigItem list = functionInfo["extensions"];
272  if (list.hasKey("enabled")) {
273  std::cout << "Having enabled extensions" << std::endl;
274  for (vit=list["enabled"].begin();vit!=list["enabled"].end();vit++) {
275  enabledExtensions.insert(vit.base()->getString());
276  }
277  }
278  if (list.hasKey("disabled")) {
279  for (vit=list["disabled"].begin();vit!=list["disabled"].end();vit++) {
280  disabledExtensions.insert(vit.base()->getString());
281  }
282  }
283  }
284 
285  if (functionInfo.hasKey("uniforms")) {
286  ConfigItem list = functionInfo["uniforms"];
287  for (mit = list.beginMap(); mit != list.endMap(); ++mit) {
288  string type = mit->first;
289  string type_name = "";
290  std::size_t a_pos = type.find("[]");
291  bool is_array = false;
292  if (a_pos != string::npos) {
293  type_name = type.substr(0, a_pos);
294  is_array = true;
295  } else {
296  type_name = type;
297  }
298  for (vit = list[type].begin(); vit != list[type].end(); ++vit) {
299  ConfigItem &item = *vit;
300  string name = item["name"].getString();
301  if (is_array) {
302  stringstream s;
303  int num = 1;
304  if (item.hasKey("arraySize")) {
305  num = (int) options[item["arraySize"].getString()];
306  }
307  s << "[" << num << "]";
308  name.append(s.str());
309  }
310  uniforms.insert((GLSLUniform) {type_name, name});
311  }
312  }
313  }
314 
315  if (functionInfo.hasKey("varyings")) {
316  ConfigItem list = functionInfo["varyings"];
317  for (mit = list.beginMap(); mit != list.endMap(); ++mit) {
318  string type = mit->first;
319  string type_name = "";
320  std::size_t a_pos = type.find("[]");
321  bool is_array = false;
322  if (a_pos != string::npos) {
323  type_name = type.substr(0, a_pos);
324  is_array = true;
325  } else {
326  type_name = type;
327  }
328  for (vit = list[type].begin(); vit != list[type].end(); ++vit) {
329  ConfigItem &item = *vit;
330  string name = item["name"].getString();
331  if (is_array) {
332  stringstream s;
333  int num = 1;
334  if (item.hasKey("arraySize")) {
335  num = (int) options[item["arraySize"].getString()];
336  }
337  s << "[" << num << "]";
338  name.append(s.str());
339  }
340  varyings.insert((GLSLAttribute) {type_name, name});
341  }
342  }
343  }
344 
345  if (functionInfo.hasKey("attributes")) {
346  ConfigItem list = functionInfo["attributes"];
347  for (mit = list.beginMap(); mit != list.endMap(); ++mit) {
348  string type = mit->first;
349  string type_name = "";
350  std::size_t a_pos = type.find("[]");
351  bool is_array = false;
352  if (a_pos != string::npos) {
353  type_name = type.substr(0, a_pos);
354  is_array = true;
355  } else {
356  type_name = type;
357  }
358  for (vit = list[type].begin(); vit != list[type].end(); ++vit) {
359  ConfigItem &item = *vit;
360  string name = item["name"].getString();
361  if (is_array) {
362  stringstream s;
363  int num = 1;
364  if (item.hasKey("arraySize")) {
365  num = (int) options[item["arraySize"].getString()];
366  }
367  s << "[" << num << "]";
368  name.append(s.str());
369  }
370  attributes.insert((GLSLAttribute) {type_name, name});
371  }
372  }
373  }
