113 
114  QFile xmlfile((QString) s_scenename.c_str());
115  if (!xmlfile.open(QIODevice::WriteOnly |QIODevice::Append |
116  QIODevice::Text)) {
117  std::cout<<"something went wrong, while generating the *scene file"

121  QTextStream out(&xmlfile);

147  }

194  std::cout<<"num Matirials:"<<materials.size()<<std::endl;
195  for(iter = materials.begin(); iter != materials.end(); iter++) {
196  if(!generate(&((*iter).material), &out, &v_fullfilenames,

200  }
201  out <<" </materiallist>\n";
202 
203 
204  interfaces::GraphicData options = control->graphics->getGraphicOptions();

206  LOG_ERROR("Save:: faild generating graphic options");
207  }
208 
209  out <<"</SceneFile>"<<"\n";
210 
211 
212  xmlfile.close();
213  if (!v_fullfilenames.empty()) {
214  Zipit zipfile(s_zipfile);
215  LOG_WARN("Save: save to: %s", s_zipfile.c_str());
216  //removing doubleentries,
217  for (unsigned int i=0; i<v_fullfilenames.size(); i++) {
218  for (unsigned int j=i+1; j<v_fullfilenames.size(); j++) {
219  if (v_fullfilenames[i]==v_fullfilenames[j]) {

221  j--;//and decrementing the counter j, because there may be more

223  }
224  }
225  }
226  for (unsigned int i=0; i<v_fullfilenames.size(); i++) {
227  s_filename = v_fullfilenames[i];
228  utils::removeFilenamePrefix(&s_filename);
229  v_filenames.push_back(s_filename);
230  }
231  if (zipfile.addToZip(v_filenames, v_fullfilenames)!=0) {
232  return 0;
233  }
234  }
235  return 1;
236  }

265  param.v_controllerList.push_back(control->controllers->getFullController(it->index));

276  void Save::writeConfigMap(const configmaps::ConfigMap &cfg, QTextStream * out,
277  std::string tag, bool handleSensor, int depth) {
278 
279  configmaps::ConfigMap::const_iterator it;
280  configmaps::ConfigVector::const_iterator it2;
281  bool close = false;
282  char text[255];
283 
284  if(handleSensor && depth == 0) {
285  it = cfg.find("name");
286  std::string name = it->second.toString();
287  it = cfg.find("type");
288  std::string type = it->second.toString();
289  // FIXME: potential Buffer Overflow!!!!
290  sprintf(text, " <sensor name=\"%s\" type=\"%s\">", name.c_str(),
291  type.c_str());
292 
293  //LOG_DEBUG("%s", text);
294 
295  *out <<text<<"\n";
296 
297  close = true;
298  depth+=3;
299  }
300  else if(depth==0) {
301  //LOG_DEBUG(" <%s>", tag.c_str());
302  *out << " <" << QString(tag.c_str()) << ">\n";
303  depth+=3;
304  close = true;
305  }
306 
307  for(int i=0; i<depth*2; i++) text[i] = ' ';
308 
309  for(it=cfg.begin(); it!=cfg.end(); ++it) {
310  if(handleSensor && (depth < 3 && (it->first == "name" || it->first == "type"))) continue;
311  for(it2=it->second.begin(); it2!=it->second.end(); ++it2) {
312  if(it2->size() == 0) {
313  sprintf(text+depth*2, "<%s>%s</%s>", it->first.c_str(),
314  it2->toString().c_str(), it->first.c_str());
315  *out << text << "\n";
316  //LOG_DEBUG("%s", text);
317  }
318  else {
319  sprintf(text+depth*2, "<%s>", it->first.c_str());
320  *out << text << "\n";
321  //LOG_DEBUG("%s", text);
322  writeConfigMap((configmaps::ConfigMap&)(*it2), out, tag, handleSensor, depth+1);
323  sprintf(text+depth*2, "</%s>", it->first.c_str());
324  *out << text << "\n";
325  //LOG_DEBUG("%s", text);
326  }
327  }
328  }
329 
330  if(close) {
331  *out << " </"<< QString(tag.c_str()) <<">\n";
332  //LOG_DEBUG(" </%s>", tag.c_str());
333  }
334  }

363  unsigned int Save::generate(interfaces::NodeData *nodeData,
364  QTextStream *out,
365  std::vector<std::string> *v_filenames) {
366  configmaps::ConfigMap config;
367 
368  if(nodeData->relative_id) { // handle relative positioning
369  interfaces::NodeData parentNode;
370  parentNode = control->nodes->getFullNode(nodeData->relative_id);
371  interfaces::getRelFromAbs(parentNode, nodeData);
372  }
373  nodeData->toConfigMap(&config, true);
374  nodeData->getFilesToSave(v_filenames);
375 
376  { // handle material
377  unsigned long mat_id;
378  std::vector<material_map>::iterator iter;
379  material_map tmp_mat_map;
380 
381  mat_id = 0;
382  for(iter = materials.begin(); iter != materials.end(); iter++) {
383  if((*iter).material == nodeData->material) {
384  mat_id = (*iter).material_id; break;}
385  }
386  if(!mat_id) {
387  tmp_mat_map.material = nodeData->material;
388  mat_id = tmp_mat_map.material_id = next_material_id;
389  materials.push_back(tmp_mat_map);
390  next_material_id++;
391  }
392  config["material_id"] = mat_id;
393  }
394 
395  writeConfigMap(config, out, std::string("node"));
396  return 1;
397  }
