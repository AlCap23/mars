231  QPainterPath MarsStyle::roundRectPath(const QRect &rect)
232  {
233  int radius = qMin(rect.width(), rect.height()) / 4;
234  int diam = 2 * radius;
235 
236  int x1, y1, x2, y2;
237  rect.getCoords(&x1, &y1, &x2, &y2);
238 
239  QPainterPath path;
240  path.moveTo(x2, y1 + radius);
241  path.arcTo(QRect(x2 - diam, y1, diam, diam), 0.0, +90.0);
242  path.lineTo(x1 + radius, y1);
243  path.arcTo(QRect(x1, y1, diam, diam), 90.0, +90.0);
244  path.lineTo(x1, y2 - radius);
245  path.arcTo(QRect(x1, y2 - diam, diam, diam), 180.0, +90.0);
246  path.lineTo(x1 + radius, y2);
247  path.arcTo(QRect(x2 - diam, y2 - diam, diam, diam), 270.0, +90.0);
248  path.closeSubpath();
249  return path;
250  }
