128  void MainConsole::add(const std::string &message,
129  data_broker::MessageType type) {
130  // output to the console widget, if it is loaded
131  bool showInWidget = consoleWidget;
132 
133  //showing messages on stderr is nice because you can read
134  //the error/debug messages even if the simulator crashes.
135 
136  if (showInWidget) {
137  int wait = 100;
138  con_data da;
139  da.type = type;
140  while(wait < 20) {
141  consoleLock.lock();
142  if(messages.size() < MAX_DE)
143  wait = 20;
144  consoleLock.unlock();
145  if(wait++ < 20) {
146 #ifdef WIN32
147  Sleep(10);
148 #else
149  usleep(10000);
150 #endif
151  }
152  }
153  consoleLock.lock();
154  da.message = message;
155  messages.push_back(da);
156  consoleLock.unlock();
157  }
158  if (showOnStdError.bValue)
159  {
160  // output to stderr
161  switch(type) {
162  case data_broker::DB_MESSAGE_TYPE_FATAL:
163  case data_broker::DB_MESSAGE_TYPE_ERROR:
164 #ifndef WIN32
165  fprintf(stderr, "\033[1;31merror: %s\033[0m\n", message.c_str());
166 #else
167  fprintf(stderr, "error: %s\n", message.c_str());
168 #endif
169  break;
170  case data_broker::DB_MESSAGE_TYPE_WARNING:
171 #ifndef WIN32
172  fprintf(stderr, "\033[0;32mmessage: %s\033[0m\n", message.c_str());
173 #else
174  fprintf(stderr, "message: %s\n", message.c_str());
175 #endif
176  break;
177  case data_broker::DB_MESSAGE_TYPE_INFO:
178  case data_broker::DB_MESSAGE_TYPE_DEBUG:
179 #ifndef WIN32
180  fprintf(stderr, "\033[1;34mwarning: %s\033[0m\n", message.c_str());
181 #else
182  fprintf(stderr, "warning: %s\n", message.c_str());
183 #endif
184  break;
185  default:
186  fprintf(stderr, "???: %s\n", message.c_str());
187  break;
188  }
189  }
190  }

219  void MainConsole::timerEvent(QTimerEvent* event) {
220  (void)event;
221  if (consoleWidget == NULL)
222  return;
223  consoleLock.lock();
224  while(messages.size() > 0) {
225  if(messages[0].type == data_broker::DB_MESSAGE_TYPE_ERROR)
226  consoleWidget->setTextColor(QColor(212, 148, 90));
227  else if(messages[0].type == data_broker::DB_MESSAGE_TYPE_WARNING)
228  consoleWidget->setTextColor(QColor(90, 148, 212));
229  else
230  consoleWidget->setTextColor(QColor(90, 200, 70));
231 
232  consoleWidget->append(QString(messages[0].message.data()));
233  messages.erase(messages.begin());
234  }
235  consoleLock.unlock();
236  }
