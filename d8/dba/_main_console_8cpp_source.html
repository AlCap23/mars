127  void MainConsole::add(const std::string &message,
128  data_broker::MessageType type) {
129  // output to the console widget, if it is loaded
130  bool showInWidget = consoleWidget;
131 
132  //showing messages on stderr is nice because you can read
133  //the error/debug messages even if the simulator crashes.
134 
135  if (showInWidget) {
136  int wait = 100;
137  con_data da;
138  da.type = type;
139  while(wait < 20) {
140  consoleLock.lock();
141  if(messages.size() < MAX_DE)
142  wait = 20;
143  consoleLock.unlock();
144  if(wait++ < 20) {
145 #ifdef WIN32
146  Sleep(10);
147 #else
148  usleep(10000);
149 #endif
150  }
151  }
152  consoleLock.lock();
153  da.message = message;
154  messages.push_back(da);
155  consoleLock.unlock();
156  }
157  if (showOnStdError.bValue)
158  {
159  // output to stderr
160  switch(type) {
161  case data_broker::DB_MESSAGE_TYPE_FATAL:
162  case data_broker::DB_MESSAGE_TYPE_ERROR:
163 #ifndef WIN32
164  fprintf(stderr, "\033[1;31merror: %s\033[0m\n", message.c_str());
165 #else
166  fprintf(stderr, "error: %s\n", message.c_str());
167 #endif
168  break;
169  case data_broker::DB_MESSAGE_TYPE_WARNING:
170 #ifndef WIN32
171  fprintf(stderr, "\033[0;32mmessage: %s\033[0m\n", message.c_str());
172 #else
173  fprintf(stderr, "message: %s\n", message.c_str());
174 #endif
175  break;
176  case data_broker::DB_MESSAGE_TYPE_INFO:
177  case data_broker::DB_MESSAGE_TYPE_DEBUG:
178 #ifndef WIN32
179  fprintf(stderr, "\033[1;34mwarning: %s\033[0m\n", message.c_str());
180 #else
181  fprintf(stderr, "warning: %s\n", message.c_str());
182 #endif
183  break;
184  default:
185  fprintf(stderr, "???: %s\n", message.c_str());
186  break;
187  }
188  }
189  }

218  void MainConsole::timerEvent(QTimerEvent* event) {
219  (void)event;
220  if (consoleWidget == NULL)
221  return;
222  consoleLock.lock();
223  while(messages.size() > 0) {
224  if(messages[0].type == data_broker::DB_MESSAGE_TYPE_FATAL)
225  consoleWidget->setTextColor(QColor(255, 48, 9));
226  else if(messages[0].type == data_broker::DB_MESSAGE_TYPE_ERROR)
227  consoleWidget->setTextColor(QColor(212, 148, 90));
228  else if(messages[0].type == data_broker::DB_MESSAGE_TYPE_WARNING)
229  consoleWidget->setTextColor(QColor(90, 148, 212));
230  else
231  consoleWidget->setTextColor(QColor(90, 200, 70));
232 
233  consoleWidget->append(QString(messages[0].message.data()));
234  messages.erase(messages.begin());
235  }
236  consoleLock.unlock();
237  }
