189  void Dialog_Import_Mesh::fill() {
190  std::vector<std::string> nameString;
191  QStringList enumNames;
192  filled = false;
193 
194  QString filename = myFileName->value().toString();
195  if (filename.isEmpty()) {
196  initialized = false;
197  return;
198  }
199  osg::ref_ptr<osg::Node> osgReadNode;
200  osgReadNode = osgDB::readNodeFile(filename.toStdString());
201  fprintf(stderr, "import: %s\n", filename.toStdString().c_str());
202  // only start if file has been read correctly
203  if (osgReadNode == NULL) {
204  initialized = false;
205  return;
206  }
207 
208  allNodes.clear();
209 
210  // convert node to group for getting the group functions
211  // don't use Node if conversion failed
212  osg::ref_ptr<osg::Geode> osgGeodeFromRead = NULL;
213  osg::ref_ptr<osg::Group> osgGroupFromRead = NULL;
214 
215  // when importing .obj files all nodes are imported as osg::Group (with
216  // all objects being its children)
217  if ((osgGroupFromRead = osgReadNode->asGroup()) != 0) {
218  fprintf(stderr, "\n");
219  // get childrens names
220  for (unsigned int i = 0 ; i < osgGroupFromRead->getNumChildren(); i++){
221  fprintf(stderr, "Dialog_Import_Mesh num child: %d\n", i);
222  osg::ref_ptr<osg::Node> createdNode = osgGroupFromRead->getChild(i);
223  // search for same names
224  fprintf(stderr, " createdNode->getName() = %s\n",
225  createdNode->getName().c_str());
226  bool found = false;
227  for(unsigned int j=0; j<nameString.size();j++){
228  if ((nameString[j] == createdNode->getName())) {
229  found = true;
230  }
231  }
232  if(!found)
233  nameString.push_back(createdNode->getName());
234  }
235  }
236  // when importing .stl files the mesh is stored as an osg::Geode
237  else if ((osgGeodeFromRead = osgReadNode->asGeode()) != 0) {
238  fprintf(stderr, "\n");
239  // clean the filename of everything path related and just keep the pure
240  // file name
241  std::string name = filename.toStdString();
242  const size_t last_slash_idx = name.find_last_of("\\/");
243  if (std::string::npos != last_slash_idx) {
244  name.erase(0, last_slash_idx + 1);
245  }
246  //the name for the new node is the filename of the STL file
247  nameString.push_back(name);
248 
249  }
250 
251  // combine nodes with same name
252  // go through the namelist
253  for (unsigned int j=0; j<nameString.size();j++){
254  osg::ref_ptr<osg::PositionAttitudeTransform> transform =
255  new osg::PositionAttitudeTransform();
256  bool found = false;
257  // create a groupnode for each name in the list
258  osg::ref_ptr<osg::Group>createdGroup = new osg::Group;
259  if (osgGroupFromRead != 0) {
260  for (unsigned int i = j ; i < osgGroupFromRead->getNumChildren(); i ++){
261  // add all nodes with same name to the group
262  osg::ref_ptr<osg::Node> createdNode = osgGroupFromRead->getChild(i);
263  if (createdNode->getName() == nameString[j]) {
264  createdGroup->addChild(createdNode.get());
265  found = true;
266  }
267  else {
268  if (found){
269  i = osgGroupFromRead->getNumChildren();
270  found = false;
271  }
272  }
273  }
274  }
275  else if (osgGeodeFromRead != 0) {
276  //if the read file is a .STL file the node was read as geode
277  createdGroup->addChild(osgReadNode);
278  }
279 
280  // attach the group to a transform ndoe
281  transform->addChild(createdGroup.get());
282  // create bounding box to get center point
283  osg::ComputeBoundsVisitor cbbv;
284  createdGroup->accept(cbbv);
285  osg::BoundingBox bb = cbbv.getBoundingBox();
286  utils::Vector ex;
287  // compute bounding box has to be done in this way
288  (fabs(bb.xMax()) > fabs(bb.xMin())) ? ex.x() = fabs(bb.xMax() - bb.xMin())
289  : ex.x() = fabs(bb.xMin() - bb.xMax());
290  (fabs(bb.yMax()) > fabs(bb.yMin())) ? ex.y() = fabs(bb.yMax() - bb.yMin())
291  : ex.y() = fabs(bb.yMin() - bb.yMax());
292  (fabs(bb.zMax()) > fabs(bb.zMin())) ? ex.z() = fabs(bb.zMax() - bb.zMin())
293  : ex.z() = fabs(bb.zMin() - bb.zMax());
294 
295  // set pivot point to the center of node instead of the file origin
296  transform->setPivotPoint(bb.center());
297  transform->setPosition(bb.center());
298 
299  // set nodes position
300  osg::Vec3 oPos = transform->getPosition();
301  utils::Vector pos;
302  pos.x() = oPos[0];
303  pos.y() = oPos[1];
304  pos.z() = oPos[2];
305 
306  // construct NodeData for each object and initialize it
307  interfaces::NodeData thisNode;
308  interfaces::MaterialData thisMaterial;
309  //ZERO_NODE_STRUCT(thisNode);
310  //ZERO_MATERIAL_STRUCT(thisMaterial);
311 
312  dcm->setMaterial(&material);
313 
314  thisNode.rot = utils::Quaternion(1,0,0,0);
315  thisNode.pos = pos;
316  thisNode.name = nameString[j];
317  thisNode.origName = nameString[j];
318  thisNode.ext = ex;
319  thisNode.physicMode = interfaces::NODE_TYPE_BOX;
320  thisNode.groupID = 0;
321  thisNode.density = 1;
322  thisNode.mass = 0;
323  thisNode.movable = true;
324  thisNode.filename = myFileName->value().toString().toStdString();
325  thisNode.material = thisMaterial;
326  thisNode.pivot = pos;
327  thisNode.visual_offset_pos = utils::Vector(0,0,0);
328  thisNode.visual_offset_rot = utils::Quaternion(1,0,0,0);
329  thisNode.visual_size = ex;
330  thisNode.relative_id = 0;
331  // attach node to allNode vector
332  allNodes.push_back(thisNode);
333  // add node name to dialogs node list
334  enumNames << QString::fromStdString(nameString[j]);
335  }
336 
337  meshes->setAttribute("enumNames", enumNames);
338  meshes->setValue(0);
339  if(allNodes.size() > 0) {
340  wno->init(allNodes[0].c_params);
341  }
342  /*
343  if (allNodes.size() == 1)
344  remove_mesh->setEnabled(false);
345  */
346  control->graphics->preview(interfaces::PREVIEW_CREATE,false,allNodes);
347 
348  user_input = true;
349 
350  filled = true;
351  initialized = true;
352  item_changed();
353  }
