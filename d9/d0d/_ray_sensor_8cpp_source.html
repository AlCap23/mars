55  RaySensor::RaySensor(ControlCenter *control, RayConfig config):
56  BasePolarIntersectionSensor(config.id, config.name, config.width,
57  config.height, config.opening_width,
58  config.opening_height),
59  SensorInterface(control), config(config) {
60 
61  updateRate = config.updateRate;
62  orientation.setIdentity();
63  maxDistance = config.maxDistance;
64  this->attached_node = config.attached_node;
65 
66  std::string groupName, dataName;
67  drawStruct draw;
68  draw_item item;
69  int i;
70  Vector tmp;
71  have_update = false;
72  for(int i = 0; i < 3; ++i)
73  positionIndices[i] = -1;
74  for(int i = 0; i < 4; ++i)
75  rotationIndices[i] = -1;
76 
77  control->nodes->addNodeSensor(this);
78  bool erg = control->nodes->getDataBrokerNames(attached_node, &groupName, &dataName);
79  (void)erg;
80  assert(erg);
81  if(control->dataBroker->registerTimedReceiver(this, groupName, dataName,"mars_sim/simTimer",updateRate)) {
82  }
83 
84  position = control->nodes->getPosition(attached_node);
85  orientation = control->nodes->getRotation(attached_node);
86 
87  //Drawing Stuff
88  if(config.draw_rays) {
89  draw.ptr_draw = (DrawInterface*)this;
90  item.id = 0;
91  item.type = DRAW_LINE;
92  item.draw_state = DRAW_STATE_CREATE;
93  item.point_size = 1;
94  item.myColor.r = 1;
95  item.myColor.g = 0;
96  item.myColor.b = 0;
97  item.myColor.a = 1;
98  item.texture = "";
99  item.t_width = item.t_height = 0;
100  item.get_light = 0.0;
101 
102 
103  double rad_steps = getCols(); //rad_angle/(sReal)(sensor.resolution-1);
104  double rad_start = -((rad_steps-1)/2.0)*stepX; //Starting to Left, because 0 is in front and rock convention posive CCW //(M_PI-rad_angle)/2;
105  if(rad_steps == 1){
106  rad_start = 0;
107  }
108 
109  //printf("Rad Start: %f, rad_steps: %f, stepX: %f\n",rad_start,rad_steps,stepX);
110 
111  for(i=0; i<rad_steps; i++){
112  tmp = Vector(cos(rad_start+i*stepX),
113  sin(rad_start+i*stepX), 0);
114  directions.push_back(tmp);
115  tmp = (orientation * tmp);
116  item.start = position;
117  item.end = (orientation * tmp);
118  item.end *= data[i];
119  draw.drawItems.push_back(item); // TEST
120  }
121 
122  if(control->graphics)
123  control->graphics->addDrawItems(&draw);
124 
125 
126  assert(rad_steps == data.size());
127  }
128  }

153  void RaySensor::receiveData(const data_broker::DataInfo &info,
154  const data_broker::DataPackage &package,
155  int callbackParam) {
156  CPP_UNUSED(info);
157  CPP_UNUSED(callbackParam);
158  long id;
159  package.get(0, &id);
160 
161  if(positionIndices[0] == -1) {
162  positionIndices[0] = package.getIndexByName("position/x");
163  positionIndices[1] = package.getIndexByName("position/y");
164  positionIndices[2] = package.getIndexByName("position/z");
165  rotationIndices[0] = package.getIndexByName("rotation/x");
166  rotationIndices[1] = package.getIndexByName("rotation/y");
167  rotationIndices[2] = package.getIndexByName("rotation/z");
168  rotationIndices[3] = package.getIndexByName("rotation/w");
169  }
170  for(int i = 0; i < 3; ++i)
171  {
172  package.get(positionIndices[i], &position[i]);
173  }
174 
175  package.get(rotationIndices[0], &orientation.x());
176  package.get(rotationIndices[1], &orientation.y());
177  package.get(rotationIndices[2], &orientation.z());
178  package.get(rotationIndices[3], &orientation.w());
179 
180  have_update = true;
181  }

183  void RaySensor::update(std::vector<draw_item>* drawItems) {
184  unsigned int i;
185  if(config.draw_rays) {
186  if(have_update) {
187  control->nodes->updateRay(attached_node);
188  have_update = false;
189  }
190 
191 
192  if(!(*drawItems)[0].draw_state) {
193  for(i=0; i<data.size(); i++) {
194  (*drawItems)[i].draw_state = DRAW_STATE_UPDATE;
195  (*drawItems)[i].start = position;
196  (*drawItems)[i].end = (orientation * directions[i]);
197  (*drawItems)[i].end *= data[i];
198 
199  (*drawItems)[i].end += (*drawItems)[i].start;
200  }
201  }
202  }
