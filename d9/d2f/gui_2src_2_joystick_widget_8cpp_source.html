162  void JoystickWidget::calculateCoordinates(void) {
163  x_coor = (double)(ballCenter.x()-125.0)/125.0;
164  y_coor = (double)(125.0-ballCenter.y())/125.0;
165 
166  height = 0;
167  maxHeight = 1;
168  if (ballCenter.y() < painterCenter.y()-90)
169  maxHeight = 100;
170  else if (ballCenter.y() > painterCenter.y()+70)
171  maxHeight = 80;
172  else
173  maxHeight = 88 - (ballCenter.y()-painterCenter.y())/9;
174 
175  height = distance(painterCenter, ballCenter);
176  if (height > maxHeight)
177  height = maxHeight;
178  strength = height/maxHeight;
179  }
