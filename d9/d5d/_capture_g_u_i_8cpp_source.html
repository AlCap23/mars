101  void CaptureGUI::updateGUI(void) {
102  if(camera < cameras.size() && take_events) {
103  take_events = false;
104  if(update_frame_box) frameBox->setValue(cameras[camera]->getFrameRate());
105  update_frame_box = false;
106  bool capturing = cameras[camera]->isCapturing();
107 
108  if(capturing)
109  status->setValue(cameras[camera]->getState());

111  status->setValue(tr("Not capturing"));

113  take_events = true;

135  void CaptureGUI::removeCamera(CaptureConfig* camera) {
136  unsigned int i = 0;
137  std::vector<CaptureConfig*>::iterator iter;
138  QStringList enumNames = winIDCombo->attributeValue("enumNames").toStringList();
139  bool found;
140 
141  // first find the correct index
142  for(iter=cameras.begin(); iter!=cameras.end(); ++iter, ++i)
143  if((*iter) == camera) break;
144 
145  if(iter!=cameras.end()) {
146  // we have the camera in the list and can remove it
147  cameras.erase(iter);
148  // select box remove i
149  found = false;
150  if(this->camera == i) {
151  found = true;
152  this->camera = 0;
153  }
154  enumNames.removeAt(i);
155  winIDCombo->setAttribute("enumNames", enumNames);
156  if(found) winIDCombo->setValue(this->camera);
157  }
158  }
