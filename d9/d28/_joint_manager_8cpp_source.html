65  unsigned long JointManager::addJoint(JointData *jointS, bool reload) {
66  JointInterface *newJointInterface;
67  std::vector<SimNode*>::iterator iter;
68  SimNode *node1 = 0;
69  SimNode *node2 = 0;
70  NodeInterface *i_node1 = 0;
71  NodeInterface *i_node2 = 0;
72  Vector an;
73 
74  if (!reload) {
75  iMutex.lock();
76  simJointsReload.push_back(*jointS);
77  iMutex.unlock();
78  }
79 
80  //if(jointS->axis1.lengthSquared() < Vector::EPSILON && jointS->type != JOINT_TYPE_FIXED) {
81  if(jointS->axis1.squaredNorm() < EPSILON && jointS->type != JOINT_TYPE_FIXED) {
82  LOG_ERROR("Cannot create joint without axis1");
83  return 0;
84  }
85 
86  // create an interface object to the physics
87  newJointInterface = PhysicsMapper::newJointPhysics(control->sim->getPhysics());
88  // reset the anchor
89  //if node index is 0, the node connects to the environment.
90  node1 = control->nodes->getSimNode(jointS->nodeIndex1);
91  if (node1) i_node1 = node1->getInterface();
92  node2 = control->nodes->getSimNode(jointS->nodeIndex2);
93  if (node2) i_node2 = node2->getInterface();
94 
95  // ### important! how to deal with different load options? ###
96  //if (load_option == OPEN_INITIAL)
97  //jointS->angle1_offset = jointS->angle2_offset = 0;
98  if (jointS->anchorPos == ANCHOR_NODE1) {
99  assert(node1);
100  jointS->anchor = node1->getPosition();
101  } else if (jointS->anchorPos == ANCHOR_NODE2) {
102  assert(node2);
103  jointS->anchor = node2->getPosition();
104  } else if (jointS->anchorPos == ANCHOR_CENTER) {
105  assert(node1);
106  assert(node2);
107  jointS->anchor = (node1->getPosition() + node2->getPosition()) / 2.;
108  }
109 
110  // create the physical node data
111  if (newJointInterface->createJoint(jointS, i_node1, i_node2)) {
112  // put all data to the correct place
113  iMutex.lock();
114  // set the next free id
115  jointS->index = next_joint_id;
116  next_joint_id++;
117  SimJoint* newJoint = new SimJoint(control, *jointS);
118  newJoint->setAttachedNodes(node1, node2);
119  // newJoint->setSJoint(*jointS);
120  newJoint->setInterface(newJointInterface);
121  simJoints[jointS->index] = newJoint;
122  iMutex.unlock();
123  control->sim->sceneHasChanged(false);
124  return jointS->index;
125  } else {
126  std::cerr << "JointManager: Could not create new joint (JointInterface::createJoint() returned false)." << std::endl;
127  // if no node was created in physics
128  // delete the objects
129  delete newJointInterface;
130  // and return false
131  return 0;
132  }
133  }
