121  void OsgMaterialManager::updateShadowSamples() {
122  static int count = 0;
123  osg::Vec2 v;
124  double x1, y1, r1, r2;
125  double scale1 = 1./shadowSamples.iValue;
126  unsigned char *data = noiseImage->data();
127  int sampleX = 0, sampleY = 0;
128  double noise = 0.5;
129  for(int i=0; i<128; ++i) {
130  for(int l=0; l<128; ++l) {
131  if(!count) {
132  r1 = ((double) rand()/RAND_MAX)*2-1; // -1 to 1
133  r2 = ((double) rand()/RAND_MAX)*2-1;
134  x1 = scale1*0.5+scale1*sampleX+r1*scale1*0.5*noise;
135  y1 = scale1*0.5+scale1*sampleY+r2*scale1*0.5*noise;
136  r1 = (sqrt(y1)*cos(6.28*x1)*0.5 + .5)*255;
137  r2 = (sqrt(y1)*sin(6.28*x1)*0.5 + .5)*255;
138  data[i*128*4+l*4+0] = (unsigned char) r1;
139  data[i*128*4+l*4+1] = (unsigned char) r2;
140  }
141  data[i*128*4+l*4+2] = (unsigned char) (((double)rand()/RAND_MAX)*255);
142  data[i*128*4+l*4+3] = (unsigned char) (((double)rand()/RAND_MAX)*255);
143  if(++sampleX == shadowSamples.iValue) {
144  sampleX = 0;
145  }
146  }
147  if(++sampleY == shadowSamples.iValue) {
148  sampleY = 0;
149  }
150  }
151  noiseImage->dirty();
152  //count = 1;
153  }

174  void OsgMaterialManager::createMaterial(const std::string &name,
175  const configmaps::ConfigMap &map) {
176  std::map<std::string, osg::ref_ptr<OsgMaterial> >::iterator it;
177  it = materialMap.find(name);
178  if(it == materialMap.end()) {
179  OsgMaterial *m = new OsgMaterial(resPath.sValue+"/mars/osg_material_manager/resources");
180  m->setShadowTextureSize(shadowTextureSize);
181  m->setMaterial(map);
182  m->setUseShader(useShader);
183  m->setNoiseImage(noiseImage.get());
184  m->setShadowSamples(shadowSamples.iValue);
185  materialMap[name] = m;
186  mainStateGroup->addChild(m);
187  }
188  }
