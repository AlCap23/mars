279  this->sMotor = sMotor;
280  if(sMotor.type == MOTOR_TYPE_PID ||
281  sMotor.type == MOTOR_TYPE_PID_FORCE) {
282  desired_position = sMotor.value;
283  }
284  else if(sMotor.type == MOTOR_TYPE_DC) {
285  actual_velocity = sMotor.value;

287  // we can initialize the motor here

314  void SimMotor::update(sReal time_ms) {
315  // updates the motor
316  // a timing should be added
317  sReal er = 0;
318  sReal vel = 0;
319  time = time_ms;// / 1000;
320  sReal play_position = 0.0;
321 
322  if(activated) {
323  if(myPlayJoint) play_position = myPlayJoint->getActualAngle1();
324  if(myJoint) {
325  if(sMotor.axis == 1)
326  actual_position = myJoint->getActualAngle1();
327  else
328  actual_position = myJoint->getActualAngle2();
329  actual_position += play_position;
330 
331  switch (sMotor.type) {
332  case MOTOR_TYPE_PID:
333  {
334  if(desired_position>sMotor.maxValue)
335  desired_position = sMotor.maxValue;
336  if(desired_position<sMotor.minValue)
337  desired_position = sMotor.minValue;
338  /*
339  if(desired_position>2*M_PI) desired_position=0;
340  else if(desired_position>M_PI) desired_position=-2*M_PI+desired_position;
341  else if(desired_position<-2*M_PI) desired_position=0;
342  else if(desired_position<-M_PI) desired_position=2*M_PI+desired_position;
343  */
344  er = desired_position - actual_position;
345  if(er > M_PI)
346  er = -2*M_PI+er;
347  else
348  if(er < -M_PI)
349  er = 2*M_PI+er;
350 
351  integ_error += er*time;
352 
353  //anti wind up, this code limits the integral error
354  //part of the pid to the maximum velocity. This makes
355  //the pid react way faster. This also eleminates the
356  //overshooting errors seen before in the simulation
357  double iPart = integ_error * sMotor.i;
358  if(iPart > sMotor.maxSpeed)
359  {
360  iPart = sMotor.maxSpeed;
361  integ_error = sMotor.maxSpeed / sMotor.i;
362  }
363 
364  if(iPart < -sMotor.maxSpeed)
365  {
366  iPart = -sMotor.maxSpeed;
367  integ_error = -sMotor.maxSpeed / sMotor.i;
368  }
369 
370  // set desired velocity. @todo add inertia
371  vel = desired_velocity;
372  // P part of the motor
373  vel += er * sMotor.p;
374  // I part of the motor
375  vel += iPart;
376  // D part of the motor
377  vel += ((er - last_error)/time) * sMotor.d;
378  last_error = er;
379  }
380  break;
381  case MOTOR_TYPE_DC:
382  vel = actual_velocity;
383  break;
384  case MOTOR_TYPE_PID_FORCE:
385  if(desired_position>2*M_PI) desired_position=0;
386  else if(desired_position>M_PI) desired_position=-2*M_PI+desired_position;
387  else if(desired_position<-2*M_PI) desired_position=0;
388  else if(desired_position<-M_PI) desired_position=2*M_PI+desired_position;
389 
390  er = desired_position - actual_position;
391  if(er > M_PI) er = -2*M_PI+er;
392  else if(er < -M_PI) er = 2*M_PI+er;
393  integ_error += er*time;
394  // P part of the motor
395  torque = er * sMotor.p;
396  // I part of the motor
397  torque += integ_error * sMotor.i;
398  // D part of the motor
399  torque += ((er - last_error)/time) * sMotor.d;
400  last_error = er;
401  if(torque > sMotor.maxEffort) torque = sMotor.maxEffort;
402  else if(torque < -sMotor.maxEffort) torque = -sMotor.maxEffort;
403  break;
404  case MOTOR_TYPE_UNDEFINED:
405  break;
406  }
407 
408  if(sMotor.type == MOTOR_TYPE_PID_FORCE) {
409  if(sMotor.axis == 1) {
410  myJoint->setTorque(torque);
411  }
412  else if(sMotor.axis == 2) {
413  myJoint->setTorque2(torque);
414  }
415  }
416  else {
417  // calculate current
418  torque = myJoint->getMotorTorque();
419  joint_velocity = myJoint->getVelocity();
420  current = (kXY*fabs(torque*joint_velocity) +
421  kX*fabs(torque) +
422  kY*fabs(joint_velocity) + k);
423  if(current < 0.0)
424  current = 0.0;
425  if(vel > sMotor.maxSpeed)
426  vel = sMotor.maxSpeed;
427  else
428  if(vel < -sMotor.maxSpeed)
429  vel = -sMotor.maxSpeed;
430 
431  if(sMotor.axis == 1) {
432  myJoint->setVelocity(vel);
433  }
434  else if(sMotor.axis == 2) {
435  myJoint->setVelocity2(vel);
436  }
437  }
438  }
439  }
