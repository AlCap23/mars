36  double SpaceClimberCurrent(double* torque, double* velocity,
37  std::vector<interfaces::sReal>* c) {
38  return fabs((*c)[0]*fabs((*torque)*(*velocity)) +
39  (*c)[1]*fabs((*torque)) +
40  (*c)[2]*fabs((*velocity)) + (*c)[3]);
41  }

266  void SimMotor::runEffortController(sReal time) {
267  // limit to range of motion
268  controlValue = std::max(sMotor.minValue,
269  std::min(controlValue, sMotor.maxValue));
270 
271  if(controlValue > 2*M_PI)
272  controlValue = 0;
273  else if(controlValue > M_PI)
274  controlValue = -2*M_PI + controlValue;
275  else if(controlValue < -2*M_PI)
276  controlValue = 0;
277  else if(controlValue < -M_PI)
278  controlValue = 2*M_PI + controlValue;
279 
280  error = controlValue - *position;
281  if(error > M_PI) error = -2*M_PI + error;
282  else if(error < -M_PI) error = 2*M_PI + error;
283  integ_error += error * time;
284  // P part of the motor
285  effort = error * sMotor.p;
286  // I part of the motor
287  effort += integ_error * sMotor.i;
288  // D part of the motor
289  effort += ((error - last_error)/time) * sMotor.d;
290  last_error = error;
291  effort = std::max(-sMotor.maxEffort, std::min(effort, sMotor.maxEffort));
292  }

298  void SimMotor::runPositionController(sReal time) {
299  // the following implements a simple PID controller using the value
300  // pointed to by controlParameter
301 
302  controlValue = mimic_multiplier * controlValue + mimic_offset;
303 
304  // limit to range of motion
305  controlValue = std::max(sMotor.minValue,
306  std::min(controlValue, sMotor.maxValue));
307 
308  // calculate control values
309  error = controlValue - *position;
310  if (std::abs(error) < 0.000001)
311  error = 0.0;
312 
313  // FIXME: not sure if this makes sense, because it forbids turning
314  // motors in the same direction for multiple revolutions
315  // and can possibly lead to the motor turning in the wrong direction
316  // beyond its limit, as limit checking was done BEFORE??
317  //if(er > M_PI)
318  // er = -2*M_PI+er;
319  //else
320  //if(er < -M_PI)
321  // er = 2*M_PI+er;
322 
323  integ_error += error*time;
324 
325  //anti wind up, this code limits the integral error
326  //part of the pid to the maximum velocity. This makes
327  //the pid react way faster. This also eleminates the
328  //overshooting errors seen before in the simulation
329  double iPart = integ_error * sMotor.i;
330  if(iPart > sMotor.maxSpeed)
331  {
332  iPart = sMotor.maxSpeed;
333  integ_error = sMotor.maxSpeed / sMotor.i;
334  }
335 
336  if(iPart < -sMotor.maxSpeed)
337  {
338  iPart = -sMotor.maxSpeed;
339  integ_error = -sMotor.maxSpeed / sMotor.i;
340  }
341 
342  // set desired velocity. @todo add inertia
343  velocity = 0; // by setting a different value we could specify a minimum
344  // P part of the motor
345  velocity += error * sMotor.p;
346  // I part of the motor
347  velocity += iPart;
348  // D part of the motor
349  velocity += ((error - last_error)/time) * sMotor.d;
350  last_error = error;
351  }

353  void SimMotor::update(sReal time_ms) {
354  time = time_ms;// / 1000;
355  sReal play_position = 0.0;
356 
357  // if the attached joint does not exist (any more)
358  if (!myJoint) deactivate();
359 
360  if(active) {
361  // set play offset to 0
362  if(myPlayJoint) play_position = myPlayJoint->getPosition();
363 
364  refreshPosition();
365  *position += play_position;
366 
367  // call control function for current motor type
368  (this->*runController)(time_ms);
369 
370  // cap speed
371  tmpmaxspeed = getMomentaryMaxSpeed();
372  velocity = std::max(-tmpmaxspeed, std::min(velocity, tmpmaxspeed));
373  // cap effort
374  tmpmaxeffort = getMomentaryMaxEffort();
375  effort = std::max(-tmpmaxeffort, std::min(effort, tmpmaxeffort));
376  myJoint->setEffortLimit(tmpmaxeffort, axis);
377 
378  for(std::map<std::string, SimMotor*>::iterator it = mimics.begin();
379  it != mimics.end(); ++it) {
380  it->second->setControlValue(controlValue);
381  //it->second->setControlValue(*position);
382  }
383 
384 
385  // estimate motor parameters based on achieved status
386  estimateCurrent();
387  estimateTemperature(time_ms);
388 
389  // pass speed (position/speed control) or torque to the attached
390  // joint's setSpeed1/2 or setTorque1/2 methods
391  (myJoint->*setJointControlParameter)(*controlParameter, axis);
392  //for mimic in myJoint->mimics:
393  // mimic->*setJointControlParameter)(mimic_multiplier*controlParameter, axis);
394  }
395  }

404  void SimMotor::estimateTemperature(sReal time_ms) {
405  temperature = temperature - calcHeatDissipation(time_ms) + calcHeatProduction(time_ms);
406  }
407 
408  /*
409  * Calculates the heat energy dissipating in the update interval.
410  * Normally, heat transfer would be calculated as
411  * P = k*A*(Ti - Ta)/d
412  * where k is the thermal conductivity, A the surface area,
413  * d the distance (material thickness) and Ti the temperature inside
414  * and Ta the ambient temperature. Since k, A and d are all constant
415  * in our case, heatTransferCoefficient = k*A/d and thus:
416  */
417  sReal SimMotor::calcHeatDissipation(sReal time_ms) const {
418  return (heatTransferCoefficient * (temperature - ambientTemperature))*time_ms/1000.0;
419  }
420 
421  /*
422  * Calculates the heat energy lost from motor activity in the update interval.
423  * heatlossCoefficient would be the fraction of the energy the motor gets from the
424  * power supply which gets transformed into heat energy.
425  */
426  sReal SimMotor::calcHeatProduction(sReal time_ms) const {
427  return (current * voltage * heatlossCoefficient)*time_ms/1000.0;
428  }

438  void SimMotor::initCurrentEstimation() {
439  kXY = 100.0*((0.00002) / ((9.81*0.07)*(2*M_PI/60)));
440  kX = 0.00512 / (9.81*0.07);
441  kY = 100.0*(0.00006 / (2*M_PI/60));
442  k = 0.025;
443  std::vector<sReal>* spaceclimber_coefficients = new std::vector<sReal>;
444  spaceclimber_coefficients->push_back(kXY);
445  spaceclimber_coefficients->push_back(kX);
446  spaceclimber_coefficients->push_back(kY);
447  spaceclimber_coefficients->push_back(k);
448  current_coefficients = spaceclimber_coefficients;
449  current = 0;
450  }
