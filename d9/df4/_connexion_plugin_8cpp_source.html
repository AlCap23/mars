123  void ConnexionPlugin::preGraphicsUpdate() {
124  double data[7];
125  Quaternion q(1.0, 0.0, 0.0, 0.0);
126 
127  camMutex.lock();
128  sReal tmpCamState[7];
129  tmpCamState[0] = camState[0];
130  tmpCamState[1] = camState[1];
131  tmpCamState[2] = camState[2];
132  tmpCamState[3] = camState[3];
133  tmpCamState[4] = camState[4];
134  tmpCamState[5] = camState[5];
135  tmpCamState[6] = camState[6];
136  Vector trans(camState[0], camState[1], camState[2]);
137  Quaternion qRot(camState[6], camState[3], camState[4], camState[5]);
138  camState[0] = 0.0;
139  camState[1] = 0.0;
140  camState[2] = 0.0;
141  camState[3] = 0.0; //q.x
142  camState[4] = 0.0; //q.y
143  camState[5] = 0.0; //q.z
144  camState[6] = 1.0; //q.w
145  camMutex.unlock();
146 
147  if (object_mode == 1) {
148  interfaces::GraphicsWindowInterface *gw = control->graphics->get3DWindow(win_id);
149 
150  if (gw) {
151  if (resetCam) {
152  gw->getCameraInterface()->updateViewportQuat(tmpCamState[0],
153  tmpCamState[1],
154  tmpCamState[2],
155  tmpCamState[3],
156  tmpCamState[4],
157  tmpCamState[5],
158  tmpCamState[6]);
159  resetCam = false;
160  }
161  gw->getCameraInterface()->getViewportQuat(data, data+1, data+2,
162  data+3, data+4, data+5,
163  data+6);
164  q = Quaternion(data[6], data[3], data[4], data[5]);
165  trans = q*trans;
166  q = q * qRot;
167 
168  data[0] += trans.x();
169  data[1] += trans.y();
170  data[2] += trans.z();
171  gw->getCameraInterface()->updateViewportQuat(data[0], data[1], data[2],
172  q.x(), q.y(), q.z(), q.w());
173  }
174  }
175  else if (object_mode == 2) {
176  interfaces::GraphicsWindowInterface *gw = control->graphics->get3DWindow(win_id);
177 
178  if (gw) {
179  gw->getCameraInterface()->getViewportQuat(data, data+1, data+2,
180  data+3, data+4, data+5,
181  data+6);
182  q = Quaternion(data[6], data[3], data[4], data[5]);
183  trans = q*trans;
184  }
185  core_objects_exchange node;
186  control->nodes->getNodeExchange(object_id, &node);
187  Quaternion tmpQ(node.rot);
188  Vector tmpV = node.pos;
189 
190  //trans = QVRotate(tmpQ, trans);
191  Quaternion qi = q;
192  qi.x() *= -1;
193  qi.y() *= -1;
194  qi.z() *= -1;
195  qRot = q * qRot * qi;
196  //tmpQ = quad_state;
197  tmpQ = qRot * tmpQ;
198  //tmpQ = tmpQ*qi;
199 
200  tmpV += trans;
201  NodeData my_node;
202  my_node.index = object_id;
203  my_node.pos = tmpV;
204  my_node.rot = tmpQ;
205  control->nodes->editNode(&my_node, EDIT_NODE_POS | EDIT_NODE_MOVE_ALL);
206  control->nodes->editNode(&my_node, EDIT_NODE_ROT | EDIT_NODE_MOVE_ALL);
207  }
208  }

244  void ConnexionPlugin::run() {
245 
246  while (run_thread) {
247  getValue(motion, newValues);
248 
249  motion[0] *= sensitivity[0]*use_axis[0] * 0.01;
250  motion[1] *= sensitivity[1]*use_axis[1] * 0.01;
251  motion[2] *= sensitivity[2]*use_axis[2] * 0.01;
252  motion[3] *= sensitivity[3]*use_axis[3] * 0.001;
253  motion[4] *= sensitivity[4]*use_axis[4] * 0.001;
254  motion[5] *= sensitivity[5]*use_axis[5] * 0.001;
255 
256  updateCam(motion);
257 
258  if (newValues->button1 == 1) {
259  camReset();
260  }
261 
262  msleep(20);
263  }
264  LOG_INFO("%s: ... stopped",name.c_str());
265 
266  thread_closed = true;
267  }

270  void ConnexionPlugin::updateCam(sReal motion[6]) {
271  sReal tx, ty, tz;
272  sReal rx, ry, rz;
273 
274  Quaternion q1, q2, q3;
275  Vector move, x_axis, y_axis, z_axis;
276 
277  tx = motion[0]; //x-axis
278  ty = motion[1]; //y-axis
279  tz = motion[2]; //z-axis
280  rx = motion[3]; //tilt
281  ry = motion[4]; //roll
282  rz = motion[5]; //spin
283 
284  if (fabs(tx) < 0.0001) tx = 0.0;
285  if (fabs(ty) < 0.0001) ty = 0.0;
286  if (fabs(tz) < 0.0001) tz = 0.0;
287 
288  x_axis.x() = 1;
289  x_axis.y() = 0;
290  x_axis.z() = 0;
291 
292  y_axis.x() = 0;
293  y_axis.y() = 1;
294  y_axis.z() = 0;
295 
296  z_axis.x() = 0;
297  z_axis.y() = 0;
298  z_axis.z() = 1;
299 
300  camMutex.lock();
301  Quaternion tmpQuatState(camState[6], camState[3],
302  camState[4], camState[5]);
303 
304  x_axis = tmpQuatState * x_axis;
305  y_axis = tmpQuatState * y_axis;
306  z_axis = tmpQuatState * z_axis;
307 
308  qFromAxisAndAngle(q1, x_axis, rx);
309  qFromAxisAndAngle(q2, y_axis, ry);
310  qFromAxisAndAngle(q3, z_axis, -rz);
311 
312  tmpQuatState = q1 * q2* q3 * tmpQuatState;
313 
314  camState[3] = tmpQuatState.x();
315  camState[4] = tmpQuatState.y();
316  camState[5] = tmpQuatState.z();
317  camState[6] = tmpQuatState.w();
318 
319  x_axis *= tx;
320  y_axis *= ty;
321  z_axis *= -tz;
322 
323  move.x() = x_axis.x() + y_axis.x() + z_axis.x(); // left / right
324  move.y() = x_axis.y() + y_axis.y() + z_axis.y(); // forward / backward
325  move.z() = x_axis.z() + y_axis.z() + z_axis.z(); // up / down
326 
327  camState[0] += move.x(); //x-axis
328  camState[1] += move.y(); //y-axis
329  camState[2] += move.z(); //z-axis
330  camMutex.unlock();
331  }

348  void ConnexionPlugin::qFromAxisAndAngle(Quaternion &q, Vector vec,
349  sReal angle) {
350  sReal l = vec.x()*vec.x() + vec.y()*vec.y() + vec.z()*vec.z();
351  if (l > 0.0) {
352  angle *= 0.5;
353  q.w() = cos(angle);
354  l = sin(angle) * sqrt(l);
355  q.x() = vec.x()*l;
356  q.y() = vec.y()*l;
357  q.z() = vec.z()*l;
358  }
359  else {
360  q.w() = 1;
361  q.x() = 0;
362  q.y() = 0;
363  q.z() = 0;
364  }
365  }
