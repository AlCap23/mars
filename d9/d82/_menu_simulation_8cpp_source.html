159  void MenuSimulation::menuAction(int action, bool checked) {
160  (void)checked;
161 
162  switch (action) {
163  case GUI_ACTION_NODE_TREE: menu_nodes(); break;
164  case GUI_ACTION_JOINT_TREE: menu_joints(); break;
165  case GUI_ACTION_LIGHT_TREE: menu_lights(); break;
166  case GUI_ACTION_MOTOR_TREE: menu_motors(); break;
167  case GUI_ACTION_SENSOR_TREE: menu_sensors(); break;
168  case GUI_ACTION_CONTROLLER_TREE: menu_controllers(); break;
169  case GUI_ACTION_DISTANCE: menu_distance(); break;
170  case GUI_ACTION_SELECTION: menu_selection(); break;
171  case GUI_ACTION_JOYSTICK: menu_joystick(); break;
172  case GUI_ACTION_MOTOR_CONTROL: menu_motorControl(); break;
173  case GUI_ACTION_IMPORT_MESH: menu_importMesh(); break;
174  case GUI_ACTION_APPLY_FORCE: menu_applyForce(); break;
175  case GUI_ACTION_APPLY_TORQUE: menu_applyTorque(); break;
176  case GUI_ACTION_EDIT_GRAPHICS: menu_graphicsOptions(); break;
177  case GUI_ACTION_RESCALE_ENV: menu_rescaleEnvironment(); break;
178  case GUI_ACTION_SIM_START:
179  control->sim->StartSimulation();
180  /*
181  if(!(control->sim->isSimRunning()))
182  control->sim->startStopTrigger();
183  */
184  break;
185  case GUI_ACTION_SIM_PAUSE:
186  control->sim->StopSimulation();
187  /*
188  if(control->sim->isSimRunning())
189  control->sim->startStopTrigger();
190  */
191  break;
192  case GUI_ACTION_SIM_STEP:
193  if(!(control->sim->isSimRunning()))
194  control->sim->singleStep();
195  break;
196  default: break;
197  }
198  }
