117  void MenuSimulation::menuAction(int action, bool checked) {
118  (void)checked;
119 
120  switch (action) {
121  case GUI_ACTION_CONTROLLER_TREE: menu_controllers(); break;
122  case GUI_ACTION_DISTANCE: menu_distance(); break;
123  case GUI_ACTION_MOTOR_CONTROL: menu_motorControl(); break;
124  case GUI_ACTION_APPLY_FORCE: menu_applyForce(); break;
125  case GUI_ACTION_APPLY_TORQUE: menu_applyTorque(); break;
126  case GUI_ACTION_RESCALE_ENV: menu_rescaleEnvironment(); break;
127  case GUI_ACTION_SIM_START:
128  control->sim->StartSimulation();
129  /*
130  if(!(control->sim->isSimRunning()))
131  control->sim->startStopTrigger();
132  */
133  break;
134  case GUI_ACTION_SIM_PAUSE:
135  control->sim->StopSimulation();
136  /*
137  if(control->sim->isSimRunning())
138  control->sim->startStopTrigger();
139  */
140  break;
141  case GUI_ACTION_SIM_STEP:
142  if(!(control->sim->isSimRunning()))
143  control->sim->singleStep();
144  break;
145  default: break;
146  }
147  }
