151  void ObstacleGenerator::createObstacleField() {
152  // initial calculations
153  double obstacle_length = params["mean_obstacle_length"];
154  if (!bool_params["use_boxes"]) {obstacle_length = params["mean_obstacle_width"];}
155  double field_length = params["field_length"];
156  double field_width = params["field_width"];
157  if (bool_params["use_grid"]) {
158  field_length = obstacle_length * params["field_length"];
159  field_width = params["mean_obstacle_width"] * params["field_width"];
160  }
161 
162  // create inclined box as slope if an angle is given
163  if (bool_params["support_platform"]) {
164  double platform_length = 0, platform_x_position = 0;
165  if ((params["incline_angle"] > sigma) or (params["incline_angle"] < -sigma)) {
166  Vector boxposition(params["field_distance"] + 0.5 * cos(degToRad(params["incline_angle"])) * field_length + 0.5 * sin(degToRad(params["incline_angle"])),
167  0,
168  params["ground_level"] - 0.5 + sin(degToRad(params["incline_angle"])) * 0.5 * field_length + 0.5 * (1 - cos(degToRad(params["incline_angle"]))));
169  Vector boxsize(field_length, field_width, 1.0);
170  Vector boxorientation(0, -params["incline_angle"], 0);
171  NodeData box("incline", boxposition, eulerToQuaternion(boxorientation));
172  box.initPrimitive(NODE_TYPE_BOX, boxsize, 1.0);
173  box.material.texturename = textures["ground"];
174  if (textures["ground_bump"] != "") {
175  box.material.bumpmap = textures["ground_bump"];
176  }
177  if (textures["ground_norm"] != "") {
178  box.material.normalmap = textures["ground_norm"];
179  }
180  box.material.diffuseFront = Color(1.0, 1.0, 1.0, 1.0);
181  box.material.tex_scale = field_width/2.0;
182  control->nodes->addNode(&box, false);
183  oldNodeIDs.push_back(box.index);
184  // control->nodes->createPrimitiveNode("incline", NODE_TYPE_BOX,
185  // false, boxposition, boxsize, 1.0, eulerToQuaternion(boxorientation), false));
186  platform_length = 2.0 + params["field_distance"];
187  platform_x_position = -1.0 + 0.5 * params["field_distance"];
188  }
189  else { //if there is no inclined surface
190  platform_length = 2.0 + params["field_distance"] + field_length;
191  platform_x_position = -1.0 + 0.5 * (params["field_distance"] + field_length);
192  }
193 
194  Vector platformsize(platform_length, field_width, 1.0);
195  Vector platformposition(platform_x_position, 0.0, -0.5 + params["ground_level"]);
196  Vector platformorientation(0.0, 0.0, 0.0);
197  NodeData platform("platform", platformposition, eulerToQuaternion(platformorientation));
198  platform.initPrimitive(NODE_TYPE_BOX, platformsize, 1.0);
199  platform.material.texturename = textures["ground"];
200  if (textures["ground_bump"] != "") {
201  platform.material.bumpmap = textures["ground_bump"];
202  }
203  if (textures["ground_norm"] != "") {
204  platform.material.normalmap = textures["ground_norm"];
205  }
206  platform.material.diffuseFront = Color(1.0, 1.0, 1.0, 1.0);
207  //platform.material.tex_scale = field_width/2.0;
208  control->nodes->addNode(&platform, false);
209  oldNodeIDs.push_back(platform.index);
210  }
211  // create obstacles
212  if (bool_params["use_grid"]) {
213  params["obstacle_number"] = params["field_width"]*params["field_length"];
214  for (int w = 0; w < params["field_width"]; w++) {
215  for (int l = 0; l < params["field_length"]; l++) {
216  std::string name = "obstacle_";
217  std::string numstring = "";
218  std::istringstream(numstring) >> l >> w;
219  name = name.append(numstring);
220 
221  //create position
222  double pos_x = (0.5+l) * obstacle_length;
223  double pos_y = -0.5 * field_width + (0.5+w) * params["mean_obstacle_width"];
224 
225  //create size
226  double height = random_normal_number(params["mean_obstacle_height"], params["std_obstacle_height"],
227  params["min_obstacle_height"], params["max_obstacle_height"]);
228 
229  //create obstacle
230  createObstacle(name, pos_x, pos_y, params["mean_obstacle_width"], params["mean_obstacle_length"], height);
231  }
232  }
233  }
234  else {
235  int n = static_cast<int>(params["obstacle_number"]);
236  for (int i = 0; i < n; i++) {
237  std::string name = "obstacle_";
238  std::string numstring = "";
239  std::istringstream(numstring) >> i;
240  name = name.append(numstring);
241 
242  //create position
243  double pos_x = random_number(0, params["field_length"], 3);
244  double pos_y = random_number(-0.5 * params["field_width"], 0.5 * params["field_width"], 3);
245 
246  //create size
247  double height = random_normal_number(params["mean_obstacle_height"], params["std_obstacle_height"],
248  params["min_obstacle_height"], params["max_obstacle_height"]);
249  double radius = random_normal_number(params["mean_obstacle_width"], params["std_obstacle_width"],
250  params["min_obstacle_width"], params["max_obstacle_width"]);
251  double length = random_normal_number(params["mean_obstacle_length"], params["std_obstacle_width"],
252  params["min_obstacle_width"], params["max_obstacle_width"]);
253 
254  //create obstacle
255  createObstacle(name, pos_x, pos_y, radius, length, height);
256  }
257  }

260  void ObstacleGenerator::createObstacle(std::string name, double pos_x, double pos_y, double width, double length, double height) {
261  Quaternion orientation(1.0, 0.0, 0.0, 0.0);
262  double pos_z=params["ground_level"];
263  // NOTE: lots of geometrical problems related to rotation can be avoided if the object
264  // origins are on ground level, thus boxes just as capsules are twice their actual height
265  // thereby placing the origin on the ground. This should be irrelevant for most cases.
266  // Changes that would have to be made are marked with "2H".
267 
268  // 2H: raise boxes
269  //if (bool_params["use_boxes"]) {
270  // pos_z = 0.5 * height;
271  //}
272  if ((params["incline_angle"] > sigma) or (params["incline_angle"] < -sigma)) {
273  // The following was an experimental implementation to stretch inclined obstacles
274  // without support platform down to the ground level. Not working yet and unnecessary.
275  //if (bool_params["use_grid"] && !bool_params["support_platform"]) {
286  pos_z += sin(degToRad(params["incline_angle"])) * pos_x;
287  pos_x *= cos(degToRad(params["incline_angle"]));
288  //}
289  if (bool_params["incline_obstacles"]) {
290  Vector incline(0, -params["incline_angle"], 0);
291  orientation = eulerToQuaternion(incline);
292  // 2H correct for height difference of middle to front to avoid floating edges
293  // pos_z += (height/2.0)/cos(degToRad(params["incline_angle"])) - (height/2.0);
294  } else {
295  // 2H: correct for height difference due to rotation point not on ground level
296  //pos_z -= tan(degToRad(params["incline_angle"])) * 0.5 * length;
297  // 2H: correct for x position due to rotation point not on ground level
298  // pos_x -= ???
299  }
300  }
301  pos_x += params["field_distance"];
302  Vector position(pos_x, pos_y, pos_z);
303  Vector size(0, 0, 0);
304  if (bool_params["use_boxes"]) {
305  size[0] = length;
306  size[1] = width;
307  size[2] = height*2;
308  } else {
309  size[0] = width / 2.0;
310  if (height<width) {
311  size[1] = sigma;
312  } else {
313  size[1] = 2*height-width;
314  }
315  }
316  NodeData obstacle(name, position, orientation);
317  if (bool_params["use_boxes"]) {
318  obstacle.initPrimitive(NODE_TYPE_BOX, size, 1.0);
319  }
320  else {
321  obstacle.initPrimitive(NODE_TYPE_CAPSULE, size, 1.0);
322  }
323  obstacle.material.texturename = textures["obstacle"];
324  obstacle.material.diffuseFront = Color(1.0, 1.0, 1.0, 1.0);
325  if (textures["obstacle_bump"] != "") {
326  obstacle.material.bumpmap = textures["obstacle_bump"];
327  }
328  if (textures["obstacle_bump"] != "") {
329  obstacle.material.normalmap = textures["obstacle_norm"];
330  }
331  control->nodes->addNode(&obstacle, false);
332  oldNodeIDs.push_back(obstacle.index);
333  }
