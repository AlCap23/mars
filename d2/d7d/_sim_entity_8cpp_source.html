182  void SimEntity::setInitialPose(bool reset) {
183 
184  if(control && (config.find("rootNode") != config.end())) {
185  NodeId id = getNode((std::string)config["rootNode"]);
186  NodeData rootNode = control->nodes->getFullNode(id);
187  utils::Quaternion tmpQ(1, 0, 0, 0);
188  utils::Vector tmpV;
189  if(config.find("position") != config.end()) {
190  rootNode.pos.x() = config["position"][0];
191  rootNode.pos.y() = config["position"][1];
192  rootNode.pos.z() = config["position"][2];
193  control->nodes->editNode(&rootNode, EDIT_NODE_POS | EDIT_NODE_MOVE_ALL);
194  }
195  if(config.find("rotation") != config.end()) {
196  // check if euler angles or quaternion is provided; rotate around z
197  // if only one angle is provided
198  switch (config["rotation"].size()) {
199  case 1: tmpV[0] = 0;
200  tmpV[1] = 0;
201  tmpV[2] = config["rotation"][0];
202  tmpQ = utils::eulerToQuaternion(tmpV);
203  break;
204  case 3: tmpV[0] = config["rotation"][0];
205  tmpV[1] = config["rotation"][1];
206  tmpV[2] = config["rotation"][2];
207  tmpQ = utils::eulerToQuaternion(tmpV);
208  break;
209  case 4: tmpQ.x() = (sReal)config["rotation"][1];
210  tmpQ.y() = (sReal)config["rotation"][2];
211  tmpQ.z() = (sReal)config["rotation"][3];
212  tmpQ.w() = (sReal)config["rotation"][0];
213  break;
214  }
215  rootNode.rot = tmpQ;
216  control->nodes->editNode(&rootNode, EDIT_NODE_ROT | EDIT_NODE_MOVE_ALL);
217  }
218  // check if there is an anchor / parent (anchor is deprecated...)
219  std::string parentname = "";
220  if (config.hasKey("parent")) {
221  parentname << config["parent"];
222  }
223  if (config.hasKey("anchor") && (std::string)config["anchor"] != "none") { // backwards compatibility
224  parentname << config["anchor"];
225  }
226  if(!parentname.empty()) {
227  if (reset) {
228  fprintf(stderr, "Resetting initial entity pose.\n");
229  std::map<unsigned long, std::string>::iterator it;
230  JointId anchorJointId = 0;
231  for (it=jointIds.begin(); it!=jointIds.end(); ++it) {
232  if (it->second == "anchor_"+name) {
233  anchorJointId = it->first;
234  break;
235  }
236  }
237  SimJoint* anchorjoint = control->joints->getSimJoint(anchorJointId);
238  if (anchorjoint != NULL) {
239  anchorjoint->reattachJoint();
240  }
241  else fprintf(stderr, "Could not reset anchor of entity %s.\n", name.c_str());
242  }
243  else {
244  unsigned long parentid;
245  if (parentname == "world") {
246  parentid = 0;
247  } else { // if entity gets attached to other entity
248  std::string entityname, linkname;
249  unsigned int splitsignpos = parentname.find("::");
250  if (splitsignpos >= 0) {
251  entityname = parentname.substr(0, splitsignpos);
252  linkname = parentname.substr(splitsignpos+2);
253  SimEntity* parententity = control->entities->getEntity(entityname);
254  if (parententity != 0) {
255  unsigned long linkid = parententity->getNode(linkname);
256  if (linkid > 0) {
257  parentid = linkid;
258  } else
259  fprintf(stderr, "No valid link id in parent entity.\n");
260  } else
261  fprintf(stderr, "No valid entity in scene.\n");
262  }
263  }
264  JointData anchorjoint;
265  anchorjoint.nodeIndex1 = id;
266  anchorjoint.nodeIndex2 = parentid;
267  anchorjoint.type = JOINT_TYPE_FIXED;
268  anchorjoint.name = "anchor_"+name;
269  JointId anchorJointId = control->joints->addJoint(&anchorjoint);
270  addJoint(anchorJointId, anchorjoint.name);
271  }
272  }
273  // set Joints
274  configmaps::ConfigVector::iterator it;
275  configmaps::ConfigMap::iterator joint_it;
276  for (it = config["poses"].begin(); it!= config["poses"].end(); ++it) {
277  if ((std::string)(*it)["name"] == (std::string)config["pose"]) {
278  for (joint_it = (*it)["joints"].beginMap();
279  joint_it!= (*it)["joints"].endMap(); ++joint_it) {
280  //fprintf(stderr, "setMotorValue: joint: %s, id: %lu, value: %f\n", ((std::string)joint_it->first).c_str(), motorIDMap[joint_it->first], (double)joint_it->second);
281  control->motors->setMotorValue(getMotor(joint_it->first),
282  joint_it->second);
283  }
284  }
285  }
286  }
287  }
