93  void SelectionTree::fill(unsigned long id, QTreeWidgetItem *current) {
94  QStringList temp;
95 
96  for (unsigned int i = 0; i < simNodes.size(); i++) {
97  if (simNodes[i].index == id) {
98  bool found = false;
99  for (unsigned int k = 0; k < present.size() && !found; k++) {
100  if (present[k] == id) found = true;
101  }
102  if (found == false) {
103  temp << QString::number(id) + ":" + QString::fromStdString(simNodes[i].name);
104  QTreeWidgetItem *next = new QTreeWidgetItem(temp);
105  // todo: should read the selected state in the simulation
106  nodeItemMap[id] = next;
107  current->addChild(next);
108  current = next;
109  present.push_back(id);
110  }
111  simNodes.erase(simNodes.begin()+i);
112  break;
113  }
114  }
115 
116  vector<unsigned long> children = control->nodes->getConnectedNodes(id);
117  vector<unsigned long> newNodes;
118 
119  for(size_t i=0; i<children.size(); i++) {
120  bool found = false;
121  for(size_t k = 0; k<present.size() && !found; k++) {
122  if(present[k] == children[i]) {
123  found = true;
124  }
125  }
126  if(!found) {
127  newNodes.push_back(children[i]);
128  }
129  }
130 
131  for (unsigned int i = 0; i < newNodes.size(); i++) {
132  fill(newNodes[i], current);
133  }
134  }

520  void SelectionTree::updateNodeMap(ConfigMap &map) {
521  Vector p(map["position"]["x"], map["position"]["y"],
522  map["position"]["z"]);
523  Quaternion q;
524  q.x() = map["rotation"]["x"];
525  q.y() = map["rotation"]["y"];
526  q.z() = map["rotation"]["z"];
527  q.w() = map["rotation"]["w"];
528  sRotation r = quaternionTosRotation(q);
529  map["rotation"]["alpha"] = r.alpha;
530  map["rotation"]["beta"] = r.beta;
531  map["rotation"]["gamma"] = r.gamma;
532  if(!map.hasKey("cfdir1")) {
533  map["contact"]["cfdir1"]["x"] = 0.;
534  map["contact"]["cfdir1"]["y"] = 0.;
535  map["contact"]["cfdir1"]["z"] = 0.;
536  }
537  std::vector<std::string> move {"cmax_num_contacts", "cerp", "ccfm", "cfriction1", "cfriction2", "cmotion1", "cmotion2", "cfds1", "cfds2", "cbounce", "cbounce_vel", "capprox", "coll_bitmask", "cfdir1"};
538  for(auto it: move) {
539  if(map.hasKey(it)) {
540  map["contact"][it] = map[it];
541  map.erase(it);
542  }
543  }
544  move = {"material", "cullMask", "brightness", "pivot", "visualsize", "visualscale", "visualposition", "visualrotation", "filename", "origname", "shadow_id", "shadowcaster", "shadowreceiver"};
545  for(auto it: move) {
546  if(map.hasKey(it)) {
547  map["visual"][it] = map[it];
548  map.erase(it);
549  }
550  }
551  if((ulong)map["relativeid"] != 0) {
552  NodeData parent = control->nodes->getFullNode(map["relativeid"]);
553  // inverse calculation to convert world to relative
554  p = parent.rot.inverse() * (p-parent.pos);
555  q = parent.rot.inverse() * q;
556  }
557  map["pose"]["local"]["relativeid"] = map["relativeid"];
558  map.erase("relativeid");
559  map["pose"]["world"]["position"] = map["position"];
560  map.erase("position");
561  map["pose"]["world"]["rotation"] = map["rotation"];
562  map.erase("rotation");
563  r = quaternionTosRotation(q);
564  map["pose"]["local"]["rotation"]["x"] = q.x();
565  map["pose"]["local"]["rotation"]["y"] = q.y();
566  map["pose"]["local"]["rotation"]["z"] = q.z();
567  map["pose"]["local"]["rotation"]["w"] = q.w();
568  map["pose"]["local"]["rotation"]["alpha"] = r.alpha;
569  map["pose"]["local"]["rotation"]["beta"] = r.beta;
570  map["pose"]["local"]["rotation"]["gamma"] = r.gamma;
571  map["pose"]["local"]["position"]["x"] = p.x();
572  map["pose"]["local"]["position"]["y"] = p.y();
573  map["pose"]["local"]["position"]["z"] = p.z();
574  }

578  unsigned long id;
579  QList<QTreeWidgetItem*> selectedItems = treeWidget->selectedItems();
580  std::vector<unsigned long> ids;
581  for(size_t i=0; i<selectedItems.size(); ++i) {
582  n = selectedItems[i]->text(0).indexOf(":");
583  QTreeWidgetItem *parent = selectedItems[i]->parent();

585  while(parent->parent()) parent = parent->parent();

644  void SelectionTree::valueChanged(std::string name, std::string value) {
645  if(name.empty()) return;
646  // remove name prefix, otherwise the name could influence the pattern matching
647  // done in the edit mehtods
648  size_t p = name.find('/', 3);
649  if(p != std::string::npos) {
650  name = name.substr(p);
651  }
652  //fprintf(stderr, "get feedback: %s\n", name.c_str());
653  if(editCategory == 1) {
654  // convert local / global pose
655  if(matchPattern("*pose/", name) && !matchPattern("*relativeid", name)) {
656  if(matchPattern("*local/", name)) {
657  p = name.find("local/");
658  if(p!=std::string::npos) {
659  name = name.substr(p+5);
660  }
661  // calculate global pose
662  nodeData = control->nodes->getFullNode(nodeData.index);
663  ConfigMap map;
664  nodeData.toConfigMap(&map, false, true);
665  updateNodeMap(map);
666  ConfigMap &posMap = map["pose"]["local"]["position"];
667  ConfigMap &qMap = map["pose"]["local"]["rotation"];
668  Vector pos(posMap["x"], posMap["y"], posMap["z"]);
669  Quaternion q;
670  q.x() = qMap["x"];
671  q.y() = qMap["y"];
672  q.z() = qMap["z"];
673  q.w() = qMap["w"];
674  sRotation r = quaternionTosRotation(q);
675  double v = atof(value.c_str());
676  bool euler = false;
677  bool editPos = true;
678  if(matchPattern("*position/x", name)) pos.x() = v;
679  if(matchPattern("*position/y", name)) pos.y() = v;
680  if(matchPattern("*position/z", name)) pos.z() = v;
681  if(matchPattern("*rotation/x", name)) q.x() = v, editPos=false;
682  if(matchPattern("*rotation/y", name)) q.y() = v, editPos=false;
683  if(matchPattern("*rotation/z", name)) q.z() = v, editPos=false;
684  if(matchPattern("*rotation/w", name)) q.w() = v, editPos=false;
685  if(matchPattern("*rotation/alpha", name)) r.alpha = v, euler=true, editPos=false;
686  if(matchPattern("*rotation/beta", name)) r.beta = v, euler=true, editPos=false;
687  if(matchPattern("*rotation/gamma", name)) r.gamma = v, euler=true, editPos=false;
688  if(euler) {
689  q = eulerToQuaternion(r);
690  }
691  /* this transformation is already done in editNode
692  if((ulong)map["pose"]["local"]["relativeid"] != 0) {
693  NodeData parent = control->nodes->getFullNode(map["relativeid"]);
694  pos = parent.rot.inverse() * (pos-parent.pos);
695  q = parent.rot.inverse() * q;
696  }
697  */
698  nodeData.pos = pos;
699  nodeData.rot = q;
700  if(editPos) {
701  control->nodes->editNode(&nodeData, EDIT_NODE_POS);
702  }
703  else {
704  control->nodes->editNode(&nodeData, EDIT_NODE_ROT);
705  }
706  }
707  else {
708  p = name.find("world/");
709  if(p!=std::string::npos) {
710  name = name.substr(p+5);
711  }
712  control->nodes->edit(nodeData.index, name, value);
713  }
714  }
715  else {
716  control->nodes->edit(nodeData.index, name, value);
717  }
718  // todo: update map
719  nodeData = control->nodes->getFullNode(nodeData.index);
720  ConfigMap map;
721  nodeData.toConfigMap(&map, false, true);
722  updateNodeMap(map);
723  dw->updateConfigMap(nodeData.name, map);
724  }
725  else if(editCategory == 2) {
726  control->joints->edit(jointData.index, name, value);
727  }
728  else if(editCategory == 3) {
729  control->motors->edit(motorData.index, name, value);
730  }
731  else if(editCategory == 5) {
732  control->graphics->editMaterial(currentMaterial["name"], name, value);
733  // update material internal
734  std::vector<interfaces::MaterialData> mList;
735  mList = control->graphics->getMaterialList();
736  for(size_t i=0; i<mList.size(); ++i) {
737  if(mList[i].name == (std::string)currentMaterial["name"]) {
738  configmaps::ConfigMap map;
739  mList[i].toConfigMap(&map);
740  materialMap[mList[i].name] = map;
741  break;
742  }
743  }
744  }
745  else if(editCategory == 6) {
746  control->graphics->editLight(currentLight["index"], name, value);
747  std::vector<interfaces::LightData*> simLights;
748  control->graphics->getLights(&simLights);
749  for(size_t i=0; i<simLights.size(); ++i) {
750  if(simLights[i]->index == (unsigned long)currentLight["index"]) {
751  configmaps::ConfigMap map;
752  simLights[i]->toConfigMap(&map);
753  lightMap[simLights[i]->name] = map;
754  break;
755  }
756  }

759  control->graphics->edit(name, value);
