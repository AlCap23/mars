84  void SelectionTree::fill(unsigned long id, QTreeWidgetItem *current) {
85  QStringList temp;
86 
87  for (unsigned int i = 0; i < simNodes.size(); i++) {
88  if (simNodes[i].index == id) {
89  bool found = false;
90  for (unsigned int k = 0; k < present.size() && !found; k++) {
91  if (present[k] == id) found = true;
92  }
93  if (found == false) {
94  temp << QString::number(id) + ":" + QString::fromStdString(simNodes[i].name);
95  QTreeWidgetItem *next = new QTreeWidgetItem(temp);
96  // todo: should read the selected state in the simulation
97  nodeItemMap[id] = next;
98  current->addChild(next);
99  current = next;
100  present.push_back(id);
101  }
102  simNodes.erase(simNodes.begin()+i);
103  break;
104  }
105  }
106 
107  vector<unsigned long> children = control->nodes->getConnectedNodes(id);
108  vector<unsigned long> newNodes;
109 
110  for(size_t i=0; i<children.size(); i++) {
111  bool found = false;
112  for(size_t k = 0; k<present.size() && !found; k++) {
113  if(present[k] == children[i]) {
114  found = true;
115  }
116  }
117  if(!found) {
118  newNodes.push_back(children[i]);
119  }
120  }
121 
122  for (unsigned int i = 0; i < newNodes.size(); i++) {
123  fill(newNodes[i], current);
124  }
125  }

180  void SelectionTree::selectNodes(void) {
181  if(selectAllowed) { // handle selection state
182  int n;
183  unsigned long id;
184  QList<QTreeWidgetItem*> selectedItems = treeWidget->selectedItems();
185  std::vector<unsigned long> ids;
186  for(size_t i=0; i<selectedItems.size(); ++i) {
187  n = selectedItems[i]->text(0).indexOf(":");
188  QTreeWidgetItem *parent = selectedItems[i]->parent();
189  if(!parent) continue;
190  while(parent->parent()) parent = parent->parent();
191  if(parent->text(0) == "nodes") {
192  ids.push_back(selectedItems[i]->text(0).left(n).toULong());
193  }
194  }
195  bool selected;
196  /* todo: get drawid2 should be used, so far we assume every node
197  uses two ids */
198  unsigned long drawID;
199  for(size_t i=0; i<simNodes.size(); ++i) {
200  drawID = control->nodes->getDrawID(simNodes[i].index);
201  if(!drawID) continue;
202  selected = false;
203  for(size_t k=0; k<ids.size(); ++k) {
204  if(ids[k] == simNodes[i].index) {
205  selected = true;
206  ids.erase(ids.begin()+k);
207  break;
208  }
209  }
210  control->graphics->setDrawObjectSelected(drawID, selected);
211  control->graphics->setDrawObjectSelected(drawID+1, selected);
212  }
213  }
214 
215  { // handle config map gui
216  QTreeWidgetItem* currentItem = treeWidget->currentItem();
217  if(currentItem) {
218  QTreeWidgetItem *parent = currentItem->parent();
219  if(!parent) return;
220  while(parent->parent()) parent = parent->parent();
221  int n = currentItem->text(0).indexOf(":");
222  if(n>-1) {
223  unsigned long id = currentItem->text(0).left(n).toULong();
224  if(parent->text(0) == "nodes") {
225  std::vector<std::string> editPattern;
226  editPattern.push_back("*/position/*");
227  editPattern.push_back("*/extend/*");
228  editPattern.push_back("*/material");
229  editPattern.push_back("*/c*");
230  nodeData = control->nodes->getFullNode(id);
231  configmaps::ConfigMap map = nodeData.map;
232  if(!map.hasKey("cfdir1")) {
233  map["cfdir1"]["x"] = 0.;
234  map["cfdir1"]["y"] = 0.;
235  map["cfdir1"]["z"] = 0.;
236  }
237  nodeData.toConfigMap(&map, false, true);
238  dw->setConfigMap(nodeData.name, map, editPattern);
239  editCategory = 1;
240  }
241  else if(parent->text(0) == "joints") {
242  std::vector<std::string> editPattern;
243  // fake pattern to disable everthing
244  editPattern.push_back("//");
245  jointData = control->joints->getFullJoint(id);
246  configmaps::ConfigMap map;
247  jointData.toConfigMap(&map);
248  dw->setConfigMap(jointData.name, map, editPattern);
249  editCategory = 2;
250  }
251  else if(parent->text(0) == "motors") {
252  std::vector<std::string> editPattern;
253  // fake pattern to disable everthing
254  editPattern.push_back("//");
255  motorData = control->motors->getFullMotor(id);
256  configmaps::ConfigMap map;
257  motorData.toConfigMap(&map);
258  dw->setConfigMap(motorData.name, map, editPattern);
259  editCategory = 3;
260  }
261  }
262  else if(parent->text(0) == "controllers") {
263  editCategory = 4;
264  }
265  else if(parent->text(0) == "materials") {
266  std::vector<std::string> editPattern;
267  // fake pattern to disable everthing
268  editPattern.push_back("*/ambientColor/*");
269  editPattern.push_back("*/diffuseColor/*");
270  editPattern.push_back("*/specularColor/*");
271  editPattern.push_back("*/emissionColor/*");
272  editPattern.push_back("*/diffuseTexture");
273  editPattern.push_back("*/normalTexture");
274  //editPattern.push_back("*/displacementTexture");
275  editPattern.push_back("*/bumpNorFac");
276  editPattern.push_back("*/transparency");
277  editPattern.push_back("*/shininess");
278  editPattern.push_back("*/tex_scale");
279 
280  currentMaterial = materialMap[currentItem->text(0).toStdString()];
281  configmaps::ConfigMap map = defaultMaterial;
282  map.append(currentMaterial);
283  dw->setConfigMap(currentMaterial["name"], map,
284  editPattern);
285  editCategory = 5;
286  }

359  void SelectionTree::addCoreExchange(const std::vector<interfaces::core_objects_exchange> &objects, std::string category) {
360  std::vector<interfaces::core_objects_exchange>::const_iterator it;
361  std::vector<std::string> path;
362  std::vector<std::string>::iterator pt;
363  std::map<std::string, QTreeWidgetItem*> treeMap;
364  QTreeWidgetItem *current, *top, *next;
365  QStringList temp;
366  temp << category.c_str();
367  top = new QTreeWidgetItem(temp);
368  treeWidget->addTopLevelItem(top);
369  for(it=objects.begin(); it!=objects.end(); ++it) {
370  path = utils::explodeString('/', it->name);
371  current = top;
372  for(pt=path.begin(); pt!=path.end(); ++pt) {
373  temp.clear();
374  if(pt==path.end()-1) {
375  temp << QString::number(it->index) + ":" + QString::fromStdString(*pt);
376  current->addChild(new QTreeWidgetItem(temp));
377  }
378  else {
379  if(treeMap.find(*pt) == treeMap.end()) {
380  temp << QString::fromStdString(*pt);
381  next = new QTreeWidgetItem(temp);
382  current->addChild(next);
383  treeMap[*pt] = next;
384  }
385  current = treeMap[*pt];
386  }
387  }
388  }
