82  NodeType NodeData::typeFromString(const std::string& str) {
83  //search in array with strings
84  int type;
85  for (type = 0; type < NUMBER_OF_NODE_TYPES; ++type) {
86  if (str == sTypeNames[type])
87  return (NodeType)type;
88  }
89 
90  if(sscanf(str.c_str(), "%d", &type)) {
91  if(type > 0 && type < NUMBER_OF_NODE_TYPES) return (NodeType)type;
92  }
93 
94 
95  std::cerr << "NodeData::typeFromString(): unknown node type \"" << str << "\"." << std::endl;
96  throw("unknown Primitive type string");

108  if((it = config->find("mass")) != config->end()) {
109  mass = it->second[0].getDouble();
110  // use some epsilon here
111  if(fabs(mass) > 0.000000001) check = true;
112  }
113 
114  if((it = config->find("density")) != config->end()) {
115  density = it->second[0].getDouble();
116  // use some epsilon here

122

124  noPhysical = it->second[0].getBool();

127

129  movable = it->second[0].getBool();

132 
133  /* there are valid cases were pyhsical objects have no mass

135  if(!check) {
136  LOG_WARN("no mass nor density given for node %s.", name.c_str());

139

142  density = 0.0;

144  }
145  */
146  } // handle node mass
