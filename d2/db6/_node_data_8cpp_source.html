81  NodeType NodeData::typeFromString(const std::string& str) {
82  //search in array with strings
83  int type;
84  for (type = 0; type < NUMBER_OF_NODE_TYPES; ++type) {
85  if (str == sTypeNames[type])
86  return (NodeType)type;
87  }
88 
89  if(sscanf(str.c_str(), "%d", &type)) {
90  if(type > 0 && type < NUMBER_OF_NODE_TYPES) return (NodeType)type;
91  }
92

94  std::cerr << "NodeData::typeFromString(): unknown node type \"" << str << "\"." << std::endl;

98  bool NodeData::fromConfigMap(ConfigMap *config,
99  std::string filenamePrefix,
100  LoadCenter *loadCenter) {
101  ConfigMap::iterator it;
102  bool check = false, massDensity = false;
103  bool needMass = true;
104 
105  name = trim(config->get("name", name));
106 
107  { // handle node mass
108  if((it = config->find("mass")) != config->end()) {
109  mass = it->second;
110  // use some epsilon here
111  if(fabs(mass) > 0.000000001) check = true;
112  }
113 
114  if((it = config->find("density")) != config->end()) {
115  density = it->second;
116  // use some epsilon here
117  if(fabs(density) > 0.000000001) {
118  if(check) massDensity = true;
119  check = true;
120  }
121  }
122 
123  if((it = config->find("noPhysical")) != config->end()) {
124  noPhysical = it->second;
125  if(noPhysical) needMass = false;
126  }
127 
128  if((it = config->find("movable")) != config->end()) {
129  movable = it->second;
130  if(!movable) needMass = false;
131  }
132 
133  /* there are valid cases were pyhsical objects have no mass
134  if(needMass) {
135  if(!check) {
136  LOG_WARN("no mass nor density given for node %s.", name.c_str());
137  }
138  else if(massDensity) {
139 
140  LOG_WARN("mass and density given for node %s. Using only the mass.",
141  name.c_str());
142  density = 0.0;
143  }
144  }
145  */
146  } // handle node mass
147 
148  { // handle node type
149  if((it = config->find("physicmode")) != config->end()) {
150  std::string typeName = (std::string)it->second;
151  try {
152  physicMode = typeFromString(trim(typeName));
153  } catch(...) {
154  //throw SceneParseException("invalid physicmode for node", -1);
155  LOG_ERROR("could not get type for node: %s", name.c_str());
156  }
157  }
158 
159  origName = trim(config->get("origname", origName));
160  filename = trim(config->get("filename", filename));
161 
162  if(filename == "PRIMITIVE") {
163  if(origName.empty()) {
164  origName = toString(physicMode);
165  }
166  /* this is a valid case
167  else if(origName != toString(physicMode)) {
168  std::string tmp = toString(physicMode);
169  LOG_WARN("origname set to \"%s\" for primitive in node \"%s\" with physicMode \"%s\"",
170  origName.c_str(), name.c_str(),
171  tmp.c_str() );
172  }
173  */
174  }
175  } // handle node type

215 
216  if((it = config->find("visualsize")) != config->end()) {

218  }
219 
220  if((it = config->find("visualscale")) != config->end()) {
221  vectorFromConfigItem(it->second, &visual_scale);
222  }
223  else {
224  visual_scale = Vector(1.0, 1.0, 1.0);
225  }
226

269  if(!filenamePrefix.empty()) {
