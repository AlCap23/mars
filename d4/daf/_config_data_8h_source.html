355  inline bool testType(ItemType _type) {
356  if(type == UNDEFINED_TYPE) {
357  switch(_type) {
358  case INT_TYPE:
359  return parseInt();
360  case DOUBLE_TYPE:
361  return parseDouble();
362  case UINT_TYPE:
363  return parseUInt();
364  case ULONG_TYPE:
365  return parseULong();
366  case STRING_TYPE:
367  return parseString();
368  case BOOL_TYPE:

374  else if(type == _type) return true;
375  return false;

502  inline std::string toString() const {
503  char text[64];
504 
505  if(type == UNDEFINED_TYPE) {
506  return sValue;
507  } else if(type == INT_TYPE) {
508  sprintf(text, "%d", iValue);
509  return text;
510  } else if(type == UINT_TYPE) {
511  sprintf(text, "%u", uValue);
512  return text;
513  } else if(type == DOUBLE_TYPE) {
514  sprintf(text, "%g", dValue);
515  return text;
516  } else if(type == ULONG_TYPE) {
517  sprintf(text, "%lu", luValue);
518  return text;
519  } else if(type == STRING_TYPE) {
520  return sValue;
521  } else if(type == BOOL_TYPE) {
522  if(iValue) return "true";
523  else return "false";
524  }
525  return std::string();
526  }
