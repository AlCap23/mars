119  void MultiResHeightMapRenderer::prepare() {
120  //recalcSteps();
121  double **heightData = new double*[getLowResVertexCntY()];
122  for(int y = 0; y < getLowResVertexCntY(); ++y) {
123  heightData[y] = new double[getLowResVertexCntX()];
124  for(int x = 0; x < getLowResVertexCntX(); ++x) {
125  heightData[y][x] = -1;

138  int d = 1;

276 
277  GLuint *indices = (GLuint*)glMapBuffer(GL_ELEMENT_ARRAY_BUFFER,
278  GL_WRITE_ONLY);

280  dataMutex.lock();
281  if(vertices) {
282  memcpy(vertices, highVertexCopy, listSubTiles.size()*numVertices*sizeof(VertexData));

284  if(indices) {
285  memcpy(indices, highIndexCopy, highIndicesToDrawBuffer*sizeof(GLuint));

287  glUnmapBuffer(GL_ARRAY_BUFFER);
288  glUnmapBuffer(GL_ELEMENT_ARRAY_BUFFER);
289  glBindBuffer(GL_ARRAY_BUFFER, vboIds[0]);

305 
306  highIndicesToDraw = highIndicesToDrawBuffer;

308  glBindBuffer(GL_ARRAY_BUFFER, 0);

317  render(true);

321  void MultiResHeightMapRenderer::handleCamPos(double x, double y, Tile *tile) {
322  // get cell in main tile
323  //fprintf(stderr, "check: %lu\n", tile);
324  //fprintf(stderr, "%g %g %g %g\n", tile->xPos, tile->yPos, x, y);
325  std::vector<Tile*>::iterator it, it2;
326  double cellStepX = tile->width/3;
327  double cellStepY = tile->height/3;
328  double x2 = x - 0.5*cellStepX;
329  double y2 = y - 0.5*cellStepY;
330  int cellx = floor((-tile->xPos + x2) / cellStepX);
331  int celly = floor((-tile->yPos + y2) / cellStepY);
332  int cellx2 = floor((-tile->xPos + x + 0.5*cellStepX) / cellStepX);
333  int celly2 = floor((-tile->yPos + y + 0.5*cellStepY) / cellStepY);
334  std::vector<Tile*> tiles;
335 
336  // todo: num cell tiles is everywhere in this file hardcoded with 3
337  bool skip = false;
338  if(x-tile->xPos+0.5*cellStepX < 0 ||
339  x-tile->xPos-0.5*cellStepX > tile->width ||
340  y-tile->yPos+0.5*cellStepY < 0 ||
341  y-tile->yPos-0.5*cellStepY > tile->height) {
342  skip = true;
343  }
344  if(!skip) {
345  for(int i=0; i<4; ++i) {
346  int cy, cx;
347  if(i==0) {
348  cx = cellx;
349  cy = celly;
350  if(cx < 0 || cy < 0) continue;
351  if(cx > 2) cx = 2;
352  if(cy > 2) cy = 2;
353  }
354  else if(i==1) {
355  cx = cellx2;
356  cy = celly;
357  if(cy < 0) continue;
358  if(cx < 0) cx = 0;
359  if(cx > 2) continue;
360  if(cy > 2) cy = 2;
361  }
362  else if(i==2) {
363  cx = cellx;
364  cy = celly2;
365  if(cx < 0) continue;
366  if(cy < 0) cy = 0;
367  if(cy > 2) continue;
368  if(cx > 2) cx = 2;
369  }
370  else if(i==3) {
371  cx = cellx2;
372  cy = celly2;
373  if(cx > 2 || cy > 2) continue;
374  if(cx < 0) cx = 0;
375  if(cy < 0) cy=0;
376  }
377  // now check that we don't already have this cell in this loop
378  bool found = false;
379  for(it=tiles.begin(); it!=tiles.end(); ++it) {
380  if((*it)->x == cx && (*it)->y == cy) {
381  found = true;
382  break;
383  }
384  }
385  if(found) continue;
386  // now check wether the subtile already exists
387  for(it=tile->subTiles.begin(); it!=tile->subTiles.end(); ++it) {
388  if(*it != 0) {
389  if((*it)->x == cx && (*it)->y == cy) {
390  tiles.push_back(*it);
391  found = true;
392  break;
393  }
394  }
395  }
396  if(found) continue;
397  // else create it
398  cutTile(cx, cy, tile);
399  //fprintf(stderr, "create: cx: %d cy: %d px: %g py: %g w: %g %lu %g %g\n", cx, cy, tile->xPos, tile->yPos, tile->width/3, tile, x, y);
400  //if(depth < 6)
401  Tile *newTile = createSubTile(cx, cy, tile);
402  if(newTile) tiles.push_back(newTile);
403  else {
404  fprintf(stderr, "----- we just created an hole!!!!!!!\n");
405  }
406  }
407  }
408  // now remove the subtiles that are not any more in use
409  for(it=tile->subTiles.begin(); it!=tile->subTiles.end(); ++it) {
410  //fprintf(stderr, "check remove\n");
411  bool found = false;
412  for(it2 = tiles.begin(); it2 != tiles.end(); ++it2) {
413  if(*it2 == *it) {
414  found = true;
415  }
416  }
417  if(!found) {
418  drawPatch((*it)->x, (*it)->y, tile);
419  //fprintf(stderr, "clear: cx: %d cy: %d px: %g py: %g w: %g %lu\n", (*it)->x, (*it)->y, tile->xPos, tile->yPos, tile->width/3, tile);
420  clearTile(*it);
421  }
422 
423  }
424  tile->subTiles.swap(tiles);
425  }

557  double MultiResHeightMapRenderer::getHeight(double x, double y) {
558  // need to replaced by image
559  //if(!isInitialized) return 0;
560  int stepx = targetWidth / img->width;
561  int stepy = targetHeight / img->height;
562 
563  x = img->width*x/targetWidth;
564  y = img->height*y/targetHeight;
565  if(x < 0) x = 0;
566  if(y < 0) y = 0;
567  int gridX = floor(x);
568  int gridY = floor(y);
569  if(gridX > img->width-1) gridX = img->width-1;
570  if(gridY > img->height-1) gridY = img->height-1;
571  int gridX2 = gridX+1;
572  if(gridX2 > img->width-1) gridX2 = gridX;
573  int gridY2 = gridY+1;
574  if(gridY2 > img->height-1) gridY2 = gridY;
575  double dx = (x - gridX);
576  double dy = (y - gridY);
577  double cornerHeights[4];
578  double imageMaxValue = pow(2., img->depth);
579  CvScalar s;
580  s = cvGet2D(img, gridY, gridX);
581  cornerHeights[0] = s.val[0] / imageMaxValue;
582  s = cvGet2D(img, gridY2, gridX);
583  cornerHeights[1] = s.val[0] / imageMaxValue;
584  s = cvGet2D(img, gridY, gridX2);
585  cornerHeights[2] = s.val[0] / imageMaxValue;
586  s = cvGet2D(img, gridY2, gridX2);
587  cornerHeights[3] = s.val[0] / imageMaxValue;
588  double height = (cornerHeights[0] * (1-dx) * (1-dy) +
589  cornerHeights[1] * (1-dx) * dy +
590  cornerHeights[2] * dx * (1-dy) +
591  cornerHeights[3] * dx * dy);
592  return height*scaleZ;
593  // return mainTile->getHeight(x, y);
594  }

596  double Tile::getHeight(double x, double y) {
597  //if(!isInitialized) return 0;
598  double x2 = (x-xPos)/width*cols;
599  double y2 = (y-yPos)/height*rows;
600  if(x2 < 0) x2 = 0;
