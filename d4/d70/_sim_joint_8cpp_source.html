203  void SimJoint::update(sReal calc_ms){
204  CPP_UNUSED(calc_ms);
205  if (my_interface) {
206  // update the position and rotation of the node
207  actualAngle1 = (sJoint.angle1_offset + invert*my_interface->getPosition());
208  actualAngle2 = (sJoint.angle2_offset + invert*my_interface->getPosition2());
209 
210  my_interface->getAnchor(&anchor);
211  my_interface->getAxis(&axis1);
212  my_interface->getAxis2(&axis2);
213  my_interface->getForce1(&f1);
214  my_interface->getForce2(&f2);
215  my_interface->getTorque1(&t1);
216  my_interface->getTorque2(&t2);
217  my_interface->update();
218  my_interface->getAxisTorque(&axis1_torque);
219  my_interface->getAxis2Torque(&axis2_torque);
220  my_interface->getJointLoad(&joint_load);
221  axis1_torque *= invert;
222  axis2_torque *= invert;
223  joint_load *= invert;
224  speed1 = invert*my_interface->getVelocity();
225  speed2 = invert*my_interface->getVelocity2();
226  motor_torque = invert*my_interface->getMotorTorque();
227  }
228  }

230  void SimJoint::setSJoint(const JointData &sJoint) {
231  this->sJoint = sJoint;
232  id = sJoint.index;
233  anchor = sJoint.anchor;
234  axis1 = sJoint.axis1;
235  axis2 = sJoint.axis2;
236  actualAngle1 = sJoint.angle1_offset;
237  actualAngle2 = sJoint.angle2_offset;
238  f1.x() = f1.y() = f1.z() = 0;
239  f2.x() = f2.y() = f2.z() = 0;
240  t1.x() = t1.y() = t1.z() = 0;
241  t2.x() = t2.y() = t2.z() = 0;
242  axis1_torque.x() = axis1_torque.y() = axis1_torque.z() = 0;
243  axis2_torque.x() = axis2_torque.y() = axis2_torque.z() = 0;
244  joint_load.x() = joint_load.y() = joint_load.z() = 0;
245  speed1 = speed2 = 0;
246  motor_torque = 0;
247  lowStop1 = sJoint.lowStopAxis1;
248  highStop1 = sJoint.highStopAxis1;
249  lowStop2 = sJoint.lowStopAxis2;
250  highStop2 = sJoint.highStopAxis2;
251  if(sJoint.invertAxis) {
252  invert = -1;
253  }
254  else {
255  invert = 1;
256  }

342  void SimJoint::reattacheJoint(void) {
343  Vector pos;
344 
345  sJoint.angle1_offset = actualAngle1;
346  sJoint.angle2_offset = actualAngle2;
347  if(sJoint.anchorPos == ANCHOR_NODE1) {
348  pos = snode1->getPosition();
349  setAnchor(pos);
350  }
351  else if(sJoint.anchorPos == ANCHOR_NODE2) {
352  pos = snode2->getPosition();
353  setAnchor(pos);
354  }
355  else if(sJoint.anchorPos == ANCHOR_CENTER) {
356  pos = (snode1->getPosition() + snode2->getPosition()) / 2.;
357  setAnchor(pos);
358  }
359  else if(my_interface) {
360  my_interface->reattacheJoint();
361  my_interface->getAnchor(&sJoint.anchor);
362  }
363 
364  sJoint.axis1 = snode1->getRotation()*axis1InNode1;
365  if(my_interface) my_interface->setAxis(sJoint.axis1);
366 
367  node1ToAnchor = sJoint.anchor - snode1->getPosition();
368  node1ToAnchor = snode1->getRotation().inverse()*node1ToAnchor;
369  }

423  void SimJoint::setOfflineValue(sReal value) {
424  if(snode2) {
425  sReal tmp = value;
426  value -= actualAngle1;
427  actualAngle1 = tmp;
428 
429  Vector pivot = snode1->getPosition()+snode1->getRotation()*node1ToAnchor;
430  Vector axis = snode1->getRotation()*axis1InNode1;
431 
432  if(sJoint.type == JOINT_TYPE_HINGE) {
433  Quaternion q = angleAxisToQuaternion(-value*invert, axis);
434  control->nodes->rotateNode(snode2->getID(), pivot, q, sJoint.index);
435  }
436  else if(sJoint.type == JOINT_TYPE_SLIDER) {
437  Vector pos2 = snode2->getPosition();
438  axis *= invert*value / axis.norm();
439  pos2 += axis;
440  control->nodes->positionNode(snode2->getID(), pos2, sJoint.index);
441  }
442  }
443  }
