80  void DialogJoystick::moveSlot(double x, double y, double strength) {
81  if (x == 0 && y == 0)
82  return;
83 
84  std::vector<unsigned long> selectedIds = nodeList->selectedNodes();
85 
86  int changes = interfaces::EDIT_NODE_POS;
87  if (move_all->value().toBool())
88  changes |= interfaces::EDIT_NODE_MOVE_ALL;
89  interfaces::NodeData node;
90  for (unsigned int i = 0; i < selectedIds.size(); i++) {
91  node = control->nodes->getFullNode(selectedIds[i]);
92  switch (plane->value().toInt()) {
93  case 0:
94  node.pos.x() += x*strength;
95  node.pos.y() += y*strength;
96  break;
97  case 1:
98  node.pos.z() += y*strength;
99  node.pos.y() += x*strength;
100  break;
101  case 2:
102  node.pos.x() += x*strength;
103  node.pos.z() += y*strength;
104  break;
105  case 3: // custom plane
106  break;
107  default:
108  break;
109  }
110  control->nodes->editNode(&node, changes);
111  }
112  }
