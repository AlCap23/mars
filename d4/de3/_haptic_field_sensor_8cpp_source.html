169  void HapticFieldSensor::receiveData(const data_broker::DataInfo &info,
170  const data_broker::DataPackage &package, int callbackParam) {
171  if (contactForceIndex == -1) {
172  contactForceIndex = package.getIndexByName("contactForce");
173  contactIndex = package.getIndexByName("contact");
174  }
175  package.get(contactForceIndex, &contactForce);
176  package.get(contactIndex, &contact);
177  if (contact) {
178  computeForces();
179  } else {
180  for (std::vector<double>::iterator it = forces.begin(); it < forces.end(); ++it) {
181  *it = 0;
182  }
183  }
184 
185  if (positionIndices[0] == -1) {
186  positionIndices[0] = package.getIndexByName("position/x");
187  positionIndices[1] = package.getIndexByName("position/y");
188  positionIndices[2] = package.getIndexByName("position/z");
189  rotationIndices[0] = package.getIndexByName("rotation/x");
190  rotationIndices[1] = package.getIndexByName("rotation/y");
191  rotationIndices[2] = package.getIndexByName("rotation/z");
192  rotationIndices[3] = package.getIndexByName("rotation/w");
193  }
194  for (int i = 0; i < 3; ++i) {
195  package.get(positionIndices[i], &position[i]);
196  }
197 
198  package.get(rotationIndices[0], &orientation.x());
199  package.get(rotationIndices[1], &orientation.y());
200  package.get(rotationIndices[2], &orientation.z());
201  package.get(rotationIndices[3], &orientation.w());
202 
203  haveUpdate = true;

253  void HapticFieldSensor::computeForces() {
254  // FIXME: add mutex here?
255  double weightSum = 0;
256  double weight = 0;
257  double distance = 0;
258  Vector pos, ray;
259  int i = 0;
260  //fprintf(stderr, "weights:\n");
261  for (int c = 0; c < config.cols; ++c) {
262  for (int r = 0; r < config.rows; ++r) {
263  i = c*config.rows+r;
264  pos = position + orientation*(sensorpoints.at(i));
265  ray = orientation*this->ray;
266  distance = control->sim->getPhysics()->getVectorCollision(pos, ray);
267  weight = 1 - distance/maxDistance; // = (maxDistance-distance)/maxDistance
268  weights.at(c*config.rows+r) = weight;
269 // fprintf(stderr, "%6g ", weight);
270  weightSum += weight;
271  }
272 // fprintf(stderr, "\n");
273  }
274  double forceQuant = contactForce / weightSum;
275  for (size_t i = 0; i < forces.size(); ++i) {
276  forces.at(i) = weights.at(i) * forceQuant;
277  }
278  }
