302  BaseSensor* SensorManager::createAndAddSensor(const std::string &type_name,
303  BaseConfig *config,
304  bool reload){
305  assert(config);
306  std::map<const std::string,BaseSensor* (*)(ControlCenter*, BaseConfig*)>::iterator it = availableSensors.find(type_name);
307  if(it == availableSensors.end()){
308  std::cerr << "Could not load unknown Sensor with name: \"" << type_name.c_str()<< "\"" << std::endl;
309  return 0;
310  }
311 
312  int id = -1;
313  iMutex.lock();
314  id = next_sensor_id++;
315  iMutex.unlock();
316 
317  if(config->name.empty()){
318  std::stringstream str;
319  str << "SENSOR-" << id;
320  config->name = str.str();
321  }
322  config->id = id;
323  BaseSensor *sensor = ((*it).second)(this->control,config);
324  iMutex.lock();
325  simSensors[id] = sensor;
326  iMutex.unlock();
327 
328  if(!reload) {
329  simSensorsReload.push_back(SensorReloadHelper(type_name, config));
330  }
331 
332  return sensor;
333  }
