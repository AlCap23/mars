65  double NodeConstraint::getAttribute() const {
66  double attr = 0.;
67  interfaces::NodeData n = control->nodes->getFullNode(nodeId);
68  if(n.relative_id) {
69  interfaces::NodeData rel = control->nodes->getFullNode(n.relative_id);
70  interfaces::getRelFromAbs(rel, &n);
71  }
72  utils::sRotation rot = utils::quaternionTosRotation(n.rot);
73  switch(nodeAttribute) {
74  case ATTRIBUTE_POSITION_X:
75  attr = n.pos.x();
76  break;
77  case ATTRIBUTE_POSITION_Y:
78  attr = n.pos.y();
79  break;
80  case ATTRIBUTE_POSITION_Z:
81  attr = n.pos.z();
82  break;
83  case ATTRIBUTE_ROTATION_X:
84  attr = rot.alpha;
85  break;
86  case ATTRIBUTE_ROTATION_Y:
87  attr = rot.beta;
88  break;
89  case ATTRIBUTE_ROTATION_Z:
90  attr = rot.gamma;
91  break;
92  case ATTRIBUTE_SIZE_X:
93  attr = n.ext.x();
94  break;
95  case ATTRIBUTE_SIZE_Y:
96  attr = n.ext.y();
97  break;
98  case ATTRIBUTE_SIZE_Z:
99  attr = n.ext.z();
100  break;
101  default:
102  LOG_ERROR("NodeConstraint: attribute %d not supported.",
103  nodeAttribute);
104  break;
105  }
106  return attr + offset;
107  }

109  void NodeConstraint::setAttribute(double value) {
110  value -= offset;
111  interfaces::NodeData n = control->nodes->getFullNode(nodeId);
112  if(n.relative_id) {
113  interfaces::NodeData rel = control->nodes->getFullNode(n.relative_id);
114  interfaces::getRelFromAbs(rel, &n);
115  }
116  utils::sRotation rot = utils::quaternionTosRotation(n.rot);
117  switch(nodeAttribute) {
118  case ATTRIBUTE_POSITION_X:
119  {
120  n.pos.x() = value;
121  control->nodes->editNode(&n, interfaces::EDIT_NODE_POS);
122  break;
123  }
124  case ATTRIBUTE_POSITION_Y:
125  {
126  n.pos.y() = value;
127  control->nodes->editNode(&n, interfaces::EDIT_NODE_POS);
128  break;
129  }
130  case ATTRIBUTE_POSITION_Z:
131  {
132  n.pos.z() = value;
133  control->nodes->editNode(&n, interfaces::EDIT_NODE_POS);
134  break;
135  }
136  case ATTRIBUTE_ROTATION_X:
137  {
138  rot.alpha -= value;
139  rot.beta = rot.gamma = 0.;
140  utils::Quaternion tmp = utils::eulerToQuaternion(rot);
141  n.rot = (n.rot * tmp).normalized();
142  control->nodes->editNode(&n, interfaces::EDIT_NODE_ROT);
143  break;
144  }
145  case ATTRIBUTE_ROTATION_Y:
146  {
147  rot.beta -= value;
148  rot.alpha = rot.gamma = 0.;
149  utils::Quaternion tmp = utils::eulerToQuaternion(rot);
150  n.rot = (n.rot * tmp).normalized();
151  control->nodes->editNode(&n, interfaces::EDIT_NODE_ROT);
152  break;
153  }
154  case ATTRIBUTE_ROTATION_Z:
155  {
156  rot.gamma -= value;
157  rot.alpha = rot.beta = 0.;
158  utils::Quaternion tmp = utils::eulerToQuaternion(rot);
159  n.rot = (n.rot * tmp).normalized();
160  control->nodes->editNode(&n, interfaces::EDIT_NODE_ROT);
161  break;
162  }
163  case ATTRIBUTE_SIZE_X:
164  {
165  n.ext.x() = value;
166  control->nodes->editNode(&n, interfaces::EDIT_NODE_SIZE);
167  break;
168  }
169  case ATTRIBUTE_SIZE_Y:
170  {
171  n.ext.y() = value;
172  control->nodes->editNode(&n, interfaces::EDIT_NODE_SIZE);
173  break;
174  }
175  case ATTRIBUTE_SIZE_Z:
176  {
177  n.ext.z() = value;
178  control->nodes->editNode(&n, interfaces::EDIT_NODE_SIZE);
179  break;
180  }
181  default:
182  LOG_ERROR("NodeConstraint: attribute %d not supported.", nodeAttribute);
183  }
184  }

186  void NodeConstraint::cfgUpdateProperty(cfgPropertyStruct property) {
187  if(property.propertyType != cfg_manager::doubleProperty) {
188  LOG_ERROR("ConstraintPlugin: Currently only double properties are supported.");
189  return;
190  }
191 
192  double currentValue = getAttribute();
193  double diff = (property.dValue - oldValue) * nodeFactor;
194  setAttribute(currentValue + diff);
195  oldValue = property.dValue;
196  }

198  void NodeConstraint::reset() {
199  double newValue = 0.;
200  if(control->cfg) {
201  control->cfg->getPropertyValue(paramId, "value", &newValue);
202  }
203  double currentValue = getAttribute();
204  double diff = (newValue - refValue) * nodeFactor;
205  setAttribute(currentValue + diff);
206  oldValue = newValue;
207  }
