125  inline bool testType(ItemType _type) {
126  if(type == UNDEFINED_TYPE) {
127  switch(_type) {

272  inline std::string toString() const {
273  char text[64];
274 
275  if(type == UNDEFINED_TYPE) {
276  return sValue;
277  } else if(type == INT_TYPE) {
278  sprintf(text, "%d", iValue);
279  return text;
280  } else if(type == UINT_TYPE) {
281  sprintf(text, "%u", uValue);
282  return text;
283  } else if(type == DOUBLE_TYPE) {
284  sprintf(text, "%g", dValue);
285  return text;
286  } else if(type == ULONG_TYPE) {
287  sprintf(text, "%lu", luValue);
288  return text;
289  } else if(type == STRING_TYPE) {
290  return sValue;
291  } else if(type == BOOL_TYPE) {
292  if(iValue) return "true";
293  else return "false";
294  }
295  return std::string();
296  }

365  inline bool parseBool() {
366  if(type != UNDEFINED_TYPE) {
367  throw std::runtime_error("ConfigAtom parsing wrong type line ...");
368  }
369  sValue = trim(sValue);
370  if(sValue == "true" || sValue == "True" || sValue == "TRUE") {
371  iValue = 1;
372  return parsed = true;
373  }
374 
375  if(sValue == "false" || sValue == "False" || sValue == "FALSE") {
376  iValue = 0;
377  return parsed = true;
378  }
379 
380  return parsed = sscanf(sValue.c_str(), "%d", &iValue);
381  }
